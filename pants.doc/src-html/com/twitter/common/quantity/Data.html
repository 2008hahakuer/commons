018    package com.twitter.common.quantity;
019    
020    /**
021     * Provides a unit to allow conversions and unambiguous passing around of data {@link Amount}s.
022     * The kilo/mega/giga/... hierarchy is built on base 2 so that the hierarchy increases by a factor
023     * of 1024 instead of 1000 as typical in metric units.  Additionally, units are divided in 2
024     * hierarchies one based on bits and the other on bytes.  Thus {@link #Kb} represents kilobits; so
025     * 1 Kb = 1024 bits, and {@link #KB} represents kilobytes so 1 KB = 1024 bytes or 8192 bits.
026     */
027    public enum Data implements Unit<Data> {
028      BITS(1),
029      Kb(1024, BITS),
030      Mb(1024, Kb),
031      Gb(1024, Mb),
032      BYTES(8, BITS),
033      KB(1024, BYTES),
034      MB(1024, KB),
035      GB(1024, MB),
036      TB(1024, GB),
037      PB(1024, TB);
038    
039      private final double multiplier;
040    
041      private Data(double multiplier) {
042        this.multiplier = multiplier;
043      }
044    
045      private Data(double multiplier, Data base) {
046        this(multiplier * base.multiplier);
047      }
048    
049      @Override
050      public double multiplier() {
051        return multiplier;
052      }
053    }
