018    package com.twitter.common.util.logging;
019    
020    import com.google.common.base.Preconditions;
021    
022    import java.io.IOException;
023    import java.io.InputStream;
024    import java.util.logging.LogManager;
025    
026    /**
027     * A custom java.util.logging configuration class that loads the logging configuration from a
028     * properties file resource (as opposed to a file as natively supported by LogManager via
029     * java.util.logging.config.file).  By default this configurator will look for the resource at
030     * /logging.properties but the resource path can be overridden by setting the system property with
031     * key {@link #LOGGING_PROPERTIES_RESOURCE_PATH java.util.logging.config.resource}.  To install this
032     * configurator you must specify the following system property:
033     * java.util.logging.config.class=com.twitter.common.util.logging.ResourceLoggingConfigurator
034     *
035     * @author John Sirois
036     */
037    public class ResourceLoggingConfigurator {
038    
039      /**
040       * A system property that controls where ResourceLoggingConfigurator looks for the logging
041       * configuration on the process classpath.
042       */
043      public static final String LOGGING_PROPERTIES_RESOURCE_PATH = "java.util.logging.config.resource";
044    
045      public ResourceLoggingConfigurator() throws IOException {
046        String loggingPropertiesResourcePath =
047            System.getProperty(LOGGING_PROPERTIES_RESOURCE_PATH, "/logging.properties");
048        InputStream loggingConfig = getClass().getResourceAsStream(loggingPropertiesResourcePath);
049        Preconditions.checkNotNull(loggingConfig,
050            "Could not locate logging config file at resource path: %s", loggingPropertiesResourcePath);
051        LogManager.getLogManager().readConfiguration(loggingConfig);
052      }
053    }
