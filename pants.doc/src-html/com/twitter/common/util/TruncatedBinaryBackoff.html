018    package com.twitter.common.util;
019    
020    import com.google.common.base.Preconditions;
021    import com.twitter.common.quantity.Amount;
022    import com.twitter.common.quantity.Time;
023    
024    /**
025     * A BackoffStrategy that implements truncated binary exponential backoff.
026     */
027    public class TruncatedBinaryBackoff implements BackoffStrategy {
028      private final long initialBackoffMs;
029      private final long maxBackoffIntervalMs;
030    
031      /**
032       * Creates a new TruncatedBinaryBackoff that will start by backing off for {@code initialBackoff}
033       * and then backoff of twice as long each time its called until reaching the {@code maxBackoff} at
034       * which point it will always wait for that amount of time.
035       *
036       * @param initialBackoff the intial amount of time to backoff
037       * @param maxBackoff the maximum amount of time to backoff
038       */
039      public TruncatedBinaryBackoff(Amount<Long, Time> initialBackoff,
040          Amount<Long, Time> maxBackoff) {
041        Preconditions.checkNotNull(initialBackoff);
042        Preconditions.checkNotNull(maxBackoff);
043        Preconditions.checkArgument(initialBackoff.getValue() > 0);
044        Preconditions.checkArgument(maxBackoff.compareTo(initialBackoff) >= 0);
045        initialBackoffMs = initialBackoff.as(Time.MILLISECONDS);
046        maxBackoffIntervalMs = maxBackoff.as(Time.MILLISECONDS);
047      }
048    
049      @Override
050      public long calculateBackoffMs(long lastBackoffMs) {
051        Preconditions.checkArgument(lastBackoffMs >= 0);
052        return (lastBackoffMs == 0) ? initialBackoffMs
053            : Math.min(maxBackoffIntervalMs, lastBackoffMs * 2);
054      }
055    }
