028    package com.twitter.common.util;
029    
030    /**
031     * Generic range normalizer class. Values must be positive.
032     *
033     * @author Abdur Chowdhury
034     */
035    public class RangeNormalizer {
036      public RangeNormalizer(double minA, double maxA, double minB, double maxB) {
037        _minA = minA;
038        _maxA = maxA;
039        _minB = minB;
040        _maxB = maxB;
041        _denominator = (_maxA - _minA);
042        _B = (_maxB - _minB);
043        _midB = minB + (_B / 2f);
044      }
045    
046      public double normalize(double value) {
047        // if no input range, return a mid range value
048        if (_denominator == 0) {
049          return _midB;
050        }
051    
052        return ((value - _minA) / _denominator) * _B + _minB;
053      }
054    
055      public static double normalize(double value, double minA, double maxA, double minB, double maxB) {
056        // if the source min and max are equal, don't return 0, return something
057        // in the target range (perhaps this "default" should be another argument)
058        if (minA == maxA) {
059          return minB;
060        }
061    
062        return ((value - minA) / (maxA - minA)) * (maxB - minB) + minB;
063      }
064    
065      public static float normalizeToStepDistribution(double rating) {
066        int integerRating = (int) Math.round(rating);
067    
068        if (integerRating == 2) {
069          integerRating = 1;
070        } else if (integerRating == 4) {
071          integerRating = 3;
072        } else if (integerRating == 6) {
073          integerRating = 5;
074        } else if (integerRating == 8) {
075          integerRating = 7;
076        } else if (integerRating == 9) {
077          integerRating = 10;
078        }
079    
080        return (float) integerRating;
081      }
082    
083      // *******************************************************************
084      private double _denominator;
085      private double _B;
086      private double _minA = Double.MIN_VALUE;
087      private double _maxA = Double.MAX_VALUE;
088      private double _minB = Double.MIN_VALUE;
089      private double _maxB = Double.MAX_VALUE;
090      private double _midB = Double.MAX_VALUE;
091    }
