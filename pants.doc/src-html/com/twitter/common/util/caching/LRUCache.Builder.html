018    package com.twitter.common.util.caching;
019    
020    import com.google.common.base.Preconditions;
021    import com.google.common.collect.ImmutableList;
022    import com.twitter.common.base.Closure;
023    import com.twitter.common.base.MorePreconditions;
024    import com.twitter.common.collections.Pair;
025    import com.twitter.common.stats.Stats;
026    
027    import java.util.Collection;
028    import java.util.Collections;
029    import java.util.LinkedHashMap;
030    import java.util.Map;
031    import java.util.concurrent.atomic.AtomicLong;
032    
033    /**
034     * A cache with a fixed maximum size, evicting items that were used least-recently.
035     * WARNING: This is not thread-safe.  If you wish to get a thread-safe version of a constructed
036     * LRUCache, you must wrap it with {@link Collections#synchronizedMap(java.util.Map)}.
037     *
038     * @author William Farner
039     */
040    public class LRUCache<K, V> implements Cache<K, V> {
041    
042      private Map<K, V> map;
043    
044      private final AtomicLong accesses;
045      private final AtomicLong misses;
046    
047      /**
048       * Creates a new bounded cache with the given load factor.
049       *
050       * @param name Unique name for this cache.
051       * @param maxCapacity Maximum capacity for the cache, after which items will be evicted.
052       * @param loadFactor Load factor for the cache.
053       * @param makeSynchronized Whether the underlying map should be synchronized.
054       * @param evictionListener Listener to be notified when an element is evicted, or {@code null} if
055       *    eviction notifications are not needed.
056       */
057      private LRUCache(final String name, final int maxCapacity, float loadFactor,
058          boolean makeSynchronized, final Closure<Pair<K, V>> evictionListener) {
059        map = new LinkedHashMap<K, V>(maxCapacity, loadFactor, true /* Access order. */) {
060          @Override public boolean removeEldestEntry(Map.Entry<K, V> entry) {
061            boolean evict = size() > maxCapacity;
062            if (evict && evictionListener != null) {
063              evictionListener.execute(Pair.of(entry.getKey(), entry.getValue()));
064            }
065            return evict;
066          }
067        };
068    
069        if (makeSynchronized) {
070          map = Collections.synchronizedMap(map);
071        }
072    
073        accesses = Stats.exportLong(name + "_lru_cache_accesses");
074        misses = Stats.exportLong(name + "_lru_cache_misses");
075      }
076    
077      public static <K, V> Builder<K, V> builder() {
078        return new Builder<K, V>();
079      }
080    
081      public static class Builder<K, V> {
082        private String name = null;
083    
084        private int maxSize = 1000;
085    
086        // Sadly, LinkedHashMap doesn't expose this, so the default is pulled from the javadoc.
087        private float loadFactor = 0.75F;
088    
089        private boolean makeSynchronized = true;
090    
091        private Closure<Pair<K, V>> evictionListener = null;
092    
093        public Builder<K, V> name(String name) {
094          this.name = MorePreconditions.checkNotBlank(name);
095          return this;
096        }
097    
098        public Builder<K, V> maxSize(int maxSize) {
099          Preconditions.checkArgument(maxSize > 0);
100          this.maxSize = maxSize;
101          return this;
102        }
103    
104        public Builder<K, V> loadFactor(float loadFactor) {
105          this.loadFactor = loadFactor;
106          return this;
107        }
108    
109        public Builder<K, V> makeSynchronized(boolean makeSynchronized) {
110          this.makeSynchronized = makeSynchronized;
111          return this;
112        }
113    
114        public Builder<K, V> evictionListener(Closure<Pair<K, V>> evictionListener) {
115          this.evictionListener = evictionListener;
116          return this;
117        }
118    
119        public LRUCache<K, V> build() {
120          return new LRUCache<K, V>(name, maxSize, loadFactor, makeSynchronized, evictionListener);
121        }
122      }
123    
124      @Override
125      public V get(K key) {
126        accesses.incrementAndGet();
127        V value = map.get(key);
128        if (value == null) {
129          misses.incrementAndGet();
130        }
131        return value;
132      }
133    
134      @Override
135      public void put(K key, V value) {
136        map.put(key, value);
137      }
138    
139      @Override
140      public void delete(K key) {
141        map.remove(key);
142      }
143    
144      public int size() {
145        return map.size();
146      }
147    
148      @Override
149      public String toString() {
150        return String.format("size: %d, accesses: %s, misses: %s",
151            map.size(),
152            accesses,
153            misses);
154      }
155    
156      public Collection<V> copyValues() {
157        synchronized(map) {
158          return ImmutableList.copyOf(map.values());
159        }
160      }
161    
162      public long getAccesses() {
163        return accesses.longValue();
164      }
165    
166      public long getMisses() {
167        return misses.longValue();
168      }
169    
170      public double getHitRate() {
171        double numAccesses = accesses.longValue();
172        return numAccesses == 0 ? 0 : (numAccesses - misses.longValue()) / numAccesses;
173      }
174    }
