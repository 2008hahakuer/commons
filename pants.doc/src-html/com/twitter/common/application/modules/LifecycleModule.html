018    package com.twitter.common.application.modules;
019    
020    import java.lang.annotation.Annotation;
021    
022    import com.google.inject.AbstractModule;
023    import com.google.inject.Binder;
024    import com.google.inject.Inject;
025    import com.google.inject.Key;
026    import com.google.inject.Singleton;
027    
028    import com.twitter.common.application.ActionRegistry;
029    import com.twitter.common.application.StartupStage;
030    import com.twitter.common.application.Lifecycle;
031    import com.twitter.common.application.ActionController;
032    import com.twitter.common.application.ShutdownStage;
033    
034    /**
035     * Binding module for startup and shutdown controller and registries.
036     *
037     * Bindings provided by this module:
038     * <ul>
039     *   <li>{@code @StartupStage LifecycleActionController} - Action controller for application
040     *       startup.
041     *   <li>{@code @StartupStage ActionRegistry} - Action registry for application startup.
042     *   <li>{@code @ShutdownStage LifecycleActionController} - Action controller for application
043     *       shutdown.
044     *   <li>{@code @ShutdownStage ActionRegistry} - Action registry for application shutdown.
045     * </ul>
046     *
047     * @author William Farner
048     */
049    public class LifecycleModule extends AbstractModule {
050    
051      @Override
052      protected void configure() {
053        bind(Lifecycle.class).in(Singleton.class);
054        bindController(StartupStage.class, binder());
055        bindController(ShutdownStage.class, binder());
056    
057        requestStaticInjection(Init.class);
058      }
059    
060      private static void bindController(Class<? extends Annotation> annotationType, Binder binder) {
061        Key<ActionController> controllerKey =
062            Key.get(ActionController.class, annotationType);
063        binder.bind(controllerKey).to(ActionController.class).in(Singleton.class);
064        binder.bind(Key.get(ActionRegistry.class, annotationType)).to(controllerKey);
065      }
066    
067      public static class Init {
068        @Inject
069        private static void setupShutdownHook(
070            @ShutdownStage final ActionController actionController) {
071    
072          Runtime.getRuntime().addShutdownHook(new Thread(new Runnable() {
073            @Override public void run() {
074              actionController.execute();
075            }
076          }, actionController.getClass().getSimpleName() + "-ShutdownHook"));
077        }
078      }
079    }
