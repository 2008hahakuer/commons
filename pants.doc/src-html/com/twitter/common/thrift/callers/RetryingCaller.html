018    package com.twitter.common.thrift.callers;
019    
020    import com.google.common.annotations.VisibleForTesting;
021    import com.google.common.base.Function;
022    import com.google.common.base.Joiner;
023    import com.google.common.base.Predicate;
024    import com.google.common.base.Throwables;
025    import com.google.common.collect.ImmutableSet;
026    import com.google.common.collect.Iterables;
027    import com.google.common.collect.Lists;
028    import com.google.common.collect.MapMaker;
029    import com.twitter.common.quantity.Amount;
030    import com.twitter.common.quantity.Time;
031    import com.twitter.common.stats.StatsProvider;
032    import com.twitter.common.thrift.TResourceExhaustedException;
033    import org.apache.thrift.async.AsyncMethodCallback;
034    
035    import javax.annotation.Nullable;
036    import java.lang.reflect.Method;
037    import java.util.List;
038    import java.util.Map;
039    import java.util.concurrent.atomic.AtomicInteger;
040    import java.util.concurrent.atomic.AtomicLong;
041    import java.util.logging.Logger;
042    
043    /**
044    * A caller that will retry calls to the wrapped caller.
045    *
046    * @author William Farner
047    */
048    public class RetryingCaller extends CallerDecorator {
049      private static final Logger LOG = Logger.getLogger(RetryingCaller.class.getName());
050    
051      @VisibleForTesting
052      public static final Amount<Long, Time> NONBLOCKING_TIMEOUT = Amount.of(-1L, Time.MILLISECONDS);
053    
054      private final StatsProvider statsProvider;
055      private final String serviceName;
056      private final int retries;
057      private final ImmutableSet<Class<? extends Exception>> retryableExceptions;
058      private final boolean debug;
059    
060      /**
061       * Creates a new retrying caller. The retrying caller will attempt to call invoked methods on the
062       * underlying caller at most {@code retries} times.  A retry will be performed only when one of
063       * the {@code retryableExceptions} is caught.
064       *
065       * @param decoratedCall The caller to decorate with retries.
066       * @param async Whether the caller is asynchronous.
067       * @param statsProvider The stat provider to export retry statistics through.
068       * @param serviceName The service name that calls are being invoked on.
069       * @param retries The maximum number of retries to perform.
070       * @param retryableExceptions The exceptions that can be retried.
071       * @param debug Whether to include debugging information when retries are being performed.
072       */
073      public RetryingCaller(Caller decoratedCall, boolean async, StatsProvider statsProvider,
074          String serviceName, int retries, ImmutableSet<Class<? extends Exception>> retryableExceptions,
075          boolean debug) {
076        super(decoratedCall, async);
077        this.statsProvider = statsProvider;
078        this.serviceName = serviceName;
079        this.retries = retries;
080        this.retryableExceptions = retryableExceptions;
081        this.debug = debug;
082      }
083    
084      private final Map<Method, AtomicLong> stats =
085          new MapMaker().makeComputingMap(new Function<Method, AtomicLong>() {
086            @Override public AtomicLong apply(Method method) {
087              // Thrift does not support overloads - so just the name disambiguates all calls.
088              return statsProvider.makeCounter(serviceName + "_" + method.getName() + "_retries");
089            }
090          });
091    
092      @Override public Object call(final Method method, final Object[] args,
093          @Nullable final AsyncMethodCallback callback,
094          @Nullable final Amount<Long, Time> connectTimeoutOverride) throws Throwable {
095        final AtomicLong retryCounter = stats.get(method);
096        final AtomicInteger attempts = new AtomicInteger();
097        final List<Throwable> exceptions = Lists.newArrayList();
098    
099        final ResultCapture capture = new ResultCapture() {
100          @Override public void success() {
101            // No-op.
102          }
103    
104          @Override public boolean fail(Throwable t) {
105            if (!isRetryable(t)) {
106              if (debug) {
107                LOG.warning(String.format(
108                    "Call failed with un-retryable exception of [%s]: %s, previous exceptions: %s",
109                    t.getClass().getName(), t.getMessage(), combineStackTraces(exceptions)));
110              }
111    
112              return true;
113            } else if (attempts.get() >= retries) {
114              exceptions.add(t);
115    
116              if (debug) {
117                LOG.warning(String.format("Retried %d times, last error: %s, exceptions: %s",
118                    attempts.get(), t, combineStackTraces(exceptions)));
119              }
120    
121              return true;
122            } else {
123              exceptions.add(t);
124    
125              if (isAsync() && attempts.incrementAndGet() <= retries) {
126                try {
127                  retryCounter.incrementAndGet();
128                  // override connect timeout in ThriftCaller to prevent blocking for a connection
129                  // for async retries (since this is within the callback in the selector thread)
130                  invoke(method, args, callback, this, NONBLOCKING_TIMEOUT);
131                } catch (Throwable throwable) {
132                  return fail(throwable);
133                }
134              }
135    
136              return false;
137            }
138          }
139        };
140    
141        boolean continueLoop;
142        do {
143          try {
144            // If this is an async call, the looping will be handled within the capture.
145            return invoke(method, args, callback, capture, connectTimeoutOverride);
146          } catch (Throwable t) {
147            if (!isRetryable(t)) {
148              Throwable propagated = t;
149    
150              if (!exceptions.isEmpty() && (t instanceof TResourceExhaustedException)) {
151                // If we've been trucking along through retries that have had remote call failures
152                // and we suddenly can't immediately get a connection on the next retry, throw the
153                // previous remote call failure - the idea here is that the remote call failure is
154                // more interesting than a transient inability to get an immediate connection.
155                propagated = exceptions.remove(exceptions.size() - 1);
156              }
157    
158              if (isAsync()) {
159                callback.onError(propagated);
160              } else {
161                throw propagated;
162              }
163            }
164          }
165    
166          continueLoop = !isAsync() && attempts.incrementAndGet() <= retries;
167          if (continueLoop) retryCounter.incrementAndGet();
168        } while (continueLoop);
169    
170        Throwable lastRetriedException = Iterables.getLast(exceptions);
171        if (debug) {
172          if (!exceptions.isEmpty()) {
173            LOG.warning(
174                String.format("Retried %d times, last error: %s, previous exceptions: %s",
175                    attempts.get(), lastRetriedException, combineStackTraces(exceptions)));
176          } else {
177            LOG.warning(
178                String.format("Retried 1 time, last error: %s", lastRetriedException));
179          }
180        }
181    
182        if (!isAsync()) throw lastRetriedException;
183        return null;
184      }
185    
186      private boolean isRetryable(Throwable throwable) {
187        return isRetryable.get(throwable.getClass());
188      }
189    
190      private final Map<Class<? extends Throwable>, Boolean> isRetryable =
191          new MapMaker().makeComputingMap(new Function<Class<? extends Throwable>, Boolean>() {
192            @Override public Boolean apply(Class<? extends Throwable> exceptionClass) {
193              return isRetryable(exceptionClass);
194            }
195          });
196    
197      private boolean isRetryable(final Class<? extends Throwable> exceptionClass) {
198        if (retryableExceptions.contains(exceptionClass)) {
199          return true;
200        }
201        return Iterables.any(retryableExceptions, new Predicate<Class<? extends Exception>>() {
202          @Override public boolean apply(Class<? extends Exception> retryableExceptionClass) {
203            return retryableExceptionClass.isAssignableFrom(exceptionClass);
204          }
205        });
206      }
207    
208      private static final Joiner STACK_TRACE_JOINER = Joiner.on('\n');
209    
210      private static String combineStackTraces(List<Throwable> exceptions) {
211        if (exceptions.isEmpty()) {
212          return "none";
213        } else {
214          return STACK_TRACE_JOINER.join(Iterables.transform(exceptions,
215              new Function<Throwable, String>() {
216                private int index = 1;
217                @Override public String apply(Throwable exception) {
218                  return String.format("[%d] %s",
219                      index++, Throwables.getStackTraceAsString(exception));
220                }
221              }));
222        }
223      }
