018    package com.twitter.common.thrift.callers;
019    
020    import com.google.common.base.Preconditions;
021    import com.twitter.common.quantity.Amount;
022    import com.twitter.common.quantity.Time;
023    import org.apache.thrift.async.AsyncMethodCallback;
024    
025    import javax.annotation.Nullable;
026    import java.lang.reflect.Method;
027    
028    /**
029    * A caller that invokes a method on an object.
030    *
031    * @author William Farner
032    */
033    public interface Caller {
034    
035      /**
036       * Invokes a method on an object, using the given arguments.  The method call may be
037       * asynchronous, in which case {@code callback} will be non-null.
038       *
039       * @param method The method being invoked.
040       * @param args The arguments to call {@code method} with.
041       * @param callback The callback to use if the method is asynchronous.
042       * @param connectTimeoutOverride Optional override for the default connection timeout.
043       * @return The return value from invoking the method.
044       * @throws Throwable Exception, as prescribed by the method's contract.
045       */
046      public Object call(Method method, Object[] args, @Nullable AsyncMethodCallback callback,
047          @Nullable Amount<Long, Time> connectTimeoutOverride) throws Throwable;
048    
049      /**
050       * Captures the result of a request, whether synchronous or asynchronous.  It should be expected
051       * that for every request made, exactly one of these methods will be called.
052       */
053      static interface ResultCapture {
054        /**
055         * Called when the request completed successfully.
056         */
057        void success();
058    
059        /**
060         * Called when the request failed.
061         *
062         * @param t Throwable that was caught.  Must never be null.
063         * @return {@code true} if a wrapped callback should be notified of the failure,
064         *    {@code false} otherwise.
065         */
066        boolean fail(Throwable t);
067      }
068    
069      /**
070       * A callback that adapts a {@link ResultCapture} with an {@link AsyncMethodCallback} while
071       * maintaining the AsyncMethodCallback interface.  The wrapped callback will handle invocation
072       * of the underlying callback based on the return values from the ResultCapture.
073       */
074      static class WrappedMethodCallback implements AsyncMethodCallback {
075        private final AsyncMethodCallback wrapped;
076        private final ResultCapture capture;
077    
078        private boolean callbackTriggered = false;
079    
080        public WrappedMethodCallback(AsyncMethodCallback wrapped, ResultCapture capture) {
081          this.wrapped = wrapped;
082          this.capture = capture;
083        }
084    
085        private void callbackTriggered() {
086          Preconditions.checkState(!callbackTriggered, "Each callback may only be triggered once.");
087          callbackTriggered = true;
088        }
089    
090        @Override @SuppressWarnings("unchecked") public void onComplete(Object o) {
091          capture.success();
092          wrapped.onComplete(o);
093          callbackTriggered();
094        }
095    
096        @Override public void onError(Throwable t) {
097          if (capture.fail(t)) {
098            wrapped.onError(t);
099            callbackTriggered();
100          }
101        }
102      }
103    }
