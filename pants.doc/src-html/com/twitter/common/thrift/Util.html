047      public static Function<ServiceInstance, InetSocketAddress> GET_ADDRESS =
048          new Function<ServiceInstance, InetSocketAddress>() {
049            @Override public InetSocketAddress apply(ServiceInstance serviceInstance) {
050              Endpoint endpoint = serviceInstance.getServiceEndpoint();
051              return InetSocketAddress.createUnresolved(endpoint.getHost(), endpoint.getPort());
052            }
053          };
054    
055      public static final Predicate<ServiceInstance> IS_ALIVE = new Predicate<ServiceInstance>() {
056        @Override public boolean apply(ServiceInstance serviceInstance) {
057          switch (serviceInstance.getStatus()) {
058            case ALIVE:
059              return true;
060    
061            // We'll be optimistic here and let MTCP's ranking deal with
062            // unhealthy services in a WARNING state.
063            case WARNING:
064              return true;
065    
066            // Services which are just starting up, on the other hand... are much easier to just not
067            // send requests to.  The STARTING state is useful to distinguish from WARNING or ALIVE:
068            // you exist in ZooKeeper, but don't yet serve traffic.
069            case STARTING:
070            default:
071              return false;
072          }
073        }
074      };

076      /**
077       * Pretty-prints a thrift object contents.
078       *
079       * @param t The thrift object to print.
080       * @return The pretty-printed version of the thrift object.
081       */
082      public static String prettyPrint(TBase t) {
083        return t == null ? "null" : printTbase(t, 0);
084      }
085    
086      /**
087       * Prints an object contained in a thrift message.
088       *
089       * @param o The object to print.
090       * @param depth The print nesting level.
091       * @return The pretty-printed version of the thrift field.
092       */
093      private static String printValue(Object o, int depth) {
094        if (o == null) {
095          return "null";
096        } else if (TBase.class.isAssignableFrom(o.getClass())) {
097          return "\n" + printTbase((TBase) o, depth + 1);
098        } else if (Map.class.isAssignableFrom(o.getClass())) {
099          return printMap((Map) o, depth + 1);
100        } else if (String.class == o.getClass()) {
101          return '"' + o.toString() + '"';
102        } else {
103          return o.toString();
104        }
105      }
106    
107      private static final String METADATA_MAP_FIELD_NAME = "metaDataMap";
108    
109      /**
110       * Prints a TBase.
111       *
112       * @param t The object to print.
113       * @param depth The print nesting level.
114       * @return The pretty-printed version of the TBase.
115       */
116      private static String printTbase(TBase t, int depth) {
117        List<String> fields = Lists.newArrayList();
118        for (Map.Entry<? extends TFieldIdEnum, FieldMetaData> entry :
119            FieldMetaData.getStructMetaDataMap(t.getClass()).entrySet()) {
120          @SuppressWarnings("unchecked")
121          Object value = t.getFieldValue(entry.getKey());
122          fields.add(tabs(depth) + entry.getValue().fieldName + ": " + printValue(value, depth));
123        }
124    
125        return Joiner.on("\n").join(fields);
126      }
127    
128      /**
129       * Prints a map in a style that is consistent with TBase pretty printing.
130       *
131       * @param map The map to print
132       * @param depth The print nesting level.
133       * @return The pretty-printed version of the map.
134       */
135      private static String printMap(Map<?, ?> map, int depth) {
136        List<String> entries = Lists.newArrayList();
137        for (Map.Entry entry : map.entrySet()) {
138          entries.add(tabs(depth) + printValue(entry.getKey(), depth)
139              + " = " + printValue(entry.getValue(), depth));
140        }
141    
142        return entries.isEmpty() ? "{}"
143            : String.format("{\n%s\n%s}", Joiner.on(",\n").join(entries), tabs(depth - 1));
144      }
145    
146      private static String tabs(int n) {
147        return Strings.repeat("  ", n);
148      }

151        // Utility class.
152      }
