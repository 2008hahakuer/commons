018    package com.twitter.common.net;
019    
020    import com.google.common.base.Joiner;
021    import com.google.common.base.Preconditions;
022    import com.google.common.collect.Lists;
023    
024    import java.net.URI;
025    import java.net.URISyntaxException;
026    import java.net.URL;
027    import java.util.Arrays;
028    import java.util.List;
029    import java.util.logging.Logger;
030    import java.util.regex.Matcher;
031    import java.util.regex.Pattern;
032    
033    /**
034     * @author William Farner
035     */
036    public class UrlHelper {
037    
038      private static final Logger LOG = Logger.getLogger(UrlHelper.class.getName());
039    
040      /**
041       * Gets the domain from {@code url}.
042       *
043       * @param url A url.
044       * @return The domain portion of the URL, or {@code null} if the url is invalid.
045       */
046      public static String getDomain(String url) {
047        try {
048          return getDomainChecked(url);
049        } catch (URISyntaxException e) {
050          LOG.finest("Malformed url: " + url);
051          return null;
052        }
053      }
054    
055      /**
056       * Gets the domain from {@code uri}, and throws an exception if it's not a valid uri.
057       *
058       * @param url A url.
059       * @throws URISyntaxException if url is not a valid {@code URI}
060       * @return The domain portion of the given url, or {@code null} if the host is undefined.
061       */
062      public static String getDomainChecked(String url) throws URISyntaxException {
063        Preconditions.checkNotNull(url);
064        url = addProtocol(url);
065        return new URI(url).getHost();
066      }
067    
068      /**
069       * Gets the path from {@code url}.
070       *
071       * @param url A url.
072       * @return The path portion of the URL, or {@code null} if the url is invalid.
073       */
074      public static String getPath(String url) {
075        Preconditions.checkNotNull(url);
076        url = addProtocol(url);
077        try {
078          return new URI(url).getPath();
079        } catch (URISyntaxException e) {
080          LOG.info("Malformed url: " + url);
081          return null;
082        }
083      }
084    
085      /**
086       * Strips URL parameters from a url.
087       * This will remove anything after and including a question mark in the URL.
088       *
089       * @param url The URL to strip parameters from.
090       * @return The original URL with parameters stripped, which will be the original URL if no
091       *   parameters were found.
092       */
093      public static String stripUrlParameters(String url) {
094        Preconditions.checkNotNull(url);
095        int paramStartIndex = url.indexOf("?");
096        if (paramStartIndex == -1) {
097          return url;
098        } else {
099          return url.substring(0, paramStartIndex);
100        }
101      }
102    
103      /**
104       * Convenience method that calls #stripUrlParameters(String) for a URL.
105       *
106       * @param url The URL to strip parameters from.
107       * @return The original URL with parameters stripped, which will be the original URL if no
108       *   parameters were found.
109       */
110      public static String stripUrlParameters(URL url) {
111        return stripUrlParameters(url.toString());
112      }
113    
114      private static final Pattern URL_PROTOCOL_REGEX =
115          Pattern.compile("^https?://", Pattern.CASE_INSENSITIVE);
116    
117      /**
118       * Checks whether a URL specifies its protocol, prepending http if it does not.
119       *
120       * @param url The URL to fix.
121       * @return The URL with the http protocol specified if no protocol was already specified.
122       */
123      public static String addProtocol(String url) {
124        Preconditions.checkNotNull(url);
125        Matcher matcher = URL_PROTOCOL_REGEX.matcher(url);
126        if (!matcher.find()) {
127          url = "http://" + url;
128        }
129        return url;
130      }
131    
132      /**
133       * Gets the domain levels for a host.
134       * For example, sub1.sub2.domain.co.uk would return
135       * [sub1.sub2.domain.co.uk, sub2.domain.co.uk, domain.co.uk, co.uk, uk].
136       *
137       *
138       * @param host The host to peel subdomains off from.
139       * @return The domain levels in this host.
140       */
141      public static List<String> getDomainLevels(String host) {
142        Preconditions.checkNotNull(host);
143    
144        // Automatically include www prefix if not present.
145        if (!host.startsWith("www")) {
146          host = "www." + host;
147        }
148    
149        Joiner joiner = Joiner.on(".");
150        List<String> domainParts = Lists.newLinkedList(Arrays.asList(host.split("\\.")));
151        List<String> levels = Lists.newLinkedList();
152    
153        while (domainParts.size() > 1) {
154          levels.add(joiner.join(domainParts));
155          domainParts.remove(0);
156        }
157    
158        return levels;
159      }
160    }
