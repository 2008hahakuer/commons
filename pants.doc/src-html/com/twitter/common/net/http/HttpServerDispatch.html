018    package com.twitter.common.net.http;
019    
020    import com.google.common.base.Preconditions;
021    import com.google.common.collect.Lists;
022    import com.google.common.collect.Sets;
023    import com.twitter.common.net.http.handlers.TextResponseHandler;
024    import org.mortbay.jetty.AbstractConnector;
025    import org.mortbay.jetty.Connector;
026    import org.mortbay.jetty.Server;
027    import org.mortbay.jetty.nio.SelectChannelConnector;
028    import org.mortbay.jetty.servlet.Context;
029    import org.mortbay.jetty.servlet.ServletHolder;
030    
031    import javax.annotation.Nullable;
032    import javax.servlet.http.HttpServlet;
033    import javax.servlet.http.HttpServletRequest;
034    import java.io.IOException;
035    import java.util.List;
036    import java.util.Map;
037    import java.util.Random;
038    import java.util.Set;
039    import java.util.logging.Level;
040    import java.util.logging.Logger;
041    
042    /**
043     * A simple multi-threaded HTTP server dispatcher.  Supports registering handlers for different
044     * URI paths, which will be called when a request is received.
045     *
046     * @author William Farner
047     */
048    public class HttpServerDispatch {
049      private static final Logger LOG = Logger.getLogger(HttpServerDispatch.class.getName());
050    
051      // Registered handlers. Used only for display.
052      private final Set<String> registeredHandlers = Sets.newHashSet();
053    
054      private Server server;
055      private Context context;
056      private int port;
057    
058      /**
059       * Opens the HTTP server on the given port.
060       *
061       * @param port The port to listen on.
062       * @return {@code true} if the server started successfully on the port, {@code false} otherwise.
063       */
064      public boolean listen(int port) {
065        return listen(port, port);
066      }
067    
068      /**
069       * Opens the HTTP server on random port within the given range.
070       *
071       * @param minPort The minimum port number to listen on.
072       * @param maxPort The maximum port number to listen on.
073       * @return {@code true} if the server started successfully on the port, {@code false} otherwise.
074       */
075      public synchronized boolean listen(int minPort, int maxPort) {
076        boolean state = !isStarted();
077        Preconditions.checkState(state,
078            "HttpServerDispatch has already been started on port: %d", port);
079    
080        Connector connector = openConnector(minPort, maxPort);
081        if (connector == null) return false; // Couldn't open a server port.
082        port = connector.getLocalPort();
083    
084        server = new Server();
085        server.addConnector(connector);
086        context = new Context(server, "/", Context.NO_SESSIONS);
087        context.addServlet(new ServletHolder(new RootHandler()), "/");
088    
089        try {
090          server.start();
091          LOG.info("HTTP server is listening on port " + port);
092          return true;
093        } catch (Exception e) {
094          LOG.log(Level.SEVERE, "HTTP server failed to start on port " + connector.getLocalPort(), e);
095          return false;
096        }
097      }
098    
099      public synchronized boolean isStarted() {
100        return (server != null) && server.isStarted();
101      }
102    
103      public synchronized int getPort() {
104        Preconditions.checkState(isStarted(), "HttpServer must be started before port can be determined");
105        return port;
106      }
107    
108      private Connector openConnector(int minPort, int maxPort) {
109        if (minPort != 0 || maxPort != 0) {
110          Preconditions.checkState(minPort > 0, "Invalid port range.");
111          Preconditions.checkState(maxPort > 0, "Invalid port range.");
112          Preconditions.checkState(minPort <= maxPort, "Invalid port range.");
113        }
114        int attempts = 0;
115        int port;
116    
117        int maxAttempts = minPort == maxPort ? 1 : 5;
118        while (++attempts <= maxAttempts) {
119          if (minPort == maxPort) {
120            port = minPort;
121          } else {
122            port = minPort + new Random().nextInt(maxPort - minPort);
123          }
124          LOG.info("Attempting to listen on port " + port);
125    
126          try {
127            // TODO(John Sirois): consider making Connector impl parametrizable
128            AbstractConnector connector = new SelectChannelConnector();
129            connector.setPort(port);
130            // Create the server with a maximum TCP backlog of 50, meaning that when the request queue
131            // exceeds 50, subsequent connections may be rejected.
132            connector.setAcceptQueueSize(50);
133            connector.open();
134            return connector;
135          } catch (IOException e) {
136            LOG.log(Level.INFO, "Failed to create HTTP server on port " + port);
137          }
138        }
139        return null;
140      }
141    
142      /**
143       * Stops the HTTP server.
144       */
145      public synchronized void stop() {
146        if (isStarted()) {
147          try {
148            server.stop();
149          } catch (Exception e) {
150            LOG.log(Level.SEVERE, "Error stopping HTTPServer on " + port, e);
151          }
152        }
153      }
154    
155      /**
156       * Registers a URI handler, replacing the existing handler if it exists.
157       *
158       * @param path The URI path that the handler should be called for.
159       * @param handler The handler to call.
160       * @param initParams An optional map of servlet init parameter names and their values.
161       * @param silent Whether to display the registered handler in the root "/" response.
162       *               Useful for handlers that you want to avoid accidental clicks on.
163       */
164      public synchronized void registerHandler(String path, HttpServlet handler,
165          @Nullable Map<String, String> initParams, boolean silent) {
166        Preconditions.checkNotNull(path);
167        Preconditions.checkNotNull(handler);
168        Preconditions.checkState(path.length() > 0);
169        Preconditions.checkState(path.charAt(0) == '/');
170    
171        if (silent) {
172          registeredHandlers.remove(path);
173        } else {
174          registeredHandlers.add(path);
175        }
176    
177        ServletHolder servletHolder = new ServletHolder(handler);
178        if (initParams != null) {
179          servletHolder.setInitParameters(initParams);
180        }
181        context.addServlet(servletHolder, path.replaceFirst("/?$", "/*"));
182      }
183    
184      public Context getRootContext() {
185        Preconditions.checkState(context != null, "Context is not yet available. " +
186            "Ensure that listen(...) is called prior to calling this method.");
187        return context;
188      }
189    
190      /**
191       * The root handler, which will display the paths at which all handlers are registered.
192       */
193      private class RootHandler extends TextResponseHandler {
194        public RootHandler() {
195          super("text/html");
196        }
197    
198        @Override
199        public Iterable<String> getLines(HttpServletRequest request) {
200          List<String> lines = Lists.newArrayList();
201          lines.add("<html>");
202          for (String handler : registeredHandlers) {
203            lines.add(String.format("<a href='%s'>%s</a><br />", handler, handler));
204          }
205          lines.add("</html>");
206          return lines;
207        }
208      }
