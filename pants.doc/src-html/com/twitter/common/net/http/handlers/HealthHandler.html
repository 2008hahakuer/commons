018    package com.twitter.common.net.http.handlers;
019    
020    import com.google.common.base.Preconditions;
021    import com.google.inject.Inject;
022    import com.google.inject.name.Named;
023    import com.twitter.common.base.ExceptionalSupplier;
024    
025    import javax.servlet.ServletException;
026    import javax.servlet.http.HttpServlet;
027    import javax.servlet.http.HttpServletRequest;
028    import javax.servlet.http.HttpServletResponse;
029    import java.io.IOException;
030    import java.io.PrintWriter;
031    import java.util.logging.Level;
032    import java.util.logging.Logger;
033    
034    /**
035     * A servlet that provides a crude mechanism for monitoring a service's health.  If the servlet
036     * returns {@link #IS_HEALTHY} then the containing service should be deemed healthy.
037     *
038     * @author John Sirois
039     */
040    public class HealthHandler extends HttpServlet {
041    
042      /**
043       * A {@literal @Named} binding key for the Healthz servlet health checker.
044       */
045      public static final String HEALTH_CHECKER_KEY =
046          "com.twitter.common.net.http.handlers.Healthz.checker";
047    
048      /**
049       * The plain text response string this servlet returns in the body of its responses to health
050       * check requests when its containing service is healthy.
051       */
052      public static final String IS_HEALTHY = "OK";
053    
054      private static final String IS_NOT_HEALTHY = "SICK";
055    
056      private static final Logger LOG = Logger.getLogger(HealthHandler.class.getName());
057    
058      private final ExceptionalSupplier<Boolean, ?> healthChecker;
059    
060      /**
061       * Constructs a new Healthz that uses the given {@code healthChecker} to determine current health
062       * of the service for at the point in time of each GET request.  The given {@code healthChecker}
063       * should return {@code true} if the service is healthy and {@code false} otherwise.  If the
064       * {@code healthChecker} returns null or throws the service is considered unhealthy.
065       *
066       * @param healthChecker a supplier that is called to perform a health check
067       */
068      @Inject
069      public HealthHandler(@Named(HEALTH_CHECKER_KEY) ExceptionalSupplier<Boolean, ?> healthChecker) {
070        this.healthChecker = Preconditions.checkNotNull(healthChecker);
071      }
072    
073      @Override
074      protected void doGet(HttpServletRequest req, HttpServletResponse resp)
075          throws ServletException, IOException {
076    
077        resp.setContentType("text/plain");
078        PrintWriter writer = resp.getWriter();
079        try {
080          writer.println(Boolean.TRUE.equals(healthChecker.get()) ? IS_HEALTHY : IS_NOT_HEALTHY);
081        } catch (Exception e) {
082          writer.println(IS_NOT_HEALTHY);
083          LOG.log(Level.WARNING, "Health check failed.", e);
084        }
085      }
086    }
