018    package com.twitter.common.net.http.handlers;
019    
020    import javax.servlet.ServletException;
021    import javax.servlet.http.HttpServlet;
022    import javax.servlet.http.HttpServletRequest;
023    import javax.servlet.http.HttpServletResponse;
024    import java.io.IOException;
025    import java.io.PrintWriter;
026    
027    /**
028     * A handler that responds to all requests in HTML format.
029     *
030     * @author William Farner
031     */
032    public abstract class TextResponseHandler extends HttpServlet {
033      private final String textContentType;
034    
035      public TextResponseHandler() {
036        this("text/plain");
037      }
038    
039      public TextResponseHandler(String textContentType) {
040        this.textContentType = textContentType;
041      }
042    
043      /**
044       * Returns the lines to be printed as the body of the response.
045       *
046       * @return An iterable collection of lines to respond to the request with.
047       */
048      public abstract Iterable<String> getLines(HttpServletRequest request);
049    
050      @Override
051      protected void doGet(HttpServletRequest req, HttpServletResponse resp)
052          throws ServletException, IOException {
053    
054        resp.setContentType(textContentType);
055        resp.setStatus(HttpServletResponse.SC_OK);
056        PrintWriter responseBody = resp.getWriter();
057        for (String line : getLines(req)) {
058          responseBody.println(line);
059        }
060        responseBody.close();
061      }
062    }
