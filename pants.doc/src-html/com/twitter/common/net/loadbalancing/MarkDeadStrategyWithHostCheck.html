001    package com.twitter.common.net.loadbalancing;
002    
003    import java.util.Map;
004    
005    import com.google.common.base.Function;
006    import com.google.common.base.Predicate;
007    import com.google.common.collect.Maps;
008    
009    import com.twitter.common.util.BackoffDecider;
010    
011    /**
012     * A load balancing strategy that extends the functionality of the mark dead strategy by
013     * integrating a hostChecker that allows hosts to transition out of a dead state
014     * if the most recent connection to the host was successful.
015     *
016     * @param <S> typically socket address of a backend host.
017     * @author Krishna Gade
018     */
019    public class MarkDeadStrategyWithHostCheck<S> extends MarkDeadStrategy<S> {
020    
021      /**
022       * LiveHostChecker implements Filter to determine whether a host is alive based on the
023       * result of the most recent connection attempt to that host. It keeps a map of
024       * backend -> last connection result, which gets updated every time someone tries to
025       * add to connection result.
026       */
027      protected static class LiveHostChecker<S> implements Predicate<S> {
028        private final Map<S, ConnectionResult> lastConnectionResult = Maps.newHashMap();
029    
030        /**
031         * Adds the connection result of this backend to the last connection result map.
032         *
033         * @param backend typically the socket address of the backend.
034         * @param result result of what happened when the client tried to connect to this backend.
035         */
036        public void addConnectResult(S backend, ConnectionResult result) {
037          lastConnectionResult.put(backend, result);
038        }
039    
040        /**
041         * Checks if the last connection result for this backend and returns {@code true} if it
042         * was {@link ConnectionResult#SUCCESS} otherwise returns {@code false}.
043         *
044         * @param backend typically the socket address of the backend.
045         */
046        @Override public boolean apply(S backend) {
047          ConnectionResult result = lastConnectionResult.get(backend);
048          return result != null && result == ConnectionResult.SUCCESS;
049        }
050      }
051    
052      // Reference to the host checker we pass to the super class.
053      // We keep it here to avoid casting on every access to it.
054      protected final LiveHostChecker<S> liveHostChecker;
055    
056      /**
057       * Creates a mark dead strategy with the given wrapped strategy and backoff decider factory.
058       * It uses a hostChecker {@link Predicate} that allows hosts to transition out
059       * of a dead state if the most recent connection to the host was successful.
060       *
061       * @param wrappedStrategy one of the implementations of the load balancing strategy.
062       * @param backoffFactory backoff decider factory per host.
063       */
064      public MarkDeadStrategyWithHostCheck(LoadBalancingStrategy<S> wrappedStrategy,
065          Function<S, BackoffDecider> backoffFactory) {
066        super(wrappedStrategy, backoffFactory, new LiveHostChecker<S>());
067        // Casting to LiveHostChecker is safe here as that's the only predicate that we pass to super.
068        this.liveHostChecker = ((LiveHostChecker<S>) hostChecker);
069      }
070    
071    
072      /**
073       * Overrides the base class implementation by adding this connection result to the
074       * host checker.
075       *
076       * @param backendKey typically the socket address of the backend.
077       * @param result result of what happened when the client tried to connect to this backend.
078       * @param connectTimeNanos time took to connect to the backend in nano seconds.
079       */
080      @Override
081      public void addConnectResult(S backendKey, ConnectionResult result, long connectTimeNanos) {
082        liveHostChecker.addConnectResult(backendKey, result);
083        super.addConnectResult(backendKey, result, connectTimeNanos);
084      }
085    }
