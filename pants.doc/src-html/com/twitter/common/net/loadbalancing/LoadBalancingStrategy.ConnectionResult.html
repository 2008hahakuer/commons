018    package com.twitter.common.net.loadbalancing;
019    
020    import com.twitter.common.base.Closure;
021    import com.twitter.common.net.pool.ResourceExhaustedException;
022    import com.twitter.common.net.loadbalancing.RequestTracker.RequestResult;
023    
024    import java.util.Collection;
025    import java.util.Set;
026    
027    /**
028     * A strategy for balancing request load among backends.
029     *
030     * Strategies should be externally synchronized, and therefore do not have to worry about reentrant
031     * access.
032     *
033     * @author William Farner
034     */
035    public interface LoadBalancingStrategy<K> {
036    
037      /**
038       * Offers a set of backends that the load balancer should choose from to distribute load amongst.
039       *
040       * @param offeredBackends Backends to choose from.
041       * @param onBackendsChosen A callback that should be notified when the offered backends have been
042       *     (re)chosen from.
043       */
044      public void offerBackends(Set<K> offeredBackends, Closure<Collection<K>> onBackendsChosen);
045    
046      /**
047       * Gets the next backend that a request should be sent to.
048       *
049       * @return Next backend to send a request.
050       * @throws ResourceExhaustedException If there are no available backends.
051       */
052      public K nextBackend() throws ResourceExhaustedException;
053    
054      /**
055       * Offers information about a connection result.
056       *
057       * @param key Backend key.
058       * @param result Connection result.
059       * @param connectTimeNanos Time spent waiting for connection to be established.
060       */
061      public void addConnectResult(K key, ConnectionResult result, long connectTimeNanos);
062    
063      /**
064       * Offers information about a connection that was returned.
065       *
066       * @param key Backend key.
067       */
068      public void connectionReturned(K key);
069    
070      /**
071       * Offers information about a request result.
072       *
073       * @param key Backend key.
074       * @param result Request result.
075       * @param requestTimeNanos Time spent waiting for a connection to be established.
076       */
077      public void addRequestResult(K key, RequestResult result, long requestTimeNanos);
078    
079      enum ConnectionResult {
080        FAILED,
081        TIMEOUT,
082        SUCCESS
083      }
084    }
