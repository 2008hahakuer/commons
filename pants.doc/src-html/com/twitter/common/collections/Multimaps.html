018    package com.twitter.common.collections;
019    
020    import java.util.Collection;
021    import java.util.Set;
022    import java.util.SortedSet;
023    
024    import com.google.common.base.Preconditions;
025    import com.google.common.base.Predicate;
026    import com.google.common.collect.ArrayListMultimap;
027    import com.google.common.collect.Multimap;
028    import com.google.common.collect.Sets;
029    
030    /**
031     * Utility class for functions related to Multimaps in the java collections library.
032     *
033     * @author William Farner
034     */
035    public final class Multimaps {
036    
037      private Multimaps() {
038        // Utility.
039      }
040    
041      /**
042       * Prunes a multimap based on a predicate, returning the pruned values.  The input map will be
043       * modified.
044       *
045       * @param map The multimap to prune.
046       * @param filterRule The pruning rule.  When the predicate returns {@code false} for an entry, it
047       *    will be pruned, otherwise it will be retained.
048       * @param <K> The key type in the multimap.
049       * @param <V> The value type in the multimap.
050       * @return A new multimap, containing the pruned keys/values.
051       */
052      public static <K, V> Multimap<K, V> prune(Multimap<K, V> map,
053          Predicate<Collection<V>> filterRule) {
054        Preconditions.checkNotNull(map);
055        Preconditions.checkNotNull(filterRule);
056        Set<K> prunedKeys = Sets.newHashSet();
057        for (K key : map.keySet()) {
058          if (!filterRule.apply(map.get(key))) {
059            prunedKeys.add(key);
060          }
061        }
062    
063        Multimap<K, V> pruned = ArrayListMultimap.create();
064        for (K key : prunedKeys) {
065          pruned.putAll(key, map.removeAll(key));
066        }
067    
068        return pruned;
069      }
070    
071      /**
072       * Convenience method to prune key/values pairs where the size of the value collection is below a
073       * threshold.
074       *
075       * @param map The multimap to prune.
076       * @param minSize The minimum size for retained value collections.
077       * @param <K> The key type in the multimap.
078       * @param <V> The value type in the multimap.
079       * @return A new multimap, containing the pruned keys/values.
080       */
081      public static <K, V> Multimap<K, V> prune(Multimap<K, V> map, final int minSize) {
082        return prune(map, new Predicate<Collection<V>>() {
083          @Override public boolean apply(Collection<V> input) {
084            return input.size() >= minSize;
085          }
086        });
087      }
088    
089      /**
090       * Returns the set of keys associated with groups of a size greater than or equal to a given size.
091       *
092       * @param map The multimap to search.
093       * @param minSize The minimum size to return associated keys for.
094       * @param <K> The key type for the multimap.
095       * @return The keys associated with groups of size greater than or equal to {@code minSize}.
096       */
097      public static <K> Set<K> getLargeGroups(Multimap<K, ?> map, int minSize) {
098        Set<K> largeKeys = Sets.newHashSet();
099        for (K key : map.keySet()) {
100          if (map.get(key).size() >= minSize) {
101            largeKeys.add(key);
102          }
103        }
104    
105        return largeKeys;
106      }
107    
108      /**
109       * Returns the set of keys associated with the largest values in the multimap.
110       *
111       * @param map The multimap to search.
112       * @param topValues Number of groupings to find the keys for.
113       * @return The keys associated with the largest groups, of maximum size {@code topValues}.
114       */
115      public static <K> Set<K> getLargestGroups(Multimap<K, ?> map, int topValues) {
116        /**
117         * A grouping of values in the multimap.
118         */
119        class Grouping implements Comparable<Grouping> {
120          private K key;
121          private int size;
122    
123          public Grouping(K key, int size) {
124            this.key = key;
125            this.size = size;
126          }
127    
128          @Override
129          public int hashCode() {
130            return size;
131          }
132          @Override
133          public int compareTo(Grouping grouping) {
134            return size - grouping.size;
135          }
136          @Override
137          public boolean equals(Object o) {
138            if (!(o instanceof Grouping)) {
139              return false;
140            }
141            Grouping other = (Grouping) o;
142            return key.equals(other.key);
143          }
144        }
145    
146        SortedSet<Grouping> topGroups = Sets.newTreeSet();
147        for (K key : map.keySet()) {
148          topGroups.add(new Grouping(key, map.get(key).size()));
149    
150          // Remove the smallest value.
151          if (topGroups.size() > topValues) {
152            topGroups.remove(topGroups.first());
153          }
154        }
155    
156        Set<K> topKeys = Sets.newHashSet();
157        for (Grouping group : topGroups) {
158          topKeys.add(group.key);
159        }
160    
161        return topKeys;
162      }
163    }
