018    package com.twitter.common.stats;
019    
020    import com.google.common.annotations.VisibleForTesting;
021    import com.google.common.base.Function;
022    import com.google.common.base.Preconditions;
023    import com.google.common.collect.ImmutableSet;
024    import com.google.common.collect.Iterables;
025    import com.google.common.collect.MapMaker;
026    import com.google.common.util.concurrent.ThreadFactoryBuilder;
027    import com.google.inject.Inject;
028    import com.google.inject.name.Named;
029    import com.twitter.common.base.Command;
030    import com.twitter.common.collections.BoundedQueue;
031    import com.twitter.common.application.ActionRegistry;
032    import com.twitter.common.quantity.Amount;
033    import com.twitter.common.quantity.Time;
034    import com.twitter.common.util.Clock;
035    
036    import java.util.Map;
037    import java.util.Set;
038    import java.util.concurrent.ScheduledExecutorService;
039    import java.util.concurrent.ScheduledThreadPoolExecutor;
040    import java.util.concurrent.atomic.AtomicBoolean;
041    import java.util.logging.Level;
042    import java.util.logging.Logger;
043    
044    /**
045     * A simple in-memory repository for exported variables.
046     *
047     * @author John Sirois
048     */
049    public class TimeSeriesRepositoryImpl implements TimeSeriesRepository {
050    
051      private static final Logger LOG = Logger.getLogger(TimeSeriesRepositoryImpl.class.getName());
052    
053      /**
054       * {@literal @Named} binding key for the sampling period.
055       */
056      public static final String SAMPLE_PERIOD =
057          "com.twitter.common.stats.TimeSeriesRepositoryImpl.SAMPLE_PERIOD";
058    
059      /**
060       * {@literal @Named} binding key for the maximum number of retained samples.
061       */
062      public static final String SAMPLE_RETENTION_PERIOD =
063          "com.twitter.common.stats.TimeSeriesRepositoryImpl.SAMPLE_RETENTION_PERIOD";
064    
065      private final SlidingStats scrapeDuration = new SlidingStats("variable_scrape", "micros");
066    
067      // We store TimeSeriesImpl, which allows us to add samples.
068      private final Map<String, TimeSeriesImpl> timeSeries;
069      private final BoundedQueue<Number> timestamps;
070    
071      private final Amount<Long, Time> samplePeriod;
072      private final int retainedSampleLimit;
073    
074      @Inject
075      public TimeSeriesRepositoryImpl(@Named(SAMPLE_PERIOD) Amount<Long, Time> samplePeriod,
076          @Named(SAMPLE_RETENTION_PERIOD) final Amount<Long, Time> retentionPeriod) {
077        this.samplePeriod = Preconditions.checkNotNull(samplePeriod);
078        Preconditions.checkArgument(samplePeriod.getValue() > 0, "Sample period must be positive.");
079        Preconditions.checkNotNull(retentionPeriod);
080        Preconditions.checkArgument(retentionPeriod.getValue() > 0,
081            "Sample retention period must be positive.");
082    
083        retainedSampleLimit = (int) (retentionPeriod.as(Time.SECONDS) / samplePeriod.as(Time.SECONDS));
084        Preconditions.checkArgument(retainedSampleLimit > 0,
085            "Sample retention period must be greater than sample period.");
086    
087        timeSeries = new MapMaker().makeComputingMap(
088            new Function<String, TimeSeriesImpl>() {
089              @Override public TimeSeriesImpl apply(final String name) {
090                TimeSeriesImpl timeSeries = new TimeSeriesImpl(name);
091    
092                // Backfill so we have data for pre-accumulated timestamps.
093                int numTimestamps = timestamps.size();
094                if (numTimestamps != 0) {
095                  for (int i = 1; i < numTimestamps; i++) {
096                    timeSeries.addSample(0L);
097                  }
098                }
099    
100                return timeSeries;
101              }
102            });
103    
104        timestamps = new BoundedQueue<Number>(retainedSampleLimit);
105      }
106    
107      /**
108       * Starts the variable sampler, which will fetch variables {@link Stats} on the given period.
109       *
110       */
111      @Override
112      public void start(ActionRegistry shutdownRegistry) {
113        Preconditions.checkNotNull(shutdownRegistry);
114        Preconditions.checkNotNull(samplePeriod);
115        Preconditions.checkArgument(samplePeriod.getValue() > 0);
116    
117        final ScheduledExecutorService executor = new ScheduledThreadPoolExecutor(1 /* One thread. */,
118            new ThreadFactoryBuilder().setNameFormat("VariableSampler-%d").setDaemon(true).build());
119    
120        final AtomicBoolean shouldSample = new AtomicBoolean(true);
121        final Runnable sampler = new Runnable() {
122          @Override public void run() {
123            if (shouldSample.get()) {
124              try {
125                runSampler(Clock.SYSTEM_CLOCK);
126              } catch (Exception e) {
127                LOG.log(Level.SEVERE, "ignoring runSampler failure", e);
128              }
129            }
130          }
131        };
132    
133        executor.scheduleAtFixedRate(sampler, samplePeriod.getValue(), samplePeriod.getValue(),
134            samplePeriod.getUnit().getTimeUnit());
135        shutdownRegistry.addAction(new Command() {
136          @Override
137          public void execute() throws RuntimeException {
138            shouldSample.set(false);
139            executor.shutdown();
140            LOG.info("Variable sampler shut down");
141          }
142        });
143      }
144    
145      @VisibleForTesting
146      synchronized void runSampler(Clock clock) {
147        timestamps.add(clock.nowMillis());
148    
149        long startNanos = clock.nowNanos();
150        for (RecordingStat<? extends Number> var : Stats.getNumericVariables()) {
151          timeSeries.get(var.getName()).addSample(var.sample());
152        }
153        scrapeDuration.accumulate(
154            Amount.of(clock.nowNanos() - startNanos, Time.NANOSECONDS).as(Time.MICROSECONDS));
155      }
156    
157      @Override
158      public synchronized Set<String> getAvailableSeries() {
159        return ImmutableSet.copyOf(timeSeries.keySet());
160      }
161    
162      @Override
163      public synchronized TimeSeries get(String name) {
164        if (!timeSeries.containsKey(name)) return null;
165        return timeSeries.get(name);
166      }
167    
168      @Override
169      public synchronized Iterable<Number> getTimestamps() {
170        return Iterables.unmodifiableIterable(timestamps);
171      }
172    
173      private class TimeSeriesImpl implements TimeSeries {
174        private final String name;
175        private final BoundedQueue<Number> samples;
176    
177        TimeSeriesImpl(String name) {
178          this.name = name;
179          samples = new BoundedQueue<Number>(retainedSampleLimit);
180        }
181    
182        @Override public String getName() {
183          return name;
184        }
185    
186        void addSample(Number value) {
187          samples.add(value);
188        }
189    
190        @Override public Iterable<Number> getSamples() {
191          return Iterables.unmodifiableIterable(samples);
192        }
193      }
194    }
