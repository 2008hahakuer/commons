018    package com.twitter.common.stats;
019    
020    import com.google.common.collect.ImmutableList;
021    import com.twitter.common.quantity.Amount;
022    import com.twitter.common.quantity.Data;
023    import com.twitter.common.quantity.Time;
024    
025    import java.lang.management.ClassLoadingMXBean;
026    import java.lang.management.GarbageCollectorMXBean;
027    import java.lang.management.ManagementFactory;
028    import java.lang.management.MemoryMXBean;
029    import java.lang.management.MemoryUsage;
030    import java.lang.management.OperatingSystemMXBean;
031    import java.util.Arrays;
032    
033    /**
034     * Convenience class to export statistics about the JVM.
035     *
036     * TODO(William Farner): Some of these are fixed values, make sure to export them so that they are not
037     * collected as time series.
038     *
039     * @author William Farner
040     */
041    public class JvmStats {
042    
043      private static final long BYTES_PER_MB = Amount.of(1L, Data.MB).as(Data.BYTES);
044      private static final double SECS_PER_NANO =
045          ((double) 1) / Amount.of(1L, Time.SECONDS).as(Time.NANOSECONDS);
046    
047      private JvmStats() {
048        // Utility class.
049      }
050    
051      /**
052       * Exports stats related to the JVM and runtime environment.
053       */
054      public static void export() {
055        final OperatingSystemMXBean osMbean = ManagementFactory.getOperatingSystemMXBean();
056        if (osMbean instanceof com.sun.management.OperatingSystemMXBean) {
057          final com.sun.management.OperatingSystemMXBean sunOsMbean =
058              (com.sun.management.OperatingSystemMXBean) osMbean;
059    
060          Stats.exportAll(
061              ImmutableList.<Stat<? extends Number>>builder()
062              .add(new StatImpl<Long>("system_free_physical_memory_mb") {
063                @Override public Long read() {
064                  return sunOsMbean.getFreePhysicalMemorySize() / BYTES_PER_MB;
065                }
066              })
067              .add(new StatImpl<Long>("system_free_swap_mb") {
068                @Override public Long read() {
069                  return sunOsMbean.getFreeSwapSpaceSize() / BYTES_PER_MB;
070                }
071              })
072              .add(
073              Rate.of(
074                  new StatImpl<Long>("process_cpu_time_nanos") {
075                    @Override public Long read() {
076                      return sunOsMbean.getProcessCpuTime();
077                    }
078                  }).withName("process_cpu_cores_utilized").withScaleFactor(SECS_PER_NANO).build())
079              .build());
080        }
081    
082        final Runtime runtime = Runtime.getRuntime();
083        final ClassLoadingMXBean classLoadingBean = ManagementFactory.getClassLoadingMXBean();
084        final MemoryMXBean memoryBean = ManagementFactory.getMemoryMXBean();
085        final MemoryUsage heapUsage = ManagementFactory.getMemoryMXBean().getHeapMemoryUsage();
086        final MemoryUsage nonHeapUsage = ManagementFactory.getMemoryMXBean().getNonHeapMemoryUsage();
087    
088        Stats.exportAll(ImmutableList.<Stat<? extends Number>>builder()
089          .add(new StatImpl<Long>("jvm_time_ms") {
090            @Override public Long read() { return System.currentTimeMillis(); }
091          })
092          .add(new StatImpl<Integer>("jvm_threads_active") {
093            @Override public Integer read() { return Thread.activeCount(); }
094          })
095          .add(new StatImpl<Integer>("jvm_available_processors") {
096            @Override public Integer read() { return runtime.availableProcessors(); }
097          })
098          .add(new StatImpl<Long>("jvm_memory_free_mb") {
099            @Override public Long read() { return runtime.freeMemory() / BYTES_PER_MB; }
100          })
101          .add(new StatImpl<Long>("jvm_memory_max_mb") {
102            @Override public Long read() { return runtime.maxMemory() / BYTES_PER_MB; }
103          })
104          .add(new StatImpl<Long>("jvm_memory_mb_total") {
105            @Override public Long read() { return runtime.totalMemory() / BYTES_PER_MB; }
106          })
107          .add(new StatImpl<Integer>("jvm_class_loaded_count") {
108            @Override public Integer read() { return classLoadingBean.getLoadedClassCount(); }
109          })
110          .add(new StatImpl<Long>("jvm_class_total_loaded_count") {
111            @Override public Long read() { return classLoadingBean.getTotalLoadedClassCount(); }
112          })
113          .add(new StatImpl<Long>("jvm_class_unloaded_count") {
114            @Override public Long read() { return classLoadingBean.getUnloadedClassCount(); }
115          })
116          .add(new StatImpl<Long>("jvm_gc_collection_time_ms") {
117            @Override public Long read() {
118              long collectionTimeMs = 0;
119              for (GarbageCollectorMXBean bean : ManagementFactory.getGarbageCollectorMXBeans()) {
120                collectionTimeMs += bean.getCollectionTime();
121              }
122              return collectionTimeMs;
123            }
124          })
125          .add(new StatImpl<Long>("jvm_gc_collection_count") {
126            @Override public Long read() {
127              long collections = 0;
128              for (GarbageCollectorMXBean bean : ManagementFactory.getGarbageCollectorMXBeans()) {
129                collections += bean.getCollectionCount();
130              }
131              return collections;
132            }
133          })
134          .add(new StatImpl<Long>("jvm_memory_heap_mb_used") {
135            @Override public Long read() { return heapUsage.getUsed() / BYTES_PER_MB; }
136          })
137          .add(new StatImpl<Long>("jvm_memory_heap_mb_committed") {
138            @Override public Long read() { return heapUsage.getCommitted() / BYTES_PER_MB; }
139          })
140          .add(new StatImpl<Long>("jvm_memory_heap_mb_max") {
141            @Override public Long read() { return heapUsage.getMax() / BYTES_PER_MB; }
142          })
143          .add(new StatImpl<Long>("jvm_memory_non_heap_mb_used") {
144            @Override public Long read() { return nonHeapUsage.getUsed() / BYTES_PER_MB; }
145          })
146          .add(new StatImpl<Long>("jvm_memory_non_heap_mb_committed") {
147            @Override public Long read() { return nonHeapUsage.getCommitted() / BYTES_PER_MB; }
148          })
149          .add(new StatImpl<Long>("jvm_memory_non_heap_mb_max") {
150            @Override public Long read() { return nonHeapUsage.getMax() / BYTES_PER_MB; }
151          })
152          .add(new StatImpl<Long>("jvm_uptime_secs") {
153            @Override public Long read() {
154              return ManagementFactory.getRuntimeMXBean().getUptime() / 1000;
155            }
156          })
157          .add(new StatImpl<Double>("system_load_avg") {
158            @Override public Double read() { return osMbean.getSystemLoadAverage(); }
159          })
160        .build());
161    
162        Stats.exportString(
163            new StatImpl<String>("jvm_input_arguments") {
164              @Override public String read() {
165                return ManagementFactory.getRuntimeMXBean().getInputArguments().toString();
166              }
167            }
168        );
169    
170        for (final String property : System.getProperties().stringPropertyNames()) {
171          Stats.exportString(
172              new StatImpl<String>("jvm_prop_" + Stats.normalizeName(property)) {
173                @Override public String read() { return System.getProperty(property); }
174              });
175        }
176      }
177    }
