018    package com.twitter.common.stats;
019    
020    import com.google.common.base.Preconditions;
021    
022    public class PrintableHistogram {
023      private double[] bucketBoundaries;
024      private int[] bucketCounts;
025      private int totalCount = 0;
026    
027      /**
028       * Creates a histogram with the given bucket boundaries.  The boundaries
029       * 0 and infinity are implicitly added.
030       *
031       * @param buckets Boundaries for histogram buckets.
032       */
033      public PrintableHistogram(double ... buckets) {
034        Preconditions.checkState(buckets[0] != 0);
035    
036        bucketBoundaries = new double[buckets.length + 2];
037        bucketBoundaries[0] = 0;
038        bucketCounts = new int[buckets.length + 2];
039        for (int i = 0; i < buckets.length; i++) {
040          if (i > 0) {
041            Preconditions.checkState(buckets[i] > buckets[i - 1],
042                "Bucket %f must be greater than %f.", buckets[i], buckets[i - 1]);
043          }
044          bucketCounts[i] = 0;
045          bucketBoundaries[i + 1] = buckets[i];
046        }
047    
048        bucketBoundaries[bucketBoundaries.length - 1] = Integer.MAX_VALUE;
049      }
050    
051      public void addValue(double value) {
052        addValue(value, 1);
053      }
054    
055      public void addValue(double value, int count) {
056        Preconditions.checkState(value >= 0);
057        Preconditions.checkState(count >= 0);
058        Preconditions.checkState(bucketBoundaries.length > 1);
059        int bucketId = -1;
060        for (double boundary : bucketBoundaries) {
061          if (value <= boundary) {
062            break;
063          }
064          bucketId++;
065        }
066    
067        bucketId = Math.max(0, bucketId);
068        bucketId = Math.min(bucketCounts.length - 1, bucketId);
069        bucketCounts[bucketId] += count;
070        totalCount += count;
071      }
072    
073      public double getBucketRatio(int bucketId) {
074        Preconditions.checkState(bucketId >= 0);
075        Preconditions.checkState(bucketId < bucketCounts.length);
076        return (double) bucketCounts[bucketId] / totalCount;
077      }
078    
079      public String toString() {
080        StringBuilder display = new StringBuilder();
081        display.append("Histogram: ");
082        for (int bucketId = 0; bucketId < bucketCounts.length - 1; bucketId++) {
083          display.append(String.format("\n(%g - %g]\n\t",
084              bucketBoundaries[bucketId], bucketBoundaries[bucketId + 1]));
085          for (int i = 0; i < getBucketRatio(bucketId) * 100; i++) {
086            display.append('#');
087          }
088          display.append(
089              String.format(" %.2g%% (%d)", getBucketRatio(bucketId) * 100, bucketCounts[bucketId]));
090        }
091    
092        return display.toString();
093      }
094    }
