018    package com.twitter.common.stats;
019    
020    import com.twitter.common.application.ActionRegistry;
021    
022    import java.util.Set;
023    
024    /**
025     * A repository for time series data.
026     *
027     * @author William Farner
028     */
029    public interface TimeSeriesRepository {
030    
031      /**
032       * Starts the time series sampler.
033       *
034       * @param shutdownRegistry An action registry that the repository can use to register a shutdown
035       *    for the sampler.
036       */
037      public void start(ActionRegistry shutdownRegistry);
038    
039      /**
040       * Fetches the names of all available time series.
041       *
042       * @return Available time series, which can then be obtained by calling {@link #get(String)}.
043       */
044      public Set<String> getAvailableSeries();
045    
046      /**
047       * Fetches a time series by name.
048       *
049       * @param name The name of the time series to fetch.
050       * @return The time series registered with the given name, or {@code null} if no such time series
051       *     has been registered.
052       */
053      public TimeSeries get(String name);
054    
055      /**
056       * Gets an ordered iterable of the timestamps that all timeseries were sampled at.
057       *
058       * @return All current timestamps.
059       */
060      public Iterable<Number> getTimestamps();
061    }
