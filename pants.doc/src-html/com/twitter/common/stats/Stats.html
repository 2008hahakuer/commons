022    import com.google.common.base.Supplier;
023    import com.google.common.collect.ImmutableList;
024    import com.google.common.collect.Maps;
025    import com.twitter.common.base.MorePreconditions;
026    
027    import java.util.Arrays;
028    import java.util.Collections;
029    import java.util.LinkedHashMap;
030    import java.util.Map;
031    import java.util.concurrent.atomic.AtomicInteger;
032    import java.util.concurrent.atomic.AtomicLong;
033    import java.util.logging.Logger;

036    /**
037     * Manages {@link Stat}s that should be exported for monitoring.
038     *
039     * Statistic names may only contain {@code [A-Za-z0-9_]},
040     * all other chars will be logged as a warning and replaced with underscore on export.
041     *
042     * @author John Sirois
043     */
044    public class Stats {
045    
046      private static final Logger LOG = Logger.getLogger(Stats.class.getName());
047      private static final Pattern NOT_NAME_CHAR = Pattern.compile("[^A-Za-z0-9_]");
048    
049      private static final Map<String, Stat> VAR_MAP =
050          Collections.synchronizedMap(Maps.<String, Stat>newHashMap());
051    
052      // Store stats in the order they were registered, so that derived variables are
053      // sampled after their inputs.
054      private static final Map<String, RecordingStat<? extends Number>> NUMERIC_STATS =
055          Collections.synchronizedMap(new LinkedHashMap<String, RecordingStat<? extends Number>>());
056    
057      public static String normalizeName(String name) {
058        return NOT_NAME_CHAR.matcher(name).replaceAll("_");
059      }
060    
061      private static String validateName(String name) {
062        String normalized = normalizeName(name);
063        if (!name.equals(normalized)) {
064          LOG.warning("Invalid stat name " + name + " exported as " + normalized);
065        }
066        return normalized;
067      }
068    
069      /**
070       * A {@code StatsProvider} that exports gauge-style stats to the global {@link Stat}s repository
071       * for time series tracking.
072       */
073      public static final StatsProvider STATS_PROVIDER = new StatsProvider() {
074        @Override public <T extends Number> Stat<T> makeGauge(String name, final Supplier<T> gauge) {
075          return Stats.export(new StatImpl<T>(name) {
076            @Override public T read() {
077              return gauge.get();
078            }
079          });
080        }
081    
082        @Override public AtomicLong makeCounter(final String name) {
083          return Stats.exportLong(name);
084        }
085    
086        @Override public RequestTimer makeRequestTimer(String name) {
087          return new RequestStats(name);
088        }
089      };
090    
091      /**
092       * Exports a stat for tracking.
093       * if the stat provided implements the internal {@link RecordingStat} interface, it will be
094       * registered for time series collection and returned.  If a {@link RecordingStat} with the same
095       * name as the provided stat has already been exported, the previously-exported stat will be
096       * returned and no additional registration will be performed.
097       *
098       * @param var The variable to export.
099       * @param <T> The value exported by the variable.
100       * @return A reference to the stat that was stored.  The stat returned may not be equal to the
101       *    stat provided.  If a variable was already returned with the same
102       */
103      public static synchronized <T extends Number> Stat<T> export(Stat<T> var) {
104        Preconditions.checkNotNull(var);
105        MorePreconditions.checkNotBlank(var.getName());
106    
107        if (var instanceof RecordingStat) {
108          String validatedName = validateName(var.getName());
109          @SuppressWarnings("unchecked")
110          Stat<T> stat = (Stat<T>) NUMERIC_STATS.get(validatedName);
111          if (stat != null) {
112            LOG.warning("Re-using already registered variable for key " + validatedName);
113            return stat;
114          } else {
115            NUMERIC_STATS.put(validatedName, (RecordingStat<T>) var);
116    
117            exportStatic(var);
118            return var;
119          }
120        } else {
121          return export(new RecordingStatImpl<T>(var));
122        }
123      }
124    
125      /**
126       * Exports a string stat.
127       * String-based statistics will not be registered for time series collection.
128       *
129       * @param var Stat to export.
130       * @return A reference back to {@code var}, or the variable that was already registered under the
131       *    same name as {@code var}.
132       */
133      public static Stat<String> exportString(Stat<String> var) {
134        return exportStatic(var);
135      }
136    
137      /**
138       * Adds a collection of stats for export.
139       *
140       * @param vars The variables to add.
141       */
142      public static void exportAll(Iterable<Stat<? extends Number>> vars) {
143        for (Stat<? extends Number> var : vars) {
144          export(var);
145        }
146      }
147    
148      /**
149       * Exports an {@link AtomicInteger}, which will be included in time series tracking.
150       *
151       * @param name The name to export the stat with.
152       * @param intVar The variable to export.
153       * @return A reference to the {@link AtomicInteger} provided.
154       */
155      public static AtomicInteger export(final String name, final AtomicInteger intVar) {
156        export(new SampledStat<Integer>(name, 0) {
157          @Override public Integer doSample() { return intVar.get(); }
158        });
159    
160        return intVar;
161      }
162    
163      /**
164       * Creates and exports an {@link AtomicInteger}.
165       *
166       * @param name The name to export the stat with.
167       * @return A reference to the {@link AtomicInteger} created.
168       */
169      public static AtomicInteger exportInt(final String name) {
170        return export(name, new AtomicInteger());
171      }
172    
173      /**
174       * Exports an {@link AtomicLong}, which will be included in time series tracking.
175       *
176       * @param name The name to export the stat with.
177       * @param longVar The variable to export.
178       * @return A reference to the {@link AtomicLong} provided.
179       */
180      public static AtomicLong export(final String name, final AtomicLong longVar) {
181        export(new StatImpl<Long>(name) {
182          @Override public Long read() { return longVar.get(); }
183        });
184    
185        return longVar;
186      }
187    
188      /**
189       * Creates and exports an {@link AtomicLong}.
190       *
191       * @param name The name to export the stat with.
192       * @return A reference to the {@link AtomicLong} created.
193       */
194      public static AtomicLong exportLong(final String name) {
195        return export(name, new AtomicLong());
196      }
197    
198      /**
199       * Exports a 'static' statistic, which will not be registered for time series tracking.
200       *
201       * @param var Variable to statically export.
202       * @return A reference back to the provided {@link Stat}.
203       */
204      public static synchronized <T> Stat<T> exportStatic(Stat<T> var) {
205        Preconditions.checkNotNull(var);
206        MorePreconditions.checkNotBlank(var.getName());
207    
208        String validatedName = validateName(var.getName());
209        @SuppressWarnings("unchecked")
210        Stat<T> displaced = (Stat<T>) VAR_MAP.put(validatedName, var);
211        if (displaced != null) {
212          LOG.warning("Warning - exported variable collision on " + validatedName);
213        }
214    
215        return var;
216      }
217    
218      /**
219       * Fetches all registered stat.
220       *
221       * @return An iterable of all registered stats.
222       */
223      public static Iterable<? extends Stat> getVariables() {
224        synchronized(VAR_MAP) {
225          return ImmutableList.copyOf(VAR_MAP.values());
226        }
227      }
228    
229      static Iterable<RecordingStat<? extends Number>> getNumericVariables() {
230        synchronized(NUMERIC_STATS) {
231          return ImmutableList.copyOf(NUMERIC_STATS.values());
232        }
233      }
234    
235      @VisibleForTesting
236      public static void flush() {
237        VAR_MAP.clear();
238        NUMERIC_STATS.clear();
239      }
240    
241      @SuppressWarnings("unchecked")
242      public static <T> Stat<T> getVariable(String name) {
243        MorePreconditions.checkNotBlank(name);
244        return VAR_MAP.get(name);
245      }
