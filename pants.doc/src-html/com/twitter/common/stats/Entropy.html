018    package com.twitter.common.stats;
019    
020    import com.google.common.base.Preconditions;
021    
022    /**
023     * Calculate the entropy of a discrete distribution of <T>.
024     *
025     * @author Gilad Mishne
026     */
027    public class Entropy<T> {
028      private final CounterMap<T> counts = new CounterMap<T>();
029      private int total = 0;
030    
031      private static double Log2(double n) {
032        return Math.log(n) / Math.log(2);
033      }
034    
035      public Entropy(Iterable<T> elements) {
036        Preconditions.checkNotNull(elements);
037        for (T element : elements) {
038          counts.incrementAndGet(element);
039          total++;
040        }
041      }
042    
043      public double entropy() {
044        double entropy = 0;
045        for (int count: counts.values()) {
046          double prob = (double) count / total;
047          entropy -= prob * Log2(prob);
048        }
049        return entropy;
050      }
051    
052      public double perplexity() {
053        return Math.pow(2, entropy());
054      }
055    }
