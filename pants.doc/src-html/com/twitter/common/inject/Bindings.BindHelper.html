018    package com.twitter.common.inject;
019    
020    import com.google.common.base.Preconditions;
021    import com.google.inject.AbstractModule;
022    import com.google.inject.Binder;
023    import com.google.inject.Key;
024    import com.google.inject.binder.LinkedBindingBuilder;
025    import com.google.inject.multibindings.Multibinder;
026    import com.google.inject.name.Names;
027    
028    /**
029     * A utility that helps with guice bindings.
030     *
031     * @author John Sirois
032     */
033    public final class Bindings {
034    
035      /**
036       * Equivalent to calling {@code requireBinding(binder, Key.get(required, Names.named(namedKey)))}.
037       */
038      public static void requireNamedBinding(Binder binder, Class<?> required, String namedKey) {
039        requireBinding(binder, Key.get(Preconditions.checkNotNull(required),
040            Names.named(Preconditions.checkNotNull(namedKey))));
041      }
042    
043      /**
044       * Equivalent to calling {@code requireBinding(binder, Key.get(required))}.
045       */
046      public static void requireBinding(Binder binder, Class<?> required) {
047        requireBinding(binder, Key.get(Preconditions.checkNotNull(required)));
048      }
049    
050      /**
051       * Registers {@code required} as non-optional dependency in the {@link com.google.inject.Injector}
052       * associated with {@code binder}.
053       *
054       * @param binder a guice binder to require bindings against
055       * @param required the dependency that is required
056       */
057      public static void requireBinding(Binder binder, final Key<?> required) {
058        Preconditions.checkNotNull(binder);
059        Preconditions.checkNotNull(required);
060    
061        binder.install(new AbstractModule() {
062          @Override protected void configure() {
063            requireBinding(required);
064          }
065        });
066      }
067    
068      /**
069       * A guice binding helper that allows for any combination of Class, TypeLiteral or Key binding
070       * without forcing guiced implementation to provide all the overloaded binding methods they would
071       * otherwise have to.
072       *
073       * @param <T> the type this helper can be used to bind implementations for
074       */
075      public interface BindHelper<T> {
076    
077        /**
078         * Associates this BindHelper with an Injector instance.
079         *
080         * @param binder the binder for the injector implementations will be bound in
081         * @return a binding builder that can be used to bind an implementation with
082         */
083        LinkedBindingBuilder<T> with(Binder binder);
084      }
085    
086      /**
087       * Creates a BindHelper for the given binding key that can be used to bind a single instance.
088       *
089       * @param key the binding key the returned BindHelper can be use to bind implementations for
090       * @param <T> the type the returned BindHelper can be used to bind implementations for
091       * @return a BindHelper that can be used to bind an implementation with
092       */
093      public static <T> BindHelper<T> binderFor(final Key<T> key) {
094        return new BindHelper<T>() {
095          public LinkedBindingBuilder<T> with(Binder binder) {
096            return binder.bind(key);
097          }
098        };
099      }
100    
101      /**
102       * Creates a BindHelper for the given type that can be used to add a binding of to a set.
103       *
104       * @param type the type the returned BindHelper can be use to bind implementations for
105       * @param <T> the type the returned BindHelper can be used to bind implementations for
106       * @return a BindHelper that can be used to bind an implementation with
107       */
108      public static <T> BindHelper<T> multiBinderFor(final Class<T> type) {
109        return new BindHelper<T>() {
110          public LinkedBindingBuilder<T> with(Binder binder) {
111            return Multibinder.newSetBinder(binder, type).addBinding();
112          }
113        };
114      }
115    
116      private Bindings() {
117        // utility
118      }
119    }
