018    package com.twitter.common.args;
019    
020    import javax.annotation.Nullable;
021    
022    import com.google.common.annotations.VisibleForTesting;
023    import com.google.common.base.Preconditions;
024    
025    /**
026     * Wrapper class for the value of an argument.  For proper behavior, an {@code Arg} should always
027     * be annotated with {@link CmdLine}, which will define the command line interface to the argument.
028     *
029     * @author William Farner
030     */
031    public class Arg<T> {
032    
033      private final T defaultValue;
034      private T value;
035      private boolean hasDefault = true;
036      private boolean valueApplied = false;
037    
038      /**
039       * Creates an arg that has no default value, meaning that its value can only ever be retrieved
040       * if it has been externally set.
041       */
042      public Arg() {
043        this(null);
044        this.hasDefault = false;
045      }
046    
047      /**
048       * Creates an arg that has a default value, and may optionally be set.
049       *
050       * @param defaultValue The default value for the arg.
051       */
052      public Arg(@Nullable T defaultValue) {
053        this.defaultValue = defaultValue;
054        this.value = defaultValue;
055      }
056    
057      synchronized void set(T value) {
058        Preconditions.checkState(!valueApplied, "A value cannot be applied twice to an argument.");
059        valueApplied = true;
060        this.value = value;
061      }
062    
063      @VisibleForTesting
064      synchronized void reset() {
065        valueApplied = false;
066        this.value = hasDefault ? defaultValue : null;
067      }
068    
069      /**
070       * Gets the value of the argument.  If a value has not yet been applied to the argument, or the
071       * argument did not provide a default value, {@link IllegalStateException} will be thrown.
072       *
073       * @return The argument value.
074       */
075      public synchronized T get() {
076        // TODO(William Farner): This has a tendency to break bad-arg reporting by ArgScanner.  Fix.
077        Preconditions.checkState(valueApplied || hasDefault,
078            "A value may only be retrieved from a variable that has a default or has been set.");
079        return uncheckedGet();
080      }
081    
082      /**
083       * Gets the value of the argument, without checking whether a default was available or if a
084       * value was applied.
085       *
086       * @return The argument value.
087       */
088      synchronized T uncheckedGet() {
089        return value;
090      }
091    
092      /**
093       * Convenience factory method to create an arg that has no default value.
094       *
095       * @param <T> Type of arg value.
096       * @return A new arg.
097       */
098      public static <T> Arg<T> create() {
099        return new Arg<T>();
100      }
101    
102      /**
103       * Convenience factory method to create an arg with a default value.
104       *
105       * @param value Default argument value.
106       * @param <T> Type of arg value.
107       * @return A new arg.
108       */
109      public static <T> Arg<T> create(T value) {
110        return new Arg<T>(value);
111      }
112    }
