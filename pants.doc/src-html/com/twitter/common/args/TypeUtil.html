018    package com.twitter.common.args;
019    
020    import java.lang.reflect.Field;
021    import java.lang.reflect.ParameterizedType;
022    import java.lang.reflect.Type;
023    import java.util.Arrays;
024    import java.util.List;
025    
026    import com.google.common.base.Function;
027    import com.google.common.base.Preconditions;
028    import com.google.common.collect.Lists;
029    
030    /**
031     * Utility class to extract generic type information.
032     *
033     * TODO(William Farner): Move this into a common library, integrate with EasyMockTest.Clazz.
034     *
035     * @author William Farner
036     */
037    public class TypeUtil {
038    
039      private static final Function<Type, Class<?>> GET_CLASS = new Function<Type, Class<?>>() {
040        @Override public Class<?> apply(Type type) {
041          if (type instanceof ParameterizedType) {
042            return GET_CLASS.apply(((ParameterizedType) type).getRawType());
043          }
044    
045          return (Class) type;
046        }
047      };
048    
049      private TypeUtil() {
050        // Utility.
051      }
052    
053      /**
054       * Convenience method to call {@link #getTypeParamClasses(Type)} on a {@link Field}.
055       *
056       * @param field The field to extract type parameters from.
057       * @return The raw classes of types that {@code field} is parameterized with.
058       */
059      public static List<Class<?>> getTypeParamClasses(Field field) {
060        return getTypeParamClasses(field.getGenericType());
061      }
062    
063      /**
064       * Gets the clases that a type is type-parameterized with, in declaration order.
065       *
066       * @param type The type to extract type parameters from.
067       * @return The raw classes of types that {@code field} is parameterized with.
068       */
069      public static List<Class<?>> getTypeParamClasses(Type type) {
070        return Lists.transform(Arrays.asList(
071            ((ParameterizedType) type).getActualTypeArguments()), GET_CLASS);
072      }
073    
074      /**
075       * Convenience method to call {@link #getTypeParamClasses(Field)}, with the requirement that there
076       * is exactly one type parameter on the field.
077       *
078       * @param field The field to extract type parameters from.
079       * @return The raw classes of types that {@code field} is parameterized with.
080       */
081      public static Class getTypeParamClass(Field field) {
082        List<Class<?>> typeParams = getTypeParamClasses(field);
083        Preconditions.checkArgument(typeParams.size() == 1,
084            "Expected exactly one type parameter for field " + field);
085        return typeParams.get(0);
086      }
087    
088      /**
089       * Gets the type parameter from a field.  Assumes that there is at least one type parameter.
090       *
091       * @param field The field to extract the type parameter from.
092       * @return The field type parameter.
093       */
094      public static Type getTypeParam(Field field) {
095        return ((ParameterizedType) field.getGenericType()).getActualTypeArguments()[0];
096      }
097    
098      /**
099       * A function to extract the type parameter class from fields.
100       */
101      public static Function<Field, Class> GET_TYPE_PARAM_CLASS = new Function<Field, Class>() {
102        @Override public Class apply(Field field) {
103          return getTypeParamClass(field);
104        }
105      };
106    }
