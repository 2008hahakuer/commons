018    package com.twitter.common.args.parsers;
019    
020    import java.util.List;
021    import java.util.regex.Matcher;
022    import java.util.regex.Pattern;
023    
024    import com.twitter.common.args.Parsers.Parser;
025    import com.twitter.common.quantity.Amount;
026    import com.twitter.common.quantity.Unit;
027    
028    import static com.google.common.base.Preconditions.checkArgument;
029    import static com.twitter.common.args.Parsers.checkedGet;
030    
031    /**
032     * amount parser.
033     *
034     * @author William Farner
035     */
036    public class AmountParser extends TypeParameterizedParser<Amount> {
037    
038      public AmountParser() {
039        super(Amount.class, 2);
040      }
041    
042      @SuppressWarnings("unchecked")
043      @Override
044      Amount doParse(String raw, List<Class<?>> paramParsers) {
045        final Parser parser = checkedGet(paramParsers.get(0));
046    
047        Matcher matcher = Pattern.compile("(\\d+)([A-Za-z]+)").matcher(raw);
048        checkArgument(matcher.matches(),
049            "Value '" + raw + "' must be of the format 1ns, 4mb, etc.");
050    
051        Number number = (Number) parser.parse(null, matcher.group(1));
052        String unitRaw = matcher.group(2);
053    
054        Unit unit = (Unit) checkedGet(Unit.class).parse(null, unitRaw);
055        checkArgument(unit.getClass() == paramParsers.get(1), String.format(
056            "Unit type (%s) does not match argument type (%s).",
057            unit.getClass(), paramParsers.get(1)));
058    
059        Class numberClass = paramParsers.get(0);
060        if (numberClass == Integer.class) {
061          return Amount.of(number.intValue(), unit);
062        } else if (numberClass == Double.class) {
063          return Amount.of(number.doubleValue(), unit);
064        } else if (numberClass == Long.class) {
065          return Amount.of(number.longValue(), unit);
066        } else if (numberClass == Byte.class) {
067          return Amount.of(number.byteValue(), unit);
068        } else if (numberClass == Short.class) {
069          return Amount.of(number.shortValue(), unit);
070        } else if (numberClass == Float.class) {
071          return Amount.of(number.floatValue(), unit);
072        }
073    
074        throw new IllegalArgumentException("Unrecognized number class "
075            + numberClass.getCanonicalName());
076      }
077    }
