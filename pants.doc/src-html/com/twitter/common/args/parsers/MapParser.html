018    package com.twitter.common.args.parsers;
019    
020    import java.util.List;
021    import java.util.Map;
022    
023    import com.google.common.base.Splitter;
024    import com.google.common.collect.ImmutableList;
025    import com.google.common.collect.ImmutableMap;
026    
027    import com.twitter.common.args.Parsers;
028    import com.twitter.common.args.Parsers.Parser;
029    
030    import static com.google.common.base.Preconditions.checkArgument;
031    import static com.twitter.common.args.Parsers.checkedGet;
032    
033    /**
034     * Map parser.
035     *
036     * @author William Farner
037     */
038    public class MapParser extends TypeParameterizedParser<Map> {
039    
040      private static final Splitter KEY_VALUE_SPLITTER = Splitter.on("=")
041          .trimResults().omitEmptyStrings();
042    
043      public MapParser() {
044        super(Map.class, 2);
045      }
046    
047      @SuppressWarnings("unchecked")
048      @Override
049      Map doParse(String raw, List<Class<?>> paramParsers) {
050        final Parser keyParser = checkedGet(paramParsers.get(0));
051        final Parser valueParser = checkedGet(paramParsers.get(1));
052    
053        ImmutableMap.Builder map = ImmutableMap.builder();
054        for (String keyAndValue : Parsers.MULTI_VALUE_SPLITTER.split(raw)) {
055          List<String> fields = ImmutableList.copyOf(KEY_VALUE_SPLITTER.split(keyAndValue));
056          checkArgument(fields.size() == 2,
057              "Failed to parse key/value pair " + keyAndValue);
058    
059          map.put(keyParser.parse(null, fields.get(0)), valueParser.parse(null, fields.get(1)));
060        }
061    
062        return map.build();
063      }
064    }
