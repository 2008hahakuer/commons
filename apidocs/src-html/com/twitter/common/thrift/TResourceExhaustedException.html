018    package com.twitter.common.thrift;
019    
020    import org.apache.thrift.TException;
021    
022    /**
023     * @author Adam Samet
024     *
025     * This is exception is thrown when there are no available instances of a thrift backend
026     * service to serve requests.
027     */
028    public class TResourceExhaustedException extends TException {
029    
030      private static final long serialVersionUID = 1L;
031    
032      public TResourceExhaustedException(String message) {
033        super(message);
034      }
035    
036      public TResourceExhaustedException(Throwable cause) {
037        super(cause);
038      }
039    
040      public TResourceExhaustedException(String message, Throwable cause) {
041        super(message, cause);
042      }
043    }
