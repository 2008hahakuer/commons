081      public static final ExceptionalFunction<ServerSetup, TServer, TTransportException>
082          THREADPOOL_SERVER = new ExceptionalFunction<ServerSetup, TServer, TTransportException>() {
083            @Override public TServer apply(ServerSetup setup) throws TTransportException {
084              TThreadPoolServer.Options options = new TThreadPoolServer.Options();
085              if (setup.getNumThreads() > 0) {
086                options.minWorkerThreads = setup.getNumThreads();
087                options.maxWorkerThreads = setup.getNumThreads();
088              }
089    
090              try {
091                setup.setSocket(new ServerSocket(setup.getPort()));
092              } catch (IOException e) {
093                throw new TTransportException("Failed to create server socket on port " +
094                    setup.getPort(), e);
095              }
096    
097              TServerSocket unmonitoredSocket = null;
098              TMonitoredServerSocket monitoredSocket = null;
099              if (setup.isMonitored()) {
100                monitoredSocket = new TMonitoredServerSocket(setup.getSocket(),
101                      setup.getSocketTimeout().as(Time.MILLISECONDS), setup.getMonitor());
102              } else {
103                unmonitoredSocket = new TServerSocket(setup.getSocket(),
104                    setup.getSocketTimeout().as(Time.MILLISECONDS));
105              }
106    
107              TTransportFactory transportFactory = new TTransportFactory();
108              TProcessor processor = setup.getProcessor();
109              if (setup.isMonitored()) {
110                processor = new TMonitoredProcessor(processor, monitoredSocket,
111                    setup.getMonitor());
112              }
113    
114              TServerSocket socket = setup.isMonitored() ? monitoredSocket : unmonitoredSocket;
115              return new TThreadPoolServer(processor, socket, transportFactory, transportFactory,
116                  setup.getProtoFactory(), setup.getProtoFactory(), options);
117            }
118          };
119      public static final ExceptionalFunction<ServerSetup, TServer, TTransportException>
120          NONBLOCKING_SERVER = new ExceptionalFunction<ServerSetup, TServer, TTransportException>() {
121            @Override public TServer apply(ServerSetup setup) throws TTransportException {
122              TNonblockingServerSocket socket = setup.getSocketTimeout() == null
123                  ? new TNonblockingServerSocket(setup.getPort())
124                  : new TNonblockingServerSocket(setup.getPort(),
125                      setup.getSocketTimeout().as(Time.MILLISECONDS));
126    
127              setup.setSocket(getServerSocketFor(socket));
128    
129              // just to grab defaults
130              THsHaServer.Options options = new THsHaServer.Options();
131              if (setup.getNumThreads() > 0) {
132                options.workerThreads = setup.getNumThreads();
133              }
134    
135              // default queue size to num threads:  max response time becomes double avg service time
136              final BlockingQueue<Runnable> queue =
137                new ArrayBlockingQueue<Runnable>(setup.getQueueSize() > 0 ? setup.getQueueSize()
138                    : options.workerThreads);
139              final ThreadPoolExecutor invoker = new ThreadPoolExecutor(options.workerThreads,
140                  options.workerThreads, options.stopTimeoutVal, options.stopTimeoutUnit, queue);
141    
142              final String serverName = (setup.getName() != null ? setup.getName() : "no-name");
143              Stats.export(new StatImpl<Integer>(serverName + "_thrift_server_active_threads") {
144                @Override public Integer read() { return invoker.getActiveCount(); }
145              });
146              Stats.export(new StatImpl<Integer>(serverName + "_thrift_server_queue_size") {
147                @Override public Integer read() { return queue.size(); }
148              });
149    
150              return new THsHaServer(new TProcessorFactory(setup.getProcessor()), socket,
151                  new TFramedTransport.Factory(),
152                  setup.getProtoFactory(), setup.getProtoFactory(), invoker,
153                  new TNonblockingServer.Options());
154            }
155          };

157      /**
158       * Thrift doesn't provide access to socket it creates,
159       * this is the only way to know what ephemeral port we bound to.
160       * TODO:  Patch thrift to provide access so we don't have to do this.
161       */
162      @VisibleForTesting
163      static ServerSocket getServerSocketFor(TNonblockingServerSocket thriftSocket)
164          throws TTransportException {
165        try {
166          Field field = TNonblockingServerSocket.class.getDeclaredField("serverSocket_");
167          field.setAccessible(true);
168          return (ServerSocket) field.get(thriftSocket);
169        } catch (NoSuchFieldException e) {
170          throw new TTransportException("Couldn't get listening port", e);
171        } catch (SecurityException e) {
172          throw new TTransportException("Couldn't get listening port", e);
173        } catch (IllegalAccessException e) {
174          throw new TTransportException("Couldn't get listening port", e);
175        }
176      }

206      /**
207       * Creates a new thrift server with the provided configuration.
208       *
209       * @param name Name for the server.
210       * @param version Version identifier.
211       * @param protoFactorySupplier Supplier to build the protocol factory to use.
212       * @param serverSupplier Function to build a TServer object based on the server setup.
213       */
214      public ThriftServer(String name, String version, Supplier<TProtocolFactory> protoFactorySupplier,
215          ExceptionalFunction<ServerSetup, TServer, TTransportException> serverSupplier) {
216        this.name = Preconditions.checkNotNull(name);
217        this.version = Preconditions.checkNotNull(version);
218        this.protoFactorySupplier = Preconditions.checkNotNull(protoFactorySupplier);
219        this.serverSupplier = Preconditions.checkNotNull(serverSupplier);
220      }

222      /**
223       * Starts the server.
224       * This may be called at any point except when the server is already alive.  That is, it's
225       * allowable to start, stop, and re-start the server.
226       *
227       * @param port The port to listen on.
228       * @param processor The processor to handle requests.
229       */
230      public void start(int port, TProcessor processor) {
231        start(new ServerSetup(name, port, processor, protoFactorySupplier.get()));
232      }
233    
234      /**
235       * Starts the server.
236       * This may be called at any point except when the server is already alive.  That is, it's
237       * allowable to start, stop, and re-start the server.
238       *
239       * @param setup options for server
240       */
241      public void start(ServerSetup setup) {
242        Preconditions.checkNotNull(setup.getProcessor());
243        Preconditions.checkState(status != Status.ALIVE, "Server must only be started once.");
244        setStatus(Status.ALIVE);
245        try {
246          doStart(setup);
247        } catch (TTransportException e) {
248          LOG.log(Level.SEVERE, "Failed to open thrift socket.", e);
249          setStatus(Status.DEAD);
250        }
251      }
252    
253      @VisibleForTesting
254      protected void doStart(ServerSetup setup) throws TTransportException {
255        serverSetup = setup;
256        server = serverSupplier.apply(setup);
257        serverStartNanos = System.nanoTime();
258        LOG.info("Starting thrift server on port " + getListeningPort());
259    
260        listeningThread = new ThreadFactoryBuilder().setDaemon(false).build().newThread(new Runnable() {
261          @Override public void run() {
262            try {
263              server.serve();
264            } catch (Throwable t) {
265              LOG.log(Level.WARNING,
266                  "Uncaught exception while attempting to handle service requests: " + t, t);
267              setStatus(Status.DEAD);
268            }
269          }
270        });
271    
272        listeningThread.start();
273      }
274    
275      public int getListeningPort() {
276        Preconditions.checkState(serverSetup != null);
277        Preconditions.checkState(status == Status.ALIVE);
278        Preconditions.checkState(serverSetup.getSocket() != null);
279        return serverSetup.getSocket().getLocalPort();
280      }

312      /**
313       * Notification to the server that a shutdown request has been made, and the server is no longer
314       * processing requests.  The implementer may veto the shutdown by throwing an exception.  A veto
315       * would suggest a failure to terminate backend connections in a timely manner.
316       *
317       * @throws Exception If the shutdown request could not be honored.
318       */
319      protected abstract void tryShutdown() throws Exception;
320    
321      /**
322       * Attempts to shut down the server.
323       * The server may be shut down at any time, though the request will be ignored if the server is
324       * already stopped.
325       */
326      @Override
327      public void shutdown() {
328        if (status == Status.STOPPED) {
329          LOG.info("Server already stopped, shutdown request ignored.");
330          return;
331        }
332    
333        LOG.info("Received shutdown request, stopping server.");
334        setStatus(Status.STOPPING);
335    
336        if (server != null) server.stop();
337        server = null;
338    
339        // TODO(William Farner): Figure out what happens to queued / in-process requests when the server is
340        // stopped.  Might want to allow a sleep period for the active requests to be completed.
341    
342        try {
343          tryShutdown();
344        } catch (Exception e) {
345          LOG.log(Level.WARNING, "Service handler vetoed shutdown request.", e);
346          setStatus(Status.WARNING);
347          return;
348        }
349    
350        setStatus(Status.STOPPED);
351      }
352    
353      /**
354       * Attempts to shut down this server, and waits for the shutdown operation to complete.
355       *
356       * @param timeout Maximum amount of time to wait for shutdown before giving up.  a timeout of
357       *    zero means wait forever.
358       *
359       * @throws InterruptedException If interrupted while waiting for shutdown.
360       */
361      public void awaitShutdown(Amount<Long, Time> timeout) throws InterruptedException {
362        Preconditions.checkNotNull(timeout);
363        shutdown();
364    
365        if (listeningThread != null) {
366          listeningThread.join(timeout.as(Time.MILLISECONDS));
367        }
368      }

419            int numThreads, Amount<Integer, Time> socketTimeout,
420            TrafficMonitor<InetSocketAddress> monitor) {
421          this(name, port, processor, protoFactory, numThreads, -1, socketTimeout, monitor);
422        }
423    
424        public ServerSetup(String name, int port, TProcessor processor, TProtocolFactory protoFactory,
425              int numThreads, int queueSize, Amount<Integer, Time> socketTimeout,
426              TrafficMonitor<InetSocketAddress> monitor) {
427          Preconditions.checkArgument(port >= 0 && port < 0xFFFF, "Invalid port: " + port);
428          Preconditions.checkArgument(numThreads != 0);
429          Preconditions.checkArgument(queueSize != 0);
430          if (socketTimeout != null) Preconditions.checkArgument(socketTimeout.getValue() >= 0);
431          this.name = name;
432          this.port = port;
433          this.processor = processor;
434          this.protoFactory = protoFactory;
435          this.numThreads = numThreads;
436          this.queueSize = queueSize;
437          this.socketTimeout = socketTimeout;
438          this.monitor = monitor;
439        }

482          return monitor;
483        }
