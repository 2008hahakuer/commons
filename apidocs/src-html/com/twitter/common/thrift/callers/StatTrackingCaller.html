017    package com.twitter.common.thrift.callers;
018    
019    import com.google.common.base.Function;
020    import com.google.common.collect.MapMaker;
021    import com.twitter.common.quantity.Amount;
022    import com.twitter.common.quantity.Time;
023    import com.twitter.common.stats.StatsProvider;
024    import com.twitter.common.thrift.TResourceExhaustedException;
025    import com.twitter.common.thrift.TTimeoutException;
026    import org.apache.thrift.async.AsyncMethodCallback;
027    
028    import javax.annotation.Nullable;
029    import java.lang.reflect.Method;
030    import java.util.Map;
031    import java.util.concurrent.TimeUnit;
032    import java.util.concurrent.TimeoutException;
033    
034    /**
035     * A caller that exports statistics about calls made to the wrapped caller.
036     *
037     * @author William Farner
038     */
039    public class StatTrackingCaller extends CallerDecorator {
040    
041      private final StatsProvider statsProvider;
042      private final String serviceName;
043    
044      private final Map<Method, StatsProvider.RequestTimer> stats =
045          new MapMaker().makeComputingMap(new Function<Method, StatsProvider.RequestTimer>() {
046            @Override public StatsProvider.RequestTimer apply(Method method) {
047              // Thrift does not support overloads - so just the name disambiguates all calls.
048              return statsProvider.makeRequestTimer(serviceName + "_" + method.getName());
049            }
050          });
051    
052      /**
053       * Creates a new stat tracking caller, which will export stats to the given {@link StatsProvider}.
054       *
055       * @param decoratedCaller The caller to decorate with a deadline.
056       * @param async Whether the caller is asynchronous.
057       * @param statsProvider The stat provider to export statistics to.
058       * @param serviceName The name of the service that methods are being called on.
059       */
060      public StatTrackingCaller(Caller decoratedCaller, boolean async, StatsProvider statsProvider,
061          String serviceName) {
062        super(decoratedCaller, async);
063    
064        this.statsProvider = statsProvider;
065        this.serviceName = serviceName;
066      }
067    
068      @Override
069      public Object call(Method method, Object[] args, @Nullable AsyncMethodCallback callback,
070          @Nullable Amount<Long, Time> connectTimeoutOverride) throws Throwable {
071        final StatsProvider.RequestTimer requestStats = stats.get(method);
072        final long startTime = System.nanoTime();
073    
074        ResultCapture capture = new ResultCapture() {
075          @Override public void success() {
076            requestStats.requestComplete(TimeUnit.NANOSECONDS.toMicros(
077                System.nanoTime() - startTime));
078          }
079    
080          @Override public boolean fail(Throwable t) {
081            // TODO(John Sirois): the ruby client reconnects for timeouts too - this provides a natural
082            // backoff mechanism - consider how to plumb something similar.
083            if (t instanceof TTimeoutException || t instanceof TimeoutException) {
084              requestStats.incTimeouts();
085              return true;
086            }
087    
088            // TODO(John Sirois): consider ditching reconnects since its nearly redundant with errors as
089            // it stands.
090            if (!(t instanceof TResourceExhaustedException)) {
091              requestStats.incReconnects();
092            }
093            // TODO(John Sirois): provide more detailed stats: track counts for distinct exceptions types,
094            // track retries-per-method, etc...
095            requestStats.incErrors();
096            return true;
097          }
098        };
099    
100        return invoke(method, args, callback, capture, connectTimeoutOverride);
101      }
102    }
