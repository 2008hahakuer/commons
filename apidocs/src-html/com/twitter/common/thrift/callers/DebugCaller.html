017    package com.twitter.common.thrift.callers;
018    
019    import com.google.common.base.Joiner;
020    import com.google.common.base.Throwables;
021    import com.twitter.common.quantity.Amount;
022    import com.twitter.common.quantity.Time;
023    import org.apache.thrift.async.AsyncMethodCallback;
024    
025    import javax.annotation.Nullable;
026    import java.lang.reflect.InvocationTargetException;
027    import java.lang.reflect.Method;
028    import java.util.logging.Logger;
029    
030    /**
031     * A caller that reports debugging information about calls.
032     *
033     * @author William Farner
034     */
035    public class DebugCaller extends CallerDecorator {
036      private static final Logger LOG = Logger.getLogger(DebugCaller.class.getName());
037      private static final Joiner ARG_JOINER = Joiner.on(", ");
038    
039      /**
040       * Creates a new debug caller.
041       *
042       * @param decoratedCaller The caller to decorate with debug information.
043       * @param async Whether the caller is asynchronous.
044       */
045      public DebugCaller(Caller decoratedCaller, boolean async) {
046        super(decoratedCaller, async);
047      }
048    
049      @Override
050      public Object call(final Method method, final Object[] args,
051          @Nullable AsyncMethodCallback callback, @Nullable Amount<Long, Time> connectTimeoutOverride)
052          throws Throwable {
053        ResultCapture capture = new ResultCapture() {
054          @Override public void success() {
055            // No-op.
056          }
057    
058          @Override public boolean fail(Throwable t) {
059            StringBuilder message = new StringBuilder("Thrift call failed: ");
060            message.append(method.getName()).append("(");
061            ARG_JOINER.appendTo(message, args);
062            message.append(")");
063            LOG.warning(message.toString());
064    
065            return true;
066          }
067        };
068    
069        try {
070          return invoke(method, args, callback, capture, connectTimeoutOverride);
071        } catch (Throwable t) {
072          capture.fail(t);
073          throw t;
074        }
075      }
076    }
