018    package com.twitter.common.thrift.callers;
019    
020    import com.google.common.base.Function;
021    import com.google.common.collect.Lists;
022    import com.twitter.common.net.pool.Connection;
023    import com.twitter.common.net.pool.ObjectPool;
024    import com.twitter.common.quantity.Amount;
025    import com.twitter.common.quantity.Time;
026    import com.twitter.common.net.pool.ResourceExhaustedException;
027    import com.twitter.common.thrift.TResourceExhaustedException;
028    import com.twitter.common.thrift.TTimeoutException;
029    import com.twitter.common.net.loadbalancing.RequestTracker;
030    import org.apache.thrift.async.AsyncMethodCallback;
031    import org.apache.thrift.transport.TTransport;
032    
033    import javax.annotation.Nullable;
034    import java.lang.reflect.InvocationTargetException;
035    import java.lang.reflect.Method;
036    import java.net.InetSocketAddress;
037    import java.util.List;
038    import java.util.concurrent.TimeoutException;
039    import java.util.logging.Logger;
040    
041    /**
042     * A caller that issues calls to a target that is assumed to be a client to a thrift service.
043     *
044     * @author William Farner
045     */
046    public class ThriftCaller<T> implements Caller {
047      private static final Logger LOG = Logger.getLogger(ThriftCaller.class.getName());
048    
049      private final ObjectPool<Connection<TTransport, InetSocketAddress>> connectionPool;
050      private final RequestTracker<InetSocketAddress> requestTracker;
051      private final Function<TTransport, T> clientFactory;
052      private final Amount<Long, Time> timeout;
053      private final boolean debug;
054    
055      /**
056       * Creates a new thrift caller.
057       *
058       * @param connectionPool The connection pool to use.
059       * @param requestTracker The request tracker to nofify of request results.
060       * @param clientFactory Factory to use for building client object instances.
061       * @param timeout The timeout to use when requesting objects from the connection pool.
062       * @param debug Whether to use the caller in debug mode.
063       */
064      public ThriftCaller(ObjectPool<Connection<TTransport, InetSocketAddress>> connectionPool,
065          RequestTracker<InetSocketAddress> requestTracker, Function<TTransport, T> clientFactory,
066          Amount<Long, Time> timeout, boolean debug) {
067    
068        this.connectionPool = connectionPool;
069        this.requestTracker = requestTracker;
070        this.clientFactory = clientFactory;
071        this.timeout = timeout;
072        this.debug = debug;
073      }
074    
075      @Override
076      public Object call(Method method, Object[] args, @Nullable AsyncMethodCallback callback,
077          @Nullable Amount<Long, Time> connectTimeoutOverride) throws Throwable {
078    
079        final Connection<TTransport, InetSocketAddress> connection = getConnection(connectTimeoutOverride);
080        final long startNanos = System.nanoTime();
081    
082        ResultCapture capture = new ResultCapture() {
083          @Override public void success() {
084            try {
085              requestTracker.requestResult(connection.getEndpoint(),
086                  RequestTracker.RequestResult.SUCCESS, System.nanoTime() - startNanos);
087            } finally {
088              connectionPool.release(connection);
089            }
090          }
091    
092          @Override public boolean fail(Throwable t) {
093            if (debug) {
094              LOG.warning(String.format("Call to endpoint: %s failed: %s", connection, t));
095            }
096    
097            try {
098              requestTracker.requestResult(connection.getEndpoint(),
099                  RequestTracker.RequestResult.FAILED, System.nanoTime() - startNanos);
100            } finally {
101              connectionPool.remove(connection);
102            }
103            return true;
104          }
105        };
106    
107        return invokeMethod(clientFactory.apply(connection.get()), method, args, callback, capture);
108      }
109    
110      private static Object invokeMethod(Object target, Method method, Object[] args,
111          AsyncMethodCallback callback, final ResultCapture capture) throws Throwable {
112    
113        // Swap the wrapped callback out for ours.
114        if (callback != null) {
115          callback = new WrappedMethodCallback(callback, capture);
116    
117          List<Object> argsList = Lists.newArrayList(args);
118          argsList.add(callback);
119          args = argsList.toArray();
120        }
121    
122        try {
123          Object result = method.invoke(target, args);
124          if (callback == null) capture.success();
125    
126          return result;
127        } catch (InvocationTargetException t) {
128          // We allow this one to go to both sync and async captures.
129          if (callback != null) {
130            callback.onError(t.getCause());
131            return null;
132          } else {
133            capture.fail(t.getCause());
134            throw t.getCause();
135          }
136        }
137      }
138    
139      private Connection<TTransport, InetSocketAddress> getConnection(
140          Amount<Long, Time> connectTimeoutOverride)
141          throws TResourceExhaustedException, TTimeoutException {
142        try {
143          Connection<TTransport, InetSocketAddress> connection;
144          if (connectTimeoutOverride != null) {
145            connection = connectionPool.get(connectTimeoutOverride);
146          } else {
147            connection = (timeout.getValue() > 0)
148                ? connectionPool.get(timeout) : connectionPool.get();
149          }
150    
151          if (connection == null) {
152            throw new TResourceExhaustedException("no connection was available");
153          }
154          return connection;
155        } catch (ResourceExhaustedException e) {
156          throw new TResourceExhaustedException(e);
157        } catch (TimeoutException e) {
158          throw new TTimeoutException(e);
159        }
160      }
161    }
