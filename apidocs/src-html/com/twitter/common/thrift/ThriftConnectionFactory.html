043    /**
044     * A connection factory for thrift transport connections to a given host.  This connection factory
045     * is lazy and will only create a configured maximum number of active connections - where a
046     * {@link ConnectionFactory#create(com.twitter.common.quantity.Amount) created} connection that has
047     * not been {@link #destroy destroyed} is considered active.
048     *
049     * @author John Sirois
050     */
051    public class ThriftConnectionFactory
052        implements ConnectionFactory<Connection<TTransport, InetSocketAddress>> {
053    
054      public enum TransportType {
055        BLOCKING, FRAMED, NONBLOCKING;
056    
057        /**
058         * Async clients implicitly use a framed transport, requiring the server they connect to to do
059         * the same. This prevents specifying a nonblocking client without a framed transport, since
060         * that is not compatible with thrift and would simply cause the client to blow up when making a
061         * request. Instead, you must explicitly say useFramedTransport(true) for any buildAsync().
062         */
063        public static TransportType get(boolean framedTransport, boolean nonblocking) {
064          if (nonblocking) {
065            Preconditions.checkArgument(framedTransport,
066                "nonblocking client requires a server running framed transport");
067            return NONBLOCKING;
068          }
069    
070          return framedTransport ? FRAMED : BLOCKING;
071        }
072      }
073    
074      private static InetSocketAddress asEndpoint(String host, int port) {
075        MorePreconditions.checkNotBlank(host);
076        Preconditions.checkArgument(port > 0);
077        return InetSocketAddress.createUnresolved(host, port);
078      }
079    
080      private InetSocketAddress endpoint;
081      private final int maxConnections;
082      private final TransportType transportType;
083      private final Amount<Long, Time> socketTimeout;
084      private final Closure<Connection<TTransport, InetSocketAddress>> postCreateCallback;
085    
086      private final Set<Connection<TTransport, InetSocketAddress>> activeConnections =
087          Sets.newSetFromMap(
088              Maps.<Connection<TTransport, InetSocketAddress>, Boolean>newIdentityHashMap());
089      private volatile int lastActiveConnectionsSize = 0;
090    
091      private final Lock activeConnectionsWriteLock = new ReentrantLock(true);
092    
093      /**
094       * Creates a thrift connection factory with a plain socket (non-framed transport).
095       * This is the same as calling {@link #ThriftConnectionFactory(String, int, int, boolean)} with
096       * {@code framedTransport} set to {@code false}.
097       *
098       * @param host Host to connect to.
099       * @param port Port to connect on.
100       * @param maxConnections Maximum number of connections for this host:port.
101       */
102      public ThriftConnectionFactory(String host, int port, int maxConnections) {
103        this(host, port, maxConnections, TransportType.BLOCKING);
104      }

184       * If {@code framedTransport} is set to {@code true}, {@link TFramedTransport} will be used,
185       * otherwise a raw {@link TSocket} will be used.
186       * If {@code nonblocking} is set to {@code true}, {@link TNonblockingSocket} will be used,

189       *

192       * @param transportType Whether to use normal blocking, framed blocking, or non-blocking
193       *          (implicitly framed) transport.
194       * @param socketTimeout timeout on thrift i/o operations, or null to default to connectTimeout o
195       *          the blocking client.
196       */
197      public ThriftConnectionFactory(InetSocketAddress endpoint, int maxConnections,
198          TransportType transportType, Amount<Long, Time> socketTimeout) {
199              this(endpoint, maxConnections, transportType, socketTimeout,
200            Closures.<Connection<TTransport, InetSocketAddress>>noop());
201      }
202    
203      public ThriftConnectionFactory(InetSocketAddress endpoint, int maxConnections,
204          TransportType transportType, Amount<Long, Time> socketTimeout,
205              Closure<Connection<TTransport, InetSocketAddress>> postCreateCallback) {
206        Preconditions.checkArgument(maxConnections > 0, "maxConnections must be at least 1");
207        if (socketTimeout != null) {
208          Preconditions.checkArgument(socketTimeout.as(Time.MILLISECONDS) >= 0);
209        }
210    
211        this.endpoint = Preconditions.checkNotNull(endpoint);
212        this.maxConnections = maxConnections;
213        this.transportType = transportType;
214        this.socketTimeout = socketTimeout;
215        this.postCreateCallback = Preconditions.checkNotNull(postCreateCallback);
216      }
217    
218      @Override
219      public boolean mightCreate() {
220        return lastActiveConnectionsSize < maxConnections;
221      }
222    
223      /**
224       * FIXME:  shouldn't this throw TimeoutException instead of returning null
225       *         in the timeout cases as per the ConnectionFactory.create javadoc?
226       */
227      @Override
228      public Connection<TTransport, InetSocketAddress> create(Amount<Long, Time> timeout)
229          throws TTransportException, IOException {
230    
231        Preconditions.checkNotNull(timeout);
232        if (timeout.getValue() == 0) {
233          return create();
234        }
235    
236        try {
237          long timeRemainingNs = timeout.as(Time.NANOSECONDS);
238          long start = System.nanoTime();
239          if(activeConnectionsWriteLock.tryLock(timeRemainingNs, TimeUnit.NANOSECONDS)) {
240            try {
241              if (!willCreateSafe()) {
242                return null;
243              }
244    
245              timeRemainingNs -= (System.nanoTime() - start);
246    
247              return createConnection((int) TimeUnit.NANOSECONDS.toMillis(timeRemainingNs));
248            } finally {
249              activeConnectionsWriteLock.unlock();
250            }
251          } else {
252            return null;
253          }
254        } catch (InterruptedException e) {
255          return null;
256        }
257      }
258    
259      private Connection<TTransport, InetSocketAddress> create()
260          throws TTransportException, IOException {
261        activeConnectionsWriteLock.lock();
262        try {
263          if (!willCreateSafe()) {
264            return null;
265          }
266    
267          return createConnection(0);
268        } finally {
269          activeConnectionsWriteLock.unlock();
270        }
271      }
272    
273      private Connection<TTransport, InetSocketAddress> createConnection(int timeoutMillis)
274          throws TTransportException, IOException {
275        TTransport transport = createTransport(timeoutMillis);
276        if (transport == null) {
277          return null;
278        }
279    
280        Connection<TTransport, InetSocketAddress> connection =
281            new TTransportConnection(transport, endpoint);
282        postCreateCallback.execute(connection);
283        activeConnections.add(connection);
284        lastActiveConnectionsSize = activeConnections.size();
285        return connection;
286      }
287    
288      private boolean willCreateSafe() {
289        return activeConnections.size() < maxConnections;
290      }
291    
292      @VisibleForTesting
293      TTransport createTransport(int timeoutMillis) throws TTransportException, IOException {
294        TSocket socket = null;
295        if (transportType != TransportType.NONBLOCKING) {
296          // can't do a nonblocking create on a blocking transport
297          if (timeoutMillis <= 0) {
298            return null;
299          }
300    
301          socket = new TSocket(endpoint.getHostName(), endpoint.getPort(), timeoutMillis);
302        }
303    
304        try {
305          switch (transportType) {
306            case BLOCKING:
307              socket.open();
308              setSocketTimeout(socket);
309              return socket;
310            case FRAMED:
311              TFramedTransport transport = new TFramedTransport(socket);
312              transport.open();
313              setSocketTimeout(socket);
314              return transport;
315            case NONBLOCKING:
316              try {
317                return new TNonblockingSocket(endpoint.getHostName(), endpoint.getPort());
318              } catch (IOException e) {
319                throw new IOException("Failed to create non-blocking transport to " + endpoint, e);
320              }
321          }
322        } catch (TTransportException e) {
323          throw new TTransportException("Failed to create transport to " + endpoint, e);
324        }
325    
326        throw new IllegalArgumentException("unknown transport type " + transportType);
327      }
328    
329      private void setSocketTimeout(TSocket socket) {
330        if (socketTimeout != null) {
331          socket.setTimeout(socketTimeout.as(Time.MILLISECONDS).intValue());
332        }
333      }
334    
335      @Override
336      public void destroy(Connection<TTransport, InetSocketAddress> connection) {
337        activeConnectionsWriteLock.lock();
338        try {
339          boolean wasActiveConnection = activeConnections.remove(connection);
340          Preconditions.checkArgument(wasActiveConnection,
341              "connection %s not created by this factory", connection);
342          lastActiveConnectionsSize = activeConnections.size();
343        } finally {
344          activeConnectionsWriteLock.unlock();
345        }
346    
347        // We close the connection outside the critical section which means we may have more connections
348        // "active" (open) than maxConnections for a very short time
349        connection.close();
350      }
351    
352      @Override
353      public String toString() {
354        return String.format("%s[%s]", getClass().getSimpleName(), endpoint);
355      }
