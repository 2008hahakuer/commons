017    package com.twitter.common.thrift;
018    
019    import com.google.common.annotations.VisibleForTesting;
020    import com.google.common.base.Preconditions;
021    import com.google.common.base.Predicates;
022    import com.google.common.collect.ImmutableSet;
023    import com.google.common.collect.Iterables;
024    import com.twitter.common.quantity.Amount;
025    import com.twitter.common.quantity.Time;
026    import com.twitter.common.stats.Stats;
027    import com.twitter.common.stats.StatsProvider;
028    
029    /**
030     * Represents the configuration for a thrift call.  Use {@link #builder()} to create a new one or
031     * or {@link #builder(Config)} to create a new config based on another config.
032     *
033     * <p>If a deadline is specified, it acts as a global timeout for each thrift call made.
034     * Obtaining connections, performing the remote call and executing retries are all expected to
035     * complete within this deadline.  When the specified deadline is not met, an
036     * {@link TTimeoutException} will be thrown.
037     *
038     * <p>If max retries is specified as zero (never retry), then the list of retryable exceptions are
039     * ignored.  It is only when max retries is greater than zero that list of retryable exceptions is
040     * used to determine if a particular failed call should be retried.
041     *
042     * @author John Sirois
043     */
044    public class Config {
045    
046      /**
047       * Created a builder for a new {@link Config}.  Default values are as follows:
048       * <ul>
049       * <li>{@link #getRequestTimeout()} 0
050       * <li>{@link #getMaxRetries()} 0
051       * <li>{@link #getRetryableExceptions()} []
052       * <li>{@link #isDebug()} ()} false
053       * </ul>
054       */
055      public static Builder builder() {
056        return new Builder();
057      }
058    
059      /**
060       *
061       * @param config
062       * @return
063       */
064      public static Builder builder(Config config) {
065        Preconditions.checkNotNull(config);
066        return new Builder(config);
067      }
068    
069      private static final Amount<Long,Time> DEADLINE_BLOCKING = Amount.of(0L, Time.MILLISECONDS);
070    
071      @VisibleForTesting
072      static final Amount<Long,Time> DEFAULT_CONNECT_TIMEOUT = Amount.of(5L, Time.SECONDS);
073    
074      private Amount<Long, Time> requestTimeout = DEADLINE_BLOCKING;
075      private Amount<Long, Time> connectTimeout = DEFAULT_CONNECT_TIMEOUT;
076      private int maxRetries;
077      private ImmutableSet<Class<? extends Exception>> retryableExceptions = ImmutableSet.of();
078      private boolean debug = false;
079      private boolean enableStats = true;
080      private StatsProvider statsProvider = Stats.STATS_PROVIDER;
081    
082      private Config() {
083        // defaults
084      }
085    
086      private Config(Config copyFrom) {
087        requestTimeout = copyFrom.requestTimeout;
088        maxRetries = copyFrom.maxRetries;
089        retryableExceptions = copyFrom.retryableExceptions;
090        debug = copyFrom.debug;
091        statsProvider = copyFrom.statsProvider;
092      }
093    
094      /**
095       * Returns the maximum time to wait for any thrift call to complete.  A deadline of 0 means to
096       * wait forever
097       */
098      public Amount<Long, Time> getRequestTimeout() {
099        return requestTimeout;
100      }
101    
102      /**
103       * Returns the maximum time to wait for a connection to be established.  A deadline of 0 means to
104       * wait forever
105       */
106      public Amount<Long, Time> getConnectTimeout() {
107        return connectTimeout;
108      }
109    
110      /**
111       * Returns the maximum number of retries to perform for each thrift call.  A value of 0 means to
112       * never retry and in this case {@link #getRetryableExceptions()} will be an empty set.
113       */
114      public int getMaxRetries() {
115        return maxRetries;
116      }
117    
118      /**
119       * Returns the set of exceptions to retry calls for.  The returned set will only be empty if
120       * {@link #getMaxRetries()} is 0.
121       */
122      public ImmutableSet<Class<? extends Exception>> getRetryableExceptions() {
123        return retryableExceptions;
124      }
125    
126      /**
127       * Returns {@code true} if the client should log extra debugging information.  Currently this
128       * includes method call arguments when RPCs fail with exceptions.
129       */
130      public boolean isDebug() {
131        return debug;
132      }
133    
134      /**
135       * Returns {@code true} if the client should track request statistics.
136       */
137      public boolean enableStats() {
138        return enableStats;
139      }
140    
141      /**
142       * Returns the stats provider to use to record Thrift client stats.
143       */
144      public StatsProvider getStatsProvider() {
145        return statsProvider;
146      }
147    
148      // This was made public because it seems to be causing problems for scala users when it is not
149      // public.
150      public static abstract class AbstractBuilder<T extends AbstractBuilder> {
151        private final Config config;
152    
153        AbstractBuilder() {
154          this.config = new Config();
155        }
156    
157        AbstractBuilder(Config template) {
158          Preconditions.checkNotNull(template);
159          this.config = new Config(template);
160        }
161    
162        protected abstract T getThis();
163    
164        // TODO(John Sirois): extra validation or design ... can currently do stange things like:
165        // builder.blocking().withDeadline(1, TimeUnit.MILLISECONDS)
166        // builder.noRetries().retryOn(TException.class)
167    
168        /**
169         * Specifies that all calls be blocking calls with no inherent deadline.  It may be the case that
170         * underlying transports will eventually deadline, but {@link Thrift} will not enforce a dealine.
171         */
172        public final T blocking() {
173          config.requestTimeout = DEADLINE_BLOCKING;
174          return getThis();
175        }
176    
177        /**
178         * Specifies that all calls be subject to a global timeout.  This deadline includes all call
179         * activities, including obtaining a free connection and any automatic retries.
180         */
181        public final T withRequestTimeout(Amount<Long, Time> timeout) {
182          Preconditions.checkNotNull(timeout);
183          Preconditions.checkArgument(timeout.getValue() >= 0,
184              "A negative deadline is invalid: %s", timeout);
185          config.requestTimeout = timeout;
186          return getThis();
187        }
188    
189        /**
190         * Assigns the timeout for all connections established with the blocking client.
191         * On an asynchronous client this timeout is only used for the connection pool lock
192         * acquisition on initial calls (not retries, @see withRetries).  The actual network
193         * connection timeout for the asynchronous client is governed by socketTimeout.
194         *
195         * @param timeout Connection timeout.
196         * @return A reference to the builder.
197         */
198        public final T withConnectTimeout(Amount<Long, Time> timeout) {
199          Preconditions.checkNotNull(timeout);
200          Preconditions.checkArgument(timeout.getValue() >= 0,
201              "A negative deadline is invalid: %s", timeout);
202          config.connectTimeout = timeout;
203          return getThis();
204        }
205    
206        /**
207         * Specifies that no calls be automatically retried.
208         */
209        public final T noRetries() {
210          config.maxRetries = 0;
211          config.retryableExceptions = ImmutableSet.of();
212          return getThis();
213        }
214    
215        /**
216         * Specifies that failing calls meeting {@link #retryOn retry} criteria be retried up to a
217         * maximum of {@code retries} times before failing.  On an asynchronous client, these retries
218         * will be forced to be non-blocking, failing fast if they cannot immediately acquire the connection
219         * pool locks, so they only provide a best-effort retry strategy there.
220         */
221        public final T withRetries(int retries) {
222          Preconditions.checkArgument(retries >= 0, "A negative retry count is invalid: %d", retries);
223          config.maxRetries = retries;
224          return getThis();
225        }
226    
227        /**
228         * Specifies the set of exception classes that are to be considered retryable (if retries are
229         * enabled).  Any exceptions thrown by the underlying thrift call will be considered retryable
230         * if they are an instance of any one of the specified exception classes.  The set of exception
231         * classes must contain at least exception class.  To specify no retries either use
232         * {@link #noRetries()} or pass zero to {@link #withRetries(int)}.
233         */
234        public final T retryOn(Iterable<? extends Class<? extends Exception>> retryableExceptions) {
235          Preconditions.checkNotNull(retryableExceptions);
236          ImmutableSet<Class<? extends Exception>> classes =
237              ImmutableSet.copyOf(Iterables.filter(retryableExceptions, Predicates.notNull()));
238          Preconditions.checkArgument(!classes.isEmpty(),
239              "Must provide at least one retryable exception class");
240          config.retryableExceptions = classes;
241          return getThis();
242        }
243    
244        /**
245         * Specifies the set of exception classes that are to be considered retryable (if retries are
246         * enabled).  Any exceptions thrown by the underlying thrift call will be considered retryable
247         * if they are an instance of any one of the specified exception classes.  The set of exception
248         * classes must contain at least exception class.  To specify no retries either use
249         * {@link #noRetries()} or pass zero to {@link #withRetries(int)}.
250         */
251        public final T retryOn(Class<? extends Exception> exception) {
252          Preconditions.checkNotNull(exception);
253          config.retryableExceptions =
254              ImmutableSet.<Class<? extends Exception>>builder().add(exception).build();
255          return getThis();
256        }
257    
258        /**
259         * When {@code debug == true}, specifies that extra debugging information should be logged.
260         */
261        public final T withDebug(boolean debug) {
262          config.debug = debug;
263          return getThis();
264        }
265    
266        /**
267         * Disables stats collection on the client (enabled by default).
268         */
269        public T disableStats() {
270          config.enableStats = false;
271          return getThis();
272        }
273    
274        /**
275         * Registers a custom stats provider to use to track various client stats.
276         */
277        public T withStatsProvider(StatsProvider statsProvider) {
278          config.statsProvider = Preconditions.checkNotNull(statsProvider);
279          return getThis();
280        }
281    
282        protected final Config getConfig() {
283          return config;
284        }
285      }
286    
287      public static final class Builder extends AbstractBuilder<Builder> {
288        private Builder() {
289          super();
290        }
291    
292        private Builder(Config template) {
293          super(template);
294        }
295    
296        @Override
297        protected Builder getThis() {
298          return this;
299        }
300    
301        public Config create() {
302          return getConfig();
303        }
304      }
305    }
