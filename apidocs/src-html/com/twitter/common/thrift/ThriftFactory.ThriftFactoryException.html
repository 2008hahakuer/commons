078    /**
079     * A utility that provides convenience methods to build common {@link Thrift}s.
080     *
081     * The thrift factory allows you to specify parameters that define how the client connects to
082     * and communicates with servers, such as the transport type, connection settings, and load
083     * balancing.  Request-level settings like sync/async and retries should be set on the
084     * {@link Thrift} instance that this factory will create.
085     *
086     * The factory will attempt to provide reasonable defaults to allow the caller to minimize the
087     * amount of necessary configuration.  Currently, the default behavior includes:
088     *
089     * <ul>
090     *   <li> A test lease/release for each host will be performed every second
091     *      {@link #withDeadConnectionRestoreInterval(Amount)}
092     *   <li> At most 50 connections will be established to each host
093     *      {@link #withMaxConnectionsPerEndpoint(int)}
094     *   <li> Unframed transport {@link #useFramedTransport(boolean)}
095     *   <li> A load balancing strategy that will mark hosts dead and prefer least-connected hosts.
096     *      Hosts are marked dead if the most recent connection attempt was a failure or else based on
097     *      the windowed error rate of attempted RPCs.  If the error rate for a connected host exceeds
098     *      20% over the last second, the host will be disabled for 2 seconds ascending up to 10 seconds
099     *      if the elevated error rate persists.
100     *      {@link #withLoadBalancingStrategy(LoadBalancingStrategy)}
101     *   <li> Statistics are reported through {@link Stats}
102     *      {@link #withStatsProvider(StatsProvider)}
103     *   <li> A service name matching the thrift interface name {@link #withServiceName(String)}
104     * </ul>
105     *
106     * @author John Sirois
107     */
108    public class ThriftFactory<T> {
109      private static final Amount<Long,Time> DEFAULT_DEAD_TARGET_RESTORE_INTERVAL =
110          Amount.of(1L, Time.SECONDS);
111

123      private StatsProvider statsProvider = Stats.STATS_PROVIDER;

131      /**
132       * Creates a default factory that will use unframed blocking transport.
133       *
134       * @param serviceInterface The interface of the thrift service to make a client for.
135       */
136      private ThriftFactory(Class<T> serviceInterface) {
137        this.serviceInterface = Thrift.checkServiceInterface(serviceInterface);
138        this.maxConnectionsPerEndpoint = DEFAULT_MAX_CONNECTIONS_PER_ENDPOINT;
139        this.connectionRestoreInterval = DEFAULT_DEAD_TARGET_RESTORE_INTERVAL;
140        this.framedTransport = false;
141        this.monitor = new TrafficMonitor<InetSocketAddress>(serviceInterface.getName());
142        this.serviceName = serviceInterface.getEnclosingClass().getSimpleName();
143        this.sslTransport = false;
144      }
145    
146      private void checkBaseState() {
147        Preconditions.checkArgument(maxConnectionsPerEndpoint > 0,
148            "Must allow at least 1 connection per endpoint; %s specified", maxConnectionsPerEndpoint);
149      }

152        return monitor;

155      /**
156       * Creates the thrift client, and initializes connection pools.
157       *
158       * @param backends Backends to connect to.
159       * @return A new thrift client.
160       */
161      public Thrift<T> build(Set<InetSocketAddress> backends) {
162        checkBaseState();
163        MorePreconditions.checkNotBlank(backends);
164    
165        ManagedThreadPool managedThreadPool = createManagedThreadpool(backends.size());
166        LoadBalancer<InetSocketAddress> loadBalancer = createLoadBalancer();
167        Function<TTransport, T> clientFactory = getClientFactory();
168    
169        ObjectPool<Connection<TTransport, InetSocketAddress>> connectionPool =
170            createConnectionPool(backends, loadBalancer, managedThreadPool, false);
171    
172        return new Thrift<T>(managedThreadPool, connectionPool, loadBalancer, serviceName,
173            serviceInterface, clientFactory, false, sslTransport);
174      }

194        return new Thrift<T>(managedThreadPool, connectionPool, loadBalancer, serviceName,
195            serviceInterface, clientFactory, false, sslTransport);
196      }

199        return new ManagedThreadPool(serviceName, initialEndpointCount, maxConnectionsPerEndpoint);

202      /**
203       * A finite thread pool that monitors backend choice events to dynamically resize.  This
204       * {@link java.util.concurrent.ExecutorService} implementation immediately rejects requests when
205       * there are no more available worked threads (requests are not queued).
206       */
207      private static class ManagedThreadPool extends ForwardingExecutorService<ThreadPoolExecutor>
208          implements Closure<Collection<InetSocketAddress>> {
209    
210        private static final Logger LOG = Logger.getLogger(ManagedThreadPool.class.getName());
211    
212        private static ThreadPoolExecutor createThreadPool(String serviceName, int initialSize) {
213          ThreadFactory threadFactory =
214              new ThreadFactoryBuilder()
215                  .setNameFormat("Thrift[" +serviceName + "][%d]")
216                  .setDaemon(true)
217                  .build();
218          return new ThreadPoolExecutor(initialSize, initialSize, 0, TimeUnit.MILLISECONDS,
219              new SynchronousQueue<Runnable>(), threadFactory);
220        }
221    
222        private final int maxConnectionsPerEndpoint;
223    
224        public ManagedThreadPool(String serviceName, int initialEndpointCount,
225            int maxConnectionsPerEndpoint) {
226    
227          super(createThreadPool(serviceName, initialEndpointCount * maxConnectionsPerEndpoint));
228          this.maxConnectionsPerEndpoint = maxConnectionsPerEndpoint;
229          setRejectedExecutionHandler(initialEndpointCount);
230        }
231    
232        private void setRejectedExecutionHandler(int endpointCount) {
233          final String message =
234              String.format("All %d x %d connections in use", endpointCount, maxConnectionsPerEndpoint);
235          delegate.setRejectedExecutionHandler(new RejectedExecutionHandler() {
236            @Override public void rejectedExecution(Runnable runnable, ThreadPoolExecutor executor) {
237              throw new RejectedExecutionException(message);
238            }
239          });
240        }
241    
242        @Override
243        public void execute(Collection<InetSocketAddress> chosenBackends) {
244          int previousPoolSize = delegate.getMaximumPoolSize();
245          /*
246           * In the case of no available backends, we need to make sure we pass in a positive pool
247           * size to our delegate.  In particular, java.util.concurrent.ThreadPoolExecutor does not
248           * accept zero as a valid core or max pool size.
249           */
250          int backendCount = Math.max(chosenBackends.size(), 1);
251          int newPoolSize = backendCount * maxConnectionsPerEndpoint;
252    
253          if (previousPoolSize != newPoolSize) {
254            LOG.info(String.format("Re-sizing deadline thread pool from: %d to: %d",
255                previousPoolSize, newPoolSize));
256            if (previousPoolSize < newPoolSize) { // Don't cross the beams!
257              delegate.setMaximumPoolSize(newPoolSize);
258              delegate.setCorePoolSize(newPoolSize);
259            } else {
260              delegate.setCorePoolSize(newPoolSize);
261              delegate.setMaximumPoolSize(newPoolSize);
262            }
263            setRejectedExecutionHandler(backendCount);
264          }
265        }
266      }

336      private ObjectPool<Connection<TTransport, InetSocketAddress>> createConnectionPool(
337          Set<InetSocketAddress> backends, LoadBalancer<InetSocketAddress> loadBalancer,
338          Closure<Collection<InetSocketAddress>> onBackendsChosen, boolean nonblocking) {
339    
340        ImmutableMap.Builder<InetSocketAddress, ObjectPool<Connection<TTransport, InetSocketAddress>>>
341            backendBuilder = ImmutableMap.builder();
342        for (InetSocketAddress backend : backends) {
343          backendBuilder.put(backend, createConnectionPool(backend, nonblocking));
344        }
345    
346        return new MetaPool<TTransport, InetSocketAddress>(backendBuilder.build(),
347            loadBalancer, onBackendsChosen, connectionRestoreInterval);
348      }
349    
350      private ObjectPool<Connection<TTransport, InetSocketAddress>> createConnectionPool(
351          DynamicHostSet<ServiceInstance> hostSet, LoadBalancer<InetSocketAddress> loadBalancer,
352          Closure<Collection<InetSocketAddress>> onBackendsChosen,
353          final boolean nonblocking) throws ThriftFactoryException {
354    
355        Function<InetSocketAddress, ObjectPool<Connection<TTransport, InetSocketAddress>>>
356            endpointPoolFactory =
357          new Function<InetSocketAddress, ObjectPool<Connection<TTransport, InetSocketAddress>>>() {
358            @Override public ObjectPool<Connection<TTransport, InetSocketAddress>> apply(
359                InetSocketAddress endpoint) {
360              return createConnectionPool(endpoint, nonblocking);
361            }
362          };
363    
364        try {
365          return new DynamicPool<ServiceInstance, TTransport, InetSocketAddress>(hostSet,
366              endpointPoolFactory, loadBalancer, onBackendsChosen, connectionRestoreInterval,
367              Util.GET_ADDRESS, Util.IS_ALIVE);
368        } catch (DynamicHostSet.MonitorException e) {
369          throw new ThriftFactoryException("Failed to monitor host set.", e);
370        }
371      }

373      private ObjectPool<Connection<TTransport, InetSocketAddress>> createConnectionPool(
374          InetSocketAddress backend, boolean nonblocking) {
375    
376        ThriftConnectionFactory connectionFactory = new ThriftConnectionFactory(
377            backend, maxConnectionsPerEndpoint, TransportType.get(framedTransport, nonblocking),
378            socketTimeout, postCreateCallback, sslTransport);
379    
380        return new ConnectionPool<Connection<TTransport, InetSocketAddress>>(connectionFactory,
381            statsProvider);
382      }

450      private LoadBalancer<InetSocketAddress> createLoadBalancer() {
451        if (loadBalancingStrategy == null) {
452          loadBalancingStrategy = createDefaultLoadBalancingStrategy();
453        }
454    
455        return LoadBalancerImpl.create(TrafficMonitorAdapter.create(loadBalancingStrategy, monitor));
456      }
457    
458      private LoadBalancingStrategy<InetSocketAddress> createDefaultLoadBalancingStrategy() {
459        Function<InetSocketAddress, BackoffDecider> backoffFactory =
460            new Function<InetSocketAddress, BackoffDecider>() {
461              @Override public BackoffDecider apply(InetSocketAddress socket) {
462                BackoffStrategy backoffStrategy = new TruncatedBinaryBackoff(
463                    Amount.of(2L, Time.SECONDS), Amount.of(10L, Time.SECONDS));
464    
465                return BackoffDecider.builder(socket.toString())
466                    .withTolerateFailureRate(0.2)
467                    .withRequestWindow(Amount.of(1L, Time.SECONDS))
468                    .withSeedSize(5)
469                    .withStrategy(backoffStrategy)
470                    .withRecoveryType(BackoffDecider.RecoveryType.FULL_CAPACITY)
471                    .withStatsProvider(statsProvider)
472                    .build();
473              }
474        };
475    
476        return new MarkDeadStrategyWithHostCheck<InetSocketAddress>(
477            new LeastConnectedStrategy<InetSocketAddress>(), backoffFactory);
478      }

534      private static <T> Function<TTransport, T> createClientFactory(Class<T> serviceInterface) {
535        final Constructor<? extends T> implementationConstructor =
536            findImplementationConstructor(serviceInterface);
537    
538        return new Function<TTransport, T>() {
539          @Override public T apply(TTransport transport) {
540            try {
541              return implementationConstructor.newInstance(new TBinaryProtocol(transport));
542            } catch (InstantiationException e) {
543              throw new RuntimeException(e);
544            } catch (IllegalAccessException e) {
545              throw new RuntimeException(e);
546            } catch (InvocationTargetException e) {
547              throw new RuntimeException(e);
548            }
549          }
550        };
551      }
552    
553      private <T> Function<TTransport, T> createAsyncClientFactory(
554          final Class<T> serviceInterface) throws IOException {
555    
556        final TAsyncClientManager clientManager = new TAsyncClientManager();
557        Runtime.getRuntime().addShutdownHook(new Thread() {
558          @Override public void run() {
559            clientManager.stop();
560          }
561        });
562    
563        final Constructor<? extends T> implementationConstructor =
564            findAsyncImplementationConstructor(serviceInterface);
565    
566        return new Function<TTransport, T>() {
567          @Override public T apply(TTransport transport) {
568            Preconditions.checkNotNull(transport);
569            Preconditions.checkArgument(transport instanceof TNonblockingTransport,
570                "Invalid transport provided to client factory: " + transport.getClass());
571    
572            try {
573              T client = implementationConstructor.newInstance(new TBinaryProtocol.Factory(),
574                  clientManager, transport);
575    
576              if (socketTimeout != null) {
577                ((TAsyncClient) client).setTimeout(socketTimeout.as(Time.MILLISECONDS));
578              }
579    
580              return client;
581            } catch (InstantiationException e) {
582              throw new RuntimeException(e);
583            } catch (IllegalAccessException e) {
584              throw new RuntimeException(e);
585            } catch (InvocationTargetException e) {
586              throw new RuntimeException(e);
587            }
588          }
589        };
590      }

592      private static <T> Constructor<? extends T> findImplementationConstructor(
593          final Class<T> serviceInterface) {
594        Class<? extends T> implementationClass = findImplementationClass(serviceInterface);
595        try {
596          return implementationClass.getConstructor(TProtocol.class);
597        } catch (NoSuchMethodException e) {
598          throw new IllegalArgumentException("Failed to find a single argument TProtocol constructor "
599                                             + "in service client class: " + implementationClass);
600        }
601      }
602    
603      private static <T> Constructor<? extends T> findAsyncImplementationConstructor(
604          final Class<T> serviceInterface) {
605        Class<? extends T> implementationClass = findImplementationClass(serviceInterface);
606        try {
607          return implementationClass.getConstructor(TProtocolFactory.class, TAsyncClientManager.class,
608              TNonblockingTransport.class);
609        } catch (NoSuchMethodException e) {
610          throw new IllegalArgumentException("Failed to find expected constructor "
611                                             + "in service client class: " + implementationClass);
612        }
613      }
614    
615      @SuppressWarnings("unchecked")
616      private static <T> Class<? extends T> findImplementationClass(final Class<T> serviceInterface) {
617        try {
618          return (Class<? extends T>)
619              Iterables.find(ImmutableList.copyOf(serviceInterface.getEnclosingClass().getClasses()),
620                  new Predicate<Class<?>>() {
621                    @Override public boolean apply(Class<?> inner) {
622                      return !serviceInterface.equals(inner)
623                             && serviceInterface.isAssignableFrom(inner);
624                    }
625                  });
626        } catch (NoSuchElementException e) {
627          throw new IllegalArgumentException("Could not find a sibling enclosed implementation of "
628                                             + "service interface: " + serviceInterface);
629        }
630      }
