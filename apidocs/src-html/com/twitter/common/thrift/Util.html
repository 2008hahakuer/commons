044      public static Function<ServiceInstance, InetSocketAddress> GET_ADDRESS =
045          new Function<ServiceInstance, InetSocketAddress>() {
046            @Override public InetSocketAddress apply(ServiceInstance serviceInstance) {
047              Endpoint endpoint = serviceInstance.getServiceEndpoint();
048              return InetSocketAddress.createUnresolved(endpoint.getHost(), endpoint.getPort());
049            }
050          };
051    
052      public static final Predicate<ServiceInstance> IS_ALIVE = new Predicate<ServiceInstance>() {
053        @Override public boolean apply(ServiceInstance serviceInstance) {
054          switch (serviceInstance.getStatus()) {
055            case ALIVE:
056              return true;
057    
058            // We'll be optimistic here and let MTCP's ranking deal with
059            // unhealthy services in a WARNING state.
060            case WARNING:
061              return true;
062    
063            // Services which are just starting up, on the other hand... are much easier to just not
064            // send requests to.  The STARTING state is useful to distinguish from WARNING or ALIVE:
065            // you exist in ZooKeeper, but don't yet serve traffic.
066            case STARTING:
067            default:
068              return false;
069          }
070        }
071      };

073      /**
074       * Pretty-prints a thrift object contents.
075       *
076       * @param t The thrift object to print.
077       * @return The pretty-printed version of the thrift object.
078       */
079      public static String prettyPrint(TBase t) {
080        return t == null ? "null" : printTbase(t, 0);
081      }
082    
083      /**
084       * Prints an object contained in a thrift message.
085       *
086       * @param o The object to print.
087       * @param depth The print nesting level.
088       * @return The pretty-printed version of the thrift field.
089       */
090      private static String printValue(Object o, int depth) {
091        if (o == null) {
092          return "null";
093        } else if (TBase.class.isAssignableFrom(o.getClass())) {
094          return "\n" + printTbase((TBase) o, depth + 1);
095        } else if (Map.class.isAssignableFrom(o.getClass())) {
096          return printMap((Map) o, depth + 1);
097        } else if (List.class.isAssignableFrom(o.getClass())) {
098          return printList((List) o, depth + 1);
099        } else if (Set.class.isAssignableFrom(o.getClass())) {
100          return printSet((Set) o, depth + 1);
101        } else if (String.class == o.getClass()) {
102          return '"' + o.toString() + '"';
103        } else {
104          return o.toString();
105        }
106      }
107    
108      private static final String METADATA_MAP_FIELD_NAME = "metaDataMap";
109    
110      /**
111       * Prints a TBase.
112       *
113       * @param t The object to print.
114       * @param depth The print nesting level.
115       * @return The pretty-printed version of the TBase.
116       */
117      private static String printTbase(TBase t, int depth) {
118        List<String> fields = Lists.newArrayList();
119        for (Map.Entry<? extends TFieldIdEnum, FieldMetaData> entry :
120            FieldMetaData.getStructMetaDataMap(t.getClass()).entrySet()) {
121          @SuppressWarnings("unchecked")
122          boolean fieldSet = t.isSet(entry.getKey());
123          String strValue;
124          if (fieldSet) {
125            @SuppressWarnings("unchecked")
126            Object value = t.getFieldValue(entry.getKey());
127            strValue = printValue(value, depth);
128          } else {
129            strValue = "not set";
130          }
131          fields.add(tabs(depth) + entry.getValue().fieldName + ": " + strValue);
132        }
133    
134        return Joiner.on("\n").join(fields);
135      }
136    
137      /**
138       * Prints a map in a style that is consistent with TBase pretty printing.
139       *
140       * @param map The map to print
141       * @param depth The print nesting level.
142       * @return The pretty-printed version of the map.
143       */
144      private static String printMap(Map<?, ?> map, int depth) {
145        List<String> entries = Lists.newArrayList();
146        for (Map.Entry entry : map.entrySet()) {
147          entries.add(tabs(depth) + printValue(entry.getKey(), depth)
148              + " = " + printValue(entry.getValue(), depth));
149        }
150    
151        return entries.isEmpty() ? "{}"
152            : String.format("{\n%s\n%s}", Joiner.on(",\n").join(entries), tabs(depth - 1));
153      }
154    
155      /**
156       * Prints a list in a style that is consistent with TBase pretty printing.
157       *
158       * @param list The list to print
159       * @param depth The print nesting level.
160       * @return The pretty-printed version of the list
161       */
162      private static String printList(List<?> list, int depth) {
163        List<String> entries = Lists.newArrayList();
164        for (int i = 0; i < list.size(); i++) {
165          entries.add(
166              String.format("%sItem[%d] = %s", tabs(depth), i, printValue(list.get(i), depth)));
167        }
168    
169        return entries.isEmpty() ? "[]"
170            : String.format("[\n%s\n%s]", Joiner.on(",\n").join(entries), tabs(depth - 1));
171      }
172      /**
173       * Prints a set in a style that is consistent with TBase pretty printing.
174       *
175       * @param set The set to print
176       * @param depth The print nesting level.
177       * @return The pretty-printed version of the set
178       */
179      private static String printSet(Set<?> set, int depth) {
180        List<String> entries = Lists.newArrayList();
181        for (Object item : set) {
182          entries.add(
183              String.format("%sItem = %s", tabs(depth), printValue(item, depth)));
184        }
185    
186        return entries.isEmpty() ? "{}"
187            : String.format("{\n%s\n%s}", Joiner.on(",\n").join(entries), tabs(depth - 1));
188      }
189    
190      private static String tabs(int n) {
191        return Strings.repeat("  ", n);
192      }

194      private Util() {
195        // Utility class.
196      }
