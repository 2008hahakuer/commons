018    package com.twitter.common.quantity;
019    
020    import java.util.concurrent.TimeUnit;
021    
022    /**
023     * Provides a unit to allow conversions and unambiguous passing around of time {@link Amount}s.
024     */
025    public enum Time implements Unit<Time> {
026      NANOSECONDS(1, TimeUnit.NANOSECONDS, "ns"),
027      MICROSECONDS(1000, NANOSECONDS, TimeUnit.MICROSECONDS, "us"),
028      MILLISECONDS(1000, MICROSECONDS, TimeUnit.MILLISECONDS, "ms"),
029      SECONDS(1000, MILLISECONDS, TimeUnit.SECONDS, "secs"),
030      MINUTES(60, SECONDS, TimeUnit.MINUTES, "mins"),
031      HOURS(60, MINUTES, TimeUnit.HOURS, "hrs"),
032      DAYS(24, HOURS, TimeUnit.DAYS, "days");
033    
034      private final double multiplier;
035      private final TimeUnit timeUnit;
036      private final String display;
037    
038      private Time(double multiplier, TimeUnit timeUnit, String display) {
039        this.multiplier = multiplier;
040        this.timeUnit = timeUnit;
041        this.display = display;
042      }
043    
044      private Time(double multiplier, Time base, TimeUnit timeUnit, String display) {
045        this(multiplier * base.multiplier, timeUnit, display);
046      }
047    
048      @Override
049      public double multiplier() {
050        return multiplier;
051      }
052    
053      /**
054       * Returns the equivalent {@code TimeUnit}.
055       */
056      public TimeUnit getTimeUnit() {
057        return timeUnit;
058      }
059    
060      @Override
061      public String toString() {
062        return display;
063      }
064    }
