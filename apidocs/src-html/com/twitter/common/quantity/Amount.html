018    package com.twitter.common.quantity;
019    
020    import com.google.common.base.Preconditions;
021    import com.twitter.common.collections.Pair;
022    
023    /**
024     * Represents a value in a unit system and facilitates unambiguous communication of amounts.
025     * Instances are created via static factory {@code of(...)} methods.
026     *
027     * @author John Sirois
028     */
029    public abstract class Amount<T extends Number & Comparable<T>, U extends Unit<U>>
030        implements Comparable<Amount<T, U>> {
031    
032      private final Pair<T, U> amount;
033    
034      private Amount(T value, U unit) {
035        Preconditions.checkNotNull(value);
036        Preconditions.checkNotNull(unit);
037    
038        this.amount = Pair.of(value, unit);
039      }
040    
041      public T getValue() {
042        return amount.getFirst();
043      }
044    
045      public U getUnit() {
046        return amount.getSecond();
047      }
048    
049      public T as(U unit) {
050        return asUnit(unit);
051      }
052    
053      private T asUnit(Unit<?> unit) {
054        return sameUnits(unit) ? getValue() : scale(getUnit().multiplier() / unit.multiplier());
055      }
056    
057      @Override
058      public int hashCode() {
059        return amount.hashCode();
060      }
061    
062      @Override
063      public boolean equals(Object obj) {
064        if (this == obj) {
065          return true;
066        }
067        if (!(obj instanceof Amount)) {
068          return false;
069        }
070    
071        Amount<?, ?> other = (Amount<?, ?>) obj;
072        return amount.equals(other.amount) || isSameAmount(other);
073      }
074    
075      private boolean isSameAmount(Amount<?, ?> other) {
076        // Equals allows Object - so we have no compile time check that other has the right value type;
077        // ie: make sure they don't have Integer when we have Long.
078        Number value = other.getValue();
079        if (!getValue().getClass().isInstance(value)) {
080          return false;
081        }
082    
083        Unit<?> unit = other.getUnit();
084        if (!getUnit().getClass().isInstance(unit)) {
085          return false;
086        }
087    
088        @SuppressWarnings("unchecked")
089        U otherUnit = (U) other.getUnit();
090    
091        // Compare in the more precise unit (the one with the lower multiplier).
092        if (otherUnit.multiplier() > getUnit().multiplier()) {
093          return getValue().equals(other.asUnit(getUnit()));
094        } else {
095          return as(otherUnit).equals(other.getValue());
096        }
097      }
098    
099      @Override
100      public String toString() {
101        return amount.toString();
102      }
103    
104      @Override
105      public int compareTo(Amount<T, U> other) {
106        // Compare in the more precise unit (the one with the lower multiplier).
107        if (other.getUnit().multiplier() > getUnit().multiplier()) {
108          return getValue().compareTo(other.as(getUnit()));
109        } else {
110          return as(other.getUnit()).compareTo(other.getValue());
111        }
112      }
113    
114      private boolean sameUnits(Unit<? extends Unit<?>> unit) {
115        return getUnit().equals(unit);
116      }
117    
118      protected abstract T scale(double multiplier);
119    
120      public static <U extends Unit<U>> Amount<Double, U> of(double value, U unit) {
121        return new Amount<Double, U>(value, unit) {
122          @Override protected Double scale(double multiplier) {
123            return getValue() * multiplier;
124          }
125        };
126      }
127    
128      public static <U extends Unit<U>> Amount<Float, U> of(float value, U unit) {
129        return new Amount<Float, U>(value, unit) {
130          @Override protected Float scale(double multiplier) {
131            return (float) (getValue() * multiplier);
132          }
133        };
134      }
135    
136      public static <U extends Unit<U>> Amount<Long, U> of(long value, U unit) {
137        return new Amount<Long, U>(value, unit) {
138          @Override protected Long scale(double multiplier) {
139            return (long) (getValue() * multiplier);
140          }
141        };
142      }
143    
144      public static <U extends Unit<U>> Amount<Integer, U> of(int value, U unit) {
145        return new Amount<Integer, U>(value, unit) {
146          @Override protected Integer scale(double multiplier) {
147            return (int) (getValue() * multiplier);
148          }
149        };
150      }
151    }
