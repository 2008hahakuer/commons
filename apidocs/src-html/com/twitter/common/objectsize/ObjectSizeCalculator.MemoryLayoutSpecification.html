038    /**
039     * Contains utility methods for calculating the memory usage of objects. It
040     * only works on the HotSpot JVM, and infers the actual memory layout (32 bit
041     * vs. 64 bit word size, compressed object pointers vs. uncompressed) from
042     * best available indicators. It can reliably detect a 32 bit vs. 64 bit JVM.
043     * It can only make an educated guess at whether compressed OOPs are used,
044     * though; specifically, it knows what the JVM's default choice of OOP
045     * compression would be based on HotSpot version and maximum heap sizes, but if
046     * the choice is explicitly overridden with the <tt>-XX:{+|-}UseCompressedOops</tt> command line
047     * switch, it can not detect
048     * this fact and will report incorrect sizes, as it will presume the default JVM
049     * behavior.
050     *
051     * @author Attila Szegedi
052     */
053    public class ObjectSizeCalculator {
054    
055      /**
056       * Describes constant memory overheads for various constructs in a JVM implementation.
057       */
058      public interface MemoryLayoutSpecification {
059    
060        /**
061         * Returns the fixed overhead of an array of any type or length in this JVM.
062         *
063         * @return the fixed overhead of an array.
064         */
065        int getArrayHeaderSize();
066    
067        /**
068         * Returns the fixed overhead of for any {@link Object} subclass in this JVM.
069         *
070         * @return the fixed overhead of any object.
071         */
072        int getObjectHeaderSize();
073    
074        /**
075         * Returns the quantum field size for a field owned by an object in this JVM.
076         *
077         * @return the quantum field size for an object.
078         */
079        int getObjectPadding();
080    
081        /**
082         * Returns the fixed size of an object reference in this JVM.
083         *
084         * @return the size of all object references.
085         */
086        int getReferenceSize();
087    
088        /**
089         * Returns the quantum field size for a field owned by one of an object's ancestor superclasses
090         * in this JVM.
091         *
092         * @return the quantum field size for a superclass field.
093         */
094        int getSuperclassFieldPadding();
095      }
096    
097      private static class CurrentLayout {
098        private static final MemoryLayoutSpecification SPEC =
099            getEffectiveMemoryLayoutSpecification();
100      }
101    
102      /**
103       * Given an object, returns the total allocated size, in bytes, of the object
104       * and all other objects reachable from it.  Attempts to to detect the current JVM memory layout,
105       * but may fail with {@link UnsupportedOperationException};
106       *
107       * @param obj the object; can be null. Passing in a {@link java.lang.Class} object doesn't do
108       *          anything special, it measures the size of all objects
109       *          reachable through it (which will include its class loader, and by
110       *          extension, all other Class objects loaded by
111       *          the same loader, and all the parent class loaders). It doesn't provide the
112       *          size of the static fields in the JVM class that the Class object
113       *          represents.
114       * @return the total allocated size of the object and all other objects it
115       *         retains.
116       * @throws UnsupportedOperationException if the current vm memory layout cannot be detected.
117       */
118      public static long getObjectSize(Object obj) throws UnsupportedOperationException {
119        return obj == null ? 0 : new ObjectSizeCalculator(CurrentLayout.SPEC).calculateObjectSize(obj);
120      }
121    
122      // Fixed object header size for arrays.
123      private final int arrayHeaderSize;
124      // Fixed object header size for non-array objects.
125      private final int objectHeaderSize;
126      // Padding for the object size - if the object size is not an exact multiple
127      // of this, it is padded to the next multiple.
128      private final int objectPadding;
129      // Size of reference (pointer) fields.
130      private final int referenceSize;
131      // Padding for the fields of superclass before fields of subclasses are
132      // added.
133      private final int superclassFieldPadding;
134    
135      private final LoadingCache<Class<?>, ClassSizeInfo> classSizeInfos =
136          CacheBuilder.newBuilder().build(new CacheLoader<Class<?>, ClassSizeInfo>() {
137            public ClassSizeInfo load(Class<?> clazz) {
138              return new ClassSizeInfo(clazz);
139            }
140          });
141    
142    
143      private final Set<Object> alreadyVisited = Sets.newIdentityHashSet();
144      private final Deque<Object> pending = new ArrayDeque<Object>(16 * 1024);
145      private long size;
146    
147      /**
148       * Creates an object size calculator that can calculate object sizes for a given
149       * {@code memoryLayoutSpecification}.
150       *
151       * @param memoryLayoutSpecification a description of the JVM memory layout.
152       */
153      public ObjectSizeCalculator(MemoryLayoutSpecification memoryLayoutSpecification) {
154        Preconditions.checkNotNull(memoryLayoutSpecification);
155        arrayHeaderSize = memoryLayoutSpecification.getArrayHeaderSize();
156        objectHeaderSize = memoryLayoutSpecification.getObjectHeaderSize();
157        objectPadding = memoryLayoutSpecification.getObjectPadding();
158        referenceSize = memoryLayoutSpecification.getReferenceSize();
159        superclassFieldPadding = memoryLayoutSpecification.getSuperclassFieldPadding();
160      }
161    
162      /**
163       * Given an object, returns the total allocated size, in bytes, of the object
164       * and all other objects reachable from it.
165       *
166       * @param obj the object; can be null. Passing in a {@link java.lang.Class} object doesn't do
167       *          anything special, it measures the size of all objects
168       *          reachable through it (which will include its class loader, and by
169       *          extension, all other Class objects loaded by
170       *          the same loader, and all the parent class loaders). It doesn't provide the
171       *          size of the static fields in the JVM class that the Class object
172       *          represents.
173       * @return the total allocated size of the object and all other objects it
174       *         retains.
175       */
176      public synchronized long calculateObjectSize(Object obj) {
177        // Breadth-first traversal instead of naive depth-first with recursive
178        // implementation, so we don't blow the stack traversing long linked lists.
179        try {
180          for (;;) {
181            visit(obj);
182            if (pending.isEmpty()) {
183              return size;
184            }
185            obj = pending.removeFirst();
186          }
187        } finally {
188          alreadyVisited.clear();
189          pending.clear();
190          size = 0;
191        }
192      }
193    
194      private void visit(Object obj) {
195        if (alreadyVisited.contains(obj)) {
196          return;
197        }
198        final Class<?> clazz = obj.getClass();
199        if (clazz == ArrayElementsVisitor.class) {
200          ((ArrayElementsVisitor) obj).visit(this);
201        } else {
202          alreadyVisited.add(obj);
203          if (clazz.isArray()) {
204            visitArray(obj);
205          } else {
206            classSizeInfos.getUnchecked(clazz).visit(obj, this);
207          }
208        }
209      }
210    
211      private void visitArray(Object array) {
212        final Class<?> componentType = array.getClass().getComponentType();
213        final int length = Array.getLength(array);
214        if (componentType.isPrimitive()) {
215          increaseByArraySize(length, getPrimitiveFieldSize(componentType));
216        } else {
217          increaseByArraySize(length, referenceSize);
218          // If we didn't use an ArrayElementsVisitor, we would be enqueueing every
219          // element of the array here instead. For large arrays, it would
220          // tremendously enlarge the queue. In essence, we're compressing it into
221          // a small command object instead. This is different than immediately
222          // visiting the elements, as their visiting is scheduled for the end of
223          // the current queue.
224          switch (length) {
225            case 0: {
226              break;
227            }
228            case 1: {
229              enqueue(Array.get(array, 0));
230              break;
231            }
232            default: {
233              enqueue(new ArrayElementsVisitor((Object[]) array));
234            }
235          }
236        }
237      }
238    
239      private void increaseByArraySize(int length, long elementSize) {
240        increaseSize(roundTo(arrayHeaderSize + length * elementSize, objectPadding));
241      }
242    
243      private static class ArrayElementsVisitor {
244        private final Object[] array;
245    
246        ArrayElementsVisitor(Object[] array) {
247          this.array = array;
248        }
249    
250        public void visit(ObjectSizeCalculator calc) {
251          for (Object elem : array) {
252            if (elem != null) {
253              calc.visit(elem);
254            }
255          }
256        }
257      }
258    
259      void enqueue(Object obj) {
260        if (obj != null) {
261          pending.addLast(obj);
262        }
263      }
264    
265      void increaseSize(long objectSize) {
266        size += objectSize;
267      }
268    
269      @VisibleForTesting
270      static long roundTo(long x, int multiple) {
271        return ((x + multiple - 1) / multiple) * multiple;
272      }
273    
274      private class ClassSizeInfo {
275        // Padded fields + header size
276        private final long objectSize;
277        // Only the fields size - used to calculate the subclasses' memory
278        // footprint.
279        private final long fieldsSize;
280        private final Field[] referenceFields;
281    
282        public ClassSizeInfo(Class<?> clazz) {
283          long fieldsSize = 0;
284          final List<Field> referenceFields = new LinkedList<Field>();
285          for (Field f : clazz.getDeclaredFields()) {
286            if (Modifier.isStatic(f.getModifiers())) {
287              continue;
288            }
289            final Class<?> type = f.getType();
290            if (type.isPrimitive()) {
291              fieldsSize += getPrimitiveFieldSize(type);
292            } else {
293              f.setAccessible(true);
294              referenceFields.add(f);
295              fieldsSize += referenceSize;
296            }
297          }
298          final Class<?> superClass = clazz.getSuperclass();
299          if (superClass != null) {
300            final ClassSizeInfo superClassInfo = classSizeInfos.getUnchecked(superClass);
301            fieldsSize += roundTo(superClassInfo.fieldsSize, superclassFieldPadding);
302            referenceFields.addAll(Arrays.asList(superClassInfo.referenceFields));
303          }
304          this.fieldsSize = fieldsSize;
305          this.objectSize = roundTo(objectHeaderSize + fieldsSize, objectPadding);
306          this.referenceFields = referenceFields.toArray(
307              new Field[referenceFields.size()]);
308        }
309    
310        void visit(Object obj, ObjectSizeCalculator calc) {
311          calc.increaseSize(objectSize);
312          enqueueReferencedObjects(obj, calc);
313        }
314    
315        public void enqueueReferencedObjects(Object obj, ObjectSizeCalculator calc) {
316          for (Field f : referenceFields) {
317            try {
318              calc.enqueue(f.get(obj));
319            } catch (IllegalAccessException e) {
320              final AssertionError ae = new AssertionError(
321                  "Unexpected denial of access to " + f);
322              ae.initCause(e);
323              throw ae;
324            }
325          }
326        }
327      }
328    
329      private static long getPrimitiveFieldSize(Class<?> type) {
330        if (type == boolean.class || type == byte.class) {
331          return 1;
332        }
333        if (type == char.class || type == short.class) {
334          return 2;
335        }
336        if (type == int.class || type == float.class) {
337          return 4;
338        }
339        if (type == long.class || type == double.class) {
340          return 8;
341        }
342        throw new AssertionError("Encountered unexpected primitive type " +
343            type.getName());
344      }

346      @VisibleForTesting
347      static MemoryLayoutSpecification getEffectiveMemoryLayoutSpecification() {
348        final String vmName = System.getProperty("java.vm.name");
349        if (vmName == null || !vmName.startsWith("Java HotSpot(TM) ")) {
350          throw new UnsupportedOperationException(
351              "ObjectSizeCalculator only supported on HotSpot VM");
352        }
353    
354        final String dataModel = System.getProperty("sun.arch.data.model");
355        if ("32".equals(dataModel)) {
356          // Running with 32-bit data model
357          return new MemoryLayoutSpecification() {
358            @Override public int getArrayHeaderSize() {
359              return 12;
360            }
361            @Override public int getObjectHeaderSize() {
362              return 8;
363            }
364            @Override public int getObjectPadding() {
365              return 8;
366            }
367            @Override public int getReferenceSize() {
368              return 4;
369            }
370            @Override public int getSuperclassFieldPadding() {
371              return 4;
372            }
373          };
374        } else if (!"64".equals(dataModel)) {
375          throw new UnsupportedOperationException("Unrecognized value '" +
376              dataModel + "' of sun.arch.data.model system property");
377        }
378    
379        final String strVmVersion = System.getProperty("java.vm.version");
380        final int vmVersion = Integer.parseInt(strVmVersion.substring(0,
381            strVmVersion.indexOf('.')));
382        if (vmVersion >= 17) {
383          long maxMemory = 0;
384          for (MemoryPoolMXBean mp : ManagementFactory.getMemoryPoolMXBeans()) {
385            maxMemory += mp.getUsage().getMax();
386          }
387          if (maxMemory < 30L * 1024 * 1024 * 1024) {
388            // HotSpot 17.0 and above use compressed OOPs below 30GB of RAM total
389            // for all memory pools (yes, including code cache).
390            return new MemoryLayoutSpecification() {
391              @Override public int getArrayHeaderSize() {
392                return 16;
393              }
394              @Override public int getObjectHeaderSize() {
395                return 12;
396              }
397              @Override public int getObjectPadding() {
398                return 8;
399              }
400              @Override public int getReferenceSize() {
401                return 4;
402              }
403              @Override public int getSuperclassFieldPadding() {
404                return 4;
405              }
406            };
407          }
408        }
409    
410        // In other cases, it's a 64-bit uncompressed OOPs object model
411        return new MemoryLayoutSpecification() {
412          @Override public int getArrayHeaderSize() {
413            return 24;
414          }
415          @Override public int getObjectHeaderSize() {
416            return 16;
417          }
418          @Override public int getObjectPadding() {
419            return 8;
420          }
421          @Override public int getReferenceSize() {
422            return 8;
423          }
424          @Override public int getSuperclassFieldPadding() {
425            return 8;
426          }
427        };
428      }
