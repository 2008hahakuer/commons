017    package com.twitter.common.objectsize;
018    
019    import com.google.common.base.Function;
020    import com.google.common.collect.MapMaker;
021    
022    import javax.annotation.Nullable;
023    import java.lang.reflect.Array;
024    import java.lang.reflect.Field;
025    import java.lang.reflect.Modifier;
026    import java.util.ArrayDeque;
027    import java.util.Arrays;
028    import java.util.Deque;
029    import java.util.IdentityHashMap;
030    import java.util.LinkedList;
031    import java.util.List;
032    import java.util.Map;
033    import java.util.concurrent.ConcurrentMap;
034    
035    /**
036     * Contains utility methods for calculating the memory usage of objects. It
037     * assumes the memory layout used by 64-bit Oracle HotSpot VMs.
038     *
039     * @author Attila Szegedi
040     */
041    public class ObjectSizeCalculator {
042      // TODO: generalize for different VMs.
043      private static final int OBJECT_HEADER_SIZE = 16;
044      private static final int ARRAY_HEADER_SIZE = OBJECT_HEADER_SIZE + 8;
045      private static final int REFERENCE_FIELD_SIZE = 8;
046    
047      private static final ConcurrentMap<Class<?>, ClassSizeInfo> classSizeInfos =
048        new MapMaker().makeComputingMap(new Function<Class<?>, ClassSizeInfo>() {
049          public ClassSizeInfo apply(Class<?> clazz) {
050            return new ClassSizeInfo(clazz);
051          }
052        });
053    
054      /**
055       * Given an object, returns the total allocated size, in bytes, of the object and all
056       * other objects reachable from it. Assumes the 64-bit HotSpot VM.
057       *
058       * @param obj the object; can be null. Passing in a {@link java.lang.Class} object doesn't
059       * do anything special, it measures the size of all objects reachable through it (which
060       * will include its class loader, and by extension, all other Class objects loaded by
061       * the same loader, and all the parent class loaders). It doesn't provide the size of the
062       * static fields in the JVM class that the Class object represents.
063       * @return the total allocated size of the object and all other objects it retains.
064       */
065      @Nullable
066      public static long getObjectSize(Object obj) {
067        return obj == null ? 0 : new ObjectSizeCalculator().getObjectSizeInternal(obj);
068      }
069    
070      // We'd use an IdentityHashSet, if such a thing existed. Sets#newSetFromMap could be
071      // used, but don't want to create a wrapper for the simple usage we have here.
072      private final Map<Object, Object> alreadyVisited = new IdentityHashMap<Object, Object>(512 * 1024);
073      private final Deque<Object> pending = new ArrayDeque<Object>(16 * 1024);
074      private long size;
075    
076      private ObjectSizeCalculator() {
077        // Clients use getObjectSize(Object obj)
078      }
079    
080      private long getObjectSizeInternal(Object obj) {
081        for (;;) {
082          visit(obj);
083          if (pending.isEmpty()) {
084            return size;
085          }
086          obj = pending.removeFirst();
087        }
088      }
089    
090      private void visit(Object obj) {
091        if (alreadyVisited.containsKey(obj)) {
092          return;
093        }
094        final Class<?> clazz = obj.getClass();
095        if (clazz == ArrayElementsVisitor.class) {
096          ((ArrayElementsVisitor) obj).visit(this);
097        } else {
098          alreadyVisited.put(obj, obj);
099          if (clazz.isArray()) {
100            visitArray(obj);
101          } else {
102            classSizeInfos.get(clazz).visit(obj, this);
103          }
104        }
105      }
106    
107      private void visitArray(Object array) {
108        final Class<?> componentType = array.getClass().getComponentType();
109        final int length = Array.getLength(array);
110        if (componentType.isPrimitive()) {
111          increaseSize(ARRAY_HEADER_SIZE + length * getPrimitiveFieldSize(componentType));
112        } else {
113          increaseSize(ARRAY_HEADER_SIZE + length * REFERENCE_FIELD_SIZE);
114          // If we didn't use an ArrayElementsVisitor, we would be enqueueing every element
115          // of the array here instead. For large arrays, it would create a huge amount of
116          // linked nodes in the pending list. In essence, we're compressing it into a
117          // small command object instead. This is different than immediately visiting the
118          // elements, as the visiting is now scheduled for the end of the current queue.
119          switch (length) {
120            case 0: {
121              break;
122            }
123            case 1: {
124              enqueue(Array.get(array, 0));
125            }
126            default: {
127              enqueue(new ArrayElementsVisitor((Object[]) array));
128            }
129          }
130        }
131      }
132    
133      private static class ArrayElementsVisitor {
134        private final Object[] array;
135    
136        ArrayElementsVisitor(Object[] array) {
137          this.array = array;
138        }
139    
140        public void visit(ObjectSizeCalculator calc) {
141          for (Object elem : array) {
142            if (elem != null) {
143              calc.visit(elem);
144            }
145          }
146        }
147      }
148    
149      void enqueue(Object obj) {
150        if (obj != null) {
151          pending.addLast(obj);
152        }
153      }
154    
155      void increaseSize(long objectSize) {
156        size += roundToEight(objectSize);
157      }
158    
159      private static long roundToEight(long x) {
160        return ((x + 7) / 8) * 8;
161      }
162    
163      private static class ClassSizeInfo {
164        private final long objectSize;
165        private final Field[] referenceFields;
166    
167        public ClassSizeInfo(Class<?> clazz) {
168          long objectSize = 0;
169          final List<Field> referenceFields = new LinkedList<Field>();
170          for (Field f: clazz.getDeclaredFields()) {
171            if (Modifier.isStatic(f.getModifiers())) {
172              continue;
173            }
174            final Class<?> type = f.getType();
175            if (type.isPrimitive()) {
176              objectSize += getPrimitiveFieldSize(type);
177            } else {
178              f.setAccessible(true);
179              referenceFields.add(f);
180              objectSize += REFERENCE_FIELD_SIZE;
181            }
182          }
183          final Class<?> superClass = clazz.getSuperclass();
184          if (superClass != null) {
185            final ClassSizeInfo superClassInfo = classSizeInfos.get(superClass);
186            objectSize += superClassInfo.objectSize - OBJECT_HEADER_SIZE;
187            referenceFields.addAll(Arrays.asList(superClassInfo.referenceFields));
188          }
189          this.objectSize = OBJECT_HEADER_SIZE + roundToEight(objectSize);
190          this.referenceFields = referenceFields.toArray(new Field[referenceFields.size()]);
191        }
192    
193        void visit(Object obj, ObjectSizeCalculator calc) {
194          calc.increaseSize(objectSize);
195          enqueueReferencedObjects(obj, calc);
196        }
197    
198        public void enqueueReferencedObjects(Object obj, ObjectSizeCalculator calc) {
199          for (Field f : referenceFields) {
200            try {
201              calc.enqueue(f.get(obj));
202            } catch(IllegalAccessException e) {
203              final AssertionError ae = new AssertionError("Unexpected denial of access to " + f);
204              ae.initCause(e);
205              throw ae;
206            }
207          }
208        }
209      }
210    
211      private static long getPrimitiveFieldSize(Class<?> type) {
212        if (type == boolean.class || type == byte.class) {
213          return 1;
214        }
215        if (type == char.class || type == short.class) {
216          return 2;
217        }
218        if (type == int.class || type == float.class) {
219          return 4;
220        }
221        if (type == long.class || type == double.class) {
222          return 8;
223        }
224        throw new AssertionError("Encountered unexpected primitive type " + type.getName());
225      }
