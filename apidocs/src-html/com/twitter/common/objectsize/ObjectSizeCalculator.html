037    /**
038     * Contains utility methods for calculating the memory usage of objects. It
039     * only works on the HotSpot JVM, and infers the actual memory layout (32 bit
040     * vs. 64 bit word size, compressed object pointers vs. uncompressed) from
041     * best available indicators. It can reliably detect a 32 bit vs. 64 bit JVM.
042     * It can only make an educated guess at whether compressed OOPs are used,
043     * though; specifically, it knows what the JVM's default choice of OOP
044     * compression would be based on HotSpot version and maximum heap sizes, but if
045     * the choice is explicitly overridden with the <tt>-XX:{+|-}UseCompressedOops</tt> command line
046     * switch, it can not detect
047     * this fact and will report incorrect sizes, as it will presume the default JVM
048     * behavior.
049     *
050     * @author Attila Szegedi
051     */
052    public class ObjectSizeCalculator {
053      private static final MemoryLayoutSpecification LAYOUT_SPEC =
054          getEffectiveMemoryLayoutSpecification();
055      // Fixed object header size for arrays.
056      private static final int ARRAY_HEADER_SIZE = LAYOUT_SPEC.getArrayHeaderSize();
057      // Fixed object header size for non-array objects.
058      private static final int OBJECT_HEADER_SIZE =
059          LAYOUT_SPEC.getObjectHeaderSize();
060      // Padding for the object size - if the object size is not an exact multiple
061      // of this, it is padded to the next multiple. All currently supported JVMs
062      // use 8.
063      private static final int OBJECT_PADDING = 8;
064      // Size of reference (pointer) fields.
065      private static final int REFERENCE_SIZE = LAYOUT_SPEC.getReferenceSize();
066      // Padding for the fields of superclass before fields of subclasses are
067      // added.
068      private static final int SUPERCLASS_FIELD_PADDING =
069          LAYOUT_SPEC.getSuperclassFieldPadding();
070    
071      private static final ConcurrentMap<Class<?>, ClassSizeInfo> classSizeInfos =
072          new MapMaker().makeComputingMap(new Function<Class<?>, ClassSizeInfo>() {
073            public ClassSizeInfo apply(Class<?> clazz) {
074              return new ClassSizeInfo(clazz);
075            }
076          });
077    
078      /**
079       * Given an object, returns the total allocated size, in bytes, of the object
080       * and all other objects reachable from it.
081       *
082       * @param obj the object; can be null. Passing in a {@link java.lang.Class} object doesn't do
083       *          anything special, it measures the size of all objects
084       *          reachable through it (which will include its class loader, and by
085       *          extension, all other Class objects loaded by
086       *          the same loader, and all the parent class loaders). It doesn't provide the
087       *          size of the static fields in the JVM class that the Class object
088       *          represents.
089       * @return the total allocated size of the object and all other objects it
090       *         retains.
091       */
092      public static long getObjectSize(Object obj) {
093        return obj == null ? 0 : new ObjectSizeCalculator().getObjectSizeInternal(
094            obj);
095      }
096    
097      private final Set<Object> alreadyVisited = Sets.newIdentityHashSet();
098      private final Deque<Object> pending = new ArrayDeque<Object>(16 * 1024);
099      private long size;
100    
101      private ObjectSizeCalculator() {
102        // Clients use getObjectSize(Object obj)
103      }
104    
105      private long getObjectSizeInternal(Object obj) {
106        // Breadth-first traversal instead of naive depth-first with recursive
107        // implementation, so we don't blow the stack traversing long linked lists.
108        for (;;) {
109          visit(obj);
110          if (pending.isEmpty()) {
111            return size;
112          }
113          obj = pending.removeFirst();
114        }
115      }
116    
117      private void visit(Object obj) {
118        if (alreadyVisited.contains(obj)) {
119          return;
120        }
121        final Class<?> clazz = obj.getClass();
122        if (clazz == ArrayElementsVisitor.class) {
123          ((ArrayElementsVisitor) obj).visit(this);
124        } else {
125          alreadyVisited.add(obj);
126          if (clazz.isArray()) {
127            visitArray(obj);
128          } else {
129            classSizeInfos.get(clazz).visit(obj, this);
130          }
131        }
132      }
133    
134      private void visitArray(Object array) {
135        final Class<?> componentType = array.getClass().getComponentType();
136        final int length = Array.getLength(array);
137        if (componentType.isPrimitive()) {
138          increaseByArraySize(length, getPrimitiveFieldSize(componentType));
139        } else {
140          increaseByArraySize(length, REFERENCE_SIZE);
141          // If we didn't use an ArrayElementsVisitor, we would be enqueueing every
142          // element of the array here instead. For large arrays, it would
143          // tremendously enlarge the queue. In essence, we're compressing it into
144          // a small command object instead. This is different than immediately
145          // visiting the elements, as their visiting is scheduled for the end of
146          // the current queue.
147          switch (length) {
148            case 0: {
149              break;
150            }
151            case 1: {
152              enqueue(Array.get(array, 0));
153              break;
154            }
155            default: {
156              enqueue(new ArrayElementsVisitor((Object[]) array));
157            }
158          }
159        }
160      }
161    
162      private void increaseByArraySize(int length, long elementSize) {
163        increaseSize(roundTo(ARRAY_HEADER_SIZE + length * elementSize,
164            OBJECT_PADDING));
165      }
166    
167      private static class ArrayElementsVisitor {
168        private final Object[] array;
169    
170        ArrayElementsVisitor(Object[] array) {
171          this.array = array;
172        }
173    
174        public void visit(ObjectSizeCalculator calc) {
175          for (Object elem : array) {
176            if (elem != null) {
177              calc.visit(elem);
178            }
179          }
180        }
181      }
182    
183      void enqueue(Object obj) {
184        if (obj != null) {
185          pending.addLast(obj);
186        }
187      }
188    
189      void increaseSize(long objectSize) {
190        size += objectSize;
191      }
192    
193      @VisibleForTesting
194      static long roundTo(long x, int multiple) {
195        return ((x + multiple - 1) / multiple) * multiple;
196      }
197    
198      private static class ClassSizeInfo {
199        // Padded fields + header size
200        private final long objectSize;
201        // Only the fields size - used to calculate the subclasses' memory
202        // footprint.
203        private final long fieldsSize;
204        private final Field[] referenceFields;
205    
206        public ClassSizeInfo(Class<?> clazz) {
207          long fieldsSize = 0;
208          final List<Field> referenceFields = new LinkedList<Field>();
209          for (Field f : clazz.getDeclaredFields()) {
210            if (Modifier.isStatic(f.getModifiers())) {
211              continue;
212            }
213            final Class<?> type = f.getType();
214            if (type.isPrimitive()) {
215              fieldsSize += getPrimitiveFieldSize(type);
216            } else {
217              f.setAccessible(true);
218              referenceFields.add(f);
219              fieldsSize += REFERENCE_SIZE;
220            }
221          }
222          final Class<?> superClass = clazz.getSuperclass();
223          if (superClass != null) {
224            final ClassSizeInfo superClassInfo = classSizeInfos.get(superClass);
225            fieldsSize += roundTo(superClassInfo.fieldsSize,
226                SUPERCLASS_FIELD_PADDING);
227            referenceFields.addAll(Arrays.asList(superClassInfo.referenceFields));
228          }
229          this.fieldsSize = fieldsSize;
230          this.objectSize = roundTo(OBJECT_HEADER_SIZE + fieldsSize,
231              OBJECT_PADDING);
232          this.referenceFields = referenceFields.toArray(
233              new Field[referenceFields.size()]);
234        }
235    
236        void visit(Object obj, ObjectSizeCalculator calc) {
237          calc.increaseSize(objectSize);
238          enqueueReferencedObjects(obj, calc);
239        }
240    
241        public void enqueueReferencedObjects(Object obj, ObjectSizeCalculator calc) {
242          for (Field f : referenceFields) {
243            try {
244              calc.enqueue(f.get(obj));
245            } catch (IllegalAccessException e) {
246              final AssertionError ae = new AssertionError(
247                  "Unexpected denial of access to " + f);
248              ae.initCause(e);
249              throw ae;
250            }
251          }
252        }
253      }
254    
255      private static long getPrimitiveFieldSize(Class<?> type) {
256        if (type == boolean.class || type == byte.class) {
257          return 1;
258        }
259        if (type == char.class || type == short.class) {
260          return 2;
261        }
262        if (type == int.class || type == float.class) {
263          return 4;
264        }
265        if (type == long.class || type == double.class) {
266          return 8;
267        }
268        throw new AssertionError("Encountered unexpected primitive type " +
269            type.getName());
270      }

272      private abstract static class MemoryLayoutSpecification {
273        abstract int getArrayHeaderSize();
274    
275        abstract int getObjectHeaderSize();
276    
277        abstract int getReferenceSize();
278    
279        abstract int getSuperclassFieldPadding();
280      }
281    
282      private static MemoryLayoutSpecification getEffectiveMemoryLayoutSpecification() {
283        final String vmName = System.getProperty("java.vm.name");
284        if (vmName == null || !vmName.startsWith("Java HotSpot(TM) ")) {
285          throw new UnsupportedOperationException(
286              "ObjectSizeCalculator only supported on HotSpot VM");
287        }
288    
289        final String dataModel = System.getProperty("sun.arch.data.model");
290        if ("32".equals(dataModel)) {
291          // Running with 32-bit data model
292          return new MemoryLayoutSpecification() {
293            int getArrayHeaderSize() {
294              return 12;
295            }
296    
297            int getObjectHeaderSize() {
298              return 8;
299            }
300    
301            int getReferenceSize() {
302              return 4;
303            }
304    
305            int getSuperclassFieldPadding() {
306              return 4;
307            }
308          };
309        } else if (!"64".equals(dataModel)) {
310          throw new UnsupportedOperationException("Unrecognized value '" +
311              dataModel + "' of sun.arch.data.model system property");
312        }
313    
314        final String strVmVersion = System.getProperty("java.vm.version");
315        final int vmVersion = Integer.parseInt(strVmVersion.substring(0,
316            strVmVersion.indexOf('.')));
317        if (vmVersion >= 17) {
318          long maxMemory = 0;
319          for (MemoryPoolMXBean mp : ManagementFactory.getMemoryPoolMXBeans()) {
320            maxMemory += mp.getUsage().getMax();
321          }
322          if (maxMemory < 30L * 1024 * 1024 * 1024) {
323            // HotSpot 17.0 and above use compressed OOPs below 30GB of RAM total
324            // for all memory pools (yes, including code cache).
325            return new MemoryLayoutSpecification() {
326              int getArrayHeaderSize() {
327                return 16;
328              }
329    
330              int getObjectHeaderSize() {
331                return 12;
332              }
333    
334              int getReferenceSize() {
335                return 4;
336              }
337    
338              int getSuperclassFieldPadding() {
339                return 4;
340              }
341            };
342          }
343        }
344    
345        // In other cases, it's a 64-bit uncompressed OOPs object model
346        return new MemoryLayoutSpecification() {
347          int getArrayHeaderSize() {
348            return 24;
349          }
350    
351          int getObjectHeaderSize() {
352            return 16;
353          }
354    
355          int getReferenceSize() {
356            return 8;
357          }
358    
359          int getSuperclassFieldPadding() {
360            return 8;
361          }
362        };
363      }
