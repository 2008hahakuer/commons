017    package com.twitter.common.application;
018    
019    import java.util.List;
020    import java.util.logging.Level;
021    import java.util.logging.Logger;
022    
023    import com.google.common.base.Preconditions;
024    import com.google.common.collect.Lists;
025    
026    import com.twitter.common.base.Command;
027    import com.twitter.common.base.ExceptionalCommand;
028    
029    /**
030     * A shutdown action controller. It executes actions in the reverse order they were registered, and
031     * logs a warning for every shutdown action that fails, but doesn't prevent completion of subsequent
032     * actions or the normal completion of the {@code execute()} method.
033     *
034     * @author Attila Szegedi
035     */
036    public interface ShutdownRegistry {
037    
038      <E extends Exception, T extends ExceptionalCommand<E>> void addAction(T action);
039    
040      public static class ShutdownRegistryImpl implements ShutdownRegistry, Command {
041        private static final Logger LOG = Logger.getLogger(ShutdownRegistry.class.getName());
042    
043        private final List<ExceptionalCommand<? extends Exception>> actions = Lists.newLinkedList();
044    
045        private boolean completed = false;
046    
047        /**
048         * Registers an action to execute during {@link #execute()}. It is an error to call this method
049         * after calling {@link #execute()}.
050         *
051         * @param action the action to add to the list of actions to execute during execution
052         */
053        @Override
054        public synchronized <E extends Exception, T extends ExceptionalCommand<E>> void addAction(
055            T action) {
056          Preconditions.checkState(!completed);
057          actions.add(action);
058        }
059    
060        /**
061         * Executes an application lifecycle stage by executing all registered actions. This method can
062         * be called multiple times but will only execute the registered actions the first time.
063         *
064         * This sends output to System.out because logging is unreliable during JVM shutdown, which
065         * this class may be used for.
066         */
067        @Override
068        public synchronized void execute() {
069          if (!completed) {
070            LOG.info(String.format("Executing %d lifecycle commands.", actions.size()));
071            completed = true;
072            try {
073              for (ExceptionalCommand<? extends Exception> action : Lists.reverse(actions)) {
074                try {
075                  action.execute();
076                } catch (Exception e) {
077                  LOG.log(Level.WARNING, "Shutdown action failed.", e);
078                }
079              }
080            } finally {
081              actions.clear();
082            }
083          } else {
084            LOG.info("Action controller has already completed, subsequent calls ignored.");
085          }
086        }
087      }
088    }
