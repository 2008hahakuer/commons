017    package com.twitter.common.application.modules;
018    
019    import java.net.InetSocketAddress;
020    import java.net.UnknownHostException;
021    import java.util.List;
022    import java.util.Map;
023    import java.util.Set;
024    
025    import com.google.common.base.Function;
026    import com.google.common.base.Optional;
027    import com.google.common.base.Preconditions;
028    import com.google.common.base.Predicate;
029    import com.google.common.base.Predicates;
030    import com.google.common.collect.ImmutableList;
031    import com.google.common.collect.ImmutableMap;
032    import com.google.common.collect.Iterables;
033    import com.google.common.collect.Maps;
034    import com.google.inject.Inject;
035    import com.google.inject.Provider;
036    
037    import org.apache.commons.lang.builder.ToStringBuilder;
038    
039    import com.twitter.common.application.ShutdownRegistry;
040    import com.twitter.common.application.modules.LifecycleModule.LaunchException;
041    import com.twitter.common.application.modules.LifecycleModule.Service;
042    import com.twitter.common.application.modules.LifecycleModule.ServiceRunner;
043    import com.twitter.common.base.Command;
044    import com.twitter.common.net.InetSocketAddressHelper;
045    
046    /**
047     * Registry for services that should be exported from the application.
048     *
049     * Example of announcing and registering a port:
050     * <pre>
051     * class MyLauncher implements Provider<LocalService> {
052     *   public LocalService get() {
053     *     // Launch service.
054     *   }
055     * }
056     *
057     * class MyServiceModule extends AbstractModule {
058     *   public void configure() {
059     *     LifeCycleModule.bindServiceLauncher(binder(), MyLauncher.class);
060     *   }
061     * }
062     * </pre>
063     *
064     * @author William Farner
065     */
066    public class LocalServiceRegistry {
067    
068      private static final Predicate<LocalService> IS_PRIMARY = new Predicate<LocalService>() {
069        @Override public boolean apply(LocalService service) {
070          return service.primary;
071        }
072      };
073    
074      private static final Function<LocalService, InetSocketAddress> SERVICE_TO_SOCKET =
075          new Function<LocalService, InetSocketAddress>() {
076            @Override public InetSocketAddress apply(LocalService service) {
077              try {
078                return InetSocketAddressHelper.getLocalAddress(service.port);
079              } catch (UnknownHostException e) {
080                throw new RuntimeException("Failed to resolve local address for " + service, e);
081              }
082            }
083          };
084    
085      private static final Function<LocalService, String> GET_NAME =
086          new Function<LocalService, String>() {
087            @Override public String apply(LocalService service) {
088              return service.name.get();
089            }
090          };
091    
092      private final ShutdownRegistry shutdownRegistry;
093      private final Provider<Set<ServiceRunner>> runnerProvider;
094    
095      private Optional<InetSocketAddress> primarySocket = null;
096      private Map<String, InetSocketAddress> auxiliarySockets = null;
097    
098      /**
099       * Creates a new local service registry.
100       *
101       * @param runnerProvider provider of registered local services.
102       * @param shutdownRegistry Shutdown registry to tear down launched services.
103       */
104      @Inject
105      public LocalServiceRegistry(@Service Provider<Set<ServiceRunner>> runnerProvider,
106          ShutdownRegistry shutdownRegistry) {
107        this.runnerProvider = Preconditions.checkNotNull(runnerProvider);
108        this.shutdownRegistry = Preconditions.checkNotNull(shutdownRegistry);
109      }
110    
111      /**
112       * Launches the local services if not already launched, otherwise this is a no-op.
113       */
114      void ensureLaunched() {
115        if (primarySocket == null) {
116          ImmutableList.Builder<LocalService> builder = ImmutableList.builder();
117    
118          for (ServiceRunner runner : runnerProvider.get()) {
119            try {
120              LocalService service = runner.launch();
121              builder.add(service);
122              shutdownRegistry.addAction(service.shutdownCommand);
123            } catch (LaunchException e) {
124              throw new IllegalStateException("Failed to launch " + runner, e);
125            }
126          }
127    
128          List<LocalService> localServices = builder.build();
129          Iterable<LocalService> primaries = Iterables.filter(localServices, IS_PRIMARY);
130          switch (Iterables.size(primaries)) {
131            case 0:
132              primarySocket = Optional.absent();
133              break;
134    
135            case 1:
136              primarySocket = Optional.of(SERVICE_TO_SOCKET.apply(Iterables.getOnlyElement(primaries)));
137              break;
138    
139            default:
140              throw new IllegalArgumentException("More than one primary local service: " + primaries);
141          }
142    
143          Map<String, LocalService> byName;
144          try {
145            byName = Maps.uniqueIndex(
146                Iterables.filter(localServices, Predicates.not(IS_PRIMARY)), GET_NAME);
147          } catch (IllegalArgumentException e) {
148            throw new IllegalArgumentException("Auxiliary services with identical names.", e);
149          }
150    
151          auxiliarySockets = ImmutableMap.copyOf(Maps.transformValues(byName, SERVICE_TO_SOCKET));
152        }
153      }
154    
155      /**
156       * Gets the mapping from auxiliary port name to socket.
157       *
158       * @return Auxiliary port mapping.
159       */
160      public synchronized Map<String, InetSocketAddress> getAuxiliarySockets() {
161        ensureLaunched();
162        return auxiliarySockets;
163      }
164    
165      /**
166       * Gets the optional primary socket address, and returns an unresolved local socket address
167       * representing that port.
168       *
169       * @return Local socket address for the primary port.
170       * @throws IllegalStateException If the primary port was not set.
171       */
172      public synchronized Optional<InetSocketAddress> getPrimarySocket() {
173        ensureLaunched();
174        return primarySocket;
175      }
176    
177      /**
178       * An individual local service.
179       */
180      public static final class LocalService {
181        private final boolean primary;
182        private final Optional<String> name;
183        private final int port;
184        private final Command shutdownCommand;
185    
186        private LocalService(boolean primary, Optional<String> name, int port,
187            Command shutdownCommand) {
188          this.primary = primary;
189          this.name = name;
190          this.port = port;
191          this.shutdownCommand = shutdownCommand;
192        }
193    
194        @Override
195        public String toString() {
196          return new ToStringBuilder(this)
197              .append("primary", primary)
198              .append("name", name)
199              .append("port", port)
200              .toString();
201        }
202    
203        /**
204         * Creates a primary local service.
205         *
206         * @param port Service port.
207         * @param shutdownCommand A command that will shut down the service.
208         * @return A new primary local service.
209         */
210        public static LocalService primaryService(int port, Command shutdownCommand) {
211          return new LocalService(true, Optional.<String>absent(), port, shutdownCommand);
212        }
213    
214        /**
215         * Creates a named auxiliary service.
216         *
217         * @param name Service name.
218         * @param port Service port.
219         * @param shutdownCommand A command that will shut down the service.
220         * @return A new auxiliary local service.
221         */
222        public static LocalService auxiliaryService(String name, int port, Command shutdownCommand) {
223          return new LocalService(false, Optional.of(name), port, shutdownCommand);
224        }
225      }
226    }
