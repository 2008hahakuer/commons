044    /**
045     * Binding module for injections related to the in-process stats system.
046     *
047     * This modules supports two command line arguments:
048     * <ul>
049     *   <li>{@code stat_sampling_interval} - Statistic value sampling interval.
050     *   <li>{@code stat_retention_period} - Time for a stat to be retained in memory before expring.
051     * </ul>
052     *
053     * Bindings required by this module:
054     * <ul>
055     *   <li>{@code @StartupStage ActionRegistry} - Startup action registry.
056     *   <li>{@code @ShutdownStage ActionRegistry} - Shutdown hook registry.
057     *   <li>{@code BuildInfo} - Build information for the application.
058     * </ul>
059     *
060     * @author William Farner
061     */
062    public class StatsModule extends AbstractModule {
063    
064      @CmdLine(name = "stat_sampling_interval", help = "Statistic value sampling interval.")
065      private static final Arg<Amount<Long, Time>> SAMPLING_INTERVAL =
066          Arg.create(Amount.of(1L, Time.SECONDS));
067    
068      @CmdLine(name = "stat_retention_period",
069          help = "Time for a stat to be retained in memory before expiring.")
070      private static final Arg<Amount<Long, Time>> RETENTION_PERIOD =
071          Arg.create(Amount.of(1L, Time.HOURS));
072    
073      @Override
074      protected void configure() {
075        requireBinding(Key.get(ActionRegistry.class, StartupStage.class));
076        requireBinding(Key.get(ActionRegistry.class, ShutdownStage.class));
077        requireBinding(BuildInfo.class);
078    
079        // Bindings for TimeSeriesRepositoryImpl.
080        bind(new TypeLiteral<Amount<Long, Time>>() {})
081            .annotatedWith(Names.named(TimeSeriesRepositoryImpl.SAMPLE_RETENTION_PERIOD))
082            .toInstance(RETENTION_PERIOD.get());
083        bind(new TypeLiteral<Amount<Long, Time>>() {})
084            .annotatedWith(Names.named(TimeSeriesRepositoryImpl.SAMPLE_PERIOD))
085            .toInstance(SAMPLING_INTERVAL.get());
086        bind(TimeSeriesRepository.class).to(TimeSeriesRepositoryImpl.class).in(Singleton.class);
087    
088        requestStaticInjection(Init.class);
089      }
090    
091      static class Init {
092        private static final Logger LOG = Logger.getLogger(Init.class.getName());
093    
094        @Inject
095        private static void startStatsSystem(
096            @StartupStage ActionRegistry startupRegistry,
097            @ShutdownStage final ActionRegistry shutdownRegistry,
098            final BuildInfo buildInfo,
099            final TimeSeriesRepository timeSeriesRepository) {
100          startupRegistry.addAction(new Command() {
101            @Override public void execute() throws RuntimeException {
102              LOG.info("Build information: " + buildInfo.getProperties());
103              for (final BuildInfo.Key key : BuildInfo.Key.values()) {
104                Stats.exportString(new StatImpl<String>(Stats.normalizeName(key.value)) {
105                  @Override public String read() {
106                    return buildInfo.getProperties().getProperty(key.value);
107                  }
108                });
109              }
110    
111              JvmStats.export();
112              timeSeriesRepository.start(shutdownRegistry);
113            }
114          });
115        }
116      }
117    }
