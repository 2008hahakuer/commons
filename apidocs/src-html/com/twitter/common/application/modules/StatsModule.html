018    package com.twitter.common.application.modules;
019    
020    import java.util.logging.Logger;
021    
022    import com.google.inject.AbstractModule;
023    import com.google.inject.Inject;
024    import com.google.inject.Key;
025    import com.google.inject.Singleton;
026    import com.google.inject.TypeLiteral;
027    import com.google.inject.name.Names;
028    
029    import com.twitter.common.application.ShutdownStage;
030    import com.twitter.common.application.StartupStage;
031    import com.twitter.common.base.Command;
032    import com.twitter.common.application.ActionRegistry;
033    import com.twitter.common.quantity.Amount;
034    import com.twitter.common.quantity.Time;
035    import com.twitter.common.stats.JvmStats;
036    import com.twitter.common.stats.StatImpl;
037    import com.twitter.common.stats.Stats;
038    import com.twitter.common.stats.TimeSeriesRepository;
039    import com.twitter.common.stats.TimeSeriesRepositoryImpl;
040    import com.twitter.common.util.BuildInfo;
041    
042    /**
043     * Binding module for injections related to the in-process stats system.
044     *
045     * @author William Farner
046     */
047    public class StatsModule extends AbstractModule {
048    
049      @Override
050      protected void configure() {
051        requireBinding(Key.get(ActionRegistry.class, StartupStage.class));
052        requireBinding(Key.get(ActionRegistry.class, ShutdownStage.class));
053        requireBinding(BuildInfo.class);
054    
055        // Bindings for TimeSeriesRepositoryImpl.
056        bind(new TypeLiteral<Amount<Long, Time>>() {})
057            .annotatedWith(Names.named(TimeSeriesRepositoryImpl.SAMPLE_RETENTION_PERIOD))
058            .toInstance(Amount.of(1L, Time.HOURS));
059        bind(new TypeLiteral<Amount<Long, Time>>() {})
060            .annotatedWith(Names.named(TimeSeriesRepositoryImpl.SAMPLE_PERIOD))
061            .toInstance(Amount.of(1L, Time.SECONDS));
062        bind(TimeSeriesRepository.class).to(TimeSeriesRepositoryImpl.class).in(Singleton.class);
063    
064        requestStaticInjection(Init.class);
065      }
066    
067      static class Init {
068        private static final Logger LOG = Logger.getLogger(Init.class.getName());
069    
070        @Inject
071        private static void startStatsSystem(
072            @StartupStage ActionRegistry startupRegistry,
073            @ShutdownStage final ActionRegistry shutdownRegistry,
074            final BuildInfo buildInfo,
075            final TimeSeriesRepository timeSeriesRepository) {
076          startupRegistry.addAction(new Command() {
077            @Override public void execute() throws RuntimeException {
078              LOG.info("Build information: " + buildInfo.getProperties());
079              for (final BuildInfo.Key key : BuildInfo.Key.values()) {
080                Stats.exportString(new StatImpl<String>(Stats.normalizeName(key.value)) {
081                  @Override public String read() {
082                    return buildInfo.getProperties().getProperty(key.value);
083                  }
084                });
085              }
086    
087              JvmStats.export();
088              timeSeriesRepository.start(shutdownRegistry);
089            }
090          });
091        }
092      }
093    }
