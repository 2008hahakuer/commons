060    /**
061     * Binding module for injections related to the HTTP server and the default set of servlets.
062     *
063     * This module uses a single command line argument 'http_port'.  If unset, the HTTP server will
064     * be started on an ephemeral port.
065     *
066     * The default HTTP server includes several generic servlets that are useful for debugging.
067     *
068     * This class also offers several convenience methods for other modules to register HTTP servlets
069     * which will be included in the HTTP server configuration.
070     *
071     * Bindings provided by this module:
072     * <ul>
073     *   <li>{@code @CacheTemplates boolean} - True if parsed stringtemplates for servlets are cached.
074     * </ul>
075     *
076     * Bindings that may be overridden with an override module:
077     * <ul>
078     *   <li>Abort handler: called when an HTTP GET request is issued to the /abortabortabort HTTP
079     *       servlet.  May be overridden by binding to:
080     *       {@code bind(Runnable.class).annotatedWith(Names.named(AbortHandler.ABORT_HANDLER_KEY))}.
081     *   <li>Quit handler: called when an HTTP GET request is issued to the /quitquitquit HTTP
082     *       servlet.  May be overridden by binding to:
083     *       {@code bind(Runnable.class).annotatedWith(Names.named(QuitHandler.QUIT_HANDLER_KEY))}.
084     *   <li>Health checker: called to determine whether the application is healthy to serve an
085     *       HTTP GET request to /health.  May be overridden by binding to:
086     *       {@code bind(new TypeLiteral<ExceptionalSupplier<Boolean, ?>>() {}).annotatedWith(Names.named(HealthHandler.HEALTH_CHECKER_KEY))}.
087     *   <li>
088     * </ul>
089     *
090     * @author William Farner
091     */
092    public class HttpModule extends AbstractModule {
093    
094      private static final Logger LOG = Logger.getLogger(HttpModule.class.getName());
095    
096      // TODO(William Farner): Consider making this configurable if needed.
097      private static final boolean CACHE_TEMPLATES = true;
098    
099      private static final Runnable DEFAULT_ABORT_HANDLER = new Runnable() {
100          @Override public void run() {
101            LOG.info("ABORTING PROCESS IMMEDIATELY!");
102            System.exit(0);
103          }
104        };
105      private static final Supplier<Boolean> DEFAULT_HEALTH_CHECKER = new Supplier<Boolean>() {
106          @Override public Boolean get() {
107            return Boolean.TRUE;
108          }
109        };
110    
111      @Range(lower=0, upper=65535)
112      @CmdLine(name = "http_port",
113               help = "The port to start an HTTP server on.  Default value will choose a random port.")
114      protected static final Arg<Integer> HTTP_PORT = Arg.create(0);
115    
116      @Override
117      protected void configure() {
118        requireBinding(Injector.class);
119        requireBinding(Key.get(ActionRegistry.class, StartupStage.class));
120        requireBinding(Key.get(ActionRegistry.class, ShutdownStage.class));
121    
122        // Bind the default abort, quit, and health check handlers.
123        bind(Key.get(Runnable.class, Names.named(AbortHandler.ABORT_HANDLER_KEY)))
124            .toInstance(DEFAULT_ABORT_HANDLER);
125        bind(Runnable.class).annotatedWith(Names.named(QuitHandler.QUIT_HANDLER_KEY))
126            .to(DefaultQuitHandler.class);
127        bind(DefaultQuitHandler.class).in(Singleton.class);
128        bind(new TypeLiteral<ExceptionalSupplier<Boolean, ?>>() {})
129            .annotatedWith(Names.named(HealthHandler.HEALTH_CHECKER_KEY))
130            .toInstance(DEFAULT_HEALTH_CHECKER);
131    
132        // Allow template reloading in interactive mode for easy debugging of string templates.
133        bindConstant().annotatedWith(CacheTemplates.class).to(CACHE_TEMPLATES);
134    
135        bind(HttpServerDispatch.class).in(Singleton.class);
136        Registration.registerServlet(binder(), "/abortabortabort", AbortHandler.class, true);
137        Registration.registerServlet(binder(), "/contention", ContentionPrinter.class, false);
138        Registration.registerServlet(binder(), "/graphdata", TimeSeriesDataSource.class, true);
139        Registration.registerServlet(binder(), "/health", HealthHandler.class, true);
140        Registration.registerServlet(binder(), "/logconfig", LogConfig.class, false);
141        Registration.registerServlet(binder(), "/logs", LogPrinter.class, false);
142        Registration.registerServlet(binder(), "/quitquitquit", QuitHandler.class, true);
143        Registration.registerServlet(binder(), "/threads", ThreadStackPrinter.class, false);
144        Registration.registerServlet(binder(), "/vars", VarsHandler.class, false);
145    
146        GraphViewer.registerResources(binder());
147    
148        requestStaticInjection(Init.class);
149      }
150    
151      // TODO(William Farner): Use a global Multibinder for startup and shutdown actions to obviate the need
152      //    for these awkward init classes.
153      public static class Init {
154        private static final Logger LOG = Logger.getLogger(Init.class.getName());
155    
156        @Inject
157        private static void startHttpServer(
158            @StartupStage ActionRegistry startupRegistry,
159            @ShutdownStage ActionRegistry shutdownRegistry,
160            final HttpServerDispatch httpServerDispatch,
161            final Set<HttpServletConfig> httpServlets,
162            final Set<HttpAssetConfig> httpAssets,
163            final Injector injector) {
164    
165          startupRegistry.addAction(new Command() {
166            @Override public void execute() {
167              if (!httpServerDispatch.listen(HTTP_PORT.get())) {
168                LOG.severe("Failed to start HTTP server, all servlets disabled.");
169                return;
170              }
171    
172              for (HttpServletConfig config : httpServlets) {
173                HttpServlet handler = injector.getInstance(config.handlerClass);
174                httpServerDispatch.registerHandler(config.path, handler, config.params, config.silent);
175              }
176    
177              for (HttpAssetConfig config : httpAssets) {
178                httpServerDispatch.registerHandler(config.path, config.handler, null, config.silent);
179              }
180            }
181          });
182    
183          shutdownRegistry.addAction(new Command() {
184            @Override public void execute() {
185              LOG.info("Shutting down embedded http server");
186              httpServerDispatch.stop();
187            }
188          });
189        }
190      }
