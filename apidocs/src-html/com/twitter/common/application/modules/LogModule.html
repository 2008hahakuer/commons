017    package com.twitter.common.application.modules;
018    
019    import java.io.File;
020    import java.util.logging.Logger;
021    
022    import com.google.common.base.Preconditions;
023    import com.google.inject.AbstractModule;
024    import com.google.inject.Inject;
025    import com.google.inject.name.Named;
026    import com.google.inject.name.Names;
027    
028    import com.twitter.common.args.Arg;
029    import com.twitter.common.args.CmdLine;
030    import com.twitter.common.args.constraints.CanRead;
031    import com.twitter.common.args.constraints.Exists;
032    import com.twitter.common.args.constraints.IsDirectory;
033    import com.twitter.common.base.Command;
034    import com.twitter.common.logging.LogUtil;
035    import com.twitter.common.logging.RootLogConfig;
036    import com.twitter.common.net.http.handlers.LogPrinter;
037    import com.twitter.common.stats.StatImpl;
038    import com.twitter.common.stats.Stats;
039    
040    /**
041     * Binding module for logging-related bindings, such as the log directory.
042     *
043     * This module uses a single optional command line argument 'log_dir'.  If unset, the logging
044     * directory will be auto-discovered via:
045     * {@link com.twitter.common.logging.LogUtil#getLogManagerLogDir()}.
046     *
047     * Bindings provided by this module:
048     * <ul>
049     *   <li>{@code @Named(LogPrinter.LOG_DIR_KEY) File} - Log directory.
050     * </ul>
051     *
052     * Default bindings that may be overridden:
053     * <ul>
054     *   <li>Log directory: directory where application logs are written.  May be overridden by binding
055     *       to: {@code bind(File.class).annotatedWith(Names.named(LogPrinter.LOG_DIR_KEY))}.
056     * </ul>
057     *
058     * @author William Farner
059     */
060    public class LogModule extends AbstractModule {
061    
062      private static final Logger LOG = Logger.getLogger(LogModule.class.getName());
063    
064      @Exists
065      @CanRead
066      @IsDirectory
067      @CmdLine(name = "log_dir",
068               help = "The directory where application logs are written.")
069      private static final Arg<File> LOG_DIR = Arg.create(null);
070    
071      // TODO(franco): change this flag's default to true, then remove after enough forewarning.
072      @CmdLine(name = "use_glog",
073               help = "True to use the new glog-based configuration for the root logger.")
074      private static final Arg<Boolean> USE_GLOG = Arg.create(false);
075    
076      @Override
077      protected void configure() {
078        // Bind the default log directory.
079        bind(File.class).annotatedWith(Names.named(LogPrinter.LOG_DIR_KEY)).toInstance(getLogDir());
080    
081        LifecycleModule.bindStartupAction(binder(), ExportLogDir.class);
082        if (USE_GLOG.get()) {
083          RootLogConfig.configureFromFlags();
084        }
085      }
086    
087      private File getLogDir() {
088        File logDir = LOG_DIR.get();
089        if (logDir == null) {
090          logDir = LogUtil.getLogManagerLogDir();
091          LOG.info("From logging properties, parsed log directory " + logDir.getAbsolutePath());
092        }
093        return logDir;
094      }
095    
096      public static final class ExportLogDir implements Command {
097        private final File logDir;
098    
099        @Inject ExportLogDir(@Named(LogPrinter.LOG_DIR_KEY) final File logDir) {
100          this.logDir = Preconditions.checkNotNull(logDir);
101        }
102    
103        @Override public void execute() {
104          Stats.exportStatic(new StatImpl<String>("logging_dir") {
105            @Override public String read() {
106              return logDir.getAbsolutePath();
107            }
108          });
109        }
110      }
111    }
