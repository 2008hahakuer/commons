017    package com.twitter.common.application;
018    
019    import java.util.List;
020    import java.util.logging.Level;
021    import java.util.logging.Logger;
022    
023    import com.google.common.base.Preconditions;
024    import com.google.common.collect.Lists;
025    
026    import com.twitter.common.base.ExceptionalCommand;
027    
028    /**
029     * A {@link ActionRegistry} that provides a {@link #execute()} method that will ensure all
030     * registered actions are executed in the reverse order they were registered.
031     *
032     * @author John Sirois
033     */
034    public class ActionController implements ActionRegistry {
035    
036      private static final Logger LOG = Logger.getLogger(ActionController.class.getName());
037    
038      private final List<ExceptionalCommand<? extends Exception>> actions = Lists.newLinkedList();
039    
040      private boolean completed = false;
041    
042      /**
043       * Registers an action to execute during {@link #execute()}.  It is an error to call this method
044       * after calling {@link #execute()}.
045       *
046       * @param action the action to add to the list of actions to execute during execution
047       */
048      @Override
049      public synchronized <E extends Exception, T extends ExceptionalCommand<E>> void addAction(
050          T action) {
051        Preconditions.checkState(!completed);
052        actions.add(action);
053      }
054    
055      /**
056       * Executes an application lifecycle stage by executing all registered actions.  This method can
057       * be called multiple times but will only execute the registered actions the first time.
058       *
059       * This sends output to System.out because logging is unreliable during JVM shutdown, which
060       * this class may be used for.
061       */
062      public synchronized void execute() {
063        if (!completed) {
064          LOG.info(String.format("Executing %d lifecycle commands.", actions.size()));
065          completed = true;
066          for (ExceptionalCommand<? extends Exception> action : Lists.reverse(actions)) {
067            try {
068              action.execute();
069            } catch (Exception e) {
070              LOG.log(Level.WARNING, "Lifecycle action failed.", e);
071            }
072          }
073        } else {
074          LOG.info("Action controller has already completed, subsequent calls ignored.");
075        }
076      }
077    }
