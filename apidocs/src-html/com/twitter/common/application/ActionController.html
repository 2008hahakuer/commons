018    package com.twitter.common.application;
019    
020    import java.util.List;
021    import java.util.logging.Level;
022    import java.util.logging.Logger;
023    
024    import com.google.common.base.Preconditions;
025    import com.google.common.collect.Lists;
026    
027    import com.twitter.common.base.ExceptionalCommand;
028    
029    /**
030     * A {@link ActionRegistry} that provides a {@link #execute()} method that will ensure all
031     * registered actions are executed in the reverse order they were registered.
032     *
033     * @author John Sirois
034     */
035    public class ActionController implements ActionRegistry {
036    
037      private static final Logger LOG = Logger.getLogger(ActionController.class.getName());
038    
039      private final List<ExceptionalCommand<? extends Exception>> actions = Lists.newLinkedList();
040    
041      private boolean completed;
042    
043      /**
044       * Registers an action to execute during {@link #execute()}.  It is an error to call this method
045       * after calling {@link #execute()}.
046       *
047       * @param action the action to add to the list of actions to execute during execution
048       */
049      @Override
050      public synchronized <E extends Exception, T extends ExceptionalCommand<E>> void addAction(
051          T action) {
052        Preconditions.checkState(!completed);
053        actions.add(action);
054      }
055    
056      /**
057       * Executes an application lifecycle stage by executing all registered actions.  This method can
058       * be called multiple times but will only execute the registered actions the first time.
059       *
060       * This sends output to System.out because logging is unreliable during JVM shutdown, which
061       * this class may be used for.
062       */
063      public synchronized void execute() {
064        if (!completed) {
065          LOG.info(String.format("Executing %d lifecycle commands.", actions.size()));
066          completed = true;
067          for (ExceptionalCommand<? extends Exception> action : Lists.reverse(actions)) {
068            try {
069              action.execute();
070            } catch (Exception e) {
071              LOG.log(Level.WARNING, "Lifecycle action failed.", e);
072            }
073          }
074        }
075      }
076    }
