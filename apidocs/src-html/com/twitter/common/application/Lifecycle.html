017    package com.twitter.common.application;
018    
019    import java.lang.Thread.UncaughtExceptionHandler;
020    import java.util.logging.Logger;
021    
022    import com.google.inject.Inject;
023    import com.google.inject.name.Named;
024    
025    /**
026     * Application lifecycle manager, which coordinates orderly shutdown of an application.  This class
027     * is responsible for executing shutodwn commands, and can also be used to allow threads to await
028     * application shutdown.
029     *
030     * @author William Farner
031     */
032    public class Lifecycle {
033    
034      private static final Logger LOG = Logger.getLogger(Lifecycle.class.getName());
035    
036      // Monitor and state for suspending and terminating execution.
037      private final Object waitMonitor = new Object();
038      private volatile boolean destroyed = false;
039    
040      private final ActionController shutdownController;
041    
042      @Inject
043      public Lifecycle(@ShutdownStage ActionController shutdownController,
044          UncaughtExceptionHandler exceptionHandler) {
045    
046        this.shutdownController = shutdownController;
047        Thread.setDefaultUncaughtExceptionHandler(exceptionHandler);
048      }
049    
050      /**
051       * Checks whether this lifecycle is still considered alive.  The lifecycle is still alive until
052       * {@link #shutdown()} has been called and all of the actions registered with the shutdown
053       * controller have completed.
054       *
055       * @return {@code true} if the lifecycle is alive, {@code false} otherwise.
056       *
057       */
058      public final boolean isAlive() {
059        return !destroyed;
060      }
061    
062      /**
063       * Allows a caller to wait forever; typically used when all work is done in daemon threads.
064       * Will exit on interrupts.
065       */
066      public final void awaitShutdown() {
067        LOG.info("Awaiting shutdown");
068        synchronized (waitMonitor) {
069          while (!destroyed) {
070            try {
071              waitMonitor.wait();
072            } catch (InterruptedException e) {
073              LOG.info("Exiting on interrupt");
074              shutdown();
075              return;
076            }
077          }
078        }
079      }
080    
081      /**
082       * Initiates an orderly shutdown of the lifecycle's registered shutdown hooks.
083       */
084      public final void shutdown() {
085        LOG.info("Shutting down application");
086        shutdownController.execute();
087        destroyed = true;
088        synchronized (waitMonitor) {
089          waitMonitor.notifyAll();
090        }
091      }
092    }
