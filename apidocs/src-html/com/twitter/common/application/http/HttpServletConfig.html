017    package com.twitter.common.application.http;
018    
019    import javax.servlet.http.HttpServlet;
020    
021    import com.google.common.collect.ImmutableMap;
022    
023    import static com.google.common.base.Preconditions.checkNotNull;
024    import static com.twitter.common.base.MorePreconditions.checkNotBlank;
025    
026    /**
027     * An {@link javax.servlet.http.HttpServlet} configuration used to mount HTTP handlers via
028     * {@link Registration#registerServlet(com.google.inject.Binder, HttpServletConfig)}.
029     *
030     * TODO(William Farner): Move this to a more appropriate package after initial AppLauncher check-in.
031     *
032     */
033    public class HttpServletConfig {
034      public final String path;
035      public final Class<? extends HttpServlet> handlerClass;
036      public final ImmutableMap<String, String> params;
037      public final boolean silent;
038    
039      /**
040       * @param path the absolute path to mount the handler on
041       * @param servletClass the type of servlet that will render pages at {@code path}
042       * @param silent whether or not to display a link for this handler on the landing page
043       */
044      public HttpServletConfig(String path, Class<? extends HttpServlet> servletClass,
045          boolean silent) {
046        this(path, servletClass, ImmutableMap.<String, String>of(), silent);
047      }
048    
049      /**
050       * @param path the absolute path to mount the handler on
051       * @param servletClass the type of servlet that will render pages at {@code path}
052       * @param params a map of servlet init parameters to initialize the servlet with
053       * @param silent whether or not to display a link for this handler on the landing page
054       */
055      public HttpServletConfig(String path, Class<? extends HttpServlet> servletClass,
056          ImmutableMap<String, String> params, boolean silent) {
057    
058        this.path = checkNotBlank(path);
059        this.handlerClass = checkNotNull(servletClass);
060        this.params = checkNotNull(params);
061        this.silent = silent;
062      }
063    }
