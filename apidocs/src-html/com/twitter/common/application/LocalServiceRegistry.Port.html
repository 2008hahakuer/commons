017    package com.twitter.common.application;
018    
019    import java.lang.annotation.Retention;
020    import java.lang.annotation.Target;
021    import java.net.InetSocketAddress;
022    import java.net.UnknownHostException;
023    import java.util.Map;
024    import java.util.Set;
025    
026    import javax.annotation.Nullable;
027    
028    import com.google.common.base.Preconditions;
029    import com.google.common.base.Predicate;
030    import com.google.common.base.Predicates;
031    import com.google.common.base.Throwables;
032    import com.google.common.collect.ImmutableMap;
033    import com.google.common.collect.Maps;
034    import com.google.common.collect.Sets;
035    import com.google.inject.BindingAnnotation;
036    import com.google.inject.Inject;
037    
038    import com.twitter.common.base.MorePreconditions;
039    import com.twitter.common.net.InetSocketAddressHelper;
040    
041    import static java.lang.annotation.ElementType.FIELD;
042    import static java.lang.annotation.ElementType.PARAMETER;
043    import static java.lang.annotation.RetentionPolicy.RUNTIME;
044    
045    /**
046     * Registry for services that should be exported from the application.
047     *
048     * Example of announcing and registering a port:
049     * <pre>
050     * class MyLauncher extends RegisteringServiceLauncher<LaunchException> {
051     *   static final String NAME = "my_service_name";
052     *
053     *   public String getPortName() {
054     *     return NAME
055     *   }
056     *
057     *   public boolean isPrimaryService() {
058     *     return false;
059     *   }
060     *
061     *   public Integer get() {
062     *     return launchServiceAndGetPort();
063     *   }
064     * }
065     *
066     * class MyServiceModule extends AbstractModule {
067     *   public void configure() {
068     *     LifeCycleModule.bindServiceLauncher(binder(), MyLauncher.NAME, MyLauncher.class);
069     *   }
070     * }
071     * </pre>
072     *
073     * @author William Farner
074     */
075    public class LocalServiceRegistry {
076    
077      @BindingAnnotation
078      @Target({FIELD, PARAMETER}) @Retention(RUNTIME)
079      public @interface Port {}
080    
081      private final Set<String> registeredNames;
082      private final Map<String, Integer> announcedPorts = Maps.newHashMap();
083      private String primaryPort = null;
084    
085      /**
086       * Creates a new local service registry.
087       *
088       * @param registeredNames Names of ports to register.
089       */
090      @Inject
091      public LocalServiceRegistry(@Port Set<String> registeredNames) {
092        this.registeredNames = Preconditions.checkNotNull(registeredNames);
093      }
094    
095      /**
096       * Announces a port.
097       * This will fail if the port has already been announced, the port was not registered,
098       * or if another port was announced with {@code isPrimary = true}.
099       *
100       * @param portName Name of the port to announce.
101       * @param port Port number.
102       * @param isPrimary Whether this port is the primary service endppoint for the application.
103       */
104      public void announce(String portName, int port, boolean isPrimary) {
105        MorePreconditions.checkNotBlank(portName, "Port name must not be blank");
106        Preconditions.checkState(!isPrimary || (primaryPort == null),
107            String.format("Attempted to announce primary port %s, conflicts with port %s",
108                portName, primaryPort));
109        Preconditions.checkState(registeredNames.contains(portName),
110            "Attempted to announce unregistered port " + portName);
111    
112        Integer collision = announcedPorts.put(portName, port);
113        Preconditions.checkState(collision == null, "Port was announced twice: " + portName);
114    
115        if (isPrimary) {
116          primaryPort = portName;
117        }
118      }
119    
120      private void checkFullyAnnounced() {
121        Set<String> notAnnouncedPorts = Sets.difference(registeredNames, announcedPorts.keySet());
122        Preconditions.checkState(notAnnouncedPorts.isEmpty(),
123            "Attempted to get announced ports, before ports were announced: " + notAnnouncedPorts);
124      }
125    
126      /**
127       * Gets all of the announced ports.
128       * This will fail if not all of the registered ports have been announced.
129       *
130       * @return All announced ports.
131       */
132      public Map<String, Integer> getAllAnnouncedPorts() {
133        checkFullyAnnounced();
134        return ImmutableMap.copyOf(announcedPorts);
135      }
136    
137      private final Predicate<String> isPrimaryPort = new Predicate<String>() {
138        @Override public boolean apply(String portName) {
139          return portName.equals(getPrimaryPort());
140        }
141      };
142    
143      /**
144       * Convenience method to return a map identical to {@link #getAllAnnouncedPorts()} with two
145       * exceptions:
146       * <ul>
147       *   <li>The primary port (if included) will be ommitted from the map.
148       *   <li>Values are unresolved local sockets rather than just port numbers.
149       * </ul>
150       *
151       * @return Auxiliary port mapping.
152       */
153      public Map<String, InetSocketAddress> getAuxiliarySockets() {
154        return ImmutableMap.copyOf(Maps.transformValues(
155            Maps.filterKeys(getAllAnnouncedPorts(), Predicates.not(isPrimaryPort)),
156            InetSocketAddressHelper.INT_TO_INET));
157      }
158    
159      /**
160       * Gets the optionally-set primary port name.
161       *
162       * @return The primary port, or {@code null} if no primary port was announced.
163       */
164      @Nullable
165      public String getPrimaryPort() {
166        checkFullyAnnounced();
167        return primaryPort;
168      }
169    
170      /**
171       * Gets the primary port number, and returns an unresolved local socket address representing that
172       * port.
173       *
174       * @return Local socket address for the primary port.
175       * @throws IllegalStateException If the primary port was not set.
176       */
177      public InetSocketAddress getPrimarySocket() throws IllegalStateException {
178        checkFullyAnnounced();
179        String primaryPort = getPrimaryPort();
180        Preconditions.checkState(primaryPort != null, "The primary port was not set.");
181        try {
182          return InetSocketAddressHelper.getLocalAddress(announcedPorts.get(primaryPort));
183        } catch (UnknownHostException e) {
184          throw Throwables.propagate(e);
185        }
186      }
187    }
