017    package com.twitter.common.application;
018    
019    import java.util.logging.Level;
020    import java.util.logging.Logger;
021    
022    import com.google.common.collect.ImmutableList;
023    import com.google.inject.Guice;
024    import com.google.inject.Inject;
025    import com.google.inject.Injector;
026    import com.google.inject.Module;
027    import com.google.inject.Stage;
028    import com.google.inject.util.Modules;
029    
030    import com.twitter.common.application.modules.AppLauncherModule;
031    import com.twitter.common.application.modules.LifecycleModule;
032    import com.twitter.common.args.Arg;
033    import com.twitter.common.args.ArgScanner;
034    import com.twitter.common.args.ArgScanner.ArgScanException;
035    import com.twitter.common.args.CmdLine;
036    import com.twitter.common.args.constraints.NotNull;
037    
038    /**
039     * An application launcher that sets up a framework for pluggable binding modules.  This class
040     * should be called directly as the main class, with a command line argument {@code -app_class}
041     * which is the canonical class name of the application to execute.
042     *
043     * If your application uses command line arguments all {@link Arg} fields annotated with
044     * {@link CmdLine} will be discovered and command line arguments will be validated against this set,
045     * parsed and applied.
046     *
047     * A bootstrap module will be automatically applied ({@link AppLauncherModule}), which provides
048     * overridable default bindings for things like quit/abort hooks and a health check function.
049     * A {@link LifecycleModule} is also automatically applied to perform startup and shutdown
050     * actions.
051     *
052     * @author William Farner
053     */
054    public final class AppLauncher {
055    
056      private static final Logger LOG = Logger.getLogger(AppLauncher.class.getName());
057    
058      @NotNull
059      @CmdLine(name = "app_class",
060               help = "Fully-qualified name of the application class, which must implement Runnable.")
061      private static final Arg<Class<? extends Application>> APP_CLASS = Arg.create();
062    
063      @CmdLine(name = "guice_stage",
064               help = "Guice development stage to create injector with.")
065      private static final Arg<Stage> GUICE_STAGE = Arg.create(Stage.DEVELOPMENT);
066    
067      @Inject @StartupStage private ActionController startupController;
068      @Inject private Lifecycle lifecycle;
069    
070      private void run(Application application) {
071        configureInjection(application);
072    
073        LOG.info("Executing startup actions.");
074        startupController.execute();
075    
076        try {
077          application.run();
078        } finally {
079          LOG.info("Application run() exited.");
080          lifecycle.shutdown();
081        }
082      }
083    
084      private void configureInjection(Application application) {
085        Iterable<Module> modules = ImmutableList.<Module>builder()
086            .add(new LifecycleModule())
087            .add(new AppLauncherModule())
088            .addAll(application.getModules())
089            .build();
090    
091        Injector injector = Guice.createInjector(GUICE_STAGE.get(),
092            Modules.override(Modules.combine(modules)).with(application.getOverridingModules()));
093        injector.injectMembers(this);
094        injector.injectMembers(application);
095      }
096    
097      public static void main(String[] args) throws IllegalAccessException, InstantiationException {
098        try {
099          ArgScanner.parse(args);
100        } catch (ArgScanException e) {
101          exit("Failed to scan arguments", e);
102        } catch (IllegalArgumentException e) {
103          exit("Failed to apply arguments", e);
104        }
105    
106        new AppLauncher().run(APP_CLASS.get().newInstance());
107      }
108    
109      private static void exit(String message, Exception error) {
110        LOG.log(Level.SEVERE, message, error);
111        System.exit(1);
112      }
113    }
