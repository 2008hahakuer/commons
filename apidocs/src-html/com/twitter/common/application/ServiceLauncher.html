001    package com.twitter.common.application;
002    
003    import com.twitter.common.base.ExceptionalCommand;
004    
005    /**
006     * Local interface to define requirements for a closure that is responsible for launching
007     * a service at startup.
008     *
009     * @param <E> Exception type thrown during service launch.
010     */
011    public interface ServiceLauncher<E extends Exception> extends ExceptionalCommand<E> {
012    
013      /**
014       * Gets the service port name.
015       *
016       * @return Port name.
017       */
018      String getPortName();
019    
020      /**
021       * Checks whether this is the primary service in the application.  There may be at most one
022       * primary service in an application.
023       *
024       * @return {@code true} if this launcher is responsible for reporting the primary application
025       *     service, {@code false} otherwise.
026       */
027      boolean isPrimaryService();
028    }
