001    package com.twitter.common.metrics;
002    
003    import java.util.Collection;
004    
005    import com.google.common.base.Supplier;
006    
007    /**
008     * Convenience functions for working with {@link Gauge}s.
009     */
010    public final class Gauges {
011    
012      private Gauges() {
013        // Utility class.
014      }
015    
016      /**
017       * Creates a supplier that serves as an accessor for gauge values.
018       *
019       * @param gauge Gauge to turn into a supplier.
020       * @return Supplier of values from {@code gauge}.
021       */
022      public static Supplier<Number> asSupplier(final Gauge gauge) {
023        return new Supplier<Number>() {
024          @Override public Number get() {
025            return gauge.read();
026          }
027        };
028      }
029    
030      /**
031       * Registers the size of a collection as a gauge.
032       *
033       * @param registry Registry to register the gauge with.
034       * @param name Name for the gauge.
035       * @param collection Collection to register size of.
036       */
037      public static void registerSize(MetricRegistry registry, String name,
038          final Collection collection) {
039        registry.register(new AbstractGauge<Integer>(name) {
040          @Override public Integer read() {
041            return collection.size();
042          }
043        });
044      }
045    }
