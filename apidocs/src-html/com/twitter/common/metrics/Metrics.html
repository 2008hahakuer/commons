001    package com.twitter.common.metrics;
002    
003    import java.util.Map;
004    import java.util.concurrent.atomic.AtomicLong;
005    
006    import com.google.common.annotations.VisibleForTesting;
007    import com.google.common.collect.ImmutableMap;
008    import com.google.common.collect.Maps;
009    
010    /**
011     * Root metric registry.
012     */
013    public class Metrics implements MetricRegistry, MetricProvider {
014    
015      private static final Metrics ROOT = new Metrics();
016    
017      private final Map<String, Gauge> metrics = Maps.newConcurrentMap();
018    
019      @VisibleForTesting
020      Metrics() {
021        // Package private.
022      }
023    
024      /**
025       * Create a new Metrics detached from the static root.
026       *
027       * @return the detached metric registry.
028       */
029      public static Metrics createDetached() {
030        return new Metrics();
031      }
032    
033      /**
034       * Returns a handle to the root metric registry.
035       *
036       * @return Root metric registry.
037       */
038      public static Metrics root() {
039        return ROOT;
040      }
041    
042      @Override
043      public MetricRegistry scope(String name) {
044        return new ScopedMetrics(name, this);
045      }
046    
047      @Override
048      public <T extends Number> void register(Gauge<T> gauge) {
049        // TODO(wfarner): Define a policy for handling collisions.
050        metrics.put(gauge.getName(), gauge);
051      }
052    
053      @Override
054      public AtomicLong registerLong(String name) {
055        final AtomicLong gauge = new AtomicLong();
056        register(new AbstractGauge<Long>(name) {
057          @Override public Long read() {
058            return gauge.get();
059          }
060        });
061        return gauge;
062      }
063    
064      @Override
065      public Map<String, Number> sample() {
066        ImmutableMap.Builder<String, Number> samples = ImmutableMap.builder();
067        for (Map.Entry<String, Gauge> metric : metrics.entrySet()) {
068          samples.put(metric.getKey(), metric.getValue().read());
069        }
070        return samples.build();
071      }
072    }
