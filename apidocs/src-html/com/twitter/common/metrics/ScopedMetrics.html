001    package com.twitter.common.metrics;
002    
003    import java.util.concurrent.atomic.AtomicLong;
004    
005    import com.google.common.annotations.VisibleForTesting;
006    import com.google.common.base.Preconditions;
007    
008    import com.twitter.common.base.MorePreconditions;
009    
010    /**
011     * A metric registry that is a 'child' of another metric registry.
012     */
013    public class ScopedMetrics implements MetricRegistry {
014    
015      private static final String SCOPE_DELIMITER = ".";
016    
017      private final String name;
018      private final MetricRegistry parentScope;
019    
020      /**
021       * Creates a new scoped metric registry.
022       * When a gauge is registered with this registry, it will be passed directly to the parent,
023       * with the scope name applied to the gauge name.
024       *
025       * @param name Name of this scope.
026       * @param parentScope Parent scope to register gauges with.
027       */
028      @VisibleForTesting
029      ScopedMetrics(String name, MetricRegistry parentScope) {
030        this.name = MorePreconditions.checkNotBlank(name);
031        this.parentScope = Preconditions.checkNotNull(parentScope);
032      }
033    
034      @Override
035      public MetricRegistry scope(String scopeName) {
036        return new ScopedMetrics(scopeName, this);
037      }
038    
039      private String scopeName(String metricName) {
040        return name + SCOPE_DELIMITER + metricName;
041      }
042    
043      @Override
044      public <T extends Number> void register(final Gauge<T> gauge) {
045        final String scopedName = scopeName(gauge.getName());
046        parentScope.register(new AbstractGauge<T>(scopedName) {
047          @Override public T read() {
048            return gauge.read();
049          }
050        });
051      }
052    
053      @Override
054      public AtomicLong registerLong(String gaugeName) {
055        return parentScope.registerLong(scopeName(gaugeName));
056      }
057    }
