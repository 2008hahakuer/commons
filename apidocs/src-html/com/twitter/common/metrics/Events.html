001    package com.twitter.common.metrics;
002    
003    import java.util.concurrent.atomic.AtomicLong;
004    
005    /**
006     * Provides gauge composition to track per-event statistics, where a variable {@code value} is
007     * associated with each event.
008     * This will export a total of five metrics:
009     * <ul>
010     *   <li>value counter</li>
011     *   <li>event counter</li>
012     *   <li>value per-second rate</li>
013     *   <li>event per-second rate</li>
014     *   <li>ratio-of-rates: rate(value) / rate(event)</li>
015     * </ul>
016     *
017     */
018    public class Events {
019    
020      private final AtomicLong totalEvents;
021      private final AtomicLong totalValue;
022    
023      /**
024       * Creates a new event composition using custom event and value metric names.
025       *
026       * @param eventMetricName Event metric name to export.
027       * @param valueMetricName Value metric name to export.
028       * @param registry Registry to associate metrics with.
029       */
030      public Events(String eventMetricName, String valueMetricName,
031          MetricRegistry registry) {
032        totalEvents = registry.registerLong(eventMetricName);
033        totalValue = registry.registerLong(valueMetricName);
034      }
035    
036      /**
037       * Accumulates a value and increments the event counter.
038       *
039       * @param value Value to accumulate.
040       */
041      public synchronized void accumulate(long value) {
042        totalEvents.incrementAndGet();
043        totalValue.addAndGet(value);
044      }
045    }
