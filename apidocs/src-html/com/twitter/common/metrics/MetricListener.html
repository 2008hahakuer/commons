001    package com.twitter.common.metrics;
002    
003    import java.util.Map;
004    
005    /**
006     * A listener that receives updated metric samples.
007     */
008    public interface MetricListener {
009    
010      /**
011       * Notifies the listener of updated samples.
012       *
013       * @param samples Updated samples.  Samples associated with the same metric will use a consistent
014       *     key, and keys may be added or removed over the lifetime of the listener.
015       */
016      void updateStats(Map<String, Number> samples);
017    }
