001    package com.twitter.common.metrics;
002    
003    import java.util.concurrent.atomic.AtomicLong;
004    
005    /**
006     * A registry that maintains a collection of metrics.
007     */
008    public interface MetricRegistry {
009    
010      /**
011       * Returns or creates a sub-scope of this metric registry.
012       *
013       * @param name Name for the sub-scope.
014       * @return A possibly-new metric registry, whose metrics will be 'children' of this scope.
015       */
016      MetricRegistry scope(String name);
017    
018      /**
019       * Registers a new gauge.
020       *
021       * @param gauge Gauge to register.
022       * @param <T> Number type of the gauge's values.
023       */
024      <T extends Number> void register(Gauge<T> gauge);
025    
026      /**
027       * Creates a gauge and returns an {@link AtomicLong} that can be modified to update the value.
028       *
029       * @param name Name to associate with the gauge.
030       * @return Handle to modify the gauge value.
031       */
032      AtomicLong registerLong(String name);
033    }
