001    package com.twitter.common.logging;
002    
003    import java.util.logging.Formatter;
004    import java.util.logging.Level;
005    import java.util.logging.LogRecord;
006    
007    import com.google.common.base.Throwables;
008    import com.google.common.cache.CacheBuilder;
009    import com.google.common.cache.CacheLoader;
010    import com.google.common.cache.LoadingCache;
011    
012    import org.apache.commons.lang.StringUtils;
013    import org.joda.time.DateTimeZone;
014    import org.joda.time.format.DateTimeFormat;
015    import org.joda.time.format.DateTimeFormatter;
016    
017    /**
018     * Log formatter to match the format generated by glog:
019     *
020     * I0218 17:36:47.460234 (source) (message)
021     *
022     * @author William Farner
023     */
024    public class LogFormatter extends Formatter {
025    
026      private static final DateTimeFormatter DATE_TIME_FORMATTER =
027          DateTimeFormat.forPattern("MMdd HH:mm:ss.SSS").withZone(DateTimeZone.UTC);
028    
029      /**
030       * Build a level 'label' by taking the first character of the level name and making uppercase.
031       */
032      private static final LoadingCache<Level, String> LEVEL_LABELS = CacheBuilder.newBuilder().build(
033          new CacheLoader<Level, String>() {
034            public String load(Level level) {
035              return String.valueOf(level.getName().charAt(0)).toUpperCase();
036            }
037          }
038      );
039    
040      private static final int BASE_MESSAGE_LENGTH =
041            1     // Level char.
042            + 4   // Month + day
043            + 1   // space
044            + 12  // Timestamp
045            + 1   // space
046            + 6   // THREAD
047            + 4   // Room for thread ID.
048            + 1;  // space
049    
050      @Override
051      public String format(final LogRecord record) {
052        int messageLength = BASE_MESSAGE_LENGTH
053            + 2  // Colon and space
054            + record.getMessage().length();
055        if (record.getSourceClassName() != null) {
056          messageLength += record.getSourceClassName().length();
057          if (record.getSourceMethodName() != null) {
058            messageLength += 1;  // Period between class and method.
059            messageLength += record.getSourceMethodName().length();
060          }
061        }
062    
063        StringBuilder sb = new StringBuilder(messageLength)
064            .append(LEVEL_LABELS.getUnchecked(record.getLevel()))
065            .append(DATE_TIME_FORMATTER.print(record.getMillis()))
066            .append(" THREAD")
067            .append(record.getThreadID());
068    
069        if (record.getSourceClassName() != null) {
070          sb.append(' ').append(record.getSourceClassName());
071          if (record.getSourceMethodName() != null) {
072            sb.append('.').append(record.getSourceMethodName());
073          }
074        }
075    
076        sb.append(": ").append(formatMessage(record));
077        if (record.getThrown() != null) {
078          sb.append('\n').append(Throwables.getStackTraceAsString(record.getThrown()));
079        }
080    
081        return sb.append('\n').toString();
082      }
083    
084      private static String zeroPad(int value, int width) {
085        return StringUtils.leftPad(String.valueOf(value), width, '0');
086      }
087    }
