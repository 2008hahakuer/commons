018    package com.twitter.common.logging;
019    
020    import com.google.common.annotations.VisibleForTesting;
021    import org.apache.commons.lang.StringUtils;
022    import org.apache.commons.lang.SystemUtils;
023    
024    import java.io.File;
025    import java.util.logging.LogManager;
026    import java.util.logging.Logger;
027    
028    /**
029     * Logging utility functions.
030     *
031     * @author William Farner
032     */
033    public class LogUtil {
034    
035      private static final Logger LOG = Logger.getLogger(LogUtil.class.getName());
036    
037      private static final String LOG_MANAGER_FILE_PROP = "java.util.logging.FileHandler.pattern";
038    
039      @VisibleForTesting
040      static final File DEFAULT_LOG_DIR = new File("/var/log");
041    
042      /**
043       * Gets the log directory as configured with the log manager.  This will attempt to expand any
044       * directory wildcards that are included in log file property.
045       *
046       * @return The configured log directory.
047       */
048      public static File getLogManagerLogDir() {
049        return getLogManagerLogDir(LogManager.getLogManager().getProperty(LOG_MANAGER_FILE_PROP));
050      }
051    
052      /**
053       * Gets the log directory as specified in a log file pattern.  This will attempt to expand any
054       * directory wildcards that are included in log file property.
055       *
056       * @param logFilePattern The pattern to extract the log directory from.
057       * @return The configured log directory.
058       */
059      public static File getLogManagerLogDir(String logFilePattern) {
060        if (StringUtils.isEmpty(logFilePattern)) {
061          LOG.warning("Could not find log dir in logging property " + LOG_MANAGER_FILE_PROP
062                      + ", reading from " + DEFAULT_LOG_DIR);
063          return DEFAULT_LOG_DIR;
064        }
065    
066        String logDir = expandWildcard(logFilePattern, "%t", SystemUtils.JAVA_IO_TMPDIR);
067        logDir = expandWildcard(logDir, "%h", SystemUtils.USER_HOME);
068        File parent = new File(logDir).getParentFile();
069        return parent == null ? new File(".") : parent;
070      }
071    
072      /**
073       * Expands a directory path wildcard within a file pattern string.
074       * Correctly handles cases where the replacement string does and does not contain a trailing
075       * slash.
076       *
077       * @param pattern File pattern string, which may or may not contain a wildcard.
078       * @param dirWildcard Wildcard string to expand.
079       * @param replacement Path component to expand wildcard to.
080       * @return {@code pattern} with all instances of {@code dirWildcard} replaced with
081       *     {@code replacement}.
082       */
083      private static String expandWildcard(String pattern, String dirWildcard, String replacement) {
084        String replace = dirWildcard;
085        if (replacement.charAt(replacement.length() - 1) == '/') {
086          replace += '/';
087        }
088        return pattern.replaceAll(replace, replacement);
089      }
090    
091      private LogUtil() {
092        // Utility class.
093      }
094    }
