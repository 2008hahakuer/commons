018    package com.twitter.common.util.testing;
019    
020    import com.google.common.base.Preconditions;
021    import com.twitter.common.quantity.Amount;
022    import com.twitter.common.quantity.Time;
023    import com.twitter.common.util.Clock;
024    
025    import java.util.concurrent.TimeUnit;
026    
027    /**
028     * A clock for use in testing with a configurable value for {@link #nowMillis()}.
029     *
030     * @author John Sirois
031     */
032    public class FakeClock implements Clock {
033      private long nowNanos;
034    
035      /**
036       * Sets what {@link #nowMillis()} will return until this method is called again with a new value
037       * for {@code now}.
038       *
039       * @param nowMillis the current time in milliseconds
040       */
041      public void setNowMillis(long nowMillis) {
042        Preconditions.checkArgument(nowMillis >= 0);
043        this.nowNanos = TimeUnit.MILLISECONDS.toNanos(nowMillis);
044      }
045    
046      /**
047       * Advances the current time by {@code millis} milliseconds.  Time can be retarded by passing a
048       * negative value.
049       *
050       * @param period the amount of time to advance the current time by
051       */
052      public void advance(Amount<Long, Time> period) {
053        Preconditions.checkNotNull(period);
054        long newNanos = nowNanos + period.as(Time.NANOSECONDS);
055        Preconditions.checkArgument(newNanos >= 0,
056            "invalid period %s - would move current time to a negative value: %sns", period, newNanos);
057        nowNanos = newNanos;
058      }
059    
060      @Override
061      public long nowMillis() {
062        return TimeUnit.NANOSECONDS.toMillis(nowNanos);
063      }
064    
065      @Override
066      public long nowNanos() {
067        return nowNanos;
068      }
069    
070      /**
071       * Waits in fake time, immediately returning in real time; however a check of {@link #nowMillis}
072       * after this method completes will consistently reveal that {@code millis} did in fact pass while
073       * waiting.
074       *
075       * @param millis the amount of time to wait in milliseconds
076       */
077      @Override
078      public void waitFor(long millis) {
079        advance(Amount.of(millis, Time.MILLISECONDS));
080      }
081    }
