017    package com.twitter.common.util;
018    
019    import java.util.logging.Level;
020    import java.util.logging.Logger;
021    
022    import com.google.common.base.Throwables;
023    
024    import org.apache.commons.lang.builder.ToStringBuilder;
025    
026    import com.twitter.common.base.ExceptionalCommand;
027    import com.twitter.common.quantity.Amount;
028    import com.twitter.common.quantity.Time;
029    
030    import static com.google.common.base.Preconditions.checkArgument;
031    import static com.google.common.base.Preconditions.checkNotNull;
032    
033    /**
034     * A runnable task that is retried in a user-configurable fashion.
035     *
036     * @param <E> The type of exception that the ExceptionalCommand throws.
037     *
038     * @author Utkarsh Srivastava
039     */
040    public class RetryingRunnable<E extends Exception> implements Runnable {
041      private final String name;
042      private final int tryNum;
043      private final int numTries;
044      private final Amount<Long, Time> retryDelay;
045      private final ExceptionalCommand<E> task;
046      private final CommandExecutor commandExecutor;
047      private final Class<E> exceptionClass;
048    
049      private static final Logger LOG = Logger.getLogger(RetryingRunnable.class.getName());
050    
051      /**
052       * Create a Task with name {@code name} that executes at most {@code  numTries}
053       * in case of failure with an interval of {@code retryDelay} between attempts.
054       *
055       * @param name Human readable name for this task.
056       * @param task the task to execute.
057       * @param exceptionClass class of the exception thrown by the task.
058       * @param numTries the total number of times to try.
059       * @param retryDelay the delay between successive tries.
060       * @param commandExecutor Executor to resubmit retries to.
061       * @param tryNum the seq number of this try.
062       */
063      public RetryingRunnable(
064          String name,
065          ExceptionalCommand<E> task,
066          Class<E> exceptionClass,
067          int numTries,
068          Amount<Long, Time> retryDelay,
069          CommandExecutor commandExecutor,
070          int tryNum) {
071    
072        this.name = checkNotNull(name);
073        this.task = checkNotNull(task);
074        this.exceptionClass = checkNotNull(exceptionClass);
075        this.retryDelay = checkNotNull(retryDelay);
076        this.commandExecutor = checkNotNull(commandExecutor);
077        checkArgument(numTries > 0);
078        this.tryNum = tryNum;
079        this.numTries = numTries;
080      }
081    
082      /**
083       * Create a Task with name {@code name} that executes at most {@code numTries}
084       * in case of failure with an interval of {@code retryDelay} between attempts
085       *  and sets tryNum to be the first (=1).
086       *
087       * @param name Human readable name for this task.
088       * @param task the task to execute.
089       * @param exceptionClass class of the exception thrown by the task.
090       * @param numTries the total number of times to try.
091       * @param retryDelay the delay between successive tries.
092       * @param commandExecutor Executor to resubmit retries to.
093       */
094      public RetryingRunnable(
095          String name,
096          ExceptionalCommand<E> task,
097          Class<E> exceptionClass,
098          int numTries,
099          Amount<Long, Time> retryDelay,
100          CommandExecutor commandExecutor) {
101    
102        this(name, task, exceptionClass, numTries, retryDelay, commandExecutor, /*tryNum=*/ 1);
103      }
104    
105      @Override
106      public void run() {
107        try {
108          task.execute();
109        } catch (Exception e) {
110          if (e.getClass().isAssignableFrom(exceptionClass)) {
111            if (tryNum < numTries) {
112              commandExecutor.execute(name, task, exceptionClass, numTries - 1, retryDelay);
113            } else {
114              LOG.log(Level.INFO, "Giving up on task: " + name + " "
115                  + "after " + "trying " + numTries + " times" + ".", e);
116            }
117          } else {
118            LOG.log(Level.INFO, "Giving up on task: " + name + " after trying "
119                + numTries + " times. " + "due to unhandled exception ", e);
120            throw Throwables.propagate(e);
121          }
122        }
123      }
124    
125      @Override
126      public String toString() {
127        return new ToStringBuilder(this)
128            .append("name", name)
129            .append("tryNum", tryNum)
130            .append("numTries", numTries)
131            .append("retryDelay", retryDelay)
132            .toString();
133      }
134    }
