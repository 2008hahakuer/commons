017    package com.twitter.common.util;
018    
019    import java.util.logging.Level;
020    import java.util.logging.Logger;
021    
022    import com.google.common.base.Preconditions;
023    import com.google.common.base.Throwables;
024    
025    import org.apache.commons.lang.builder.ToStringBuilder;
026    
027    import com.twitter.common.base.ExceptionalCommand;
028    import com.twitter.common.quantity.Amount;
029    import com.twitter.common.quantity.Time;
030    
031    /**
032     * A runnable task that is retried in a user-configurable fashion.
033     *
034     * @param <E> The type of exception that the ExceptionalCommand throws.
035     *
036     * @author Utkarsh Srivastava
037     */
038    public class RetryingRunnable<E extends Exception> implements Runnable {
039      private final String name;
040      private final int tryNum;
041      private final int numTries;
042      private final Amount<Long, Time> retryDelay;
043      private final ExceptionalCommand<E> task;
044      private final CommandExecutor commandExecutor;
045      private final Class<E> exceptionClass;
046    
047      private static final Logger LOG = Logger.getLogger(RetryingRunnable.class.getName());
048    
049      /**
050       * Create a Task with name {@code name} that executes at most {@code  numTries}
051       * in case of failure with an interval of {@code retryDelay} between attempts.
052       *
053       * @param name Human readable name for this task.
054       * @param task the task to execute.
055       * @param exceptionClass class of the exception thrown by the task.
056       * @param numTries the total number of times to try.
057       * @param retryDelay the delay between successive tries.
058       * @param tryNum the seq number of this try.
059       * @param name
060       */
061      public RetryingRunnable(String name, ExceptionalCommand<E> task, Class<E> exceptionClass,
062          int numTries, Amount<Long, Time> retryDelay, CommandExecutor commandExecutor, int tryNum) {
063        this.name = Preconditions.checkNotNull(name);
064        this.task = Preconditions.checkNotNull(task);
065        this.exceptionClass = Preconditions.checkNotNull(exceptionClass);
066        this.retryDelay = Preconditions.checkNotNull(retryDelay);
067        this.commandExecutor = Preconditions.checkNotNull(commandExecutor);
068        Preconditions.checkArgument(numTries > 0);
069        this.tryNum = tryNum;
070        this.numTries = numTries;
071      }
072    
073      /**
074       * Create a Task with name {@code name} that executes at most {@code numTries}
075       * in case of failure with an interval of {@code retryDelay} between attempts
076       *  and sets tryNum to be the first (=1).
077       *
078       * @param name Human readable name for this task.
079       * @param task the task to execute.
080       * @param exceptionClass class of the exception thrown by the task.
081       * @param numTries the total number of times to try.
082       * @param retryDelay the delay between successive tries.
083       */
084      public RetryingRunnable(String name, ExceptionalCommand<E> task,
085          Class<E> exceptionClass, int numTries, Amount<Long, Time> retryDelay,
086          CommandExecutor commandExecutor) {
087        this(name, task, exceptionClass, numTries, retryDelay, commandExecutor,
088        /*tryNum=*/ 1);
089      }
090    
091      @Override
092      public void run() {
093        try {
094          task.execute();
095        } catch (Exception e) {
096          if (e.getClass().isAssignableFrom(exceptionClass)) {
097            if (tryNum < numTries) {
098              commandExecutor.execute(name, task, exceptionClass, numTries - 1, retryDelay);
099            } else {
100              LOG.log(Level.INFO, "Giving up on task: " + name + " "
101                  + "after " + "trying " + numTries + " times" + ".", e);
102            }
103          } else {
104            LOG.log(Level.INFO, "Giving up on task: " + name + " after trying "
105                + numTries + " times. " + "due to unhandled exception ", e);
106            throw Throwables.propagate(e);
107          }
108        }
109      }
110    
111      @Override
112      public String toString() {
113        return new ToStringBuilder(this)
114            .append("name", name)
115            .append("tryNum", tryNum)
116            .append("numTries", numTries)
117            .append("retryDelay", retryDelay)
118            .toString();
119      }
120    }
