017    package com.twitter.common.util;
018    
019    import com.twitter.common.base.ExceptionalCommand;
020    import com.twitter.common.quantity.Amount;
021    import com.twitter.common.quantity.Time;
022    
023    /**
024     * Asynchronous executor of enqueued tasks in a rate limited manner.
025     *
026     * @author Srinivasan Rajagopal
027     */
028    public interface CommandExecutor {
029    
030      /**
031       * Enqueue a task to be executed with retry semantics defined.
032       *
033       * @param name Human readable name for this task.
034       * @param task task to execute.
035       * @param exceptionClass Concrete exception type.
036       * @param maxTries num of tries in case of failure.
037       * @param retryDelay interval between retries in case of failure.
038       */
039      <E extends Exception> void execute(
040          String name,
041          ExceptionalCommand<E> task,
042          Class<E> exceptionClass,
043          int maxTries,
044          Amount<Long, Time> retryDelay);
045    }
