017    package com.twitter.common.util;
018    
019    import com.google.common.annotations.VisibleForTesting;
020    import com.google.common.base.Preconditions;
021    import com.twitter.common.base.ExceptionalSupplier;
022    import com.twitter.common.quantity.Amount;
023    import com.twitter.common.quantity.Time;
024    
025    import java.util.logging.Logger;
026    
027    /**
028     * A utility for dealing with backoffs of retryable actions.
029     *
030     * <p>TODO(John Sirois): investigate synergies with BackoffDecider & also consider options for a timeout
031     * and/or a maximum number of retries.
032     *
033     * @author John Sirois
034     */
035    public class BackoffHelper {
036      private static final Logger LOG = Logger.getLogger(BackoffHelper.class.getName());
037    
038      private static final Amount<Long,Time> DEFAULT_INITIAL_BACKOFF = Amount.of(1L, Time.SECONDS);
039      private static final Amount<Long,Time> DEFAULT_MAX_BACKOFF = Amount.of(1L, Time.MINUTES);
040    
041      private final Clock clock;
042      private final BackoffStrategy backoffStrategy;
043    
044      /**
045       * Creates a new BackoffHelper that uses truncated binary backoff starting at a 1 second backoff
046       * and maxing out at a 1 minute backoff.
047       */
048      public BackoffHelper() {
049        this(DEFAULT_INITIAL_BACKOFF, DEFAULT_MAX_BACKOFF);
050      }
051    
052      /**
053       * Creates a new BackoffHelper that uses truncated binary backoff starting at the given
054       * {@code initialBackoff} and maxing out at the given {@code maxBackoff}.
055       *
056       * @param initialBackoff the initial amount of time to back off
057       * @param maxBackoff the maximum amount of time to back off
058       */
059      public BackoffHelper(Amount<Long, Time> initialBackoff, Amount<Long, Time> maxBackoff) {
060        this(new TruncatedBinaryBackoff(initialBackoff, maxBackoff));
061      }
062    
063      /**
064       * Creates a BackoffHelper that uses the given {@code backoffStrategy} to calculate backoffs
065       * between retries.
066       *
067       * @param backoffStrategy the backoff strategy to use
068       */
069      public BackoffHelper(BackoffStrategy backoffStrategy) {
070        this(Clock.SYSTEM_CLOCK, backoffStrategy);
071      }
072    
073      @VisibleForTesting BackoffHelper(Clock clock, BackoffStrategy backoffStrategy) {
074        this.clock = Preconditions.checkNotNull(clock);
075        this.backoffStrategy = Preconditions.checkNotNull(backoffStrategy);
076      }
077    
078      /**
079       * Executes the given task using the configured backoff strategy until the task succeeds as
080       * indicated by returning {@code true}.
081       *
082       * @param task the retryable task to execute until success
083       * @throws InterruptedException if interrupted while waiting for the task to execute successfully
084       * @throws E if the task throws
085       */
086      public <E extends Exception> void doUntilSuccess(final ExceptionalSupplier<Boolean, E> task)
087          throws InterruptedException, E {
088        doUntilResult(new ExceptionalSupplier<Boolean, E>() {
089          @Override public Boolean get() throws E {
090            Boolean result = task.get();
091            return Boolean.TRUE.equals(result) ? result : null;
092          }
093        });
094      }
095    
096      /**
097       * Executes the given task using the configured backoff strategy until the task succeeds as
098       * indicated by returning a non-null value.
099       *
100       * @param task the retryable task to execute until success
101       * @return the result of the successfully executed task
102       * @throws InterruptedException if interrupted while waiting for the task to execute successfully
103       * @throws E if the task throws
104       */
105      public <T, E extends Exception> T doUntilResult(ExceptionalSupplier<T, E> task)
106          throws InterruptedException, E {
107        T result = task.get(); // give an immediate try
108        return (result != null) ? result : retryWork(task);
109      }
110    
111      private <T, E extends Exception> T retryWork(ExceptionalSupplier<T, E> work)
112          throws E, InterruptedException {
113        long currentBackoffMs = 0;
114        while (true) {
115          currentBackoffMs = backoffStrategy.calculateBackoffMs(currentBackoffMs);
116          LOG.fine("Operation failed, backing off for " + currentBackoffMs + "ms");
117          clock.waitFor(currentBackoffMs);
118    
119          T result = work.get();
120          if (result != null) {
121            return result;
122          }
123        }
124      }
125    }
