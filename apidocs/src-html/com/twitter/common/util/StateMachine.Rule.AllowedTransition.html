047    /**
048     * Represents a state machine that is not necessarily a Finite State Machine.
049     * The caller may configure the state machine to permit only known state transitions, or to only
050     * disallow known state transitions (and permit unknown transitions).
051     *
052     * @param <T> THe type of objects that the caller uses to represent states.
053     *
054     * TODO(William Farner): Consider merging the stats-tracking ala PipelineStats into this.
055     *
056     * @author William Farner
057     */
058    public class StateMachine<T> {
059      private static final Logger LOG = Logger.getLogger(StateMachine.class.getName());
060    
061      private final String name;
062    
063      // Stores mapping from states to the states that the machine is allowed to transition into.
064      private final Multimap<T, T> stateTransitions;
065    
066      private final Closure<Transition<T>> transitionCallback;
067      private final boolean throwOnBadTransition;
068    
069      private volatile T currentState;
070      private final Lock readLock;
071      private final Lock writeLock;
072    
073    
074      private StateMachine(String name,
075          T initialState,
076          Multimap<T, T> stateTransitions,
077          Closure<Transition<T>> transitionCallback,
078          boolean throwOnBadTransition) {
079        this.name = name;
080        this.currentState = initialState;
081        this.stateTransitions = stateTransitions;
082        this.transitionCallback = transitionCallback;
083        this.throwOnBadTransition = throwOnBadTransition;
084    
085        ReadWriteLock stateLock = new ReentrantReadWriteLock(true /* fair */);
086        readLock = stateLock.readLock();
087        writeLock = stateLock.writeLock();
088      }
089    
090      /**
091       * Gets the name of this state machine.
092       *
093       * @return The state machine name.
094       */
095      public String getName() {
096        return name;
097      }
098    
099      /**
100       * Fetches the state that the machine is currently in.
101       *
102       * @return Current state.
103       */
104      public T getState() {
105        return currentState;
106      }
107    
108      /**
109       * Checks that the current state is the {@code expectedState} and throws if it is not.
110       *
111       * @param expectedState The expected state
112       * @throws IllegalStateException if the current state is not the {@code expectedState}.
113       */
114      public void checkState(T expectedState) {
115        checkState(ImmutableSet.of(expectedState));
116      }
117    
118      /**
119       * Checks that the current state is one of the {@code allowedStates} and throws if it is not.
120       *
121       * @param allowedStates The allowed states.
122       * @throws IllegalStateException if the current state is not the {@code expectedState}.
123       */
124      public void checkState(Set<T> allowedStates) {
125        checkNotNull(allowedStates);
126        checkArgument(!allowedStates.isEmpty(), "At least one possible state must be provided.");
127    
128        readLock.lock();
129        try {
130          if (!allowedStates.contains(currentState)) {
131            throw new IllegalStateException(
132                String.format("In state %s, expected to be in %s.", allowedStates, currentState));
133          }
134        } finally {
135          readLock.unlock();
136        }
137      }
138    
139      /**
140       * Executes the supplied {@code work} if the state machine is in the {@code expectedState},
141       * postponing any concurrently requested {@link #transition(Object)} until after the execution of
142       * the work.
143       *
144       * @param expectedState The expected state the work should be performed in.
145       * @param work The work to perform in the {@code expectedState}.
146       * @param <O> The type returned by the unit of work.
147       * @param <E> The type of exception that may be thrown by the unit of work.
148       * @return The result of the unit of work if the current state is the {@code expectedState}.
149       * @throws IllegalStateException if the current state is not the {@code expectedState}.
150       * @throws E if the unit of work throws.
151       */
152      public <O, E extends Exception> O doInState(T expectedState, ExceptionalSupplier<O, E> work)
153          throws E {
154    
155        checkNotNull(expectedState);
156        checkNotNull(work);
157    
158        readLock.lock();
159        try {
160          checkState(expectedState);
161          return work.get();
162        } finally {
163          readLock.unlock();
164        }
165      }
166    
167      /**
168       * Transitions the machine into state {@code nextState}.
169       *
170       * @param nextState The state to move into.
171       * @throws IllegalStateTransitionException If the state transition is not allowed.
172       * @return {@code true} if the transition was allowed, {@code false} otherwise.
173       */
174      public boolean transition(T nextState) throws IllegalStateTransitionException {
175        boolean transitionAllowed = false;
176    
177        T currentCopy = currentState;
178    
179        writeLock.lock();
180        try {
181          if (stateTransitions.containsEntry(currentState, nextState)) {
182            currentState = nextState;
183            transitionAllowed = true;
184          } else if (throwOnBadTransition) {
185            throw new IllegalStateTransitionException(
186                String.format("State transition from %s to %s is not allowed.", currentState,
187                    nextState));
188          }
189        } finally {
190          writeLock.unlock();
191        }
192    
193        transitionCallback.execute(new Transition<T>(currentCopy, nextState, transitionAllowed));
194        return transitionAllowed;
195      }
196    
197      public static class IllegalStateTransitionException extends IllegalStateException {
198        public IllegalStateTransitionException(String msg) {
199          super(msg);
200        }
201      }

204       * Convenience method to create a builder object.
205       *
206       * @param <T> Type of builder to create.
207       * @param name Name of the state machine to create a builder for.
208       * @return New builder.

210      public static <T> Builder<T> builder(String name) {
211        return new Builder<T>(name);

214      /**
215       * A state and its allowed transitions (if any) and (optional) callback.
216       *
217       * @param <T> State type.
218       */
219      public static class Rule<T> {
220        private final T from;
221        private final Set<T> to;
222        private final Closure<Transition<T>> callback;
223    
224        private Rule(T from) {
225          this(from, ImmutableSet.<T>of());
226        }
227    
228        private Rule(T from, Set<T> to) {
229          this(from, to, Closures.<Transition<T>>noop());
230        }
231    
232        private Rule(T from, Set<T> to, Closure<Transition<T>> callback) {
233          this.from = checkNotNull(from);
234          this.to = checkNotNull(to);
235          this.callback = checkNotNull(callback);
236        }
237    
238        /**
239         * Associates a callback to be triggered after any attempt to transition from this state is
240         * made.
241         *
242         * @param callback Callback to signal.
243         * @return A new rule that is identical to this rule, but with the provided
244         *     callback
245         */
246        public Rule<T> withCallback(Closure<Transition<T>> callback) {
247          return new Rule<T>(from, to, callback);
248        }
249    
250        /**
251         * A helper class when building a transition rule, to define the allowed transitions.
252         *
253         * @param <T> State type.
254         */
255        public static class AllowedTransition<T> {
256          private final Rule<T> rule;
257    
258          private AllowedTransition(Rule<T> rule) {
259            this.rule = rule;
260          }
261    
262          /**
263           * Associates a single allowed transition with this state.
264           *
265           * @param state Allowed transition state.
266           * @return A new rule that identical to the original, but only allowing a transition to the
267           *     provided state.
268           */
269          public Rule<T> to(T state) {
270            return new Rule<T>(rule.from, ImmutableSet.<T>of(state), rule.callback);
271          }
272    
273          /**
274           * Associates multiple transitions with this state.
275           *
276           * @param state An allowed transition state.
277           * @param additionalStates Additional states that may be transitioned to.
278           * @return A new rule that identical to the original, but only allowing a transition to the
279           *     provided states.
280           */
281          public Rule<T> to(T state, T... additionalStates) {
282            return new Rule<T>(rule.from, ImmutableSet.copyOf(Lists.asList(state, additionalStates)));
283          }
284    
285          /**
286           * Allows no transitions to be performed from this state.
287           *
288           * @return The original rule.
289           */
290          public Rule<T> noTransitions() {
291            return rule;
292          }
293        }
294    
295        /**
296         * Creates a new transition rule.
297         *
298         * @param state State to create and associate transitions with.
299         * @param <T> State type.
300         * @return A new transition rule builder.
301         */
302        public static <T> AllowedTransition<T> from(T state) {
303          return new AllowedTransition<T>(new Rule<T>(state));
304        }
305      }

307      /**
308       * Builder to create a state machine.
309       *
310       * @param <T>
311       */
312      public static class Builder<T> {
313        private final String name;
314        private T initialState;
315        private final Multimap<T, T> stateTransitions = HashMultimap.create();
316        private final List<Closure<Transition<T>>> transitionCallbacks = Lists.newArrayList();
317        private boolean throwOnBadTransition = true;
318    
319        public Builder(String name) {
320          this.name = checkNotBlank(name);
321        }
322    
323        /**
324         * Sets the initial state for the state machine.
325         *
326         * @param state Initial state.
327         * @return A reference to the builder.
328         */
329        public Builder<T> initialState(T state) {
330          checkNotNull(state);
331          initialState = state;
332          return this;
333        }
334    
335        /**
336         * Adds a state and its allowed transitions.
337         *
338         * @param rule The state and transition rule to add.
339         * @return A reference to the builder.
340         */
341        public Builder<T> addState(Rule<T> rule) {
342          return addState(rule.callback, rule.from, rule.to);
343        }
344    
345        /**
346         * Adds a state and its allowed transitions.
347         * At least one transition state must be added, it is not necessary to explicitly add states
348         * that have no allowed transitions (terminal states).
349         *
350         * @param callback Callback to notify of any transition attempted from the state.
351         * @param state State to add.
352         * @param transitionStates Allowed transitions from {@code state}.
353         * @return A reference to the builder.
354         */
355        public Builder<T> addState(Closure<Transition<T>> callback, T state,
356            Set<T> transitionStates) {
357          checkNotNull(callback);
358          checkNotNull(state);
359    
360          Preconditions.checkArgument(Iterables.all(transitionStates, Predicates.notNull()));
361    
362          stateTransitions.putAll(state, transitionStates);
363    
364          @SuppressWarnings("unchecked")
365          Predicate<Transition<T>> filter = Transition.from(state);
366          onTransition(filter, callback);
367          return this;
368        }
369    
370        /**
371         * Varargs version of {@link #addState(com.twitter.common.base.Closure, Object, java.util.Set)}.
372         *
373         * @param callback Callback to notify of any transition attempted from the state.
374         * @param state State to add.
375         * @param transitionStates Allowed transitions from {@code state}.
376         * @return A reference to the builder.
377         */
378        public Builder<T> addState(Closure<Transition<T>> callback, T state,
379            T... transitionStates) {
380          Set<T> states = ImmutableSet.copyOf(transitionStates);
381          Preconditions.checkArgument(Iterables.all(states, Predicates.notNull()));
382    
383          return addState(callback, state, states);
384        }
385    
386        /**
387         * Adds a state and its allowed transitions.
388         * At least one transition state must be added, it is not necessary to explicitly add states
389         * that have no allowed transitions (terminal states).
390         *
391         * @param state State to add.
392         * @param transitionStates Allowed transitions from {@code state}.
393         * @return A reference to the builder.
394         */
395        public Builder<T> addState(T state, T... transitionStates) {
396          return addState(Closures.<Transition<T>>noop(), state, transitionStates);
397        }
398    
399        private void onTransition(Predicate<Transition<T>> transitionFilter,
400            Closure<Transition<T>> handler) {
401          onAnyTransition(Closures.filter(transitionFilter, handler));
402        }
403    
404        /**
405         * Adds a callback to be executed for every state transition, including invalid transitions
406         * that are attempted.
407         *
408         * @param handler Callback to notify of transition attempts.
409         * @return A reference to the builder.
410         */
411        public Builder<T> onAnyTransition(Closure<Transition<T>> handler) {
412          transitionCallbacks.add(handler);
413          return this;
414        }
415    
416        /**
417         * Adds a log message for every state transition that is attempted.
418         *
419         * @return A reference to the builder.
420         */
421        public Builder<T> logTransitions() {
422          return onAnyTransition(new Closure<Transition<T>>() {
423            @Override public void execute(Transition<T> transition) {
424              LOG.info(name + " state machine transition " + transition);
425            }
426          });
427        }
428    
429        /**
430         * Allows the caller to specify whether {@link IllegalStateTransitionException} should be thrown
431         * when a bad state transition is attempted (the default behavior).
432         *
433         * @param throwOnBadTransition Whether an exception should be thrown when a bad state transition
434         *     is attempted.
435         * @return A reference to the builder.
436         */
437        public Builder<T> throwOnBadTransition(boolean throwOnBadTransition) {
438          this.throwOnBadTransition = throwOnBadTransition;
439          return this;
440        }
441    
442        /**
443         * Builds the state machine.
444         *
445         * @return A reference to the prepared state machine.
446         */
447        public StateMachine<T> build() {
448          Preconditions.checkState(initialState != null, "Initial state must be specified.");
449          checkArgument(!stateTransitions.isEmpty(), "No state transitions were specified.");
450          return new StateMachine<T>(name,
451              initialState,
452              stateTransitions,
453              Closures.combine(transitionCallbacks),
454              throwOnBadTransition);
455        }
456      }

458      /**
459       * Representation of a state transition.
460       *
461       * @param <T> State type.
462       */
463      public static class Transition<T> {
464        private final T from;
465        private final T to;
466        private final boolean allowed;
467    
468        public Transition(T from, T to, boolean allowed) {
469          this.from = checkNotNull(from);
470          this.to = checkNotNull(to);
471          this.allowed = allowed;
472        }
473    
474        private static <T> Function<Transition<T>, T> from() {
475          return new Function<Transition<T>, T>() {
476            @Override public T apply(Transition<T> transition) {
477              return transition.from;
478            }
479          };
480        }
481    
482        private static <T> Function<Transition<T>, T> to() {
483          return new Function<Transition<T>, T>() {
484            @Override public T apply(Transition<T> transition) {
485              return transition.to;
486            }
487          };
488        }
489    
490        private static <T> Predicate<Transition<T>> oneSideFilter(
491            Function<Transition<T>, T> extractor, final T... states) {
492          checkArgument(Iterables.all(Arrays.asList(states), Predicates.notNull()));
493    
494          return Predicates.compose(Predicates.in(ImmutableSet.copyOf(states)), extractor);
495        }
496    
497        /**
498         * Creates a predicate that returns {@code true} for transitions from the given states.
499         *
500         * @param states States to filter on.
501         * @param <T> State type.
502         * @return A from-state filter.
503         */
504        public static <T> Predicate<Transition<T>> from(final T... states) {
505          return oneSideFilter(Transition.<T>from(), states);
506        }
507    
508        /**
509         * Creates a predicate that returns {@code true} for transitions to the given states.
510         *
511         * @param states States to filter on.
512         * @param <T> State type.
513         * @return A to-state filter.
514         */
515        public static <T> Predicate<Transition<T>> to(final T... states) {
516          return oneSideFilter(Transition.<T>to(), states);
517        }
518    
519        /**
520         * Creates a predicate that returns {@code true} for a specific state transition.
521         *
522         * @param from From state.
523         * @param to To state.
524         * @param <T> State type.
525         * @return A state transition filter.
526         */
527        public static <T> Predicate<Transition<T>> transition(final T from, final T to) {
528          @SuppressWarnings("unchecked")
529          Predicate<Transition<T>> fromFilter = from(from);
530          @SuppressWarnings("unchecked")
531          Predicate<Transition<T>> toFilter = to(to);
532          return Predicates.and(fromFilter, toFilter);
533        }
534    
535        public T getFrom() {
536          return from;
537        }
538    
539        public T getTo() {
540          return to;
541        }
542    
543        public boolean isAllowed() {
544          return allowed;
545        }
546    
547        /**
548         * Checks whether this transition represents a state change, which means that the 'to' state is
549         * not equal to the 'from' state, and the transition is allowed.
550         *
551         * @return {@code true} if the state was changed, {@code false} otherwise.
552         */
553        public boolean isValidStateChange() {
554          return isAllowed() && !from.equals(to);
555        }
556    
557        @Override
558        public boolean equals(Object o) {
559          if (!(o instanceof Transition)) {
560            return false;
561          }
562    
563          if (o == this) {
564            return true;
565          }
566    
567          Transition other = (Transition) o;
568          return from.equals(other.from) && to.equals(other.to);
569        }
570    
571        @Override
572        public int hashCode() {
573          return new HashCodeBuilder()
574              .append(from)
575              .append(to)
576              .toHashCode();
577        }
578    
579        @Override
580        public String toString() {
581          String str = from.toString() + " -> " + to.toString();
582          if (!isAllowed()) {
583            str += " (not allowed)";
584          }
585          return str;
586        }
587      }
