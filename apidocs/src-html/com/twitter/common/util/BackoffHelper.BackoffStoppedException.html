017    package com.twitter.common.util;
018    
019    import com.google.common.annotations.VisibleForTesting;
020    import com.google.common.base.Preconditions;
021    import com.twitter.common.base.ExceptionalSupplier;
022    import com.twitter.common.quantity.Amount;
023    import com.twitter.common.quantity.Time;
024    
025    import java.util.logging.Logger;
026    
027    /**
028     * A utility for dealing with backoffs of retryable actions.
029     *
030     * <p>TODO(John Sirois): investigate synergies with BackoffDecider.
031     *
032     * @author John Sirois
033     */
034    public class BackoffHelper {
035      private static final Logger LOG = Logger.getLogger(BackoffHelper.class.getName());
036    
037      private static final Amount<Long,Time> DEFAULT_INITIAL_BACKOFF = Amount.of(1L, Time.SECONDS);
038      private static final Amount<Long,Time> DEFAULT_MAX_BACKOFF = Amount.of(1L, Time.MINUTES);
039    
040      private final Clock clock;
041      private final BackoffStrategy backoffStrategy;
042    
043      /**
044       * Creates a new BackoffHelper that uses truncated binary backoff starting at a 1 second backoff
045       * and maxing out at a 1 minute backoff.
046       */
047      public BackoffHelper() {
048        this(DEFAULT_INITIAL_BACKOFF, DEFAULT_MAX_BACKOFF);
049      }
050    
051      /**
052       * Creates a new BackoffHelper that uses truncated binary backoff starting at the given
053       * {@code initialBackoff} and maxing out at the given {@code maxBackoff}.
054       *
055       * @param initialBackoff the initial amount of time to back off
056       * @param maxBackoff the maximum amount of time to back off
057       */
058      public BackoffHelper(Amount<Long, Time> initialBackoff, Amount<Long, Time> maxBackoff) {
059        this(new TruncatedBinaryBackoff(initialBackoff, maxBackoff));
060      }
061    
062      /**
063       * Creates a new BackoffHelper that uses truncated binary backoff starting at the given
064       * {@code initialBackoff} and maxing out at the given {@code maxBackoff}. This will either:
065       * <ul>
066       *   <li>{@code stopAtMax == true} : throw {@code BackoffExpiredException} when maxBackoff is
067       *   reached</li>
068       *   <li>{@code stopAtMax == false} : continue backing off with maxBackoff</li>
069       * </ul>
070       *
071       * @param initialBackoff the initial amount of time to back off
072       * @param maxBackoff the maximum amount of time to back off
073       * @param stopAtMax if true, this will throw {@code BackoffStoppedException} when the max backoff is
074       * reached
075       */
076      public BackoffHelper(Amount<Long, Time> initialBackoff, Amount<Long, Time> maxBackoff,
077          boolean stopAtMax) {
078        this(new TruncatedBinaryBackoff(initialBackoff, maxBackoff, stopAtMax));
079      }
080    
081      /**
082       * Creates a BackoffHelper that uses the given {@code backoffStrategy} to calculate backoffs
083       * between retries.
084       *
085       * @param backoffStrategy the backoff strategy to use
086       */
087      public BackoffHelper(BackoffStrategy backoffStrategy) {
088        this(Clock.SYSTEM_CLOCK, backoffStrategy);
089      }
090    
091      @VisibleForTesting BackoffHelper(Clock clock, BackoffStrategy backoffStrategy) {
092        this.clock = Preconditions.checkNotNull(clock);
093        this.backoffStrategy = Preconditions.checkNotNull(backoffStrategy);
094      }
095    
096      /**
097       * Executes the given task using the configured backoff strategy until the task succeeds as
098       * indicated by returning {@code true}.
099       *
100       * @param task the retryable task to execute until success
101       * @throws InterruptedException if interrupted while waiting for the task to execute successfully
102       * @throws BackoffStoppedException if the backoff stopped unsuccessfully
103       * @throws E if the task throws
104       */
105      public <E extends Exception> void doUntilSuccess(final ExceptionalSupplier<Boolean, E> task)
106          throws InterruptedException, BackoffStoppedException, E {
107        doUntilResult(new ExceptionalSupplier<Boolean, E>() {
108          @Override public Boolean get() throws E {
109            Boolean result = task.get();
110            return Boolean.TRUE.equals(result) ? result : null;
111          }
112        });
113      }
114    
115      /**
116       * Executes the given task using the configured backoff strategy until the task succeeds as
117       * indicated by returning a non-null value.
118       *
119       * @param task the retryable task to execute until success
120       * @return the result of the successfully executed task
121       * @throws InterruptedException if interrupted while waiting for the task to execute successfully
122       * @throws BackoffStoppedException if the backoff stopped unsuccessfully
123       * @throws E if the task throws
124       */
125      public <T, E extends Exception> T doUntilResult(ExceptionalSupplier<T, E> task)
126          throws InterruptedException, BackoffStoppedException, E {
127        T result = task.get(); // give an immediate try
128        return (result != null) ? result : retryWork(task);
129      }
130    
131      private <T, E extends Exception> T retryWork(ExceptionalSupplier<T, E> work)
132          throws E, InterruptedException, BackoffStoppedException {
133        long currentBackoffMs = 0;
134        while (backoffStrategy.shouldContinue()) {
135          currentBackoffMs = backoffStrategy.calculateBackoffMs(currentBackoffMs);
136          LOG.fine("Operation failed, backing off for " + currentBackoffMs + "ms");
137          clock.waitFor(currentBackoffMs);
138    
139          T result = work.get();
140          if (result != null) {
141            return result;
142          }
143        }
144        throw new BackoffStoppedException(String.format("Backoff stopped without succeeding."));
145      }
146    
147      /**
148       * Occurs after the backoff strategy should stop.
149       */
150      public static class BackoffStoppedException extends RuntimeException {
151        public BackoffStoppedException(String msg) {
152          super(msg);
153        }
154      }
155    }
