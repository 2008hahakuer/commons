017    package com.twitter.common.util;
018    
019    import java.util.concurrent.BlockingQueue;
020    import java.util.concurrent.ScheduledExecutorService;
021    import java.util.concurrent.TimeUnit;
022    import java.util.logging.Level;
023    import java.util.logging.Logger;
024    
025    import com.google.common.base.Preconditions;
026    
027    import com.twitter.common.base.ExceptionalCommand;
028    import com.twitter.common.quantity.Amount;
029    import com.twitter.common.quantity.Time;
030    
031    /**
032     * CommandExecutor that invokes {@code queueDrainer} with a best-effort
033     * mechanism to execute with a fixed interval between requests of {@code
034     * intervalBetweenRequests}.
035     *
036     * @author Srinivasan Rajagopal
037     */
038    public class RateLimitedCommandExecutor implements CommandExecutor {
039    
040      private static final Logger LOG = Logger.getLogger(RateLimitedCommandExecutor.class.getName());
041    
042      final private BlockingQueue<RetryingRunnable> blockingQueue;
043    
044      /**
045       * Create a CommandExecutor that executes enquequed tasks in the task
046       * executor with specified interval between executions.
047       *
048       * @param taskExecutor executor for periodic execution of enqueued tasks.
049       * @param intervalBetweenRequests interval between requests to rate limit
050       * request rate.
051       */
052      public RateLimitedCommandExecutor(ScheduledExecutorService taskExecutor,
053          Amount<Long, Time> intervalBetweenRequests, Runnable queueDrainer,
054          BlockingQueue<RetryingRunnable> blockingQueue) {
055        Preconditions.checkNotNull(taskExecutor);
056        Preconditions.checkNotNull(intervalBetweenRequests);
057        Preconditions.checkArgument(intervalBetweenRequests.as(Time.MILLISECONDS) > 0);
058        Preconditions.checkNotNull(queueDrainer);
059        this.blockingQueue = Preconditions.checkNotNull(blockingQueue);
060        taskExecutor.scheduleWithFixedDelay(
061            getSafeRunner(queueDrainer),
062            0,
063            intervalBetweenRequests.as(Time.MILLISECONDS),
064            TimeUnit.MILLISECONDS);
065      }
066    
067      private static Runnable getSafeRunner(final Runnable runnable) {
068        return new Runnable() {
069          @Override
070          public void run() {
071            try {
072              runnable.run();
073            } catch (RuntimeException t) {
074              LOG.log(Level.INFO, " error processing task " + runnable);
075            }
076          }
077        };
078      }
079    
080      @Override
081      public void execute(String name, ExceptionalCommand task,
082          Class exceptionClass, int numTries, Amount<Long, Time> retryDelay) {
083        blockingQueue.add(new RetryingRunnable(name, task, exceptionClass,
084            numTries, retryDelay, this));
085      }
086    }
