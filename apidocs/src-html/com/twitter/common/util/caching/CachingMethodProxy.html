019

021    import com.google.common.base.Preconditions;
022    import com.google.common.base.Predicate;
023    import com.google.common.base.Predicates;
024    import com.google.common.collect.ImmutableMap;
025    import com.google.common.collect.Maps;
026    
027    import java.lang.reflect.InvocationHandler;
028    import java.lang.reflect.InvocationTargetException;
029    import java.lang.reflect.Method;
030    import java.lang.reflect.Proxy;
031    import java.util.Arrays;

035    /**
036     * A proxy class that handles caching of return values for method calls to a wrapped object.
037     *
038     * Example usage:
039     *
040     *   Foo uncached = new Foo();
041     *   CachingMethodProxy<Foo> methodProxy = CachingMethodProxy.proxyFor(uncached, Foo.class);
042     *   Foo foo = methodProxy.getCachingProxy();
043     *   methodProxy.cache(foo.doBar(), lruCache1)
044     *              .cache(foo.doBaz(), lruCache2)
045     *              .prepare();
046     *
047     * @author William Farner
048     */
049    public class CachingMethodProxy<T> {
050    
051      // Dummy return values to return when in recording state.
052      private static final Map<Class<?>, Object> EMPTY_RETURN_VALUES =
053          ImmutableMap.<Class<?>, Object>builder()
054          .put(Boolean.TYPE, Boolean.FALSE)
055          .put(Byte.TYPE, Byte.valueOf((byte) 0))
056          .put(Short.TYPE, Short.valueOf((short) 0))
057          .put(Character.TYPE, Character.valueOf((char)0))
058          .put(Integer.TYPE, Integer.valueOf(0))
059          .put(Long.TYPE, Long.valueOf(0))
060          .put(Float.TYPE, Float.valueOf(0))
061          .put(Double.TYPE, Double.valueOf(0))
062          .build();
063      private static final Map<Class<?>, Class<?>> AUTO_BOXING_MAP =
064          ImmutableMap.<Class<?>, Class<?>>builder()
065          .put(Boolean.TYPE, Boolean.class)
066          .put(Byte.TYPE, Byte.class)
067          .put(Short.TYPE, Short.class)
068          .put(Character.TYPE, Character.class)
069          .put(Integer.TYPE, Integer.class)
070          .put(Long.TYPE, Long.class)
071          .put(Float.TYPE, Float.class)
072          .put(Double.TYPE, Double.class)
073          .build();
074    
075      // The uncached resource, whose method calls are deemed to be expensive and cacheable.
076      private final T uncached;
077    
078      // The methods that are cached, and the caches themselves.
079      private final Map<Method, MethodCache> methodCaches = Maps.newHashMap();
080      private final Class<T> type;
081    
082      private Method lastMethodCall = null;
083      private boolean recordMode = true;
084    
085      /**
086       * Creates a new caching method proxy that will wrap an object and cache for the provided methods.
087       *
088       * @param uncached The uncached object that will be reverted to when a cache entry is not present.
089       */
090      private CachingMethodProxy(T uncached, Class<T> type) {
091        this.uncached = Preconditions.checkNotNull(uncached);
092        this.type = Preconditions.checkNotNull(type);
093        Preconditions.checkArgument(type.isInterface(), "The proxied type must be an interface.");
094      }
095    
096      private static Object invokeMethod(Object subject, Method method, Object[] args)
097          throws Throwable {
098        try {
099          return method.invoke(subject, args);
100        } catch (IllegalAccessException e) {
101          throw new RuntimeException("Cannot access " + subject.getClass() + "." + method, e);
102        } catch (InvocationTargetException e) {
103          throw e.getCause();
104        }
105      }
106    
107      /**
108       * A cached method and its caching control structures.
109       *
110       * @param <K> Cache key type.
111       * @param <V> Cache value type, expected to match the return type of the method.
112       */
113      private static class MethodCache<K, V> {
114        private final Method method;
115        private final Cache<K, V> cache;
116        private final Function<Object[], K> keyBuilder;
117        private final Predicate<V> entryFilter;
118    
119        MethodCache(Method method, Cache<K, V> cache, Function<Object[], K> keyBuilder,
120            Predicate<V> entryFilter) {
121          this.method = method;
122          this.cache = cache;
123          this.keyBuilder = keyBuilder;
124          this.entryFilter = entryFilter;
125        }
126    
127        V doInvoke(Object uncached, Object[] args) throws Throwable {
128          K key = keyBuilder.apply(args);
129    
130          V cachedValue = cache.get(key);
131    
132          if (cachedValue != null) return cachedValue;
133    
134          Object fetched = invokeMethod(uncached, method, args);
135    
136          if (fetched == null) return null;
137    
138          @SuppressWarnings("unchecked")
139          V typedValue = (V) fetched;
140    
141          if (entryFilter.apply(typedValue)) cache.put(key, typedValue);
142    
143          return typedValue;
144        }
145      }
146    
147      /**
148       * Creates a new builder for the given type.
149       *
150       * @param uncached The uncached object that should be insulated by caching.
151       * @param type The interface that a proxy should be created for.
152       * @param <T> Type parameter to the proxied class.
153       * @return A new builder.
154       */
155      public static <T> CachingMethodProxy<T> proxyFor(T uncached, Class<T> type) {
156        return new CachingMethodProxy<T>(uncached, type);
157      }
158    
159      @SuppressWarnings("unchecked")
160      public T getCachingProxy() {
161        return (T) Proxy.newProxyInstance(type.getClassLoader(), new Class[] { type },
162            new InvocationHandler() {
163              @Override public Object invoke(Object proxy, Method method, Object[] args)
164                  throws Throwable {
165                return doInvoke(method, args);
166              }
167            });
168      }
169    
170      private Object doInvoke(Method method, Object[] args) throws Throwable {
171        return recordMode ? recordCall(method) : cacheRequest(method, args);
172      }
173    
174      private Object recordCall(Method method) {
175        Preconditions.checkArgument(method.getReturnType() != Void.TYPE,
176            "Void return methods cannot be cached: " + method);
177        Preconditions.checkArgument(method.getParameterTypes().length > 0,
178            "Methods with zero arguments cannot be cached: " + method);
179        Preconditions.checkState(lastMethodCall == null,
180            "No cache instructions provided for call to: " + lastMethodCall);
181    
182        lastMethodCall = method;
183    
184        Class<?> returnType = method.getReturnType();
185        return returnType.isPrimitive() ? EMPTY_RETURN_VALUES.get(returnType) : null;
186      }
187    
188      private Object cacheRequest(Method method, Object[] args) throws Throwable {
189        MethodCache cache = methodCaches.get(method);
190    
191        // Check if we are caching for this method.
192        if (cache == null) return invokeMethod(uncached, method, args);
193    
194        return cache.doInvoke(uncached, args);
195      }
196    
197      /**
198       * Instructs the proxy that cache setup is complete, and the proxy instance should begin caching
199       * and delegating uncached calls.  After this is called, any subsequent calls to any of the
200       * cache setup methods will result in an {@link IllegalStateException}.
201       */
202      public void prepare() {
203        Preconditions.checkState(!methodCaches.isEmpty(), "At least one method must be cached.");
204        Preconditions.checkState(recordMode, "prepare() may only be invoked once.");
205    
206        recordMode = false;
207      }
208    
209      public <V> CachingMethodProxy<T> cache(V value, Cache<List, V> cache) {
210        return cache(value, cache, Predicates.<V>alwaysTrue());
211      }
212    
213      public <V> CachingMethodProxy<T> cache(V value, Cache<List, V> cache,
214          Predicate<V> valueFilter) {
215        return cache(value, cache, DEFAULT_KEY_BUILDER, valueFilter);
216      }
217    
218      public <K, V> CachingMethodProxy<T> cache(V value, Cache<K, V> cache,
219          Function<Object[], K> keyBuilder) {
220        // Get the last method call and declare it the cached method.
221        return cache(value, cache, keyBuilder, Predicates.<V>alwaysTrue());
222      }
223    
224      public <K, V> CachingMethodProxy<T> cache(V value, Cache<K, V> cache,
225          Function<Object[], K> keyBuilder, Predicate<V> valueFilter) {
226        Preconditions.checkNotNull(cache);
227        Preconditions.checkNotNull(keyBuilder);
228        Preconditions.checkNotNull(valueFilter);
229    
230        Preconditions.checkState(recordMode, "Cache setup is not allowed after prepare() is called.");
231    
232        // Get the last method call and declare it the cached method.
233        Preconditions.checkState(lastMethodCall != null, "No method call captured to be cached.");
234    
235        Class<?> returnType = lastMethodCall.getReturnType();
236    
237        Preconditions.checkArgument(returnType != Void.TYPE,
238            "Cannot cache results from void method: " + lastMethodCall);
239    
240        if (returnType.isPrimitive()) {
241          // If a primitive type is returned, we need to make sure that the cache holds the boxed
242          // type for the primitive.
243          returnType = AUTO_BOXING_MAP.get(returnType);
244        }
245    
246        // TODO(William Farner): Figure out a simple way to make this possible.  Right now, since the proxy
247        //    objects return null, we get a null here and can't check the type.
248        //Preconditions.checkArgument(value.getClass() == returnType,
249        //    String.format("Cache value type '%s' does not match method return type '%s'",
250        //        value.getClass(), lastMethodCall.getReturnType()));
251    
252        methodCaches.put(lastMethodCall, new MethodCache<K, V>(lastMethodCall, cache, keyBuilder,
253            valueFilter));
254    
255        lastMethodCall = null;
256    
257        return this;
258      }
259    
260      private static final Function<Object[], List> DEFAULT_KEY_BUILDER =
261          new Function<Object[], List>() {
262            @Override public List apply(Object[] args) {
263              return Arrays.asList(args);
264            }
265          };
266    }
