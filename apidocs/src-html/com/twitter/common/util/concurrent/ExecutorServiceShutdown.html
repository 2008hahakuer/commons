017    package com.twitter.common.util.concurrent;
018    
019    import java.util.concurrent.ExecutorService;
020    import java.util.concurrent.TimeUnit;
021    import java.util.logging.Logger;
022    
023    import com.google.common.base.Preconditions;
024    
025    import com.twitter.common.base.Command;
026    import com.twitter.common.quantity.Amount;
027    import com.twitter.common.quantity.Time;
028    
029    /**
030     * An implementation of the graceful shutdown sequence recommended by {@link ExecutorService}.
031     *
032     * @author John Sirois
033     */
034    public class ExecutorServiceShutdown implements Command {
035      private static final Logger LOG = Logger.getLogger(ExecutorServiceShutdown.class.getName());
036    
037      private final ExecutorService executor;
038      private final Amount<Long, Time> gracePeriod;
039    
040      /**
041       * Creates a new {@code ExecutorServiceShutdown} command that will try to gracefully shut down the
042       * given {@code executor} when executed.  If the supplied grace period is less than or equal to
043       * zero the executor service will be asked to shut down but no waiting will be done after these
044       * requests.
045       *
046       * @param executor The executor service this command should shut down when executed.
047       * @param gracePeriod The maximum time to wait after a shutdown request before continuing to the
048       *     next shutdown phase.
049       */
050      public ExecutorServiceShutdown(ExecutorService executor, Amount<Long, Time> gracePeriod) {
051        this.executor = Preconditions.checkNotNull(executor);
052        this.gracePeriod = Preconditions.checkNotNull(gracePeriod);
053      }
054    
055      @Override
056      public void execute() {
057        executor.shutdown(); // Disable new tasks from being submitted.
058        try {
059           // Wait a while for existing tasks to terminate.
060          if (!executor.awaitTermination(gracePeriod.as(Time.MILLISECONDS), TimeUnit.MILLISECONDS)) {
061            executor.shutdownNow(); // Cancel currently executing tasks.
062            // Wait a while for tasks to respond to being cancelled.
063            if (!executor.awaitTermination(gracePeriod.as(Time.MILLISECONDS), TimeUnit.MILLISECONDS)) {
064              LOG.warning("Pool did not terminate");
065            }
066          }
067        } catch (InterruptedException ie) {
068          // (Re-)Cancel if current thread also interrupted.
069          executor.shutdownNow();
070          // Preserve interrupt status.
071          Thread.currentThread().interrupt();
072        }
073      }
074    }
