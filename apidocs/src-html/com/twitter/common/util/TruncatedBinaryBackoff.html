017    package com.twitter.common.util;
018    
019    import com.google.common.base.Preconditions;
020    import com.twitter.common.quantity.Amount;
021    import com.twitter.common.quantity.Time;
022    
023    /**
024     * A BackoffStrategy that implements truncated binary exponential backoff.
025     */
026    public class TruncatedBinaryBackoff implements BackoffStrategy {
027      private final long initialBackoffMs;
028      private final long maxBackoffIntervalMs;
029    
030      /**
031       * Creates a new TruncatedBinaryBackoff that will start by backing off for {@code initialBackoff}
032       * and then backoff of twice as long each time its called until reaching the {@code maxBackoff} at
033       * which point it will always wait for that amount of time.
034       *
035       * @param initialBackoff the intial amount of time to backoff
036       * @param maxBackoff the maximum amount of time to backoff
037       */
038      public TruncatedBinaryBackoff(Amount<Long, Time> initialBackoff,
039          Amount<Long, Time> maxBackoff) {
040        Preconditions.checkNotNull(initialBackoff);
041        Preconditions.checkNotNull(maxBackoff);
042        Preconditions.checkArgument(initialBackoff.getValue() > 0);
043        Preconditions.checkArgument(maxBackoff.compareTo(initialBackoff) >= 0);
044        initialBackoffMs = initialBackoff.as(Time.MILLISECONDS);
045        maxBackoffIntervalMs = maxBackoff.as(Time.MILLISECONDS);
046      }
047    
048      @Override
049      public long calculateBackoffMs(long lastBackoffMs) {
050        Preconditions.checkArgument(lastBackoffMs >= 0);
051        return (lastBackoffMs == 0) ? initialBackoffMs
052            : Math.min(maxBackoffIntervalMs, lastBackoffMs * 2);
053      }
054    }
