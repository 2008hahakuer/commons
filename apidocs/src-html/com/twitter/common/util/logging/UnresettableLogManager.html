017    package com.twitter.common.util.logging;
018    
019    import java.util.logging.LogManager;
020    
021    /**
022     * A LogManager which by default ignores calls to {@link #reset()}.  This is useful to avoid missing
023     * log statements that occur during vm shutdown.  The standard LogManager installs a
024     * {@link Runtime#addShutdownHook(Thread) shutdown hook} that disables logging and this subclass
025     * nullifies that shutdown hook by disabling any reset of the LogManager by default.
026     *
027     * @author John Sirois
028     */
029    public class UnresettableLogManager extends LogManager {
030    
031      /**
032       * The system property that controls which LogManager the java.util.logging subsystem should load.
033       */
034      public static final String LOGGING_MANAGER = "java.util.logging.manager";
035    
036      /**
037       * A system property which can be used to control an {@code UnresettableLogManager}'s behavior.
038       * If the UnresettableLogManager is installed, but an application still wants
039       * {@link LogManager#reset()} behavior, they can set this property to "false".
040       */
041      private static final String LOGGING_MANAGER_IGNORERESET = "java.util.logging.manager.ignorereset";
042    
043      @Override
044      public void reset() throws SecurityException {
045        if (Boolean.parseBoolean(System.getProperty(LOGGING_MANAGER_IGNORERESET, "true"))) {
046          System.err.println("UnresettableLogManager is ignoring a reset() request.");
047        } else {
048          super.reset();
049        }
050      }
051    }
