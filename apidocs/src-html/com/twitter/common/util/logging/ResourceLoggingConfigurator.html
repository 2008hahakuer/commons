017    package com.twitter.common.util.logging;
018    
019    import com.google.common.base.Preconditions;
020    
021    import java.io.IOException;
022    import java.io.InputStream;
023    import java.util.logging.LogManager;
024    
025    /**
026     * A custom java.util.logging configuration class that loads the logging configuration from a
027     * properties file resource (as opposed to a file as natively supported by LogManager via
028     * java.util.logging.config.file).  By default this configurator will look for the resource at
029     * /logging.properties but the resource path can be overridden by setting the system property with
030     * key {@link #LOGGING_PROPERTIES_RESOURCE_PATH java.util.logging.config.resource}.  To install this
031     * configurator you must specify the following system property:
032     * java.util.logging.config.class=com.twitter.common.util.logging.ResourceLoggingConfigurator
033     *
034     * @author John Sirois
035     */
036    public class ResourceLoggingConfigurator {
037    
038      /**
039       * A system property that controls where ResourceLoggingConfigurator looks for the logging
040       * configuration on the process classpath.
041       */
042      public static final String LOGGING_PROPERTIES_RESOURCE_PATH = "java.util.logging.config.resource";
043    
044      public ResourceLoggingConfigurator() throws IOException {
045        String loggingPropertiesResourcePath =
046            System.getProperty(LOGGING_PROPERTIES_RESOURCE_PATH, "/logging.properties");
047        InputStream loggingConfig = getClass().getResourceAsStream(loggingPropertiesResourcePath);
048        Preconditions.checkNotNull(loggingConfig,
049            "Could not locate logging config file at resource path: %s", loggingPropertiesResourcePath);
050        LogManager.getLogManager().readConfiguration(loggingConfig);
051      }
052    }
