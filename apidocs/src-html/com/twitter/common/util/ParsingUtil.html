017    package com.twitter.common.util;
018    
019    import com.google.common.base.Preconditions;
020    
021    import com.twitter.common.collections.Pair;
022    
023    /**
024     * Common methods for parsing configs.
025     *
026     * @author John Sirois
027     */
028    public class ParsingUtil {
029      /**
030       * Parses a string as a range between one integer and another.  The integers must be separated by
031       * a hypen character (space padding is acceptable).  Additionally, the first integer
032       * (left-hand side) must be less than or equal to the second (right-hand side).
033       *
034       * @param rangeString The string to parse as an integer range.
035       * @return A pair of the parsed integers.
036       */
037      public static Pair<Integer, Integer> parseRange(String rangeString) {
038        if (rangeString == null) return null;
039    
040        String[] startEnd = rangeString.split("-");
041        Preconditions.checkState(
042            startEnd.length == 2, "Shard range format: start-end (e.g. 1-4)");
043        int start;
044        int end;
045        try {
046          start = Integer.parseInt(startEnd[0].trim());
047          end = Integer.parseInt(startEnd[1].trim());
048        } catch (NumberFormatException e) {
049          throw new IllegalArgumentException("Failed to parse shard range.", e);
050        }
051    
052        Preconditions.checkState(
053            start <= end, "The left-hand side of a shard range must be <= the right-hand side.");
054        return Pair.of(start, end);
055      }
056    }
