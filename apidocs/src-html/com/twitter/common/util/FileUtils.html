018    package com.twitter.common.util;
019    
020    import com.google.common.io.Files;
021    
022    import java.io.File;
023    
024    /**
025     * Utilities for working with Files
026     *
027     * @author Florian Leibert
028     */
029    public class FileUtils {
030    
031      public static boolean forceDeletePath(File path) {
032        if (path == null) {
033          return false;
034        }
035        if (path.exists()) {
036          File[] files = path.listFiles();
037          for (File file : files) {
038            if (file.isDirectory()) {
039              forceDeletePath(file);
040            } else {
041              file.delete ();
042            }
043          }
044        }
045        return (path.delete());
046      }
047    }
