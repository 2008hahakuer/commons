017    package com.twitter.common.testing;
018    
019    import com.google.common.base.Preconditions;
020    import com.google.common.testing.TearDown;
021    import com.google.common.testing.TearDownAccepter;
022    
023    import com.twitter.common.application.ShutdownRegistry;
024    import com.twitter.common.base.ExceptionalCommand;
025    
026    /**
027     * An action registry suitable for use as a shutdownRegistry in tests that extend
028     * {@link com.google.common.testing.junit4.TearDownTestCase}.
029     *
030     * @author John Sirois
031     */
032    public class TearDownRegistry implements ShutdownRegistry {
033      private final TearDownAccepter tearDownAccepter;
034    
035      /**
036       * Creates a new tear down registry that delegates execution of shutdown actions to a
037       * {@code tearDownAccepter}.
038       *
039       * @param tearDownAccepter A tear down accepter that will be used to register shutdown actions
040       *     with.
041       */
042      public TearDownRegistry(TearDownAccepter tearDownAccepter) {
043        this.tearDownAccepter = Preconditions.checkNotNull(tearDownAccepter);
044      }
045    
046      @Override
047      public <E extends Exception, T extends ExceptionalCommand<E>> void addAction(final T action) {
048        tearDownAccepter.addTearDown(new TearDown() {
049          @Override public void tearDown() throws Exception {
050            action.execute();
051          }
052        });
053      }
054    }
