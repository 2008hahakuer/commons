039    /**
040     * An alternative to {@link JUnitCore} with stream capture and junit-report xml output capabilities.
041     */
042    public class JUnitConsoleRunner extends AbstractApplication {
043    
044      private static final SwappableStream<PrintStream> SWAPPABLE_OUT =
045          new SwappableStream<PrintStream>(System.out);
046    
047      private static final SwappableStream<PrintStream> SWAPPABLE_ERR =
048          new SwappableStream<PrintStream>(System.err);
049    
050      /**
051       * A stream that allows its underlying output to be swapped.
052       */
053      static class SwappableStream<T extends OutputStream> extends FilterOutputStream {
054        private final T original;
055    
056        SwappableStream(T out) {
057          super(out);
058          this.original = out;
059        }
060    
061        OutputStream swap(OutputStream out) {
062          OutputStream old = this.out;
063          this.out = out;
064          return old;
065        }
066    
067        /**
068         * Returns the original stream this swappable stream was created with.
069         */
070        public T getOriginal() {
071          return original;
072        }
073      }
074    
075      /**
076       * Captures a tests stderr and stdout streams, restoring the previous streams on {@link #close()}.
077       */
078      static class StreamCapture {
079        private final File out;
080        private OutputStream outstream;
081    
082        private final File err;
083        private OutputStream errstream;
084    
085        private int useCount;
086        private boolean closed;
087    
088        StreamCapture(File out, File err) throws IOException {
089          this.out = out;
090          this.err = err;
091        }
092    
093        void incrementUseCount() {
094          this.useCount++;
095        }
096    
097        void open() throws FileNotFoundException {
098          if (outstream == null) {
099            outstream = new FileOutputStream(out);
100          }
101          if (errstream == null) {
102            errstream = new FileOutputStream(err);
103          }
104          SWAPPABLE_OUT.swap(outstream);
105          SWAPPABLE_ERR.swap(errstream);
106        }
107    
108        void close() throws IOException {
109          if (--useCount <= 0 && !closed) {
110            if (outstream != null) {
111              Closeables.closeQuietly(outstream);
112            }
113            if (errstream != null) {
114              Closeables.closeQuietly(errstream);
115            }
116            closed = true;
117          }
118        }
119    
120        void dispose() throws IOException {
121          useCount = 0;
122          close();
123        }
124    
125        byte[] readOut() throws IOException {
126          return read(out);
127        }
128    
129        byte[] readErr() throws IOException {
130          return read(err);
131        }
132    
133        private byte[] read(File file) throws IOException {
134          Preconditions.checkState(closed, "Capture must be closed by all users before it can be read");
135          return Files.toByteArray(file);
136        }
137      }

139      /**
140       * A run listener that captures the output and error streams for each test class and makes the
141       * content of these available.
142       */
143      static class StreamCapturingListener extends ForwardingListener implements StreamSource {
144        private final Map<Class<?>, StreamCapture> captures = Maps.newHashMap();
145    
146        private final File outdir;
147    
148        StreamCapturingListener(File outdir) {
149          this.outdir = outdir;
150        }
151    
152        @Override
153        public void testRunStarted(Description description) throws Exception {
154          registerTests(description.getChildren());
155          super.testRunStarted(description);
156        }
157    
158        private void registerTests(Iterable<Description> tests) throws IOException {
159          for (Description test : tests) {
160            registerTests(test.getChildren());
161            if (Util.isRunnable(test)) {
162              StreamCapture capture = captures.get(test.getTestClass());
163              if (capture == null) {
164                String prefix = test.getClassName();
165    
166                File out = new File(outdir, prefix + ".out.txt");
167                Files.createParentDirs(out);
168    
169                File err = new File(outdir, prefix + ".err.txt");
170                Files.createParentDirs(err);
171                capture = new StreamCapture(out, err);
172                captures.put(test.getTestClass(), capture);
173              }
174              capture.incrementUseCount();
175            }
176          }
177        }
178    
179        @Override
180        public void testRunFinished(Result result) throws Exception {
181          for (StreamCapture capture : captures.values()) {
182            capture.dispose();
183          }
184          super.testRunFinished(result);
185        }
186    
187        @Override
188        public void testStarted(Description description) throws Exception {
189          captures.get(description.getTestClass()).open();
190          super.testStarted(description);
191        }
192    
193        @Override
194        public void testFinished(Description description) throws Exception {
195          captures.get(description.getTestClass()).close();
196          super.testFinished(description);
197        }
198    
199        @Override
200        public byte[] readOut(Class<?> testClass) throws IOException {
201          return captures.get(testClass).readOut();
202        }
203    
204        @Override
205        public byte[] readErr(Class<?> testClass) throws IOException {
206          return captures.get(testClass).readErr();
207        }
208      }

210      @CmdLine(name = "suppress-output", help = "Suppresses test output.")
211      private static final Arg<Boolean> SUPPRESS_OUTPUT = Arg.create(false);
212    
213      @CmdLine(name = "xmlreport", help = "Create ant compatible junit xml report files in -outdir.")
214      private static final Arg<Boolean> ANT_JUNIT_XML = Arg.create(false);
215    
216      @CmdLine(name = "outdir",
217               help = "Directory to output test captures too.  Only used if -suppress-output "
218                      + "or -xmlreport is set.")
219      private static final Arg<File> OUTDIR = Arg.create(SystemUtils.getJavaIoTmpDir());
220    
221      @NotEmpty
222      @Positional(help = "Names of junit test classes or test methods to run.")
223      private static final Arg<List<String>> TESTS = Arg.create();
224    
225      private static final Pattern METHOD_PARSER = Pattern.compile("^([^#]+)#([^#]+)$");
226    
227      @Override
228      public void run() {
229        PrintStream out = SWAPPABLE_OUT.getOriginal();
230        List<Request> requests = parseRequests(out, TESTS.get());
231    
232        final JUnitCore core = new JUnitCore();
233        ListenerRegistry listenerRegistry = new ListenerRegistry() {
234          @Override public void addListener(RunListener listener) {
235            core.addListener(listener);
236          }
237        };
238    
239        if (ANT_JUNIT_XML.get() || SUPPRESS_OUTPUT.get()) {
240          File outdir = OUTDIR.get();
241          if (!outdir.exists()) {
242            if (!outdir.mkdirs()) {
243              throw new IllegalStateException("Failed to create output directory: " + outdir);
244            }
245          }
246          StreamCapturingListener streamCapturingListener = new StreamCapturingListener(outdir);
247          listenerRegistry.addListener(streamCapturingListener);
248          listenerRegistry = streamCapturingListener;
249    
250          if (ANT_JUNIT_XML.get()) {
251            AntJunitXmlReportListener xmlReportListener =
252                new AntJunitXmlReportListener(outdir, streamCapturingListener);
253            listenerRegistry.addListener(xmlReportListener);
254          }
255        }
256    
257        listenerRegistry.addListener(new ConsoleListener(out));
258    
259        int failures = 0;
260        for (Request request : requests) {
261          Result result = core.run(request);
262          failures += result.getFailureCount();
263        }
264        exit(failures);
265      }
266    
267      private List<Request> parseRequests(PrintStream out, List<String> specs) {
268        Set<Class<?>> classes = Sets.newLinkedHashSet();
269        Set<Pair<Class<?>, String>> methods = Sets.newLinkedHashSet();
270        for (String spec : specs) {
271          Matcher matcher = METHOD_PARSER.matcher(spec);
272          try {
273            if (matcher.matches()) {
274              Class<?> testClass = Class.forName(matcher.group(1));
275              String method = matcher.group(2);
276              methods.add(Pair.<Class<?>, String>of(testClass, method));
277            } else {
278              Class<?> testClass = Class.forName(spec);
279              classes.add(testClass);
280            }
281          } catch (ClassNotFoundException e) {
282            out.printf("WARNING: Skipping %s: %s\n", spec, e);
283          }
284        }
285        List<Request> requests = Lists.newArrayList();
286        if (!classes.isEmpty()) {
287          requests.add(Request.classes(classes.toArray(new Class<?>[classes.size()])));
288        }
289        for (Pair<Class<?>, String> method : methods) {
290          Class<?> testClass = method.getFirst();
291          String testMethod = method.getSecond();
292          requests.add(Request.method(testClass, testMethod));
293        }
294        return requests;
295      }

297      /**
298       * Launcher for JUnitConsoleRunner.
299       */
300      public static void main(String[] args) {
301        System.setOut(new PrintStream(SWAPPABLE_OUT));
302        System.setErr(new PrintStream(SWAPPABLE_ERR));
303        AppLauncher.launch(JUnitConsoleRunner.class, ArgFilters.selectClass(JUnitConsoleRunner.class),
304            args);
305      }
306    
307      private static void exit(int code) {
308        // We're a main - its fine to exit.
309        // SUPPRESS CHECKSTYLE RegexpSinglelineJava
310        System.exit(code);
311      }
