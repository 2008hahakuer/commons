039    /**
040     * An alternative to {@link JUnitCore} with stream capture and junit-report xml output capabilities.
041     */
042    public class JUnitConsoleRunner {
043    
044      private static final SwappableStream<PrintStream> SWAPPABLE_OUT =
045          new SwappableStream<PrintStream>(System.out);
046    
047      private static final SwappableStream<PrintStream> SWAPPABLE_ERR =
048          new SwappableStream<PrintStream>(System.err);
049    
050      /**
051       * A stream that allows its underlying output to be swapped.
052       */
053      static class SwappableStream<T extends OutputStream> extends FilterOutputStream {
054        private final T original;
055    
056        SwappableStream(T out) {
057          super(out);
058          this.original = out;
059        }
060    
061        OutputStream swap(OutputStream out) {
062          OutputStream old = this.out;
063          this.out = out;
064          return old;
065        }
066    
067        /**
068         * Returns the original stream this swappable stream was created with.
069         */
070        public T getOriginal() {
071          return original;
072        }
073      }
074    
075      /**
076       * Captures a tests stderr and stdout streams, restoring the previous streams on {@link #close()}.
077       */
078      static class StreamCapture {
079        private final File out;
080        private OutputStream outstream;
081    
082        private final File err;
083        private OutputStream errstream;
084    
085        private int useCount;
086        private boolean closed;
087    
088        StreamCapture(File out, File err) throws IOException {
089          this.out = out;
090          this.err = err;
091        }
092    
093        void incrementUseCount() {
094          this.useCount++;
095        }
096    
097        void open() throws FileNotFoundException {
098          if (outstream == null) {
099            outstream = new FileOutputStream(out);
100          }
101          if (errstream == null) {
102            errstream = new FileOutputStream(err);
103          }
104          SWAPPABLE_OUT.swap(outstream);
105          SWAPPABLE_ERR.swap(errstream);
106        }
107    
108        void close() throws IOException {
109          if (--useCount <= 0 && !closed) {
110            if (outstream != null) {
111              Closeables.closeQuietly(outstream);
112            }
113            if (errstream != null) {
114              Closeables.closeQuietly(errstream);
115            }
116            closed = true;
117          }
118        }
119    
120        void dispose() throws IOException {
121          useCount = 0;
122          close();
123        }
124    
125        byte[] readOut() throws IOException {
126          return read(out);
127        }
128    
129        byte[] readErr() throws IOException {
130          return read(err);
131        }
132    
133        private byte[] read(File file) throws IOException {
134          Preconditions.checkState(closed, "Capture must be closed by all users before it can be read");
135          return Files.toByteArray(file);
136        }
137      }

139      /**
140       * A run listener that captures the output and error streams for each test class and makes the
141       * content of these available.
142       */
143      static class StreamCapturingListener extends ForwardingListener implements StreamSource {
144        private final Map<Class<?>, StreamCapture> captures = Maps.newHashMap();
145    
146        private final File outdir;
147    
148        StreamCapturingListener(File outdir) {
149          this.outdir = outdir;
150        }
151    
152        @Override
153        public void testRunStarted(Description description) throws Exception {
154          registerTests(description.getChildren());
155          super.testRunStarted(description);
156        }
157    
158        private void registerTests(Iterable<Description> tests) throws IOException {
159          for (Description test : tests) {
160            registerTests(test.getChildren());
161            if (Util.isRunnable(test)) {
162              StreamCapture capture = captures.get(test.getTestClass());
163              if (capture == null) {
164                String prefix = test.getClassName();
165    
166                File out = new File(outdir, prefix + ".out.txt");
167                Files.createParentDirs(out);
168    
169                File err = new File(outdir, prefix + ".err.txt");
170                Files.createParentDirs(err);
171                capture = new StreamCapture(out, err);
172                captures.put(test.getTestClass(), capture);
173              }
174              capture.incrementUseCount();
175            }
176          }
177        }
178    
179        @Override
180        public void testRunFinished(Result result) throws Exception {
181          for (StreamCapture capture : captures.values()) {
182            capture.dispose();
183          }
184          super.testRunFinished(result);
185        }
186    
187        @Override
188        public void testStarted(Description description) throws Exception {
189          captures.get(description.getTestClass()).open();
190          super.testStarted(description);
191        }
192    
193        @Override
194        public void testFinished(Description description) throws Exception {
195          captures.get(description.getTestClass()).close();
196          super.testFinished(description);
197        }
198    
199        @Override
200        public byte[] readOut(Class<?> testClass) throws IOException {
201          return captures.get(testClass).readOut();
202        }
203    
204        @Override
205        public byte[] readErr(Class<?> testClass) throws IOException {
206          return captures.get(testClass).readErr();
207        }
208      }
209    
210      private static final Pattern METHOD_PARSER = Pattern.compile("^([^#]+)#([^#]+)$");
211    
212      private final boolean suppressOutput;
213      private final boolean xmlReport;
214      private final File outdir;
215    
216      JUnitConsoleRunner(boolean suppressOutput, boolean xmlReport, File outdir) {
217        this.suppressOutput = suppressOutput;
218        this.xmlReport = xmlReport;
219        this.outdir = outdir;
220      }
221    
222      void run(Iterable<String> tests) {
223        PrintStream out = SWAPPABLE_OUT.getOriginal();
224        List<Request> requests = parseRequests(out, tests);
225    
226        final JUnitCore core = new JUnitCore();
227        ListenerRegistry listenerRegistry = new ListenerRegistry() {
228          @Override public void addListener(RunListener listener) {
229            core.addListener(listener);
230          }
231        };
232    
233        if (xmlReport || suppressOutput) {
234          if (!outdir.exists()) {
235            if (!outdir.mkdirs()) {
236              throw new IllegalStateException("Failed to create output directory: " + outdir);
237            }
238          }
239          StreamCapturingListener streamCapturingListener = new StreamCapturingListener(outdir);
240          listenerRegistry.addListener(streamCapturingListener);
241          listenerRegistry = streamCapturingListener;
242    
243          if (xmlReport) {
244            AntJunitXmlReportListener xmlReportListener =
245                new AntJunitXmlReportListener(outdir, streamCapturingListener);
246            listenerRegistry.addListener(xmlReportListener);
247          }
248        }
249    
250        listenerRegistry.addListener(new ConsoleListener(out));
251    
252        int failures = 0;
253        for (Request request : requests) {
254          Result result = core.run(request);
255          failures += result.getFailureCount();
256        }
257        exit(failures);
258      }
259    
260      private List<Request> parseRequests(PrintStream out, Iterable<String> specs) {
261        /**
262         * Datatype representing an individual test method.
263         */
264        class TestMethod {
265          private final Class<?> clazz;
266          private final String name;
267          TestMethod(Class<?> clazz, String name) {
268            this.clazz = clazz;
269            this.name = name;
270          }
271        }
272        Set<TestMethod> testMethods = Sets.newLinkedHashSet();
273        Set<Class<?>> classes = Sets.newLinkedHashSet();
274        for (String spec : specs) {
275          Matcher matcher = METHOD_PARSER.matcher(spec);
276          try {
277            if (matcher.matches()) {
278              Class<?> testClass = Class.forName(matcher.group(1));
279              if (isTest(testClass)) {
280                String method = matcher.group(2);
281                testMethods.add(new TestMethod(testClass, method));
282              }
283            } else {
284              Class<?> testClass = Class.forName(spec);
285              if (isTest(testClass)) {
286                classes.add(testClass);
287              }
288            }
289          } catch (NoClassDefFoundError e) {
290            warnNotFound(spec, out, e);
291          } catch (ClassNotFoundException e) {
292            warnNotFound(spec, out, e);
293          }
294        }
295        List<Request> requests = Lists.newArrayList();
296        if (!classes.isEmpty()) {
297          requests.add(Request.classes(classes.toArray(new Class<?>[classes.size()])));
298        }
299        for (TestMethod testMethod : testMethods) {
300          requests.add(Request.method(testMethod.clazz, testMethod.name));
301        }
302        return requests;
303      }
304    
305      private void warnNotFound(String spec, PrintStream out, Throwable t) {
306        out.printf("WARNING: Skipping %s: %s\n", spec, t);
307      }

309      /**
310       * Launcher for JUnitConsoleRunner.
311       */
312      public static void main(String[] args) {
313        System.setOut(new PrintStream(SWAPPABLE_OUT));
314        System.setErr(new PrintStream(SWAPPABLE_ERR));
315    
316        /**
317         * Command line option bean.
318         */
319        class Options {
320          private boolean suppressOutput = false;
321          private boolean xmlReport = false;
322          private File outdir = new File(System.getProperty("java.io.tmpdir"));
323          private List<String> tests = Lists.newArrayList();
324    
325          @Option(name = "-suppress-output", usage = "Suppresses test output.")
326          public void setSuppressOutput(boolean suppressOutput) {
327            this.suppressOutput = suppressOutput;
328          }
329    
330          @Option(name = "-xmlreport",
331                  usage = "Create ant compatible junit xml report files in -outdir.")
332          public void setXmlReport(boolean xmlReport) {
333            this.xmlReport = xmlReport;
334          }
335    
336          @Option(name = "-outdir",
337                  usage = "Directory to output test captures too.  Only used if -suppress-output or "
338                          + "-xmlreport is set.")
339          public void setOutdir(File outdir) {
340            this.outdir = outdir;
341          }
342    
343          @Argument(usage = "Names of junit test classes or test methods to run.",
344                    required = true,
345                    metaVar = "TESTS",
346                    handler = StringArrayOptionHandler.class)
347          public void setTests(String[] tests) {
348            this.tests = Arrays.asList(tests);
349          }
350        }
351        Options options = new Options();
352        CmdLineParser parser = new CmdLineParser(options);
353        try {
354          parser.parseArgument(args);
355        } catch (CmdLineException e) {
356          e.getParser().printUsage(System.out);
357          exit(1);
358        }
359    
360        JUnitConsoleRunner runner =
361            new JUnitConsoleRunner(options.suppressOutput, options.xmlReport, options.outdir);
362        runner.run(options.tests);
363      }
364    
365      private static boolean isTest(final Class<?> clazz) {
366        // Must be a public concrete class to be a runnable junit Test.
367        if (clazz.isInterface()
368            || Modifier.isAbstract(clazz.getModifiers())
369            || !Modifier.isPublic(clazz.getModifiers())) {
370          return false;
371        }
372    
373        // Support junit 3.x Test hierarchy.
374        if (junit.framework.Test.class.isAssignableFrom(clazz)) {
375          return true;
376        }
377    
378        // Support junit 4.x @Test annotated methods.
379        return Iterables.any(Arrays.asList(clazz.getDeclaredMethods()), new Predicate<Method>() {
380          @Override public boolean apply(Method method) {
381            return Modifier.isPublic(method.getModifiers())
382                && method.isAnnotationPresent(org.junit.Test.class);
383          }
384        });
385      }
386    
387      private static void exit(int code) {
388        // We're a main - its fine to exit.
389        // SUPPRESS CHECKSTYLE RegexpSinglelineJava
390        System.exit(code);
391      }
