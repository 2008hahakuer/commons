001    package com.twitter.common.net.http.handlers;
002    
003    import java.io.IOException;
004    import java.io.PrintWriter;
005    import java.util.Map;
006    
007    import javax.servlet.ServletException;
008    import javax.servlet.http.HttpServlet;
009    import javax.servlet.http.HttpServletRequest;
010    import javax.servlet.http.HttpServletResponse;
011    
012    import com.google.common.annotations.VisibleForTesting;
013    import com.google.common.base.Preconditions;
014    import com.google.common.base.Supplier;
015    import com.google.common.collect.Maps;
016    import com.google.gson.Gson;
017    import com.google.gson.GsonBuilder;
018    import com.google.inject.Inject;
019    
020    import com.twitter.common.stats.Stat;
021    
022    /**
023     * A servlet that returns the current value of all variables in JSON format.
024     * The format returns a JSON object with string fields and typed values:
025     * <pre>
026     *   {
027     *     "var_a": 1,
028     *     "var_b": 126.0,
029     *     "var_c": "a string value",
030     *   }
031     * </pre>
032     * If the optional URL parameter 'pretty' is used, the output will be pretty-printed
033     * (similar to the above example).
034     *
035     * @author William Farner
036     */
037    public class VarsJsonHandler extends HttpServlet {
038    
039      private final Supplier<Iterable<Stat>> statSupplier;
040    
041      /**
042       * Creates a new handler that will report stats from the provided supplier.
043       *
044       * @param statSupplier Stats supplier.
045       */
046      @Inject
047      VarsJsonHandler(Supplier<Iterable<Stat>> statSupplier) {
048        this.statSupplier = Preconditions.checkNotNull(statSupplier);
049      }
050    
051      @VisibleForTesting
052      String getBody(boolean pretty) {
053        Map<String, Object> vars = Maps.newHashMap();
054        for (Stat<?> var : statSupplier.get()) {
055          vars.put(var.getName(), var.read());
056        }
057        return getGson(pretty).toJson(vars);
058      }
059    
060      @Override
061      protected void doGet(HttpServletRequest req, HttpServletResponse resp)
062          throws ServletException, IOException {
063    
064        resp.setContentType("application/json");
065        resp.setStatus(HttpServletResponse.SC_OK);
066        PrintWriter responseBody = resp.getWriter();
067        try {
068          responseBody.print(getBody(req.getParameter("pretty") != null));
069        } finally {
070          responseBody.close();
071        }
072      }
073    
074      private Gson getGson(boolean pretty) {
075        return pretty ? new GsonBuilder().setPrettyPrinting().create() : new Gson();
076      }
077    }
