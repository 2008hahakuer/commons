017    package com.twitter.common.net.http.handlers;
018    
019    import com.google.common.annotations.VisibleForTesting;
020    import com.google.common.base.Splitter;
021    import com.google.common.collect.Iterables;
022    import com.google.common.io.ByteStreams;
023    import com.google.common.io.Closeables;
024    import com.google.common.io.InputSupplier;
025    
026    import com.twitter.common.quantity.Amount;
027    import com.twitter.common.quantity.Time;
028    
029    import javax.servlet.ServletException;
030    import javax.servlet.http.HttpServlet;
031    import javax.servlet.http.HttpServletRequest;
032    import javax.servlet.http.HttpServletResponse;
033    
034    import java.io.ByteArrayInputStream;
035    import java.io.ByteArrayOutputStream;
036    import java.io.IOException;
037    import java.io.InputStream;
038    import java.io.OutputStream;
039    import java.util.zip.GZIPInputStream;
040    import java.util.zip.GZIPOutputStream;
041    
042    import org.apache.commons.codec.binary.Base64;
043    import org.apache.commons.codec.digest.DigestUtils;
044    
045    import static com.google.common.base.Preconditions.checkNotNull;
046    
047    /**
048     * Servlet that is responsible for serving an asset.
049     *
050     * @author William Farner
051     */
052    public class AssetHandler extends HttpServlet {
053    
054      @VisibleForTesting
055      static final Amount<Integer, Time> CACHE_CONTROL_MAX_AGE_SECS = Amount.of(30, Time.DAYS);
056      private static final String GZIP_ENCODING = "gzip";
057    
058      private final StaticAsset staticAsset;
059    
060      public static class StaticAsset {
061        private final InputSupplier<? extends InputStream> inputSupplier;
062        private final String contentType;
063        private final boolean cacheLocally;
064    
065        private byte[] gzipData = null;
066        private String hash = null;
067    
068        /**
069         * Creates a new static asset.
070         *
071         * @param inputSupplier Supplier of the input stream from which to load the asset.
072         * @param contentType HTTP content type of the asset.
073         * @param cacheLocally If {@code true} the asset will be loaded once and stored in memory, if
074         *    {@code false} it will be loaded on each request.
075         */
076        public StaticAsset(InputSupplier<? extends InputStream> inputSupplier,
077            String contentType, boolean cacheLocally) {
078          this.inputSupplier = checkNotNull(inputSupplier);
079          this.contentType = checkNotNull(contentType);
080          this.cacheLocally = cacheLocally;
081        }
082    
083        public String getContentType() {
084          return contentType;
085        }
086    
087        public synchronized byte[] getRawData() throws IOException {
088          byte[] zipData = getGzipData();
089          GZIPInputStream in = new GZIPInputStream(new ByteArrayInputStream(zipData));
090          return ByteStreams.toByteArray(in);
091        }
092    
093        public synchronized byte[] getGzipData() throws IOException {
094          byte[] data = gzipData;
095          // Ensure we don't double-read after a call to getChecksum().
096          if (!cacheLocally || gzipData == null) {
097            load();
098            data = gzipData;
099          }
100          if (!cacheLocally) {
101            gzipData = null;
102          }
103    
104          return data;
105        }
106    
107        public synchronized String getChecksum() throws IOException {
108          if (hash == null) {
109            load();
110          }
111          return hash;
112        }
113    
114        private void load() throws IOException {
115          ByteArrayOutputStream gzipBaos = new ByteArrayOutputStream();
116          GZIPOutputStream gzipStream = new GZIPOutputStream(gzipBaos);
117          ByteStreams.copy(inputSupplier, gzipStream);
118          gzipStream.flush();  // copy() does not flush or close output stream.
119          gzipStream.close();
120          gzipData = gzipBaos.toByteArray();
121    
122          // Calculate a checksum of the gzipped data.
123          hash = Base64.encodeBase64String(DigestUtils.md5(gzipData)).trim();
124        }
125      }
126    
127      /**
128       * Creates a new asset handler.
129       *
130       * @param staticAsset The asset to serve.
131       */
132      public AssetHandler(StaticAsset staticAsset) {
133        this.staticAsset = checkNotNull(staticAsset);
134      }
135    
136      @Override
137      protected void doGet(HttpServletRequest req, HttpServletResponse resp)
138          throws ServletException, IOException {
139    
140        OutputStream responseBody = resp.getOutputStream();
141    
142        if (checksumMatches(req)) {
143          resp.setStatus(HttpServletResponse.SC_NOT_MODIFIED);
144        } else {
145          setPayloadHeaders(resp);
146    
147          boolean gzip = supportsGzip(req);
148          if (gzip) {
149            resp.setHeader("Content-Encoding", GZIP_ENCODING);
150          }
151    
152          InputStream in = new ByteArrayInputStream(
153              gzip ? staticAsset.getGzipData() : staticAsset.getRawData());
154          ByteStreams.copy(in, responseBody);
155        }
156    
157        Closeables.closeQuietly(responseBody);
158      }
159    
160      private void setPayloadHeaders(HttpServletResponse resp) throws IOException {
161        resp.setStatus(HttpServletResponse.SC_OK);
162        resp.setContentType(staticAsset.getContentType());
163        resp.setHeader("Cache-Control", "public,max-age=" + CACHE_CONTROL_MAX_AGE_SECS);
164    
165        String checksum = staticAsset.getChecksum();
166        if (checksum != null) {
167          resp.setHeader("ETag", checksum);
168        }
169      }
170    
171      private boolean checksumMatches(HttpServletRequest req) throws IOException {
172        // TODO(William Farner): Change this to more fully comply with
173        // http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.26
174        // Specifically - a response to 'If-None-Match: *' should include ETag as well as other
175        // cache-related headers.
176        String suppliedETag = req.getHeader("If-None-Match");
177        if ("*".equals(suppliedETag)) {
178          return true;
179        }
180    
181        String checksum = staticAsset.getChecksum();
182        // Note - this isn't a completely accurate check since the tag we end up matching against could
183        // theoretically be the actual tag with some extra characters appended.
184        return (checksum != null) && (suppliedETag != null) && suppliedETag.contains(checksum);
185      }
186    
187      private static boolean supportsGzip(HttpServletRequest req) {
188        String header = req.getHeader("Accept-Encoding");
189        return (header != null)
190            && Iterables.contains(Splitter.on(",").trimResults().split(header), GZIP_ENCODING);
191      }
192    }
