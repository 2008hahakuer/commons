055    /**
056     * HTTP handler to page through log files. Supports GET and POST requests.  GET requests are
057     * responsible for fetching chrome and javascript, while the POST requests are used to fetch actual
058     * log data.
059     *
060     * TODO(William Farner): Change all links (Next, Prev, filter) to issue AJAX requests rather than
061     *     reloading the page.
062     *
063     * @author William Farner
064     */
065    public class LogPrinter extends StringTemplateServlet {
066      private static final Logger LOG = Logger.getLogger(LogPrinter.class.getName());
067    
068      /**
069       * A {@literal @Named} binding key for the log directory to display by default.
070       */
071      public static final String LOG_DIR_KEY =
072            "com.twitter.common.net.http.handlers.LogPrinter.log_dir";
073    
074      private static final int DEFAULT_PAGE = 0;
075    
076      private static final int PAGE_CHUNK_SIZE_BYTES = Amount.of(512, Data.KB).as(Data.BYTES);
077      private static final int TAIL_START_BYTES = Amount.of(10, Data.KB).as(Data.BYTES);
078      private static final int PAGE_END_BUFFER_SIZE_BYTES = Amount.of(1, Data.KB).as(Data.BYTES);
079    
080      private static final String XML_RESP_FORMAT = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
081                                                  + "<logchunk text=\"%s\""
082                                                  + " end_pos=\"%d\">"
083                                                  + "</logchunk>";
084      private final File logDir;
085    
086      @Inject
087      public LogPrinter(@Named(LOG_DIR_KEY) File logDir, @CacheTemplates boolean cacheTemplates) {
088        super("logprinter", cacheTemplates);
089        this.logDir = Preconditions.checkNotNull(logDir);
090      }

092      /**
093       * A POST request is made from javascript, to request the contents of a log file.  In order to
094       * fulfill the request, the 'file' parameter must be set in the request.
095       *
096       * @param req Servlet request.
097       * @param resp Servlet response.
098       * @throws ServletException If there is a problem with the servlet.
099       * @throws IOException If there is a problem reading/writing data to the client.
100       */
101      @Override
102      protected void doPost(HttpServletRequest req, HttpServletResponse resp)
103          throws ServletException, IOException {
104        resp.setContentType("text/xml");
105    
106        try {
107          LogViewRequest request = new LogViewRequest(req);
108    
109          if (request.file == null) {
110            // The log file is a required parameter for POST requests.
111            resp.setStatus(HttpServletResponse.SC_BAD_REQUEST);
112            return;
113          }
114    
115          resp.setStatus(HttpServletResponse.SC_OK);
116          PrintWriter responseBody = resp.getWriter();
117    
118          String responseXml = fetchXmlLogContents(request);
119          responseBody.write(responseXml);
120          responseBody.close();
121        } catch (Exception e) {
122          LOG.log(Level.SEVERE, "Unknown exception.", e);
123          resp.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
124        }
125      }

127      /**
128       * Fetches the chrome for the page.  If a file is requested, a page will be returned that uses an
129       * AJAX request to fetch the log contents.  If no file is specified, then a file listing is
130       * displayed.
131       *
132       * @param req Servlet request.
133       * @param resp Servlet response.
134       * @throws ServletException If there is a problem with the servlet.
135       * @throws IOException If there is a problem reading/writing data to the client.
136       */
137      @Override
138      protected void doGet(final HttpServletRequest req, HttpServletResponse resp)
139          throws ServletException, IOException {
140        final LogViewRequest request = new LogViewRequest(req);
141    
142        if (request.download) {
143          if (request.file == null) {
144            resp.sendError(HttpServletResponse.SC_BAD_REQUEST, "No file requested for download.");
145            return;
146          }
147    
148          if (!request.file.isRegularFile()) {
149            resp.sendError(HttpServletResponse.SC_BAD_REQUEST, "Only regular files may be downloaded.");
150            return;
151          }
152    
153          try {
154            OutputStream out = resp.getOutputStream();
155            ServletContext context  = getServletConfig().getServletContext();
156            String mimetype = context.getMimeType(request.file.getName());
157    
158            resp.setContentType(mimetype != null ? mimetype : "application/octet-stream" );
159            resp.setContentLength((int) request.file.getFile().length());
160            resp.setHeader("Content-Disposition", String.format("attachment; filename=\"%s\"",
161                request.file.getName()));
162    
163            Files.copy(request.file.getFile(), out);
164          } catch (Exception e) {
165            resp.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, "Failed to fetch file.");
166            LOG.warning("Failed to download file " + request.file.getPath() + ": " + e.getMessage());
167          }
168        } else {
169          writeTemplate(resp, new ExceptionalClosure<StringTemplate, LogConfigException>() {
170            @Override public void execute(StringTemplate template) throws LogConfigException {
171    
172              // TODO(William Farner): Consider using unix file utility to check if the requested file is a
173              //    text file, and allow the user to download the file if it is not.
174              if (request.isFileViewRequest()) {
175                request.sendToTemplate(template);
176    
177                if (!request.tailing) {
178                  long readStartPos = getReadStartPos(request.file.getFile(), request.page);
179    
180                  if (readStartPos > 0) template.setAttribute("prev", request.page + 1);
181                  if (request.page > 0) template.setAttribute("next", request.page - 1);
182                }
183              } else {
184                // If a file was not requested, show a list of files.
185                File dir = request.getListingDir();
186    
187                List<LogFile> logFiles = Lists.newArrayList();
188                for (File file : dir.listFiles()) {
189                  logFiles.add(new LogFile(file));
190                }
191    
192                // Sort by dir/file, subsort by name.
193                Collections.sort(logFiles, new Comparator<LogFile>() {
194                  @Override public int compare(LogFile fileA, LogFile fileB) {
195                    if (fileA.isDir() == fileB.isDir()) {
196                      return fileA.file.getName().compareTo(fileB.file.getName());
197                    } else {
198                      return fileA.isDir() ? -1 : 1;
199                    }
200                  }
201                });
202    
203                template.setAttribute("dir", dir);
204                template.setAttribute("parent", dir.getParentFile());
205                template.setAttribute("files", logFiles);
206              }
207            }
208          });
209        }
210      }
211    
212      /**
213       * Gets the starting position for reading a page from a file.
214       *
215       * @param file The file to find a page within.
216       * @param page The page index, where page 0 is the last page (at the end of the file).
217       * @return The byte index that the page begins on, or 0 if an invalid page number was provided.
218       */
219      private long getReadStartPos(File file, int page) {
220        return page < 0 ? 0 : Math.max(0, file.length() - (page + 1) * PAGE_CHUNK_SIZE_BYTES);
221      }

223      /**
224       * Stores request parameters and assigns default values.
225       */
226      private class LogViewRequest {
227        public static final String DIR_PARAM = "dir";
228        public static final String FILE_PARAM = "file";
229        public static final String PAGE_PARAM = "page";
230        public static final String FILTER_PARAM = "filter";
231        public static final String TAIL_PARAM = "tail";
232        public static final String START_POS_PARAM = "start_pos";
233        public static final String DOWNLOAD_PARAM = "download";
234    
235        public final File dir;
236        public final LogFile file;
237        public final boolean download;
238        public final int page;
239        public final long startPos;
240        public final String filter;
241        public final boolean tailing;
242    
243        public LogViewRequest(HttpServletRequest req) {
244          dir = req.getParameter(DIR_PARAM) == null ? null : new File(req.getParameter(DIR_PARAM));
245          file = req.getParameter(FILE_PARAM) ==  null ? null
246              : new LogFile(req.getParameter(FILE_PARAM));
247          download = req.getParameter(DOWNLOAD_PARAM) == null ? false
248              : Boolean.parseBoolean(req.getParameter(DOWNLOAD_PARAM));
249          tailing = req.getParameter(TAIL_PARAM) != null
250                    && Boolean.parseBoolean(req.getParameter(TAIL_PARAM));
251          page = req.getParameter(PAGE_PARAM) == null ? DEFAULT_PAGE
252              : Integer.parseInt(req.getParameter(PAGE_PARAM));
253          Preconditions.checkArgument(page >= 0);
254    
255          startPos = req.getParameter(START_POS_PARAM) == null ? -1
256              : Long.parseLong(req.getParameter(START_POS_PARAM));
257          if (file != null) {
258            Preconditions.checkArgument(startPos >= -1 && startPos <= file.getFile().length());
259          }
260          filter = req.getParameter(FILTER_PARAM) == null ? "" : req.getParameter(FILTER_PARAM);
261        }
262    
263        public boolean isFileViewRequest() {
264          return file != null && file.isRegularFile();
265        }
266    
267        public File getListingDir() {
268          if (file != null && file.getFile().isDirectory()) {
269            return file.getFile();
270          } else if (dir != null) {
271            return dir;
272          } else {
273            return logDir;
274          }
275        }
276    
277        public void sendToTemplate(StringTemplate template) {
278          template.setAttribute(FILE_PARAM, file);
279          template.setAttribute(PAGE_PARAM, page);
280          template.setAttribute(FILTER_PARAM, filter);
281          template.setAttribute(TAIL_PARAM, tailing);
282        }
283      }

285      /**
286       * Class to wrap a log file and offer functions to StringTemplate via reflection.
287       */
288      private class LogFile {
289        private final File file;
290    
291        public LogFile(File file) {
292          this.file = file;
293        }
294    
295        public LogFile(String filePath) {
296          this(new File(filePath));
297        }
298    
299        public File getFile() {
300          return file;
301        }
302    
303        public boolean isDir() {
304          return !isRegularFile();
305        }
306    
307        public boolean isRegularFile() {
308          return file.isFile();
309        }
310    
311        public String getPath() {
312          return file.getAbsolutePath();
313        }
314    
315        public String getName() {
316          return file.getName();
317        }
318    
319        public String getUrlpath() throws UnsupportedEncodingException {
320          return URLEncoder.encode(getPath(), Charsets.UTF_8.name());
321        }
322    
323        public String getSize() {
324          Amount<Long, Data> length = Amount.of(file.length(), Data.BYTES);
325    
326          if (length.as(Data.GB) > 0) {
327            return length.as(Data.GB) + " GB";
328          } else if (length.as(Data.MB) > 0) {
329            return length.as(Data.MB) + " MB";
330          } else if (length.as(Data.KB) > 0) {
331            return length.as(Data.KB) + " KB";
332          } else {
333            return length.getValue() + " bytes";
334          }
335        }
336      }

338      /**
339       * Reads data from a log file and prepares an XML response which includes the (sanitized) log text
340       * and the last position read from the file.
341       *
342       * @param request The request parameters.
343       * @return A string containing the XML-formatted response.
344       * @throws IOException If there was a problem reading the file.
345       */
346      private String fetchXmlLogContents(LogViewRequest request) throws IOException {
347        RandomAccessFile seekFile = new RandomAccessFile(request.file.getFile(), "r");
348        try {
349          // Move to the approximate start of the page.
350          if (!request.tailing) {
351            seekFile.seek(getReadStartPos(request.file.getFile(), request.page));
352          } else {
353            if (request.startPos < 0) {
354              seekFile.seek(Math.max(0, request.file.getFile().length() - TAIL_START_BYTES));
355            } else {
356              seekFile.seek(request.startPos);
357            }
358          }
359    
360          byte[] buffer = new byte[PAGE_CHUNK_SIZE_BYTES];
361          int bytesRead = seekFile.read(buffer);
362          long chunkStop = seekFile.getFilePointer();
363          StringBuilder fileChunk = new StringBuilder();
364          if (bytesRead > 0) {
365            fileChunk.append(new String(buffer, 0, bytesRead));
366    
367            // Read at most 1 KB more while searching for another line break.
368            buffer = new byte[PAGE_END_BUFFER_SIZE_BYTES];
369            int newlinePos = 0;
370            bytesRead = seekFile.read(buffer);
371            if (bytesRead > 0) {
372              for (byte b : buffer) {
373                newlinePos++;
374                if (b == '\n') break;
375              }
376    
377              fileChunk.append(new String(buffer, 0, newlinePos));
378              chunkStop = seekFile.getFilePointer() - (bytesRead - newlinePos);
379            }
380          }
381    
382          return logChunkXml(filterLines(fileChunk.toString(), request.filter), chunkStop);
383        } finally {
384          seekFile.close();
385        }
386      }
387    
388      private String logChunkXml(String text, long lastBytePosition) {
389        // TODO(William Farner): There still seems to be a problem with the sanitization here, data is sent
390        // back to the client that breaks XML syntax when some non-ascii characters appear in the log
391        // (i think).
392        String sanitized = StringEscapeUtils.escapeXml(
393            StringEscapeUtils.escapeHtml(text).replaceAll("\n", "&#10;"));
394        return String.format(XML_RESP_FORMAT, sanitized , lastBytePosition);
395      }
396    
397      @VisibleForTesting
398      protected static String filterLines(String text, String filterRegexp) {
399        if (StringUtils.isEmpty(filterRegexp)) return text;
400    
401        List<String> lines = Lists.newArrayList(text.split("\n"));
402        final Pattern pattern = Pattern.compile(filterRegexp);
403    
404        Iterable<String> filtered = Iterables.filter(lines, new Predicate<String>() {
405          @Override public boolean apply(String line) {
406            return pattern.matcher(line).matches();
407          }
408        });
409    
410        return Joiner.on("\n").join(filtered);
411      }
