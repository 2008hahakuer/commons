017    package com.twitter.common.net.http.handlers;
018    
019    import com.google.common.collect.Lists;
020    import com.google.common.collect.Maps;
021    import com.google.common.collect.Sets;
022    import com.google.common.primitives.Longs;
023    
024    import javax.servlet.http.HttpServletRequest;
025    import java.lang.management.ManagementFactory;
026    import java.lang.management.ThreadInfo;
027    import java.lang.management.ThreadMXBean;
028    import java.util.List;
029    import java.util.Map;
030    import java.util.Set;
031    
032    /**
033     * HTTP request handler that prints information about blocked threads.
034     *
035     * @author William Farner
036     */
037    public class ContentionPrinter extends TextResponseHandler {
038      public ContentionPrinter() {
039        ManagementFactory.getThreadMXBean().setThreadContentionMonitoringEnabled(true);
040      }
041    
042      @Override
043      public Iterable<String> getLines(HttpServletRequest request) {
044        List<String> lines = Lists.newLinkedList();
045        ThreadMXBean bean = ManagementFactory.getThreadMXBean();
046    
047        Map<Long, StackTraceElement[]> threadStacks = Maps.newHashMap();
048        for (Map.Entry<Thread, StackTraceElement[]> entry : Thread.getAllStackTraces().entrySet()) {
049          threadStacks.put(entry.getKey().getId(), entry.getValue());
050        }
051    
052        Set<Long> lockOwners = Sets.newHashSet();
053    
054        lines.add("Locked threads:");
055        for (ThreadInfo t : bean.getThreadInfo(bean.getAllThreadIds())) {
056          switch (t.getThreadState()) {
057            case BLOCKED:
058            case WAITING:
059            case TIMED_WAITING:
060              lines.addAll(getThreadInfo(t, threadStacks.get(t.getThreadId())));
061              if (t.getLockOwnerId() != -1) lockOwners.add(t.getLockOwnerId());
062              break;
063          }
064        }
065    
066        if (lockOwners.size() > 0) {
067          lines.add("\nLock Owners");
068          for (ThreadInfo t : bean.getThreadInfo(Longs.toArray(lockOwners))) {
069            lines.addAll(getThreadInfo(t, threadStacks.get(t.getThreadId())));
070          }
071        }
072    
073        return lines;
074      }
075    
076      private static List<String> getThreadInfo(ThreadInfo t, StackTraceElement[] stack) {
077        List<String> lines = Lists.newLinkedList();
078    
079        lines.add(String.format("'%s' Id=%d %s",
080            t.getThreadName(), t.getThreadId(), t.getThreadState()));
081        lines.add("Waiting for lock: " + t.getLockName());
082        lines.add("Lock is currently held by thread: " + t.getLockOwnerName());
083        lines.add("Wait time: " + t.getBlockedTime() + " ms.");
084        for (StackTraceElement s : stack) {
085          lines.add(String.format("    " + s.toString()));
086        }
087        lines.add("\n");
088    
089        return lines;
090      }
091    }
