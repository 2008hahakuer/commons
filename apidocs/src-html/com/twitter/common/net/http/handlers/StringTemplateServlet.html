017    package com.twitter.common.net.http.handlers;
018    
019    import com.google.common.base.Preconditions;
020    import com.google.inject.BindingAnnotation;
021    import com.twitter.common.base.ExceptionalClosure;
022    import com.twitter.common.base.MorePreconditions;
023    import org.antlr.stringtemplate.AutoIndentWriter;
024    import org.antlr.stringtemplate.StringTemplate;
025    import org.antlr.stringtemplate.StringTemplateGroup;
026    
027    import javax.servlet.http.HttpServlet;
028    import javax.servlet.http.HttpServletResponse;
029    import java.io.IOException;
030    import java.lang.annotation.ElementType;
031    import java.lang.annotation.Retention;
032    import java.lang.annotation.RetentionPolicy;
033    import java.lang.annotation.Target;
034    import java.util.logging.Level;
035    import java.util.logging.Logger;
036    
037    /**
038     * A base class for servlets that render using the string template templating system.  Subclasses
039     * can call one of the {@link #writeTemplate} methods to render their content with the associated
040     * template.
041     *
042     * @author John Sirois
043     */
044    public abstract class StringTemplateServlet extends HttpServlet {
045      private static final String CONTENT_TYPE_TEXT_HTML = "text/html";
046    
047      /**
048       * A {@literal @BindingAnnotation} that allows configuration of whether or not
049       * StringTemplateServlets should cache their templates.
050       */
051      @BindingAnnotation
052      @Retention(RetentionPolicy.RUNTIME)
053      @Target({ElementType.PARAMETER, ElementType.METHOD})
054      public @interface CacheTemplates {}
055    
056      private static final Logger LOG = Logger.getLogger(StringTemplateServlet.class.getName());
057    
058      private final StringTemplateGroup group;
059      private final String templatePath;
060    
061      /**
062       * Creates a new StringTemplateServlet that expects to find its template located in the same
063       * package on the classpath at '{@code templateName}.st'.
064       *
065       * @param templateName The name of the string template to use.
066       * @param cacheTemplates {@code true} to re-use loaded templates, {@code false} to reload the
067       *     template for each request.
068       */
069      protected StringTemplateServlet(String templateName, boolean cacheTemplates) {
070        MorePreconditions.checkNotBlank(templateName);
071        String templatePath = getClass().getPackage().getName().replace('.', '/') + "/" + templateName;
072        StringTemplateGroup group = new StringTemplateGroup(templateName);
073        Preconditions.checkNotNull(group.getInstanceOf(templatePath),
074            "Failed to load template at: %s", templatePath);
075    
076        this.group = group;
077        if (!cacheTemplates) {
078          group.setRefreshInterval(0);
079        }
080        this.templatePath = templatePath;
081      }
082    
083      protected final void writeTemplate(HttpServletResponse response,
084            ExceptionalClosure<StringTemplate, ?> parameterSetter) throws IOException {
085        writeTemplate(response, CONTENT_TYPE_TEXT_HTML, HttpServletResponse.SC_OK, parameterSetter);
086      }
087    
088      protected final void writeTemplate(HttpServletResponse response, String contentType, int status,
089          ExceptionalClosure<StringTemplate, ?> parameterSetter) throws IOException {
090        Preconditions.checkNotNull(response);
091        MorePreconditions.checkNotBlank(contentType);
092        Preconditions.checkArgument(status > 0);
093        Preconditions.checkNotNull(parameterSetter);
094    
095        StringTemplate stringTemplate = group.getInstanceOf(templatePath);
096        try {
097          parameterSetter.execute(stringTemplate);
098          response.setStatus(status);
099          response.setContentType(contentType);
100          stringTemplate.write(new AutoIndentWriter(response.getWriter()));
101        } catch (Exception e) {
102          LOG.log(Level.SEVERE, "Unknown exception.", e);
103          response.sendError(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
104        }
105      }
106    }
