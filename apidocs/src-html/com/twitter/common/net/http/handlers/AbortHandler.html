017    package com.twitter.common.net.http.handlers;
018    
019    import com.google.common.base.Preconditions;
020    import com.google.inject.Inject;
021    import com.google.inject.name.Named;
022    
023    import javax.servlet.ServletException;
024    import javax.servlet.http.HttpServlet;
025    import javax.servlet.http.HttpServletRequest;
026    import javax.servlet.http.HttpServletResponse;
027    import java.io.IOException;
028    import java.io.PrintWriter;
029    import java.util.logging.Level;
030    import java.util.logging.Logger;
031    
032    /**
033     * A servlet that provides a way to remotely terminate the running process immediately.
034     *
035     * @author William Farner
036     */
037    public class AbortHandler extends HttpServlet {
038    
039      /**
040       * A {@literal @Named} binding key for the QuitHandler listener.
041       */
042      public static final String ABORT_HANDLER_KEY =
043          "com.twitter.common.net.http.handlers.AbortHandler.listener";
044    
045      private static final Logger LOG = Logger.getLogger(AbortHandler.class.getName());
046    
047      private final Runnable abortListener;
048    
049      /**
050       * Constructs a new AbortHandler that will notify the given {@code abortListener} when the servlet
051       * is accessed.  It is the responsibility of the listener to initiate an immediate shutdown of
052       * the system.
053       *
054       * @param abortListener Runnable to notify when the servlet is accessed.
055       */
056      @Inject
057      public AbortHandler(@Named(ABORT_HANDLER_KEY) Runnable abortListener) {
058        this.abortListener = Preconditions.checkNotNull(abortListener);
059      }
060    
061      @Override
062      protected void doGet(HttpServletRequest req, HttpServletResponse resp)
063          throws ServletException, IOException {
064        LOG.info(String.format("Received abort HTTP signal from %s (%s)",
065            req.getRemoteAddr(), req.getRemoteHost()));
066    
067        resp.setContentType("text/plain");
068        PrintWriter writer = resp.getWriter();
069        try {
070          writer.println("Aborting process NOW!");
071          writer.close();
072          abortListener.run();
073        } catch (Exception e) {
074          LOG.log(Level.WARNING, "Abort failed.", e);
075        }
076      }
077    }
