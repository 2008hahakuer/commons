043    /**
044     * A connection pool that picks connections from a set of backend pools.  Backend pools are selected
045     * from randomly initially but then as they are used they are ranked according to how many
046     * connections they have available and whether or not the last used connection had an error or not.
047     * In this way, backends that are responsive should get selected in preference to those that are
048     * not.
049     *
050     * <p>Non-responsive backends are monitored after a configurable period in a background thread and
051     * if a connection can be obtained they start to float back up in the rankings.  In this way,
052     * backends that are initially non-responsive but later become responsive should end up getting
053     * selected.
054     *
055     * <p> TODO(John Sirois): take a ShutdownRegistry and register a close command
056     *
057     * @author John Sirois
058     */
059    public class MetaPool<T, E> implements ObjectPool<Connection<T, E>> {
060    
061      private final Command stopCommand;
062    
063      private Map<E, ObjectPool<Connection<T, E>>> backends = null;
064      private final LoadBalancer<E> loadBalancer;
065      private final Closure<Collection<E>> onBackendsChosen;
066    
067      /**
068       * Creates a connection pool with no backends.  Backends may be added post-creation by calling
069       * {@link #setBackends<ImmutableSet>()};
070       *
071       * @param loadBalancer the load balancer to distribute requests among backends.
072       * @param onBackendsChosen a callback to notify whenever the {@code loadBalancer} chooses a new
073       *     set of backends to restrict its call distribution to
074       * @param restoreInterval the interval after a backend goes dead to begin checking the backend to
075       *     see if it has come back to a healthy state
076       */
077      public MetaPool(LoadBalancer<E> loadBalancer,
078          Closure<Collection<E>> onBackendsChosen, Amount<Long, Time> restoreInterval) {
079        this(ImmutableMap.<E, ObjectPool<Connection<T, E>>>of(), loadBalancer,
080            onBackendsChosen, restoreInterval);
081      }
082    
083      /**
084       * Creates a connection pool that balances connections across multiple backend pools.
085       *
086       * @param backends the connection pools for the backends
087       * @param onBackendsChosen a callback to notify whenever the {@code loadBalancer} chooses a new
088       *     set of backends to restrict its call distribution to
089       * @param loadBalancer the load balancer to distribute requests among backends.
090       * @param restoreInterval the interval after a backend goes dead to begin checking the backend to
091       *     see if it has come back to a healthy state
092       */
093      public MetaPool(
094          ImmutableMap<E, ObjectPool<Connection<T, E>>> backends,
095          LoadBalancer<E> loadBalancer,
096          Closure<Collection<E>> onBackendsChosen, Amount<Long, Time> restoreInterval) {
097    
098        this.loadBalancer = Preconditions.checkNotNull(loadBalancer);
099        this.onBackendsChosen = Preconditions.checkNotNull(onBackendsChosen);
100        setBackends(backends);
101    
102        Preconditions.checkNotNull(restoreInterval);
103        Preconditions.checkArgument(restoreInterval.getValue() > 0);
104        stopCommand = startDeadBackendRestorer(restoreInterval);
105      }
106    
107      /**
108       * Assigns the backend pools that this pool should draw from.
109       *
110       * @param pools New pools to use.
111       */
112      public synchronized void setBackends(Map<E, ObjectPool<Connection<T, E>>> pools) {
113        backends = Preconditions.checkNotNull(pools);
114        loadBalancer.offerBackends(pools.keySet(), onBackendsChosen);
115      }
116    
117      private Command startDeadBackendRestorer(final Amount<Long, Time> restoreInterval) {
118    
119        final AtomicBoolean shouldRestore = new AtomicBoolean(true);
120        Runnable restoreDeadBackends = new Runnable() {
121          @Override public void run() {
122            if (shouldRestore.get()) {
123              restoreDeadBackends(restoreInterval);
124            }
125          }
126        };
127        final ScheduledExecutorService scheduledExecutorService =
128            Executors.newScheduledThreadPool(1,
129                new ThreadFactoryBuilder()
130                    .setDaemon(true)
131                    .setNameFormat("MTCP-DeadBackendRestorer[%s]")
132                    .build());
133        long restoreDelay = restoreInterval.getValue();
134        scheduledExecutorService.scheduleWithFixedDelay(restoreDeadBackends, restoreDelay,
135            restoreDelay, restoreInterval.getUnit().getTimeUnit());
136    
137        return new Command() {
138          @Override public void execute() {
139            shouldRestore.set(false);
140            scheduledExecutorService.shutdownNow();
141            LOG.info("Backend restorer shut down");
142          }
143        };
144      }
145    
146      private static final Logger LOG = Logger.getLogger(MetaPool.class.getName());
147    
148      private synchronized void restoreDeadBackends(Amount<Long, Time> restoreInterval) {
149        for (E backend : backends.keySet()) {
150          try {
151            release(get(backend, restoreInterval));
152          } catch (TimeoutException e) {
153            LOG.warning("Backend restorer failed to revive backend: " + backend + " -> " + e);
154          } catch (ResourceExhaustedException e) {
155            LOG.warning("Backend restorer failed to revive backend: " + backend + " -> " + e);
156          }
157        }
158      }
159    
160      @Override
161      public Connection<T, E> get() throws ResourceExhaustedException, TimeoutException {
162        return get(ObjectPool.NO_TIMEOUT);
163      }
164    
165      @Override
166      public Connection<T, E> get(Amount<Long, Time> timeout)
167          throws ResourceExhaustedException, TimeoutException {
168    
169        return get(loadBalancer.nextBackend(), timeout);
170      }
171    
172      private static class ManagedConnection<T, E> implements Connection<T, E> {
173        private final Connection<T, E> connection;
174        private final ObjectPool<Connection<T, E>> pool;
175    
176        private ManagedConnection(Connection<T, E> connection, ObjectPool<Connection<T, E>> pool) {
177          this.connection = connection;
178          this.pool = pool;
179        }
180    
181        @Override
182        public void close() {
183          connection.close();
184        }
185    
186        @Override
187        public T get() {
188          return connection.get();
189        }
190    
191        @Override
192        public boolean isValid() {
193          return connection.isValid();
194        }
195    
196        @Override
197        public E getEndpoint() {
198          return connection.getEndpoint();
199        }
200    
201        @Override public String toString() {
202          return "ManagedConnection[" + connection.toString() + "]";
203        }
204    
205        void release(boolean remove) {
206          if (remove) {
207            pool.remove(connection);
208          } else {
209            pool.release(connection);
210          }
211        }
212      }
213    
214      private Connection<T, E> get(E backend, Amount<Long, Time> timeout)
215          throws ResourceExhaustedException, TimeoutException {
216        long startNanos = System.nanoTime();
217    
218        ObjectPool<Connection<T, E>> pool = Preconditions.checkNotNull(backends.get(backend));
219    
220        try {
221          Connection<T, E> connection = (timeout.getValue() == 0) ? pool.get() : pool.get(timeout);
222    
223          // BEWARE: We have leased a connection from the underlying pool here and must return it to the
224          // caller so they can later release it.  If we fail to do so, the connection will leak.
225          // Catching intermediate exceptions ourselves and pro-actively returning the connection to the
226          // pool before re-throwing is not a viable option since the return would have to succeed,
227          // forcing us to ignore the timeout passed in.
228    
229          try {
230            loadBalancer.connected(backend, System.nanoTime() - startNanos);
231          } catch (RuntimeException e) {
232            LOG.log(Level.WARNING, "Encountered an exception updating load balancer stats after "
233                                   + "leasing a connection - continuing", e);
234          }
235          return new ManagedConnection<T, E>(connection, pool);
236        } catch (TimeoutException e) {
237          loadBalancer.connectFailed(backend, ConnectionResult.TIMEOUT);
238          throw e;
239        } catch (ResourceExhaustedException e) {
240          loadBalancer.connectFailed(backend, ConnectionResult.FAILED);
241          throw e;
242        }
243      }
244    
245      // Locks to guard mutation of the backends set.
246      private final ReadWriteLock backendsLock = new ReentrantReadWriteLock(true);
247      private final Lock backendsReadLock = backendsLock.readLock();
248      private final Lock backendsWriteLock = backendsLock.writeLock();
249    
250      @Override
251      public void release(Connection<T, E> connection) {
252        release(connection, false);
253      }
254    
255      /**
256       * Equivalent to releasing a Connection with isValid() == false.
257       * @see {@link ObjectPool#remove(Object)}
258       */
259      @Override
260      public void remove(Connection<T, E> connection) {
261        release(connection, true);
262      }
263    
264      private void release(Connection<T, E> connection, boolean remove) {
265        backendsWriteLock.lock();
266        try {
267    
268          if (!(connection instanceof ManagedConnection)) {
269            throw new IllegalArgumentException("Connection not controlled by this connection pool: "
270                                               + connection);
271          }
272          ((ManagedConnection) connection).release(remove);
273    
274          loadBalancer.released(connection.getEndpoint());
275        } finally {
276          backendsWriteLock.unlock();
277        }
278      }
279    
280      @Override
281      public synchronized void close() {
282        stop();
283    
284        backendsReadLock.lock();
285        try {
286          for (ObjectPool<Connection<T, E>> backend : backends.values()) {
287            backend.close();
288          }
289        } finally {
290          backendsReadLock.unlock();
291        }
292      }
293    
294      /**
295       * Stops dead backend restoration attempts.
296       *
297       * <p>TODO(John Sirois): stop functionality is needed to properly implement a close that frees all
298       * pool resources; however having to expose it for subclasses is a hack that solely supports
299       * ServerSetConnectionPool - this might be made cleaner by injecting the dead pool restorer
300       * instead.
301       */
302      protected final void stop() {
303        stopCommand.execute();
304      }
