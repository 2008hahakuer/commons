018    package com.twitter.common.net.pool;
019    
020    import com.google.common.collect.ImmutableSet;
021    
022    /**
023     * A host set that can be monitored for changes.
024     *
025     * @param <T> The type that is used to identify members of the host set.
026     *
027     * @author William Farner
028     */
029    public interface DynamicHostSet<T> {
030    
031      /**
032       * Registers a monitor to receive change notices for this server set as long as this jvm process
033       * is alive.  Blocks until the initial server set can be gathered and delivered to the monitor.
034       * The monitor will be notified if the membership set or parameters of existing members have
035       * changed.
036       *
037       * @param monitor the server set monitor to call back when the host set changes
038       * @throws MonitorException if there is a problem monitoring the host set
039       */
040      public void monitor(final HostChangeMonitor<T> monitor) throws MonitorException;
041    
042      /**
043       * An interface to an object that is interested in receiving notification whenever the host set
044       * changes.
045       */
046      public static interface HostChangeMonitor<T> {
047    
048        /**
049         * Called when either the available set of services changes (when a service dies or a new
050         * instance comes on-line) or when an existing service advertises a status or health change.
051         *
052         * @param hostSet the current set of available ServiceInstances
053         */
054        void onChange(ImmutableSet<T> hostSet);
055      }
056    
057      public static class MonitorException extends Exception {
058        public MonitorException(String msg) {
059          super(msg);
060        }
061    
062        public MonitorException(String msg, Throwable cause) {
063          super(msg, cause);
064        }
065      }
066    }
