018    package com.twitter.common.net.pool;
019    
020    import com.google.common.collect.ImmutableSet;
021    import com.google.common.collect.Sets;
022    
023    import java.util.Set;
024    
025    /**
026     * Util Class for dealing with dynamic sets of hosts.
027     *
028     * @author Florian Leibert
029     * @author Jake Mannix
030     */
031    public class DynamicHostSetUtil {
032    
033      /**
034       * Gets a snapshot of a set of dynamic hosts (e.g. a ServerSet) and returns a readable copy of
035       * the underlying actual endpoints.
036       * @param hostSet
037       * @param <T>
038       * @return
039       * @throws DynamicHostSet.MonitorException
040       */
041      public static <T> ImmutableSet<T> getSnapshot(DynamicHostSet<T> hostSet)
042          throws DynamicHostSet.MonitorException {
043        final Set<T> set = Sets.newHashSet();
044        hostSet.monitor(new DynamicHostSet.HostChangeMonitor<T>() {
045          @Override
046          public void onChange(ImmutableSet<T> hostSet) {
047            set.addAll(hostSet);
048          }
049        });
050        return ImmutableSet.copyOf(set);
051      }
052    }
