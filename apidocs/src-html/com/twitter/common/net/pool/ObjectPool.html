018    package com.twitter.common.net.pool;
019    
020    import com.twitter.common.quantity.Amount;
021    import com.twitter.common.quantity.Time;
022    
023    import java.util.concurrent.TimeoutException;
024    
025    /**
026     * A generic object pool that provides object of a given type for exclusive use by the caller.
027     * Object pools generally pool expensive resources and so offer a {@link #close} method that should
028     * be used to free these resources when the pool is no longer needed.
029     *
030     * @author John Sirois
031     */
032    public interface ObjectPool<T> {
033    
034      /**
035       * Gets a resource potentially blocking for as long as it takes to either create a new one or wait
036       * for one to be {@link #release(Object) released}.  Callers must {@link #release(Object) release}
037       * the connection when they are done with it.
038       *
039       * @return a resource for exclusive use by the caller
040       * @throws ResourceExhaustedException if no resource could be obtained because this pool was
041       *     exhausted
042       * @throws TimeoutException if we timed out while trying to fetch a resource
043       */
044      T get() throws ResourceExhaustedException, TimeoutException;
045    
046      /**
047       * A convenience constant representing a no timeout.
048       */
049      Amount<Long,Time> NO_TIMEOUT = Amount.of(0L, Time.MILLISECONDS);
050    
051      /**
052       * Gets a resource; timing out if there are none available and it takes longer than specified to
053       * create a new one or wait for one to be {@link #release(Object) released}.  Callers must
054       * {@link #release (Object) release} the connection when they are done with it.
055       *
056       * @param timeout the maximum amount of time to wait
057       * @return a resource for exclusive use by the caller
058       * @throws TimeoutException if the specified timeout was reached before a resource became
059       *     available
060       * @throws ResourceExhaustedException if no resource could be obtained because this pool was
061       *     exhausted
062       */
063      T get(Amount<Long, Time> timeout) throws ResourceExhaustedException, TimeoutException;
064    
065      /**
066       * Releases a resource obtained from this pool back into the pool of available resources. It is an
067       * error to release a resource not obtained from this pool.
068       *
069       * @param resource Resource to release.
070       */
071      void release(T resource);
072    
073      /**
074       * Removes a resource obtained from this pool from its available resources.  It is an error to
075       * remove a resource not obtained from this pool.
076       *
077       * @param resource Resource to remove.
078       */
079      void remove(T resource);
080    
081      /**
082       * Disallows further gets from this pool, "closes" all idle objects and any outstanding objects
083       * when they are released.
084       */
085      void close();
086    }
