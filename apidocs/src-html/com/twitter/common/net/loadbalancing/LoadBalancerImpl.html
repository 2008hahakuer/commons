017    package com.twitter.common.net.loadbalancing;
018    
019    import com.google.common.base.Preconditions;
020    import com.google.common.collect.ImmutableSet;
021    import com.google.common.collect.Sets;
022    import com.twitter.common.base.Closure;
023    import com.twitter.common.net.pool.ResourceExhaustedException;
024    import com.twitter.common.net.loadbalancing.LoadBalancingStrategy.ConnectionResult;
025    
026    import java.util.Collection;
027    import java.util.Set;
028    import java.util.logging.Logger;
029    
030    /**
031     * Implementation of a load balancer, that uses a pluggable {@link LoadBalancingStrategy} to define
032     * actual load balancing behavior.  This class handles the responsibility of associating connections
033     * with backends.
034     *
035     * Calls to {@link #connected(Object, long)},
036     * {@link #requestResult(Object, RequestResult, long)}, and {@link #released(Object)} will not
037     * be forwarded for unknown backends/connections.
038     *
039     * @author William Farner
040     */
041    public class LoadBalancerImpl<K> implements LoadBalancer<K> {
042      private static final Logger LOG = Logger.getLogger(LoadBalancerImpl.class.getName());
043    
044      private final LoadBalancingStrategy<K> strategy;
045    
046      private Set<K> offeredBackends = ImmutableSet.of();
047    
048      /**
049       * Creates a new load balancer that will use the given strategy.
050       *
051       * @param strategy Strategy to delegate load balancing work to.
052       */
053      public LoadBalancerImpl(LoadBalancingStrategy<K> strategy) {
054        this.strategy = Preconditions.checkNotNull(strategy);
055      }
056    
057      @Override
058      public synchronized void offerBackends(Set<K> offeredBackends,
059          final Closure<Collection<K>> onBackendsChosen) {
060        this.offeredBackends = ImmutableSet.copyOf(offeredBackends);
061        strategy.offerBackends(offeredBackends, new Closure<Collection<K>>() {
062          @Override public void execute(Collection<K> chosenBackends) {
063            onBackendsChosen.execute(chosenBackends);
064          }
065        });
066      }
067    
068      @Override
069      public synchronized K nextBackend() throws ResourceExhaustedException {
070        return strategy.nextBackend();
071      }
072    
073      @Override
074      public synchronized void connected(K backend, long connectTimeNanos) {
075        Preconditions.checkNotNull(backend);
076    
077        if (!hasBackend(backend)) return;
078    
079        strategy.addConnectResult(backend, ConnectionResult.SUCCESS, connectTimeNanos);
080      }
081    
082      private boolean hasBackend(K backend) {
083        return offeredBackends.contains(backend);
084      }
085    
086      @Override
087      public synchronized void connectFailed(K backend, ConnectionResult result) {
088        Preconditions.checkNotNull(backend);
089        Preconditions.checkNotNull(result);
090        Preconditions.checkArgument(result != ConnectionResult.SUCCESS);
091    
092        if (!hasBackend(backend)) return;
093    
094        strategy.addConnectResult(backend, result, 0);
095      }
096    
097      @Override
098      public synchronized void released(K backend) {
099        Preconditions.checkNotNull(backend);
100    
101        if (!hasBackend(backend)) return;
102    
103        strategy.connectionReturned(backend);
104      }
105    
106      @Override
107      public synchronized void requestResult(K backend, RequestResult result, long requestTimeNanos) {
108        Preconditions.checkNotNull(backend);
109        Preconditions.checkNotNull(result);
110    
111        if (!hasBackend(backend)) return;
112    
113        strategy.addRequestResult(backend, result, requestTimeNanos);
114      }
115    
116      /**
117       * Convenience method to create a new load balancer.
118       *
119       * @param strategy Strategy to use.
120       * @param <K> Backend type.
121       * @return A new load balancer.
122       */
123      public static <K> LoadBalancerImpl<K>
124          create(LoadBalancingStrategy<K> strategy) {
125        return new LoadBalancerImpl<K>(strategy);
126      }
127    }
