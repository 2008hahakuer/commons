016    package com.twitter.common.net.loadbalancing;
017    
018    import java.util.Map;
019    
020    import com.google.common.base.Function;
021    import com.google.common.base.Predicate;
022    import com.google.common.collect.Maps;
023    
024    import com.twitter.common.util.BackoffDecider;
025    
026    /**
027     * A load balancing strategy that extends the functionality of the mark dead strategy by
028     * integrating a hostChecker that allows hosts to transition out of a dead state
029     * if the most recent connection to the host was successful.
030     *
031     * @param <S> typically socket address of a backend host.
032     * @author Krishna Gade
033     */
034    public class MarkDeadStrategyWithHostCheck<S> extends MarkDeadStrategy<S> {
035    
036      /**
037       * LiveHostChecker implements Filter to determine whether a host is alive based on the
038       * result of the most recent connection attempt to that host. It keeps a map of
039       * backend -> last connection result, which gets updated every time someone tries to
040       * add to connection result.
041       */
042      protected static class LiveHostChecker<S> implements Predicate<S> {
043        private final Map<S, ConnectionResult> lastConnectionResult = Maps.newHashMap();
044    
045        /**
046         * Adds the connection result of this backend to the last connection result map.
047         *
048         * @param backend typically the socket address of the backend.
049         * @param result result of what happened when the client tried to connect to this backend.
050         */
051        public void addConnectResult(S backend, ConnectionResult result) {
052          lastConnectionResult.put(backend, result);
053        }
054    
055        /**
056         * Checks if the last connection result for this backend and returns {@code true} if it
057         * was {@link ConnectionResult#SUCCESS} otherwise returns {@code false}.
058         *
059         * @param backend typically the socket address of the backend.
060         */
061        @Override public boolean apply(S backend) {
062          ConnectionResult result = lastConnectionResult.get(backend);
063          return result != null && result == ConnectionResult.SUCCESS;
064        }
065      }
066    
067      // Reference to the host checker we pass to the super class.
068      // We keep it here to avoid casting on every access to it.
069      protected final LiveHostChecker<S> liveHostChecker;
070    
071      /**
072       * Creates a mark dead strategy with the given wrapped strategy and backoff decider factory.
073       * It uses a hostChecker {@link Predicate} that allows hosts to transition out
074       * of a dead state if the most recent connection to the host was successful.
075       *
076       * @param wrappedStrategy one of the implementations of the load balancing strategy.
077       * @param backoffFactory backoff decider factory per host.
078       */
079      public MarkDeadStrategyWithHostCheck(LoadBalancingStrategy<S> wrappedStrategy,
080          Function<S, BackoffDecider> backoffFactory) {
081        super(wrappedStrategy, backoffFactory, new LiveHostChecker<S>());
082        // Casting to LiveHostChecker is safe here as that's the only predicate that we pass to super.
083        this.liveHostChecker = ((LiveHostChecker<S>) hostChecker);
084      }
085    
086    
087      /**
088       * Overrides the base class implementation by adding this connection result to the
089       * host checker.
090       *
091       * @param backendKey typically the socket address of the backend.
092       * @param result result of what happened when the client tried to connect to this backend.
093       * @param connectTimeNanos time took to connect to the backend in nano seconds.
094       */
095      @Override
096      public void addConnectResult(S backendKey, ConnectionResult result, long connectTimeNanos) {
097        liveHostChecker.addConnectResult(backendKey, result);
098        super.addConnectResult(backendKey, result, connectTimeNanos);
099      }
100    }
