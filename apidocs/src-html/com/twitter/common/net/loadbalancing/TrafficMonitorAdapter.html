017    package com.twitter.common.net.loadbalancing;
018    
019    import com.google.common.base.Preconditions;
020    import com.twitter.common.base.Closure;
021    import com.twitter.common.net.pool.ResourceExhaustedException;
022    import com.twitter.common.net.monitoring.TrafficMonitor;
023    
024    import java.util.Collection;
025    import java.util.Set;
026    
027    /**
028     * @author William Farner
029     */
030    public class TrafficMonitorAdapter<K> implements LoadBalancingStrategy<K> {
031      private final LoadBalancingStrategy<K> strategy;
032      private final TrafficMonitor<K> monitor;
033    
034      public TrafficMonitorAdapter(LoadBalancingStrategy<K> strategy, TrafficMonitor<K> monitor) {
035        this.strategy = Preconditions.checkNotNull(strategy);
036        this.monitor = Preconditions.checkNotNull(monitor);
037      }
038    
039      public static <K> TrafficMonitorAdapter<K> create(LoadBalancingStrategy<K> strategy,
040          TrafficMonitor<K> monitor) {
041        return new TrafficMonitorAdapter<K>(strategy, monitor);
042      }
043    
044      @Override
045      public void offerBackends(Set<K> offeredBackends, Closure<Collection<K>> onBackendsChosen) {
046        strategy.offerBackends(offeredBackends, onBackendsChosen);
047      }
048    
049      @Override
050      public K nextBackend() throws ResourceExhaustedException {
051        return strategy.nextBackend();
052      }
053    
054      @Override
055      public void addConnectResult(K key, ConnectionResult result, long connectTimeNanos) {
056        strategy.addConnectResult(key, result, connectTimeNanos);
057        if (result == ConnectionResult.SUCCESS) monitor.connected(key);
058      }
059    
060      @Override
061      public void connectionReturned(K key) {
062        strategy.connectionReturned(key);
063        monitor.released(key);
064      }
065    
066      @Override
067      public void addRequestResult(K key, RequestTracker.RequestResult result, long requestTimeNanos) {
068        strategy.addRequestResult(key, result, requestTimeNanos);
069        monitor.requestResult(key, result, requestTimeNanos);
070      }
071    }
