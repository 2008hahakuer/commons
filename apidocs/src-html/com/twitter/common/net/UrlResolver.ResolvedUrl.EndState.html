052    /**
053     * Class to aid in resolving URLs by following redirects, which can optionally be performed
054     * asynchronously using a thread pool.
055     *
056     * @author William Farner
057     */
058    public class UrlResolver {
059      private static final Logger LOG = Logger.getLogger(UrlResolver.class.getName());
060    
061      private static final String TWITTER_UA = "Twitterbot/0.1";
062      private static final UrlResolverUtil URL_RESOLVER =
063          new UrlResolverUtil(Functions.constant(TWITTER_UA));
064    
065      private static final ExceptionalFunction<String, String, IOException> RESOLVER =
066          new ExceptionalFunction<String, String, IOException>() {
067            @Override public String apply(String url) throws IOException {
068              return URL_RESOLVER.getEffectiveUrl(url, null);
069            }
070          };
071    
072      private static ExceptionalFunction<String, String, IOException>
073          getUrlResolver(final @Nullable ProxyConfig proxyConfig) {
074        if (proxyConfig != null) {
075          return new ExceptionalFunction<String, String, IOException>() {
076            @Override public String apply(String url) throws IOException {
077              return URL_RESOLVER.getEffectiveUrl(url, proxyConfig);
078            }
079          };
080        } else {
081          return RESOLVER;
082        }
083      }

085      private final ExceptionalFunction<String, String, IOException> resolver;
086      private final int maxRedirects;
087    
088      // Tracks the number of active tasks (threads in use).
089      private final Semaphore poolEntrySemaphore;
090      private final Integer threadPoolSize;
091    
092      // Helps with signaling the handler.
093      private final Executor handlerExecutor;
094    
095      // Manages the thread pool and task execution.
096      private ExecutorService executor;
097    
098      // Cache to store resolved URLs.
099      private final Cache<String, String> urlCache = LRUCache.<String, String>builder()
100          .maxSize(10000)
101          .makeSynchronized(true)
102          .build();

104      // Variables to track connection/request stats.
105      private AtomicInteger requestCount = new AtomicInteger(0);
106      private AtomicInteger cacheHits = new AtomicInteger(0);
107      private AtomicInteger failureCount = new AtomicInteger(0);
108      // Tracks the time (in milliseconds) required to resolve URLs.
109      private final PrintableHistogram urlResolutionTimesMs = new PrintableHistogram(
110          1, 5, 10, 25, 50, 75, 100, 150, 200, 250, 300, 500, 750, 1000, 1500, 2000);

112      private final Clock clock;
113      private final BackoffStrategy backoffStrategy;

121      /**
122       * Creates a new asynchronous URL resolver.  A thread pool will be used to resolve URLs, and
123       * resolved URLs will be announced via {@code handler}.
124       *
125       * @param maxRedirects The maximum number of HTTP redirects to follow.
126       * @param threadPoolSize The number of threads to use for resolving URLs.
127       * @param proxyConfig The proxy settings with which to make the HTTP request, or null for the
128       *    default configured proxy.
129       */
130      public UrlResolver(int maxRedirects, int threadPoolSize, @Nullable ProxyConfig proxyConfig) {
131        this(Clock.SYSTEM_CLOCK,
132            new TruncatedBinaryBackoff(Amount.of(100L, Time.MILLISECONDS), Amount.of(1L, Time.SECONDS)),
133            getUrlResolver(proxyConfig), maxRedirects, threadPoolSize);
134      }
135    
136      public UrlResolver(int maxRedirects, int threadPoolSize) {
137        this(maxRedirects, threadPoolSize, null);
138      }
139    
140      private UrlResolver(Clock clock, BackoffStrategy backoffStrategy,
141          ExceptionalFunction<String, String, IOException> resolver, int maxRedirects,
142          @Nullable Integer threadPoolSize) {
143        this.clock = clock;
144        this.backoffStrategy = backoffStrategy;
145        this.resolver = resolver;
146        this.maxRedirects = maxRedirects;
147    
148        if (threadPoolSize != null) {
149          this.threadPoolSize = threadPoolSize;
150          Preconditions.checkState(threadPoolSize > 0);
151          poolEntrySemaphore = new Semaphore(threadPoolSize);
152    
153          // Start up the thread pool.
154          reset();
155    
156          // Executor to send notifications back to the handler.  This also needs to be
157          // a daemon thread.
158          handlerExecutor =
159              Executors.newSingleThreadExecutor(new ThreadFactoryBuilder().setDaemon(true).build());
160        } else {
161          this.threadPoolSize = null;
162          poolEntrySemaphore = null;
163          handlerExecutor = null;
164        }
165      }
166    
167      public Future<ResolvedUrl> resolveUrlAsync(final String url, final ResolvedUrlHandler handler) {
168        Preconditions.checkNotNull(
169            "Asynchronous URL resolution cannot be performed without a valid handler.", handler);
170    
171        try {
172          poolEntrySemaphore.acquire();
173        } catch (InterruptedException e) {
174          LOG.log(Level.SEVERE, "Interrupted while waiting for thread to resolve URL: " + url, e);
175          return null;
176        }
177        final ListenableFutureTask<ResolvedUrl> future =
178            new ListenableFutureTask<ResolvedUrl>(
179              new Callable<ResolvedUrl>() {
180                @Override public ResolvedUrl call() {
181                  return resolveUrl(url);
182                }
183              });
184    
185        future.addListener(new Runnable() {
186          @Override public void run() {
187            try {
188              handler.resolved(future);
189            } finally {
190              poolEntrySemaphore.release();
191            }
192          }
193        }, handlerExecutor);
194    
195        executor.execute(future);
196        return future;
197      }
198    
199      private void logThreadpoolInfo() {
200        LOG.info("Shutting down thread pool, available permits: "
201                 + poolEntrySemaphore.availablePermits());
202        LOG.info("Queued threads? " + poolEntrySemaphore.hasQueuedThreads());
203        LOG.info("Queue length: " + poolEntrySemaphore.getQueueLength());
204      }
205    
206      public void reset() {
207        Preconditions.checkState(threadPoolSize != null);
208        if (executor != null) {
209          Preconditions.checkState(executor.isShutdown(),
210              "The thread pool must be shut down before resetting.");
211          Preconditions.checkState(executor.isTerminated(), "There may still be pending async tasks.");
212        }
213    
214        // Create a thread pool with daemon threads, so that they may be terminated when no
215        // application threads are running.
216        executor = Executors.newFixedThreadPool(threadPoolSize,
217            new ThreadFactoryBuilder().setDaemon(true).setNameFormat("UrlResolver[%d]").build());
218      }
219    
220      /**
221       * Terminates the thread pool, waiting at most {@code waitSeconds} for active threads to complete.
222       * After this method is called, no more URLs may be submitted for resolution.
223       *
224       * @param waitSeconds The number of seconds to wait for active threads to complete.
225       */
226      public void clearAsyncTasks(int waitSeconds) {
227        Preconditions.checkState(threadPoolSize != null,
228            "finish() should not be called on a synchronous URL resolver.");
229    
230        logThreadpoolInfo();
231        executor.shutdown(); // Disable new tasks from being submitted.
232        try {
233          // Wait a while for existing tasks to terminate
234          if (!executor.awaitTermination(waitSeconds, TimeUnit.SECONDS)) {
235            LOG.info("Pool did not terminate, forcing shutdown.");
236            logThreadpoolInfo();
237            List<Runnable> remaining = executor.shutdownNow();
238            LOG.info("Tasks still running: " + remaining);
239            // Wait a while for tasks to respond to being cancelled
240            if (!executor.awaitTermination(waitSeconds, TimeUnit.SECONDS)) {
241              LOG.warning("Pool did not terminate.");
242              logThreadpoolInfo();
243            }
244          }
245        } catch (InterruptedException e) {
246          LOG.log(Level.WARNING, "Interrupted while waiting for threadpool to finish.", e);
247          // (Re-)Cancel if current thread also interrupted
248          executor.shutdownNow();
249          // Preserve interrupt status
250          Thread.currentThread().interrupt();
251        }
252      }

254      /**
255       * Resolves a URL synchronously.
256       *
257       * @param url The URL to resolve.
258       * @return The resolved URL.
259       */
260      public ResolvedUrl resolveUrl(String url) {
261        ResolvedUrl resolvedUrl = new ResolvedUrl();
262        resolvedUrl.setStartUrl(url);
263    
264        String cached = urlCache.get(url);
265        if (cached != null) {
266          cacheHits.incrementAndGet();
267          resolvedUrl.setNextResolve(cached);
268          resolvedUrl.setEndState(EndState.CACHED);
269          return resolvedUrl;
270        }
271    
272        String currentUrl = url;
273        long backoffMs = 0L;
274        String next = null;
275        for (int i = 0; i < maxRedirects; i++) {
276          try {
277            next = resolveOnce(currentUrl);
278    
279            // If there was a 4xx or a 5xx, we''ll get a null back, so we pretend like we never advanced
280            // to allow for a retry within the redirect limit.
281            // TODO(John Sirois): we really need access to the return code here to do the right thing; ie:
282            // retry for internal server errors but probably not for unauthorized
283            if (next == null) {
284              if (i < maxRedirects - 1) { // don't wait if we're about to exit the loop
285                backoffMs = backoffStrategy.calculateBackoffMs(backoffMs);
286                try {
287                  clock.waitFor(backoffMs);
288                } catch (InterruptedException e) {
289                  Thread.currentThread().interrupt();
290                  throw new RuntimeException(
291                      "Interrupted waiting to retry a failed resolution for: " + currentUrl, e);
292                }
293              }
294              continue;
295            }
296    
297            backoffMs = 0L;
298            if (next.equals(currentUrl)) {
299              // We've reached the end of the redirect chain.
300              resolvedUrl.setEndState(EndState.REACHED_LANDING);
301              urlCache.put(url, currentUrl);
302              for (String intermediateUrl : resolvedUrl.getIntermediateUrls()) {
303                urlCache.put(intermediateUrl, currentUrl);
304              }
305              return resolvedUrl;
306            } else if (!url.equals(next)) {
307              resolvedUrl.setNextResolve(next);
308            }
309            currentUrl = next;
310          } catch (IOException e) {
311            LOG.log(Level.INFO, "Failed to resolve url: " + url, e);
312            resolvedUrl.setEndState(EndState.ERROR);
313            return resolvedUrl;
314          }
315        }
316    
317        resolvedUrl.setEndState(next == null || url.equals(currentUrl) ? EndState.ERROR
318            : EndState.REDIRECT_LIMIT);
319        return resolvedUrl;
320      }
321    
322      /**
323       * Resolves a url, following at most one redirect.  Thread-safe.
324       *
325       * @param url The URL to resolve.
326       * @return The result of following the URL through at most one redirect or null if the url could
327       *     not be followed
328       * @throws IOException If an error occurs while resolving the URL.
329       */
330      private String resolveOnce(String url) throws IOException {
331        requestCount.incrementAndGet();
332    
333        String resolvedUrl = urlCache.get(url);
334        if (resolvedUrl != null) {
335          cacheHits.incrementAndGet();
336          return resolvedUrl;
337        }
338    
339        try {
340          long startTimeMs = System.currentTimeMillis();
341          resolvedUrl = resolver.apply(url);
342          if (resolvedUrl == null) {
343            return null;
344          }
345    
346          urlCache.put(url, resolvedUrl);
347    
348          synchronized (urlResolutionTimesMs) {
349            urlResolutionTimesMs.addValue(System.currentTimeMillis() - startTimeMs);
350          }
351          return resolvedUrl;
352        } catch (IOException e) {
353          failureCount.incrementAndGet();
354          throw e;
355        }
356      }
357    
358      @Override
359      public String toString() {
360        return String.format("Cache: %s\nFailed requests: %d,\nResolution Times: %s",
361            urlCache, failureCount.get(),
362            urlResolutionTimesMs.toString());
363      }

365      /**
366       * Class to wrap the result of a URL resolution.
367       */
368      public static class ResolvedUrl {
369        public enum EndState {
370          REACHED_LANDING,
371          ERROR,
372          CACHED,
373          REDIRECT_LIMIT
374        }
375    
376        private String startUrl;
377        private final List<String> resolveChain;
378        private EndState endState;
379    
380        public ResolvedUrl() {
381          resolveChain = Lists.newArrayList();
382        }
383    
384        @VisibleForTesting
385        public ResolvedUrl(EndState endState, String startUrl, String... resolveChain) {
386          this.endState = endState;
387          this.startUrl = startUrl;
388          this.resolveChain = Lists.newArrayList(resolveChain);
389        }
390    
391        public String getStartUrl() {
392          return startUrl;
393        }
394    
395        void setStartUrl(String startUrl) {
396          this.startUrl = startUrl;
397        }
398    
399        /**
400         * Returns the last URL resolved following a redirect chain, or null if the startUrl is a
401         * landing URL.
402         */
403        public String getEndUrl() {
404          return resolveChain.isEmpty() ? null : Iterables.getLast(resolveChain);
405        }
406    
407        void setNextResolve(String endUrl) {
408          this.resolveChain.add(endUrl);
409        }
410    
411        /**
412         * Returns any immediate URLs encountered on the resolution chain.  If the startUrl redirects
413         * directly to the endUrl or they are the same the imtermediate URLs will be empty.
414         */
415        public Iterable<String> getIntermediateUrls() {
416          return resolveChain.size() <= 1 ? ImmutableList.<String>of()
417              : resolveChain.subList(0, resolveChain.size() - 1);
418        }
419    
420        public EndState getEndState() {
421          return endState;
422        }
423    
424        void setEndState(EndState endState) {
425          this.endState = endState;
426        }
427    
428        public String toString() {
429          return String.format("%s -> %s [%s, %d redirects]",
430              startUrl, Joiner.on(" -> ").join(resolveChain), endState, resolveChain.size());
431        }
432      }
433    
434      /**
435       * Interface to use for notifying the caller of resolved URLs.
436       */
437      public interface ResolvedUrlHandler {
438        /**
439         * Signals that a URL has been resolved to its target.  The implementation of this method must
440         * be thread safe.
441         *
442         * @param future The future that has finished resolving a URL.
443         */
444        public void resolved(Future<ResolvedUrl> future);
445      }
