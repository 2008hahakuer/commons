017    package com.twitter.common.text.combiner;
018    
019    import com.twitter.common.text.token.TokenProcessor;
020    import com.twitter.common.text.token.TokenStream;
021    import com.twitter.common.text.token.attribute.CharSequenceTermAttribute;
022    import com.twitter.common.text.token.attribute.TokenType;
023    import com.twitter.common.text.token.attribute.TokenTypeAttribute;
024    
025    /**
026     * Combiner that looks ahead to the next token and combines it with the current token based on
027     * specified conditions.
028     */
029    public abstract class LookAheadTokenCombiner extends TokenProcessor {
030      private TokenType type = TokenType.TOKEN;
031    
032      private final CharSequenceTermAttribute inputTermAttr;
033    
034      private final CharSequenceTermAttribute termAttr;
035      private final TokenTypeAttribute typeAttr;
036    
037      private State nextState = null;
038    
039      public LookAheadTokenCombiner(TokenStream inputStream) {
040        super(inputStream);
041    
042        inputTermAttr = inputStream.getAttribute(CharSequenceTermAttribute.class);
043    
044        termAttr = getAttribute(CharSequenceTermAttribute.class);
045        typeAttr = getAttribute(TokenTypeAttribute.class);
046      }
047    
048      @Override
049      public boolean incrementToken() {
050        if (nextState != null) {
051          restoreState(nextState);
052          nextState = null;
053          return true;
054        }
055    
056        if (!getInputStream().incrementToken()) {
057          return false;
058        }
059    
060        restoreState(getInputStream().captureState());
061    
062        // check if current token is contracted word
063        if (canBeCombinedWithNextToken(inputTermAttr.getTermCharSequence())) {
064          // check if the next token is comma or not.
065          if (getInputStream().incrementToken()) {
066            nextState = getInputStream().captureState();
067    
068            CharSequence term = inputTermAttr.getTermCharSequence();
069            if (canBeCombinedWithPreviousToken(term)) {
070              // combine
071              termAttr.setLength(termAttr.getLength() + term.length());
072              typeAttr.setType(type);
073    
074              nextState = null;
075            }
076          }
077        }
078    
079        return true;
080      }
081    
082      protected void setType(TokenType type) {
083        this.type = type;
084      }
085    
086      public abstract boolean canBeCombinedWithNextToken(CharSequence term);
087    
088      public abstract boolean canBeCombinedWithPreviousToken(CharSequence term);
089    }
