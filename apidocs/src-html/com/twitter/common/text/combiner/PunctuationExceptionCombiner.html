001    package com.twitter.common.text.combiner;
002    
003    import com.twitter.common.text.extractor.RegexExtractor;
004    import com.twitter.common.text.token.TokenStream;
005    import com.twitter.common.text.token.attribute.TokenType;
006    
007    import java.util.regex.Pattern;
008    
009    public class PunctuationExceptionCombiner extends ExtractorBasedTokenCombiner {
010    
011      private static final String PUNCTUATION_EXCEPTIONS_CHARS = "â™¥";
012      private static final String PUNCTUATION_EXCEPTION_REGEX =
013          "[" + PUNCTUATION_EXCEPTIONS_CHARS + "]+";
014      private static final Pattern PUNCTUATION_EXCEPTIONS_PATTERN =
015          Pattern.compile(PUNCTUATION_EXCEPTION_REGEX);
016    
017      protected PunctuationExceptionCombiner(TokenStream inputStream, Pattern exceptionsPattern) {
018        super(inputStream);
019        setExtractor(new RegexExtractor.Builder().setRegexPattern(exceptionsPattern, 0, 0)
020            .build());
021        setType(TokenType.TOKEN);
022      }
023    
024      public static class Builder {
025        private String exceptionChars = null;
026        private TokenStream inputStream;
027    
028        public Builder(TokenStream inputStream) {
029          this.inputStream = inputStream;
030        }
031    
032        /**
033         * Add additional exception chars. For example, to add . and ! to the list of
034         * non-punctuation chars, additionalChars should be ".!"
035         *
036         * @param additionalChars Additional characters that should not be considered punctuation
037         * @return PunctuationExceptionCombiner builder instance
038         */
039        public Builder addExceptionChars(String additionalChars) {
040          if (exceptionChars == null) {
041            exceptionChars = PUNCTUATION_EXCEPTIONS_CHARS + additionalChars;
042          } else {
043            exceptionChars += additionalChars;
044          }
045          return this;
046        }
047    
048        public PunctuationExceptionCombiner build() {
049          Pattern exceptionsPattern = PUNCTUATION_EXCEPTIONS_PATTERN;
050          if (exceptionChars != null) {
051            exceptionsPattern = Pattern.compile("[" + exceptionChars + "]+");
052          }
053          return new PunctuationExceptionCombiner(inputStream, exceptionsPattern);
054        }
055      }
056    }
