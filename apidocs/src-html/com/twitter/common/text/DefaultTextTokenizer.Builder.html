017    package com.twitter.common.text;
018    
019    import java.util.List;
020    
021    import com.google.common.base.Preconditions;
022    
023    import com.twitter.common.text.combiner.DotContractedTokenCombiner;
024    import com.twitter.common.text.combiner.EmoticonTokenCombiner;
025    import com.twitter.common.text.combiner.HashtagTokenCombiner;
026    import com.twitter.common.text.combiner.PossessiveTokenCombiner;
027    import com.twitter.common.text.combiner.StockTokenCombiner;
028    import com.twitter.common.text.combiner.URLTokenCombiner;
029    import com.twitter.common.text.combiner.UserNameTokenCombiner;
030    import com.twitter.common.text.example.TokenizerUsageExample;
031    import com.twitter.common.text.filter.PunctuationFilter;
032    import com.twitter.common.text.token.TokenStream;
033    import com.twitter.common.text.token.TokenizedCharSequence;
034    import com.twitter.common.text.token.TokenizedCharSequenceStream;
035    import com.twitter.common.text.tokenizer.LatinTokenizer;
036    
037    /**
038     * Default implementation of a tokenizer for processing tweets. For sample usage, please consult
039     * annotated code example in {@link TokenizerUsageExample}.
040     */
041    public class DefaultTextTokenizer {
042      private TokenStream tokenizationStream =
043          new TokenizedCharSequenceStream(applyDefaultChain(new LatinTokenizer.Builder().build()));
044    
045      // use Builder.
046      private DefaultTextTokenizer() { }
047    
048      public static final TokenStream applyDefaultChain(TokenStream tokenizer) {
049        return
050          // combine stock symbol
051          new StockTokenCombiner(
052            // combine emoticon like :) :-D
053            new EmoticonTokenCombiner(
054              // combine Mr/Mrs/Dr/Inc + .
055              new DotContractedTokenCombiner(
056                // combine possessive form (e.g., apple's)
057                new PossessiveTokenCombiner(
058                  // combine URL
059                  new URLTokenCombiner(
060                    // combine # + hashtag
061                    new HashtagTokenCombiner(
062                      // combine @ + user name
063                      new UserNameTokenCombiner(tokenizer)))))));
064      }
065    
066      /**
067       * Returns {@code TokenStream} to tokenize a text.
068       *
069       * @return {@code TokenStream} to tokenize the text
070       */
071      public TokenStream getDefaultTokenStream() {
072        return tokenizationStream;
073      }
074    
075      /**
076       * Tokenizes a {@code CharSequence}, and returns a {@code TokenizedCharSequence} as a result.
077       *
078       * @param input text to be tokenized
079       * @return {@code TokenizedCharSequence} instance
080       */
081      public TokenizedCharSequence tokenize(CharSequence input) {
082        Preconditions.checkNotNull(input);
083        return TokenizedCharSequence.createFrom(input, getDefaultTokenStream());
084      }
085    
086      /**
087       * Tokenizes a {@code CharSequence} into a list of Strings.
088       *
089       * @param input text to be tokenized
090       * @return a list of tokens as String objects
091       */
092      public List<String> tokenizeToStrings(CharSequence input) {
093        Preconditions.checkNotNull(input);
094        TokenStream tokenizer = getDefaultTokenStream();
095        tokenizer.reset(input);
096        return tokenizer.toStringList();
097      }
098    
099      /**
100       * Builder for {@link DefaultTextTokenizer}
101       */
102      public static final class Builder {
103        private final DefaultTextTokenizer tokenizer = new DefaultTextTokenizer();
104        private boolean keepPunctuation = false;
105    
106        /**
107         * Set to {@code true} to retain punctuation in output.
108         *
109         * @param keepPunctuation {@code true} to retain punctuation
110         * @return this Builder
111         */
112        public Builder setKeepPunctuation(boolean keepPunctuation) {
113          this.keepPunctuation = keepPunctuation;
114          return this;
115        }
116    
117        public DefaultTextTokenizer build() {
118          if (!keepPunctuation) {
119            tokenizer.tokenizationStream =
120                new PunctuationFilter(tokenizer.tokenizationStream);
121          }
122    
123          return tokenizer;
124        }
125      }
126    }
