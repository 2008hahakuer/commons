017    package com.twitter.common.text.tokenizer;
018    
019    import java.nio.CharBuffer;
020    import java.util.List;
021    import java.util.regex.Matcher;
022    import java.util.regex.Pattern;
023    
024    import com.google.common.base.Preconditions;
025    import com.google.common.collect.Lists;
026    
027    import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
028    
029    import com.twitter.common.text.token.TokenStream;
030    import com.twitter.common.text.token.attribute.CharSequenceTermAttribute;
031    import com.twitter.common.text.token.attribute.TokenType;
032    import com.twitter.common.text.token.attribute.TokenTypeAttribute;
033    
034    /**
035     * Tokenizes text based on regular expressions of word delimiters and punctuation characters.
036     */
037    public class RegexTokenizer extends TokenStream {
038      private Pattern delimiterPattern;
039      private int punctuationGroup = 0;
040      private boolean keepPunctuation = false;
041    
042      private List<CharBuffer> tokens;
043      private List<TokenType> tokenTypes;
044      private int tokenIndex = 0;
045    
046      private CharSequenceTermAttribute termAttr;
047      private OffsetAttribute offsetAttr;
048      private TokenTypeAttribute typeAttr;
049    
050      // please use Builder instead.
051      protected RegexTokenizer() {
052        termAttr = addAttribute(CharSequenceTermAttribute.class);
053        offsetAttr = addAttribute(OffsetAttribute.class);
054        typeAttr = addAttribute(TokenTypeAttribute.class);
055      }
056    
057      protected void setDelimiterPattern(Pattern delimiterPattern) {
058        this.delimiterPattern = delimiterPattern;
059      }
060    
061      protected void setPunctuationGroupInDelimiterPattern(int group) {
062        this.punctuationGroup = group;
063      }
064    
065      protected void setKeepPunctuation(boolean keepPunctuation) {
066        this.keepPunctuation = keepPunctuation;
067      }
068    
069      @Override
070      public boolean incrementToken() {
071        if (tokenIndex >= tokens.size()) {
072          return false;
073        }
074    
075        CharBuffer token = tokens.get(tokenIndex);
076    
077        termAttr.setOffset(token.position());
078        termAttr.setLength(token.limit() - token.position());
079        offsetAttr.setOffset(token.position(), token.limit());
080        typeAttr.setType(tokenTypes.get(tokenIndex));
081    
082        tokenIndex++;
083    
084        return true;
085      }
086    
087      @Override
088      public void reset(CharSequence input) {
089        // reset termAttr
090        termAttr.setCharSequence(input);
091    
092        // reset tokens
093        tokens = Lists.newArrayList();
094        tokenTypes = Lists.newArrayList();
095    
096        Matcher matcher = delimiterPattern.matcher(input);
097        int lastMatch = 0;
098    
099        while (matcher.find()) {
100          if (matcher.start() != lastMatch) {
101            tokens.add(CharBuffer.wrap(input, lastMatch, matcher.start()));
102            tokenTypes.add(TokenType.TOKEN);
103          }
104    
105          if (keepPunctuation && matcher.start(punctuationGroup) >= 0) {
106            tokens.add(CharBuffer.wrap(input, matcher.start(punctuationGroup),
107                matcher.end(punctuationGroup)));
108            tokenTypes.add(TokenType.PUNCTUATION);
109          }
110    
111          lastMatch = matcher.end();
112        }
113        if (lastMatch < input.length()) {
114          tokens.add(CharBuffer.wrap(input, lastMatch, input.length()));
115          tokenTypes.add(TokenType.TOKEN);
116        }
117    
118        // reset tokenIndex
119        tokenIndex = 0;
120      }
121    
122      /**
123       * Builder for RegexTokenizer.
124       *
125       * @author Keita Fujii
126       */
127      public static final class Builder extends AbstractBuilder<RegexTokenizer, Builder> {
128        public Builder() {
129          super(new RegexTokenizer());
130        }
131      }
132    
133      public abstract static class
134          AbstractBuilder<N extends RegexTokenizer, T extends AbstractBuilder<N, T>> {
135        private final N tokenizer;
136    
137        protected AbstractBuilder(N tokenizer) {
138          this.tokenizer = Preconditions.checkNotNull(tokenizer);
139        }
140    
141        @SuppressWarnings("unchecked")
142        protected T self() {
143          return (T) this;
144        }
145    
146        /**
147         * Sets the Regex pattern of the delimiter.
148         *
149         * An input text is tokenized by the CharSequence
150         * specified by this pattern.
151         *
152         * @param delimiterPattern Regex pattern of delimiter.
153         * @return this Builder object
154         */
155        public T setDelimiterPattern(Pattern delimiterPattern) {
156          tokenizer.setDelimiterPattern(delimiterPattern);
157          return self();
158        }
159    
160        /**
161         * Sets the ID of the group in delimiterPattern that should
162         * be handled as punctuation.
163         * For example, you can set delimiterPattern as "([.,])\\s+"
164         * and punctuationGroup as 1 in order to detect comma
165         * and period as punctuations.
166         *
167         * @param group group ID of punctuation in delimiterPattern.
168         * @return this Builder object
169         */
170        public T setPunctuationGroupInDelimiterPattern(int group) {
171          tokenizer.setPunctuationGroupInDelimiterPattern(group);
172          return self();
173        }
174    
175        /**
176         * Specifies whether to keep punctuations (which is specified
177         * by delimiterPattern and punctuationGroupInDelimiterPattern)
178         * in the output token stream.
179         *
180         * @param keepPunctuation true to keep delimiters. false otherwise.
181         * @return this Builder object.
182         */
183        public T setKeepPunctuation(boolean keepPunctuation) {
184          tokenizer.setKeepPunctuation(keepPunctuation);
185          return self();
186        }
187    
188        public N build() {
189          return tokenizer;
190        }
191      }
192    }
