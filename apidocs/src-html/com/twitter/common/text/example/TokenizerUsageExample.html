017    package com.twitter.common.text.example;
018    
019    import java.util.Iterator;
020    
021    import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
022    import org.apache.lucene.util.Attribute;
023    
024    import com.twitter.common.text.DefaultTextTokenizer;
025    import com.twitter.common.text.token.TokenStream;
026    import com.twitter.common.text.token.TokenizedCharSequence;
027    import com.twitter.common.text.token.TokenizedCharSequence.Token;
028    import com.twitter.common.text.token.attribute.CharSequenceTermAttribute;
029    import com.twitter.common.text.token.attribute.TokenTypeAttribute;
030    
031    /**
032     * Annotated example illustrating major features of {@link DefaultTextTokenizer}.
033     */
034    public class TokenizerUsageExample {
035      private static final String[] famousTweets = {
036          // http://twitter.com/#!/BarackObama/status/992176676
037          "We just made history. All of this happened because you gave your time, talent and passion."
038            + "All of this happened because of you. Thanks",
039          // http://twitter.com/#!/jkrums/status/1121915133
040          "http://twitpic.com/135xa - There's a plane in the Hudson."
041            + " I'm on the ferry going to pick up the people. Crazy.",
042          // http://twitter.com/#!/carlbildt/status/73498110629904384
043          "@khalidalkhalifa Trying to get in touch with you on an issue.",
044          // http://twitter.com/#!/SHAQ/status/75996821360615425
045          "im retiring Video: http://bit.ly/kvLtE3 #ShaqRetires"
046      };
047    
048      public static void main(String[] args) {
049        // This is the canonical way to create a token stream.
050        DefaultTextTokenizer tokenizer =
051            new DefaultTextTokenizer.Builder().setKeepPunctuation(true).build();
052        TokenStream stream = tokenizer.getDefaultTokenStream();
053    
054        // We're going to ask the token stream what type of attributes it makes available. "Attributes"
055        // can be understood as "annotations" on the original text.
056        System.out.println("Attributes available:");
057        Iterator<Class<? extends Attribute>> iter = stream.getAttributeClassesIterator();
058        while (iter.hasNext()) {
059          Class<? extends Attribute> c = iter.next();
060          System.out.println(" - " + c.getCanonicalName());
061        }
062        System.out.println("");
063    
064        // We're now going to iterate through a few tweets and tokenize each in turn.
065        for (String tweet : famousTweets) {
066          // We're first going to demonstrate the "token-by-token" method of consuming tweets.
067          System.out.println("Processing: " + tweet);
068          // Reset the token stream to process new input.
069          stream.reset(tweet);
070    
071          // Now we're going to consume tokens from the stream.
072          int tokenCnt = 0;
073          while (stream.incrementToken()) {
074            // CharSequenceTermAttribute holds the actual token text. This is preferred over
075            // TermAttribute because it avoids creating new String objects.
076            CharSequenceTermAttribute termAttribute = stream
077                .getAttribute(CharSequenceTermAttribute.class);
078    
079            // OffsetAttribute holds indexes into the original String that the current token occupies.
080            // The startOffset is character position is inclusive, the endOffset is exclusive.
081            OffsetAttribute offsetAttribute = stream.getAttribute(OffsetAttribute.class);
082    
083            // TokenTypeAttribute holds, as you'd expect, the type of the token.
084            TokenTypeAttribute typeAttribute = stream.getAttribute(TokenTypeAttribute.class);
085            System.out.println(String.format("token %2d (%3d, %3d) type: %12s, token: '%s'",
086                    tokenCnt, offsetAttribute.startOffset(), offsetAttribute.endOffset(),
087                    typeAttribute.getType().name, termAttribute.getTermCharSequence()));
088            tokenCnt++;
089          }
090          System.out.println("");
091    
092          // We're now going to demonstrate the TokenizedCharSequence API.
093          // This should produce exactly the same result as above.
094          tokenCnt = 0;
095          System.out.println("Processing: " + tweet);
096          TokenizedCharSequence tokSeq = tokenizer.tokenize(tweet);
097          for (Token tok : tokSeq.getTokens()) {
098            System.out.println(String.format("token %2d (%3d, %3d) type: %12s, token: '%s'",
099                tokenCnt, tok.getOffset(), tok.getOffset() + tok.getLength(),
100                tok.getType().name, tok.getTerm()));
101            tokenCnt++;
102          }
103          System.out.println("");
104        }
105      }
106    }
