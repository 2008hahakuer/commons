017    package com.twitter.common.text.token;
018    
019    import java.util.List;
020    
021    import com.google.common.base.Preconditions;
022    import com.google.common.collect.Lists;
023    
024    import org.apache.lucene.util.AttributeSource;
025    
026    import com.twitter.common.text.example.TokenizerUsageExample;
027    import com.twitter.common.text.token.attribute.CharSequenceTermAttribute;
028    
029    /**
030     * Abstraction to enumerate a sequence of tokens. This class represents the central abstraction in
031     * Twitter's text processing library, and is similar to Lucene's TokenStream, with the following
032     * exceptions:
033     *
034     * <ul>
035     * <li>This class assumes that the input text is a {@link CharSequence}.
036     * <li>Calls support chaining.
037     * <li>Instances are reusable.
038     * </ul>
039     *
040     * For an annotated example of how this class is used in practice, refer to
041     * {@link TokenizerUsageExample}.
042     */
043    public abstract class TokenStream extends AttributeSource {
044      /**
045       * Constructs a {@code TokenStream} using the default attribute factory.
046       */
047      public TokenStream() {
048        super();
049      }
050    
051      /**
052       * Constructs a {@code TokenStream} using the supplied {@code AttributeFactory} for creating new
053       * {@code Attribute} instances.
054       *
055       * @param factory attribute factory
056       */
057      protected TokenStream(AttributeSource.AttributeFactory factory) {
058        super(factory);
059      }
060    
061      /**
062       * Constructs a {@code TokenStream} that uses the same attributes as the supplied one.
063       *
064       * @param input attribute source
065       */
066      protected TokenStream(AttributeSource input) {
067        super(input);
068      }
069    
070      /**
071       * Consumers call this method to advance the stream to the next token.
072       *
073       * @return false for end of stream; true otherwise
074       */
075      public abstract boolean incrementToken();
076    
077      /**
078       * Resets this {@code TokenStream} (and also downstream tokens if they exist) to parse a new
079       * input.
080       *
081       * @param input new text to parse.
082       */
083      public abstract void reset(CharSequence input);
084    
085      /**
086       * Converts this token stream into a list of {@code Strings}.
087       *
088       * @return the contents of the token stream as a list of {@code Strings}.
089       */
090      public List<String> toStringList() {
091        List<String> tokens = Lists.newArrayList();
092    
093        if (hasAttribute(CharSequenceTermAttribute.class)) {
094          CharSequenceTermAttribute termAttr = getAttribute(CharSequenceTermAttribute.class);
095    
096          while (incrementToken()) {
097            tokens.add(termAttr.getTermString());
098          }
099        } else {
100          throw new UnsupportedOperationException("This instance does not support toStringList()"
101              + " because it does not support CharSequenceTermAttribute.");
102        }
103    
104        return tokens;
105      }
106    
107      /**
108       * Searches and returns an instance of a specified class in this TokenStream chain.
109       *
110       * @param cls class to search for
111       * @return instance of the class {@code cls} if found or {@code null} if not found
112       */
113      public <T extends TokenStream> T getInstanceOf(Class<T> cls) {
114        Preconditions.checkNotNull(cls);
115        if (cls.isInstance(this)) {
116          return cls.cast(this);
117        }
118        return null;
119      }
120    }
