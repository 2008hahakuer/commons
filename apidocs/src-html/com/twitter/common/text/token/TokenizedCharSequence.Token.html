017    package com.twitter.common.text.token;
018    
019    import java.nio.CharBuffer;
020    import java.util.Collections;
021    import java.util.List;
022    import java.util.Map;
023    
024    import com.google.common.base.Preconditions;
025    import com.google.common.collect.Lists;
026    import com.google.common.collect.Maps;
027    
028    import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
029    
030    import com.twitter.common.text.token.attribute.CharSequenceTermAttribute;
031    import com.twitter.common.text.token.attribute.PartOfSpeechAttribute;
032    import com.twitter.common.text.token.attribute.TokenGroupAttribute;
033    import com.twitter.common.text.token.attribute.TokenType;
034    import com.twitter.common.text.token.attribute.TokenTypeAttribute;
035    
036    /**
037     * Keeps the original text as well as its tokenized tokens.
038     */
039    public class TokenizedCharSequence implements CharSequence {
040      public static final class Token {
041        public static final int DEFAULT_PART_OF_SPEECH = -1;
042    
043        private final CharBuffer term;
044        private final TokenType type;
045        private final int pos;
046    
047        protected Token(CharBuffer term, TokenType type) {
048          this(term, type, DEFAULT_PART_OF_SPEECH);
049        }
050    
051        protected Token(CharBuffer term, TokenType type, int pos) {
052          this.term = term;
053          this.type = type;
054          this.pos = pos;
055        }
056    
057        @Override
058        public String toString() {
059          return term.toString();
060        }
061    
062        public CharSequence getTerm() {
063          return term;
064        }
065    
066        public int getOffset() {
067          return term.position();
068        }
069    
070        public int getLength() {
071          return term.limit() - term.position();
072        }
073    
074        public TokenType getType() {
075          return type;
076        }
077    
078        public int getPartOfSpeech() {
079          return pos;
080        }
081      }
082    
083      private final CharSequence term;
084      private final List<Token> tokens;
085    
086      private List<String> strTokens = null;
087      private Map<TokenType, List<Token>> typeToTokensMap = null;
088    
089      private String strValue = null;
090      private int hashCode;
091      private boolean hashCodeCalced = false;
092    
093      protected TokenizedCharSequence(CharSequence text, List<Token> tokens) {
094        this.tokens = Collections.unmodifiableList(tokens);
095        this.term = text;
096      }
097    
098      @Override
099      public char charAt(int index) {
100        return term.charAt(index);
101      }
102    
103      @Override
104      public int length() {
105        return term.length();
106      }
107    
108      @Override
109      public CharSequence subSequence(int fromIndex, int toIndex) {
110        return term.subSequence(fromIndex, toIndex);
111      }
112    
113      @Override
114      public String toString() {
115        if (strValue == null) {
116          strValue = term.toString();
117        }
118        return strValue;
119      }
120    
121      @Override
122      public boolean equals(Object obj) {
123        return (obj != null)
124          && (obj instanceof TokenizedCharSequence)
125          && ((TokenizedCharSequence) obj).term.toString().equals(this.term.toString());
126      }
127    
128      @Override
129      public int hashCode() {
130        if (!hashCodeCalced) {
131          hashCode = term.toString().hashCode();
132          hashCodeCalced = true;
133        }
134        return hashCode;
135      }
136    
137      /**
138       * Returns all tokens.
139       *
140       * @return a list of tokens as CharBuffer objects
141       */
142      public List<Token> getTokens() {
143        return tokens;
144      }
145    
146      /**
147       * Returns all tokens as String.
148       *
149       * @return a list of tokens as String objects
150       */
151      public List<String> getTokenStrings() {
152        if (strTokens == null) {
153          // lazy initialization
154          strTokens = Lists.newArrayListWithCapacity(tokens.size());
155          for (Token token : tokens) {
156            strTokens.add(token.getTerm().toString());
157          }
158        }
159    
160        return strTokens;
161      }
162    
163      /**
164       * Returns tokens of one or more specified types.
165       *
166       * @param types token type(s)
167       * @return tokens of the specified type(s)
168       */
169      public List<Token> getTokensOf(TokenType... types) {
170        if (typeToTokensMap == null) {
171          // lazy initialization
172          typeToTokensMap = Maps.newHashMap();
173    
174          for (Token token : tokens) {
175            List<Token> subtokens = typeToTokensMap.get(token.getType());
176            if (subtokens == null) {
177              subtokens = Lists.newArrayList(token);
178              typeToTokensMap.put(token.getType(), subtokens);
179            } else {
180              subtokens.add(token);
181            }
182          }
183        }
184    
185        if (types.length == 1) {
186          return typeToTokensMap.get(types[0]);
187        }
188    
189        List<Token> subtokens = Lists.newArrayList();
190        for (TokenType type : types) {
191          subtokens.addAll(typeToTokensMap.get(type));
192        }
193        return subtokens;
194      }
195    
196      /**
197       * Returns tokens of one or more specified types as Strings.
198       *
199       * @param types token type(s)
200       * @return list of tokens of specified type(s) as String objects
201       */
202      public List<String> getTokenStringsOf(TokenType... types) {
203        List<String> strSubtokens = Lists.newArrayListWithCapacity(tokens.size());
204        for (Token token : getTokensOf(types)) {
205          strSubtokens.add(token.getTerm().toString());
206        }
207    
208        return strSubtokens;
209      }
210    
211      public static final class Builder {
212        private final CharSequence origText;
213        private final List<Token> tokens;
214    
215        public Builder(CharSequence originalText) {
216          Preconditions.checkNotNull(originalText);
217          this.origText = originalText;
218          tokens = Lists.newArrayList();
219        }
220    
221        public Builder addToken(int offset, int length) {
222          addToken(offset, length, TokenType.TOKEN);
223    
224          return this;
225        }
226    
227        public Builder addToken(int offset, int length, TokenType type) {
228          addToken(offset, length, type, PartOfSpeechAttribute.UNKNOWN);
229    
230          return this;
231        }
232    
233        public Builder addToken(int offset, int length, TokenType type, int pos) {
234          Preconditions.checkArgument(offset >= 0);
235          Preconditions.checkArgument(length >= 0);
236          Preconditions.checkNotNull(type);
237          tokens.add(new Token(CharBuffer.wrap(origText, offset, offset + length), type, pos));
238    
239          return this;
240        }
241    
242        public TokenizedCharSequence build() {
243          return new TokenizedCharSequence(origText, tokens);
244        }
245      }
246    
247      public static final TokenizedCharSequence createFrom(CharSequence text,
248          TokenStream tokenizer) {
249        tokenizer.reset(text);
250        OffsetAttribute offsetAttr = tokenizer.getAttribute(OffsetAttribute.class);
251        TokenTypeAttribute typeAttr = tokenizer.getAttribute(TokenTypeAttribute.class);
252        PartOfSpeechAttribute posAttr = null;
253        if (tokenizer.hasAttribute(PartOfSpeechAttribute.class)) {
254          posAttr = tokenizer.getAttribute(PartOfSpeechAttribute.class);
255        }
256    
257        TokenizedCharSequence.Builder builder = new TokenizedCharSequence.Builder(text);
258        while (tokenizer.incrementToken()) {
259          builder.addToken(offsetAttr.startOffset(),
260                offsetAttr.endOffset() - offsetAttr.startOffset(),
261                typeAttr.getType(),
262                posAttr == null ? Token.DEFAULT_PART_OF_SPEECH : posAttr.getPOS());
263        }
264    
265        return builder.build();
266      }
267    
268      public static final List<TokenizedCharSequence> createFromTokenGroupsIn(
269          TokenStream stream) {
270        OffsetAttribute offsetAttr = stream.getAttribute(OffsetAttribute.class);
271        CharSequenceTermAttribute termAttr = stream.getAttribute(CharSequenceTermAttribute.class);
272        TokenGroupAttribute groupAttr = stream.getAttribute(TokenGroupAttribute.class);
273    
274        List<TokenizedCharSequence> groups = Lists.newArrayList();
275        while (stream.incrementToken()) {
276          Builder builder = new Builder(termAttr.getTermCharSequence());
277    
278          TokenStream groupStream = groupAttr.getTokenGroupStream();
279          OffsetAttribute groupOffsetAttr = groupStream.getAttribute(OffsetAttribute.class);
280          TokenTypeAttribute typeAttr = groupStream.getAttribute(TokenTypeAttribute.class);
281    
282          while (groupStream.incrementToken()) {
283            builder.addToken(groupOffsetAttr.startOffset() - offsetAttr.startOffset(),
284                             groupOffsetAttr.endOffset() - groupOffsetAttr.startOffset(),
285                             typeAttr.getType());
286          }
287    
288          groups.add(builder.build());
289        }
290    
291        return groups;
292      }
