017    package com.twitter.common.text.token.attribute;
018    
019    import java.io.IOException;
020    import java.io.NotSerializableException;
021    import java.io.ObjectStreamException;
022    import java.util.Collections;
023    import java.util.List;
024    
025    import com.google.common.collect.Lists;
026    
027    import org.apache.lucene.util.AttributeImpl;
028    import org.apache.lucene.util.AttributeSource;
029    
030    import com.twitter.common.text.token.TokenGroupStream;
031    
032    /**
033     * Implementation of {@link TokenGroupAttribute}.
034     * <p>
035     * Note that this class explicitly suppresses the ability for instance to be serialized, inherited
036     * via {@link AttributeImpl}.
037     */
038    public class TokenGroupAttributeImpl extends AttributeImpl implements TokenGroupAttribute {
039      private static final long serialVersionUID = 0L;
040    
041      private List<AttributeSource.State> states = Collections.emptyList();
042      private AttributeSource attributeSource;
043    
044      @Override
045      public void clear() {
046        states = Collections.emptyList();
047      }
048    
049      @Override
050      public void copyTo(AttributeImpl obj) {
051        if (obj instanceof TokenGroupAttributeImpl) {
052          TokenGroupAttributeImpl attr = (TokenGroupAttributeImpl) obj;
053          attr.setAttributeSource(attributeSource);
054          attr.setStates(Lists.newArrayList(states));
055        }
056      }
057    
058      @Override
059      public boolean equals(Object obj) {
060        return (obj instanceof TokenGroupAttributeImpl)
061          && ((TokenGroupAttributeImpl) obj).states.equals(states);
062      }
063    
064      @Override
065      public int hashCode() {
066        return states.hashCode();
067      }
068    
069      @Override
070      public boolean isEmpty() {
071        return states.isEmpty();
072      }
073    
074      @Override
075      public int size() {
076        return states.size();
077      }
078    
079      public void setStates(List<AttributeSource.State> states) {
080        this.states = states;
081      }
082    
083      public void setAttributeSource(AttributeSource source) {
084        this.attributeSource = source;
085      }
086    
087      @Override
088      public TokenGroupStream getTokenGroupStream() {
089        return new TokenGroupStream(attributeSource, states);
090      }
091    
092      // Explicitly suppress ability to serialize.
093      private void writeObject(java.io.ObjectOutputStream out) throws IOException {
094        throw new NotSerializableException();
095      }
096    
097      private void readObject(java.io.ObjectInputStream in)
098          throws IOException, ClassNotFoundException {
099        throw new NotSerializableException();
100      }
101    
102      private void readObjectNoData() throws ObjectStreamException {
103        throw new NotSerializableException();
104      }
105    }
