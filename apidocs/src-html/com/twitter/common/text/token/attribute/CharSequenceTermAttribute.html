017    package com.twitter.common.text.token.attribute;
018    
019    import org.apache.lucene.util.Attribute;
020    
021    import com.twitter.common.text.token.TokenStream;
022    
023    /**
024     * {@code TermAttribute} backed by a larger {@link CharSequence} that does not change as
025     * {@link TokenStream#incrementToken()} is called. Instead, the offset and character length are
026     * updated to reference a new span with respect to the underlying {@code CharSequence}.
027     */
028    public interface CharSequenceTermAttribute extends Attribute {
029      /**
030       * The offset is the character index, with respect to the underlying CharSequence,
031       * of the first character in the span referenced by this CharSequenceTermAttribute.
032       * The offset may point to the end of the underlying CharSequence when length is zero.
033       *
034       * @return the current offset
035       */
036      int getOffset();
037    
038      /**
039       * The length is the length in characters of the span referenced by this
040       * {@link CharSequenceTermAttribute}.
041       *
042       * @return the current length
043       */
044      int getLength();
045    
046      /**
047       * Assigns the offset to the specified value.
048       *
049       * @param offset new value for the offset, which must be at least zero, and less
050       *               than or equal to the length of the underlying {@code CharSequence}
051       * @throws IndexOutOfBoundsException if the specified offset is out of bounds
052       */
053      void setOffset(int offset);
054    
055      /**
056       * Assigns the length to the specified value.
057       *
058       * @param length new value for the length, which must be at least zero, and at most
059       *               equal to the length of the underlying {@code CharSequence}
060       * @throws IndexOutOfBoundsException if the specified length is out of bounds
061       */
062      void setLength(int length);
063    
064      /**
065       * Sets the encapsulated {@code CharSequence}.
066       *
067       * @param originalCharSequence {@code CharSequence} encapsulated by this
068       *     {@code CharSequenceAttribute}
069       */
070      void setCharSequence(CharSequence originalCharSequence);
071    
072      /**
073       * Provides access to the encapsulated {@code CharSequence}.
074       *
075       * @return the underlying {@code CharSequence} object
076       */
077      CharSequence getCharSequence();
078    
079      /**
080       * Assigns the backing {@code CharSequence} for this attribute to the specified {@code
081       * CharSequence}. The start character index is set to zero, and the end character index is set to
082       * the length of the specified {@code CharSequence}.
083       *
084       * @param seq {@code CharSequence} that will become the new underlying {@code CharSequence} for
085       *          this attribute.
086       */
087      void setTermBuffer(CharSequence seq);
088    
089      /**
090       * Assigns the backing {@code CharSequence} for this attribute to the specified {@code
091       * CharSequence}. The start character index is set to specified offset, and the end character
092       * index is set to offset plus length.
093       *
094       * @param seq {@code CharSequence} that will become the new underlying {@code CharSequence} for
095       *          this attribute.
096       * @param offset character index with respect to the specified {@code CharSequence} that will
097       *          become the new start character index for this attribute.
098       * @param length this value will be added to the specified offset value, and the result will
099       *          become the new end character index for this attribute.
100       */
101      void setTermBuffer(CharSequence seq, int offset, int length);
102    
103    
104      /**
105       * Returns the term text as a {@code CharSequence}, without needing to construct a
106       * {@code String}. This method is preferred over {@link getTermString()}.
107       *
108       * @return {@code CharSequence} representing the term text.
109       */
110      CharSequence getTermCharSequence();
111    
112      /**
113       * Returns the term text as a {@code String}.
114       * {@link getTermCharSequence()} is preferred over this method.
115       *
116       * @return {@code String} representing the term text.
117       */
118      String getTermString();
119    }
