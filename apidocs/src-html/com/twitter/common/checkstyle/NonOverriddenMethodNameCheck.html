001    package com.twitter.common.checkstyle;
002    
003    import com.puppycrawl.tools.checkstyle.api.AnnotationUtility;
004    import com.puppycrawl.tools.checkstyle.api.DetailAST;
005    import com.puppycrawl.tools.checkstyle.checks.naming.MethodNameCheck;
006    
007    /**
008     * An extension to method name check that does not check-style the name of an overridden method
009     * because the programmer does not have a choice in renaming such methods.
010     * If the interface/base class definition is checkstyled, the bad name will always be caught
011     * over there.
012     *
013     * @author Utkarsh Srivastava
014     */
015    public class NonOverriddenMethodNameCheck extends MethodNameCheck {
016      /**
017       * {@link Override Override} annotation name.
018       */
019      private static final String OVERRIDE = "Override";
020    
021      /**
022       * Fully-qualified {@link Override Override} annotation name.
023       */
024      private static final String FQ_OVERRIDE = "java.lang." + OVERRIDE;
025    
026      // Based on code in com.puppycrawl.tools.checkstyle.checks.annotation.MissingOverrideCheck
027      @Override
028      public void visitToken(DetailAST aAST) {
029        if (!AnnotationUtility.containsAnnotation(aAST, OVERRIDE)
030            && !AnnotationUtility.containsAnnotation(aAST, FQ_OVERRIDE)) {
031          super.visitToken(aAST);
032        }
033      }
034    
035      /**
036       * For human readable error messages
037       */
038      @Override
039      protected String getMessageBundle() {
040        return CheckStyleUtils.getMessageBundle(MethodNameCheck.class);
041      }
042    }
