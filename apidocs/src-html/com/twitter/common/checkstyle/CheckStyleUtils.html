016    package com.twitter.common.checkstyle;
017    
018    import com.puppycrawl.tools.checkstyle.api.Check;
019    
020    /**
021     * Utilities needed by custom checkstyle rules
022     *
023     * @author Utkarsh Srivastava
024     */
025    public final class CheckStyleUtils {
026    
027      /**
028       * Hide constructor
029       */
030      private CheckStyleUtils() {
031      }
032    
033      /**
034       * Returns the message bundle to use for a custom rule. All built-in rules in checkstyle have
035       * a message bundle in the same package as that rule. This method returns the path to that
036       * bundle.
037       *
038       * @param parentBuiltInCheckClass The class of the built-in checkstyle rule that the custom rule
039       *                                ultimately derives from
040       * @return Path to message bundle
041       */
042      public static String getMessageBundle(final Class<? extends Check> parentBuiltInCheckClass) {
043        String parentBuiltInCheckClassName = parentBuiltInCheckClass.getName();
044        // This code is copied from com.puppycrawl.tools.checkstyle.api.
045        // AbstractViolationReporter.java::getMessageBundle(String s)
046        // If that method were protected (and not package private) we could
047        // just invoke it
048        final int endIndex = parentBuiltInCheckClassName.lastIndexOf('.');
049        final String messages = "messages";
050        if (endIndex < 0) {
051          return messages;
052        }
053        final String packageName = parentBuiltInCheckClassName.substring(0, endIndex);
054        return packageName + "." + messages;
055      }
056    }
