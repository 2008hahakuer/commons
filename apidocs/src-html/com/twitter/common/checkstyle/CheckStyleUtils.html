001    package com.twitter.common.checkstyle;
002    
003    import com.puppycrawl.tools.checkstyle.api.Check;
004    
005    /**
006     * Utilities needed by custom checkstyle rules
007     *
008     * @author Utkarsh Srivastava
009     */
010    public final class CheckStyleUtils {
011    
012      /**
013       * Hide constructor
014       */
015      private CheckStyleUtils() {
016      }
017    
018      /**
019       * Returns the message bundle to use for a custom rule. All built-in rules in checkstyle have
020       * a message bundle in the same package as that rule. This method returns the path to that
021       * bundle.
022       *
023       * @param parentBuiltInCheckClass The class of the built-in checkstyle rule that the custom rule
024       *                                ultimately derives from
025       * @return Path to message bundle
026       */
027      public static String getMessageBundle(final Class<? extends Check> parentBuiltInCheckClass) {
028        String parentBuiltInCheckClassName = parentBuiltInCheckClass.getName();
029        // This code is copied from com.puppycrawl.tools.checkstyle.api.
030        // AbstractViolationReporter.java::getMessageBundle(String s)
031        // If that method were protected (and not package private) we could
032        // just invoke it
033        final int endIndex = parentBuiltInCheckClassName.lastIndexOf('.');
034        final String messages = "messages";
035        if (endIndex < 0) {
036          return messages;
037        }
038        final String packageName = parentBuiltInCheckClassName.substring(0, endIndex);
039        return packageName + "." + messages;
040      }
041    }
