017    package com.twitter.common.base;
018    
019    /**
020     * Utility functions for working with exceptional functions.
021     *
022     * @author John Sirois
023     */
024    public final class ExceptionalFunctions {
025    
026      private ExceptionalFunctions() {
027        // utility
028      }
029    
030      /**
031       * Returns an {@link ExceptionalSupplier}/{@link java.util.concurrent.Callable} object that
032       * will return the result of {@code function} applied to {@code argument}.  Evaluation is lazy
033       * and un-memoized.
034       */
035      public static <S, T, E extends Exception> CallableExceptionalSupplier<T, E> curry(
036          final ExceptionalFunction<S, T, E> function, final S argument) {
037    
038        return new CallableExceptionalSupplier<T, E>() {
039          @Override public T get() throws E {
040            return function.apply(argument);
041          }
042        };
043      }
044    }
