018    package com.twitter.common.base;
019    
020    import com.google.common.base.Preconditions;
021    import com.google.common.collect.Iterables;
022    
023    import org.apache.commons.lang.StringUtils;
024    
025    /**
026     * A utility helpful in concisely checking preconditions on arguments.  This utility is a complement
027     * to {@link com.google.common.base.Preconditions}.
028     *
029     * @author John Sirois
030     */
031    public final class MorePreconditions {
032    
033      private static final String ARG_NOT_BLANK_MSG = "Argument cannot be blank";
034    
035      private MorePreconditions() {
036        // utility
037      }
038    
039      /**
040       * Checks that a string is both non-null and non-empty.
041       *
042       * @see #checkNotBlank(String, String, Object...)
043       */
044      public static String checkNotBlank(String argument) {
045        return checkNotBlank(argument, ARG_NOT_BLANK_MSG);
046      }
047    
048      /**
049       * Checks that a string is both non-null and non-empty.
050       *
051       * @param argument the argument to validate
052       * @param message the message template for validation exception messages where %s serves as the
053       *     sole argument placeholder
054       * @param args any arguments needed by the message template
055       * @return the argument if it is valid
056       * @throws NullPointerException if the argument is null
057       * @throws IllegalArgumentException if the argument is the empty string or a pure whitespace
058       *    string
059       */
060      public static String checkNotBlank(String argument, String message, Object... args) {
061        Preconditions.checkNotNull(argument, message, args);
062        Preconditions.checkArgument(!StringUtils.isBlank(argument), message, args);
063        return argument;
064      }
065    
066      /**
067       * Checks that an Iterable is both non-null and non-empty.  This method does not check individual
068       * elements in the Iterable.
069       *
070       * @see #checkNotBlank(Iterable, String, Object...)
071       */
072      public static <S, T extends Iterable<S>> T checkNotBlank(T argument) {
073        return checkNotBlank(argument, ARG_NOT_BLANK_MSG);
074      }
075    
076      /**
077       * Checks that an Iterable is both non-null and non-empty.  This method does not check individual
078       * elements in the Iterable, it just checks that the Iterable has at least one element.
079       *
080       * @param argument the argument to validate
081       * @param message the message template for validation exception messages where %s serves as the
082       *     sole argument placeholder
083       * @param args any arguments needed by the message template
084       * @return the argument if it is valid
085       * @throws NullPointerException if the argument is null
086       * @throws IllegalArgumentException if the argument has no iterable elements
087       */
088      public static <S, T extends Iterable<S>> T checkNotBlank(T argument, String message,
089          Object... args) {
090        Preconditions.checkNotNull(argument, message, args);
091        Preconditions.checkArgument(!Iterables.isEmpty(argument), message, args);
092        return argument;
093      }
094    
095      /**
096       * Checks that a double falls within a specified range, inclusive
097       *
098       * @param argument argument to validate.
099       * @param minimum minimum possible valid value for the argument.
100       * @param maximum maximum possible valid value for the argument.
101       * @param message the message template for validation exception messages where %s serves as the
102       *                sole argument placeholder.
103       * @return the argument if it is valid.
104       * @throws IllegalArgumentException if the argument falls outside of the specified range.
105       */
106      public static double checkArgumentRange(double argument, double minimum, double maximum,
107          String message) {
108        Preconditions.checkArgument(minimum <= argument, message, argument);
109        Preconditions.checkArgument(argument <= maximum, message, argument);
110        return argument;
111      }
112    
113      /**
114       * Checks that at least one of the specified arguments is true.
115       *
116       * @param message the message for validation exception messages.
117       * @param arguments one or more arguments to check.
118       * @return true if at least one of the arguments is true.
119       * @throws IllegalArgumentException if none of the arguments are true.
120       */
121      public static boolean checkArguments(String message,
122          Boolean... arguments) {
123        for (Boolean argument : arguments) {
124          if (argument) {
125            return true;
126          }
127        }
128        throw new IllegalArgumentException(message);
129      }
130    }
