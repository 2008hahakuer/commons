017    package com.twitter.common.base;
018    
019    import com.google.common.base.Function;
020    import com.google.common.base.Preconditions;
021    import com.google.common.base.Throwables;
022    
023    /**
024     * Utilities for dealing with Closures.
025     *
026     * @author John Sirois
027     */
028    public final class Closures {
029    
030      private static final Closure NOOP = new Closure() {
031        @Override public void execute(Object item) {
032          // noop
033        }
034      };
035    
036      private Closures() {
037        // utility
038      }
039    
040      /**
041       * Converts a closure into a function returning {@code null}.
042       */
043      public static <T> Function<T, Void> asFunction(final ExceptionalClosure<T, ?> closure) {
044        Preconditions.checkNotNull(closure);
045    
046        return new Function<T, Void>() {
047          @Override public Void apply(T item) {
048            try {
049              closure.execute(item);
050            } catch (Exception e) {
051              Throwables.propagate(e);
052            }
053            return null;
054          }
055        };
056      }
057    
058      /**
059       * Returns a closure that will do nothing.
060       *
061       * @param <T> The closure argument type.
062       * @return A closure that does nothing.
063       */
064      @SuppressWarnings("unchecked")
065      public static <T> Closure<T> noop() {
066        return NOOP;
067      }
068    }
