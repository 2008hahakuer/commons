018    package com.twitter.common.base;
019    
020    import com.google.common.base.Function;
021    import com.google.common.base.Preconditions;
022    import com.google.common.base.Throwables;
023    
024    /**
025     * Utilities for dealing with Closures.
026     *
027     * @author John Sirois
028     */
029    public final class Closures {
030    
031      private static final Closure NOOP = new Closure() {
032        @Override public void execute(Object item) {
033          // noop
034        }
035      };
036    
037      private Closures() {
038        // utility
039      }
040    
041      /**
042       * Converts a closure into a function returning {@code null}.
043       */
044      public static <T> Function<T, Void> asFunction(final ExceptionalClosure<T, ?> closure) {
045        Preconditions.checkNotNull(closure);
046    
047        return new Function<T, Void>() {
048          @Override public Void apply(T item) {
049            try {
050              closure.execute(item);
051            } catch (Exception e) {
052              Throwables.propagate(e);
053            }
054            return null;
055          }
056        };
057      }
058    
059      /**
060       * Returns a closure that will do nothing.
061       *
062       * @param <T> The closure argument type.
063       * @return A closure that does nothing.
064       */
065      @SuppressWarnings("unchecked")
066      public static <T> Closure<T> noop() {
067        return NOOP;
068      }
069    }
