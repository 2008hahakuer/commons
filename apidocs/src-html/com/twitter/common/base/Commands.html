018    package com.twitter.common.base;
019    
020    import com.google.common.base.Preconditions;
021    
022    /**
023     * Utility functions for working with commands.
024     *
025     * @author John Sirois
026     */
027    public final class Commands {
028    
029      /**
030       * A command that does nothing when executed.
031       */
032      public static final Command NOOP = new Command() {
033        @Override public void execute() {
034          // noop
035        }
036      };
037    
038      private Commands() {
039        // utility
040      }
041    
042      /**
043       * Converts a command into a supplier returning null.
044       *
045       * @return A supplier whose {@link com.twitter.common.base.Supplier#get()} will cause the given
046       *     {@code command} to be executed and {@code null} to be returned.
047       */
048      public static <E extends Exception> ExceptionalSupplier<Void, E> asSupplier(
049          final ExceptionalCommand<E> command) {
050        Preconditions.checkNotNull(command);
051    
052        return new ExceptionalSupplier<Void, E>() {
053          @Override public Void get() throws E {
054            command.execute();
055            return null;
056          }
057        };
058      }
059    }
