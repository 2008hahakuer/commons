017    package com.twitter.common.stats;
018    
019    import java.lang.management.ClassLoadingMXBean;
020    import java.lang.management.GarbageCollectorMXBean;
021    import java.lang.management.ManagementFactory;
022    import java.lang.management.MemoryMXBean;
023    import java.lang.management.MemoryUsage;
024    import java.lang.management.OperatingSystemMXBean;
025    import java.lang.management.RuntimeMXBean;
026    import java.lang.management.ThreadMXBean;
027    
028    import com.google.common.base.Function;
029    import com.google.common.collect.ImmutableList;
030    
031    import com.google.common.collect.Iterables;
032    import com.twitter.common.quantity.Amount;
033    import com.twitter.common.quantity.Data;
034    import com.twitter.common.quantity.Time;
035    
036    /**
037     * Convenience class to export statistics about the JVM.
038     *
039     * TODO(William Farner): Some of these are fixed values, make sure to export them so that they are not
040     * collected as time series.
041     *
042     * @author William Farner
043     */
044    public class JvmStats {
045    
046      private static final long BYTES_PER_MB = Amount.of(1L, Data.MB).as(Data.BYTES);
047      private static final double SECS_PER_NANO =
048          ((double) 1) / Amount.of(1L, Time.SECONDS).as(Time.NANOSECONDS);
049    
050      private JvmStats() {
051        // Utility class.
052      }
053    
054      /**
055       * Exports stats related to the JVM and runtime environment.
056       */
057      public static void export() {
058        final OperatingSystemMXBean osMbean = ManagementFactory.getOperatingSystemMXBean();
059        if (osMbean instanceof com.sun.management.OperatingSystemMXBean) {
060          final com.sun.management.OperatingSystemMXBean sunOsMbean =
061              (com.sun.management.OperatingSystemMXBean) osMbean;
062    
063          Stats.exportAll(
064              ImmutableList.<Stat<? extends Number>>builder()
065              .add(new StatImpl<Long>("system_free_physical_memory_mb") {
066                @Override public Long read() {
067                  return sunOsMbean.getFreePhysicalMemorySize() / BYTES_PER_MB;
068                }
069              })
070              .add(new StatImpl<Long>("system_free_swap_mb") {
071                @Override public Long read() {
072                  return sunOsMbean.getFreeSwapSpaceSize() / BYTES_PER_MB;
073                }
074              })
075              .add(
076              Rate.of(
077                  new StatImpl<Long>("process_cpu_time_nanos") {
078                    @Override public Long read() {
079                      return sunOsMbean.getProcessCpuTime();
080                    }
081                  }).withName("process_cpu_cores_utilized").withScaleFactor(SECS_PER_NANO).build())
082              .build());
083        }
084        if (osMbean instanceof com.sun.management.UnixOperatingSystemMXBean) {
085          final com.sun.management.UnixOperatingSystemMXBean unixOsMbean =
086              (com.sun.management.UnixOperatingSystemMXBean) osMbean;
087    
088          Stats.exportAll(ImmutableList.<Stat<? extends Number>>builder()
089              .add(new StatImpl<Long>("process_max_fd_count") {
090                @Override public Long read() { return unixOsMbean.getMaxFileDescriptorCount(); }
091              }).add(new StatImpl<Long>("process_open_fd_count") {
092                @Override public Long read() { return unixOsMbean.getOpenFileDescriptorCount(); }
093              }).build());
094        }
095    
096        final Runtime runtime = Runtime.getRuntime();
097        final ClassLoadingMXBean classLoadingBean = ManagementFactory.getClassLoadingMXBean();
098        MemoryMXBean memoryBean = ManagementFactory.getMemoryMXBean();
099        final MemoryUsage heapUsage = memoryBean.getHeapMemoryUsage();
100        final MemoryUsage nonHeapUsage = memoryBean.getNonHeapMemoryUsage();
101        final ThreadMXBean threads = ManagementFactory.getThreadMXBean();
102        final RuntimeMXBean runtimeMXBean = ManagementFactory.getRuntimeMXBean();
103    
104        Stats.exportAll(ImmutableList.<Stat<? extends Number>>builder()
105          .add(new StatImpl<Long>("jvm_time_ms") {
106            @Override public Long read() { return System.currentTimeMillis(); }
107          })
108          .add(new StatImpl<Integer>("jvm_available_processors") {
109            @Override public Integer read() { return runtime.availableProcessors(); }
110          })
111          .add(new StatImpl<Long>("jvm_memory_free_mb") {
112            @Override public Long read() { return runtime.freeMemory() / BYTES_PER_MB; }
113          })
114          .add(new StatImpl<Long>("jvm_memory_max_mb") {
115            @Override public Long read() { return runtime.maxMemory() / BYTES_PER_MB; }
116          })
117          .add(new StatImpl<Long>("jvm_memory_mb_total") {
118            @Override public Long read() { return runtime.totalMemory() / BYTES_PER_MB; }
119          })
120          .add(new StatImpl<Integer>("jvm_class_loaded_count") {
121            @Override public Integer read() { return classLoadingBean.getLoadedClassCount(); }
122          })
123          .add(new StatImpl<Long>("jvm_class_total_loaded_count") {
124            @Override public Long read() { return classLoadingBean.getTotalLoadedClassCount(); }
125          })
126          .add(new StatImpl<Long>("jvm_class_unloaded_count") {
127            @Override public Long read() { return classLoadingBean.getUnloadedClassCount(); }
128          })
129          .add(new StatImpl<Long>("jvm_gc_collection_time_ms") {
130            @Override public Long read() {
131              long collectionTimeMs = 0;
132              for (GarbageCollectorMXBean bean : ManagementFactory.getGarbageCollectorMXBeans()) {
133                collectionTimeMs += bean.getCollectionTime();
134              }
135              return collectionTimeMs;
136            }
137          })
138          .add(new StatImpl<Long>("jvm_gc_collection_count") {
139            @Override public Long read() {
140              long collections = 0;
141              for (GarbageCollectorMXBean bean : ManagementFactory.getGarbageCollectorMXBeans()) {
142                collections += bean.getCollectionCount();
143              }
144              return collections;
145            }
146          })
147          .add(new StatImpl<Long>("jvm_memory_heap_mb_used") {
148            @Override public Long read() { return heapUsage.getUsed() / BYTES_PER_MB; }
149          })
150          .add(new StatImpl<Long>("jvm_memory_heap_mb_committed") {
151            @Override public Long read() { return heapUsage.getCommitted() / BYTES_PER_MB; }
152          })
153          .add(new StatImpl<Long>("jvm_memory_heap_mb_max") {
154            @Override public Long read() { return heapUsage.getMax() / BYTES_PER_MB; }
155          })
156          .add(new StatImpl<Long>("jvm_memory_non_heap_mb_used") {
157            @Override public Long read() { return nonHeapUsage.getUsed() / BYTES_PER_MB; }
158          })
159          .add(new StatImpl<Long>("jvm_memory_non_heap_mb_committed") {
160            @Override public Long read() { return nonHeapUsage.getCommitted() / BYTES_PER_MB; }
161          })
162          .add(new StatImpl<Long>("jvm_memory_non_heap_mb_max") {
163            @Override public Long read() { return nonHeapUsage.getMax() / BYTES_PER_MB; }
164          })
165          .add(new StatImpl<Long>("jvm_uptime_secs") {
166            @Override public Long read() { return runtimeMXBean.getUptime() / 1000; }
167          })
168          .add(new StatImpl<Double>("system_load_avg") {
169            @Override public Double read() { return osMbean.getSystemLoadAverage(); }
170          })
171          .add(new StatImpl<Integer>("jvm_threads_peak") {
172            @Override public Integer read() { return threads.getPeakThreadCount(); }
173          })
174          .add(new StatImpl<Long>("jvm_threads_started") {
175            @Override public Long read() { return threads.getTotalStartedThreadCount(); }
176          })
177          .add(new StatImpl<Integer>("jvm_threads_daemon") {
178            @Override public Integer read() { return threads.getDaemonThreadCount(); }
179          })
180          .add(new StatImpl<Integer>("jvm_threads_active") {
181            @Override public Integer read() { return threads.getThreadCount(); }
182          })
183          .build());
184    
185        // Export per memory pool gc time and cycle count like Ostrich
186        // This is based on code in Bridcage: https://cgit.twitter.biz/birdcage/tree/ \
187        // ostrich/src/main/scala/com/twitter/ostrich/stats/StatsCollection.scala
188        Stats.exportAll(Iterables.transform(ManagementFactory.getGarbageCollectorMXBeans(),
189            new Function<GarbageCollectorMXBean, Stat<? extends Number>>(){
190              @Override
191              public Stat<? extends Number> apply(final GarbageCollectorMXBean gcMXBean) {
192                return new StatImpl<Long>(
193                    "jvm_gc_" + Stats.normalizeName(gcMXBean.getName()) + "_collection_count") {
194                  @Override public Long read() {
195                    return gcMXBean.getCollectionCount();
196                  }
197                };
198              }
199            }
200        ));
201    
202        Stats.exportAll(Iterables.transform(ManagementFactory.getGarbageCollectorMXBeans(),
203            new Function<GarbageCollectorMXBean, Stat<? extends Number>>(){
204              @Override
205              public Stat<? extends Number> apply(final GarbageCollectorMXBean gcMXBean) {
206                return new StatImpl<Long>(
207                    "jvm_gc_" + Stats.normalizeName(gcMXBean.getName()) + "_collection_time_ms") {
208                  @Override public Long read() {
209                    return gcMXBean.getCollectionTime();
210                  }
211                };
212              }
213            }
214        ));
215    
216        Stats.exportString(
217            new StatImpl<String>("jvm_input_arguments") {
218              @Override public String read() {
219                return runtimeMXBean.getInputArguments().toString();
220              }
221            }
222        );
223    
224        for (final String property : System.getProperties().stringPropertyNames()) {
225          Stats.exportString(
226              new StatImpl<String>("jvm_prop_" + Stats.normalizeName(property)) {
227                @Override public String read() { return System.getProperty(property); }
228              });
229        }
230      }
231    }
