001    package com.twitter.common.stats;
002    
003    import java.util.concurrent.atomic.AtomicLong;
004    
005    import com.google.common.annotations.VisibleForTesting;
006    import com.google.common.base.Preconditions;
007    
008    import com.twitter.common.base.MorePreconditions;
009    import com.twitter.common.quantity.Amount;
010    import com.twitter.common.quantity.Time;
011    import com.twitter.common.util.Clock;
012    
013    /**
014     * A stat that exports the amount of time since it was last reset.
015     *
016     * @author William Farner
017     */
018    public class Elapsed {
019    
020      private final Clock clock;
021      private final AtomicLong lastEventNs = new AtomicLong();
022    
023      /**
024       * Calls {@link #Elapsed(String, Time)} using a default granularity of nanoseconds.
025       *
026       * @param name Name of the stat to export.
027       */
028      public Elapsed(String name) {
029        this(name, Time.NANOSECONDS);
030      }
031    
032      /**
033       * Creates and exports a new stat that maintains the difference between the system clock time
034       * and the time since it was last reset.  Upon export, the counter will act as though it were just
035       * reset.
036       *
037       * @param name Name of the stat to export.
038       * @param granularity Time unit granularity to export.
039       */
040      public Elapsed(String name, Time granularity) {
041        this(name, granularity, Clock.SYSTEM_CLOCK);
042      }
043    
044      @VisibleForTesting
045      Elapsed(String name, final Time granularity, final Clock clock) {
046        MorePreconditions.checkNotBlank(name);
047        Preconditions.checkNotNull(granularity);
048        this.clock = Preconditions.checkNotNull(clock);
049    
050        reset();
051    
052        Stats.export(new StatImpl<Long>(name) {
053          @Override public Long read() {
054            return Amount.of(clock.nowNanos() - lastEventNs.get(), Time.NANOSECONDS).as(granularity);
055          }
056        });
057      }
058    
059      public void reset() {
060        lastEventNs.set(clock.nowNanos());
061      }
062    }
