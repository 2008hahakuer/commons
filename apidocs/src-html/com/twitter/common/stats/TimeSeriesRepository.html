017    package com.twitter.common.stats;
018    
019    import java.util.Set;
020    
021    import com.twitter.common.application.ShutdownRegistry;
022    
023    /**
024     * A repository for time series data.
025     *
026     * @author William Farner
027     */
028    public interface TimeSeriesRepository {
029    
030      /**
031       * Starts the time series sampler.
032       *
033       * @param shutdownRegistry An action registry that the repository can use to register a shutdown
034       *    for the sampler.
035       */
036      public void start(ShutdownRegistry shutdownRegistry);
037    
038      /**
039       * Fetches the names of all available time series.
040       *
041       * @return Available time series, which can then be obtained by calling {@link #get(String)}.
042       */
043      public Set<String> getAvailableSeries();
044    
045      /**
046       * Fetches a time series by name.
047       *
048       * @param name The name of the time series to fetch.
049       * @return The time series registered with the given name, or {@code null} if no such time series
050       *     has been registered.
051       */
052      public TimeSeries get(String name);
053    
054      /**
055       * Gets an ordered iterable of the timestamps that all timeseries were sampled at.
056       *
057       * @return All current timestamps.
058       */
059      public Iterable<Number> getTimestamps();
060    }
