017    package com.twitter.common.stats;
018    
019    import java.util.concurrent.LinkedBlockingDeque;
020    
021    import com.google.common.base.Preconditions;
022    
023    /**
024     * Function to compute the moving average of a time series.
025     *
026     * @author William Farner
027     */
028    public class MovingAverage<T extends Number> extends SampledStat<Double> {
029    
030      private static final int DEFAULT_WINDOW = 10;
031      private final Stat<T> input;
032    
033      private final LinkedBlockingDeque<T> samples;
034      private double sampleSum = 0;
035    
036      private MovingAverage(String name, Stat<T> input, int windowSize) {
037        super(name, 0d);
038        Preconditions.checkArgument(windowSize > 1);
039    
040        this.input = Preconditions.checkNotNull(input);
041        this.samples = new LinkedBlockingDeque<T>(windowSize);
042        Stats.export(input);
043      }
044    
045      public static <T extends Number> MovingAverage<T> of(Stat<T> input) {
046        return MovingAverage.of(input, DEFAULT_WINDOW);
047      }
048    
049      public static <T extends Number> MovingAverage<T> of(Stat<T> input, int windowSize) {
050        return MovingAverage.of(String.format("%s_avg", input.getName()), input, windowSize);
051      }
052    
053      public static <T extends Number> MovingAverage<T> of(String name, Stat<T> input,
054          int windowSize) {
055        return new MovingAverage<T>(name, input, windowSize);
056      }
057    
058      @Override
059      public Double doSample() {
060        T sample = input.read();
061    
062        if (samples.remainingCapacity() == 0) {
063          sampleSum -= samples.removeLast().doubleValue();
064        }
065    
066        samples.addFirst(sample);
067        sampleSum += sample.doubleValue();
068    
069        return sampleSum / samples.size();
070      }
071    }
