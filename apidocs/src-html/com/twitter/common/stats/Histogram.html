001    package com.twitter.common.stats;
002    
003    /**
004     * An interface for Histogram
005     */
006    public interface Histogram {
007    
008      /**
009       * Add an entry into the histogram
010       *
011       * @param x the value to insert
012       */
013      void add(long x);
014    
015      /**
016       * Clear the histogram
017       */
018      void clear();
019    
020      /**
021       * Return the current quantiles of the histogram
022       *
023       * @param quantiles the list of quantiles that you want to compute
024       */
025      long[] getQuantiles(double[] quantiles);
026    }
