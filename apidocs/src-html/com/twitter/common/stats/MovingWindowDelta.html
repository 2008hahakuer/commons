017    package com.twitter.common.stats;
018    
019    import com.google.common.annotations.VisibleForTesting;
020    import com.google.common.base.Preconditions;
021    import com.google.common.base.Supplier;
022    
023    import java.util.concurrent.LinkedBlockingDeque;
024    
025    import com.twitter.common.base.MorePreconditions;
026    
027    
028    /**
029     * Delta over the most recent k sample periods.
030     *
031     * If you use this class with a counter, you can get the cumulation of counts in a sliding window.
032     *
033     * One sample period is the time in between doSample() calls.
034     *
035     * @author Feng Zhuge
036     */
037    public class MovingWindowDelta<T extends Number> extends SampledStat<Long> {
038      private static final int DEFAULT_WINDOW_SIZE = 60;
039      private final LinkedBlockingDeque<Long> deltaSeries;
040      private final Supplier<T> inputAccessor;
041      long sumDelta = 0l;
042      long lastInput = 0l;
043    
044      private MovingWindowDelta(String name, Supplier<T> inputAccessor, int windowSize) {
045        super(name, 0l);
046    
047        Preconditions.checkArgument(windowSize > 1);
048        Preconditions.checkNotNull(inputAccessor);
049        MorePreconditions.checkNotBlank(name);
050    
051        deltaSeries = new LinkedBlockingDeque<Long>(windowSize);
052        this.inputAccessor = inputAccessor;
053    
054        Stats.export(this);
055      }
056    
057      /**
058       * Create a new MovingWindowDelta instance.
059       *
060       * @param name The name of the value to be tracked.
061       * @param inputAccessor The accessor of the value.
062       * @param windowSize How many sample periods shall we use to calculate delta.
063       * @param <T> The type of the value.
064       * @return The created MovingWindowSum instance.
065       */
066      public static <T extends Number> MovingWindowDelta<T> of(
067          String name, Supplier<T> inputAccessor, int windowSize) {
068        return new MovingWindowDelta<T>(name, inputAccessor, windowSize);
069      }
070    
071      /**
072       * Create a new MovingWindowDelta instance using the default window size (currently 60).
073       *
074       * @param name The name of the value to be tracked.
075       * @param inputAccessor The accessor of the value.
076       * @param <T> The type of the value.
077       * @return The created MovingWindowSum instance.
078       */
079      public static <T extends Number> MovingWindowDelta<T> of(String name, Supplier<T> inputAccessor) {
080        return of(name, inputAccessor, DEFAULT_WINDOW_SIZE);
081      }
082    
083      @Override
084      public Long doSample() {
085        long lastDelta = 0l;
086        if (deltaSeries.remainingCapacity() == 0) {
087          lastDelta = deltaSeries.removeFirst();
088        }
089    
090        long newInput = inputAccessor.get().longValue();
091        long newDelta = newInput - lastInput;
092        lastInput = newInput;
093    
094        deltaSeries.addLast(newDelta);
095    
096        sumDelta += newDelta - lastDelta;
097    
098        return sumDelta;
099      }
100    }
