001    package com.twitter.common.stats;
002    
003    import java.util.Vector;
004    
005    import com.google.common.base.Preconditions;
006    
007    import com.twitter.common.util.Random;
008    
009    /**
010     * An in memory implementation of Reservoir Sampling for sampling from
011     * a population.
012     * <p>Several optimizations can be done.
013     * Especially, one can avoid rolling the dice as many times as the
014     * size of the population with an involved trick.
015     * See "Random Sampling with a Reservoir", Vitter, 1985</p>
016     * <p>TODO (delip): Fix this when the problem arises</p>
017     *
018     * @param <T> Type of the sample
019     * @author Delip Rao
020     */
021    public class ReservoirSampler<T> {
022      private final Vector<T> reservoir = new Vector<T>();
023      private final int numSamples;
024    
025      private final Random random;
026      private int numItemsSeen = 0;
027    
028      /**
029       * Create a new sampler with a certain reservoir size using
030       * a supplied random number generator.
031       *
032       * @param numSamples Maximum number of samples to
033       *                   retain in the reservoir. Must be non-negative.
034       * @param random Instance of the random number generator
035       *               to use for sampling
036       */
037      public ReservoirSampler(int numSamples, Random random) {
038        Preconditions.checkArgument(numSamples > 0,
039            "numSamples should be positive");
040        Preconditions.checkNotNull(random);
041        this.numSamples = numSamples;
042        this.random = random;
043      }
044    
045      /**
046       * Create a new sampler with a certain reservoir size using
047       * the default random number generator.
048       *
049       * @param numSamples Maximum number of samples to
050       *        retain in the reservoir. Must be non-negative.
051       */
052      public ReservoirSampler(int numSamples) {
053        this(numSamples, Random.Util.newDefaultRandom());
054      }
055    
056      /**
057       * Sample an item and store in the reservoir if needed.
058       *
059       * @param item The item to sample - may not be null.
060       */
061      public void sample(T item) {
062        Preconditions.checkNotNull(item);
063        if (reservoir.size() < numSamples) {
064          // reservoir not yet full, just append
065          reservoir.add(item);
066        } else {
067          // find a sample to replace
068          int rIndex = random.nextInt(numItemsSeen + 1);
069          if (rIndex < numSamples) {
070            reservoir.set(rIndex, item);
071          }
072        }
073        numItemsSeen++;
074      }
075    
076      /**
077       * Get samples collected in the reservoir.
078       *
079       * @return A sequence of the samples. No guarantee is provided on the order of the samples.
080       */
081      public Iterable<T> getSamples() {
082        return reservoir;
083      }
084    }
