017    package com.twitter.common.stats;
018    
019    import java.util.Set;
020    import java.util.concurrent.ScheduledExecutorService;
021    import java.util.concurrent.ScheduledThreadPoolExecutor;
022    import java.util.concurrent.atomic.AtomicBoolean;
023    import java.util.logging.Level;
024    import java.util.logging.Logger;
025    
026    import com.google.common.annotations.VisibleForTesting;
027    import com.google.common.base.Preconditions;
028    import com.google.common.cache.CacheBuilder;
029    import com.google.common.cache.CacheLoader;
030    import com.google.common.cache.LoadingCache;
031    import com.google.common.collect.ImmutableSet;
032    import com.google.common.collect.Iterables;
033    import com.google.common.util.concurrent.ThreadFactoryBuilder;
034    import com.google.inject.Inject;
035    import com.google.inject.name.Named;
036    
037    import com.twitter.common.application.ShutdownRegistry;
038    import com.twitter.common.base.Command;
039    import com.twitter.common.collections.BoundedQueue;
040    import com.twitter.common.quantity.Amount;
041    import com.twitter.common.quantity.Time;
042    import com.twitter.common.util.Clock;
043    
044    import static com.google.common.base.Preconditions.checkNotNull;
045    
046    /**
047     * A simple in-memory repository for exported variables.
048     *
049     * @author John Sirois
050     */
051    public class TimeSeriesRepositoryImpl implements TimeSeriesRepository {
052    
053      private static final Logger LOG = Logger.getLogger(TimeSeriesRepositoryImpl.class.getName());
054    
055      /**
056       * {@literal @Named} binding key for the sampling period.
057       */
058      public static final String SAMPLE_PERIOD =
059          "com.twitter.common.stats.TimeSeriesRepositoryImpl.SAMPLE_PERIOD";
060    
061      /**
062       * {@literal @Named} binding key for the maximum number of retained samples.
063       */
064      public static final String SAMPLE_RETENTION_PERIOD =
065          "com.twitter.common.stats.TimeSeriesRepositoryImpl.SAMPLE_RETENTION_PERIOD";
066    
067      private final SlidingStats scrapeDuration = new SlidingStats("variable_scrape", "micros");
068    
069      // We store TimeSeriesImpl, which allows us to add samples.
070      private final LoadingCache<String, TimeSeriesImpl> timeSeries;
071      private final BoundedQueue<Number> timestamps;
072    
073      private final StatRegistry statRegistry;
074      private final Amount<Long, Time> samplePeriod;
075      private final int retainedSampleLimit;
076    
077      @Inject
078      public TimeSeriesRepositoryImpl(
079          StatRegistry statRegistry,
080          @Named(SAMPLE_PERIOD) Amount<Long, Time> samplePeriod,
081          @Named(SAMPLE_RETENTION_PERIOD) final Amount<Long, Time> retentionPeriod) {
082        this.statRegistry = checkNotNull(statRegistry);
083        this.samplePeriod = checkNotNull(samplePeriod);
084        Preconditions.checkArgument(samplePeriod.getValue() > 0, "Sample period must be positive.");
085        checkNotNull(retentionPeriod);
086        Preconditions.checkArgument(retentionPeriod.getValue() > 0,
087            "Sample retention period must be positive.");
088    
089        retainedSampleLimit = (int) (retentionPeriod.as(Time.SECONDS) / samplePeriod.as(Time.SECONDS));
090        Preconditions.checkArgument(retainedSampleLimit > 0,
091            "Sample retention period must be greater than sample period.");
092    
093        timeSeries = CacheBuilder.newBuilder().build(
094            new CacheLoader<String, TimeSeriesImpl>() {
095              @Override public TimeSeriesImpl load(final String name) {
096                TimeSeriesImpl timeSeries = new TimeSeriesImpl(name);
097    
098                // Backfill so we have data for pre-accumulated timestamps.
099                int numTimestamps = timestamps.size();
100                if (numTimestamps != 0) {
101                  for (int i = 1; i < numTimestamps; i++) {
102                    timeSeries.addSample(0L);
103                  }
104                }
105    
106                return timeSeries;
107              }
108            });
109    
110        timestamps = new BoundedQueue<Number>(retainedSampleLimit);
111      }
112    
113      /**
114       * Starts the variable sampler, which will fetch variables {@link Stats} on the given period.
115       *
116       */
117      @Override
118      public void start(ShutdownRegistry shutdownRegistry) {
119        checkNotNull(shutdownRegistry);
120        checkNotNull(samplePeriod);
121        Preconditions.checkArgument(samplePeriod.getValue() > 0);
122    
123        final ScheduledExecutorService executor = new ScheduledThreadPoolExecutor(1 /* One thread. */,
124            new ThreadFactoryBuilder().setNameFormat("VariableSampler-%d").setDaemon(true).build());
125    
126        final AtomicBoolean shouldSample = new AtomicBoolean(true);
127        final Runnable sampler = new Runnable() {
128          @Override public void run() {
129            if (shouldSample.get()) {
130              try {
131                runSampler(Clock.SYSTEM_CLOCK);
132              } catch (Exception e) {
133                LOG.log(Level.SEVERE, "ignoring runSampler failure", e);
134              }
135            }
136          }
137        };
138    
139        executor.scheduleAtFixedRate(sampler, samplePeriod.getValue(), samplePeriod.getValue(),
140            samplePeriod.getUnit().getTimeUnit());
141        shutdownRegistry.addAction(new Command() {
142          @Override
143          public void execute() throws RuntimeException {
144            shouldSample.set(false);
145            executor.shutdown();
146            LOG.info("Variable sampler shut down");
147          }
148        });
149      }
150    
151      @VisibleForTesting
152      synchronized void runSampler(Clock clock) {
153        timestamps.add(clock.nowMillis());
154    
155        long startNanos = clock.nowNanos();
156        for (RecordingStat<? extends Number> var : statRegistry.getStats()) {
157          timeSeries.getUnchecked(var.getName()).addSample(var.sample());
158        }
159        scrapeDuration.accumulate(
160            Amount.of(clock.nowNanos() - startNanos, Time.NANOSECONDS).as(Time.MICROSECONDS));
161      }
162    
163      @Override
164      public synchronized Set<String> getAvailableSeries() {
165        return ImmutableSet.copyOf(timeSeries.asMap().keySet());
166      }
167    
168      @Override
169      public synchronized TimeSeries get(String name) {
170        if (!timeSeries.asMap().containsKey(name)) return null;
171        return timeSeries.getUnchecked(name);
172      }
173    
174      @Override
175      public synchronized Iterable<Number> getTimestamps() {
176        return Iterables.unmodifiableIterable(timestamps);
177      }
178    
179      private class TimeSeriesImpl implements TimeSeries {
180        private final String name;
181        private final BoundedQueue<Number> samples;
182    
183        TimeSeriesImpl(String name) {
184          this.name = name;
185          samples = new BoundedQueue<Number>(retainedSampleLimit);
186        }
187    
188        @Override public String getName() {
189          return name;
190        }
191    
192        void addSample(Number value) {
193          samples.add(value);
194        }
195    
196        @Override public Iterable<Number> getSamples() {
197          return Iterables.unmodifiableIterable(samples);
198        }
199      }
200    }
