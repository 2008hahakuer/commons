017    package com.twitter.common.stats;
018    
019    import com.google.common.base.Preconditions;
020    import com.google.common.collect.Maps;
021    
022    import java.util.Collection;
023    import java.util.Iterator;
024    import java.util.Map;
025    import java.util.Set;
026    import java.util.logging.Logger;
027    
028    /**
029     * A map from a key type to integers.  This simplifies the process of storing counters for multiple
030     * values of the same type.
031     */
032    public class CounterMap <K> implements Iterable<Map.Entry<K, Integer>>, Cloneable {
033      private final Map<K, Integer> map = Maps.newHashMap();
034    
035      private static Logger log = Logger.getLogger(CounterMap.class.getName());
036    
037      /**
038       * Increments the counter value associated with {@code key}, and returns the new value.
039       *
040       * @param key The key to increment
041       * @return The incremented value.
042       */
043      public int incrementAndGet(K key) {
044        return incrementAndGet(key, 1);
045      }
046    
047      /**
048       * Increments the value associated with {@code key} by {@code value}, returning the new value.
049       *
050       * @param key The key to increment
051       * @return The incremented value.
052       */
053      public int incrementAndGet(K key, int count) {
054        Integer value = map.get(key);
055        if (value == null) {
056          value = 0;
057        }
058        int newValue = count + value;
059        map.put(key, newValue);
060        return newValue;
061      }
062    
063      /**
064       * Gets the value associated with a key.
065       *
066       * @param key The key to look up.
067       * @return The counter value stored for {@code key}, or 0 if no mapping exists.
068       */
069      public int get(K key) {
070        if (!map.containsKey(key)) {
071          return 0;
072        }
073    
074        return map.get(key);
075      }
076    
077      /**
078       * Assigns a value to a key.
079       *
080       * @param key The key to assign a value to.
081       * @param newValue The value to assign.
082       */
083      public void set(K key, int newValue) {
084        Preconditions.checkNotNull(key);
085        map.put(key, newValue);
086      }
087    
088      /**
089       * Resets the value for {@code key}.  This will remove the key from the counter.
090       *
091       * @param key The key to reset.
092       */
093      public void reset(K key) {
094        map.remove(key);
095      }
096    
097      /**
098       * Gets the number of entries stored in the map.
099       *
100       * @return The size of the map.
101       */
102      public int size() {
103        return map.size();
104      }
105    
106      /**
107       * Gets an iterator for the mapped values.
108       *
109       * @return Iterator for mapped values.
110       */
111      public Iterator<Map.Entry<K, Integer>> iterator() {
112        return map.entrySet().iterator();
113      }
114    
115      public Collection<Integer> values() {
116        return map.values();
117      }
118    
119      public Set<K> keySet() {
120        return map.keySet();
121      }
122    
123      public String toString() {
124        StringBuilder strVal = new StringBuilder();
125        for (Map.Entry<K, Integer> entry : this) {
126          strVal.append(entry.getKey().toString()).append(": ").append(entry.getValue()).append('\n');
127        }
128        return strVal.toString();
129      }
130    
131      public Map<K, Integer> toMap() {
132        return map;
133      }
134    
135      @Override
136      public CounterMap<K> clone() {
137        CounterMap<K> newInstance = new CounterMap<K>();
138        newInstance.map.putAll(map);
139        return newInstance;
140      }
141    }
