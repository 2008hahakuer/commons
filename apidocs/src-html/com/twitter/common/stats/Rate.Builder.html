017    package com.twitter.common.stats;
018    
019    import com.google.common.annotations.VisibleForTesting;
020    import com.google.common.base.Preconditions;
021    import com.google.common.base.Supplier;
022    import com.google.common.base.Suppliers;
023    
024    import com.twitter.common.collections.Pair;
025    import com.twitter.common.quantity.Amount;
026    import com.twitter.common.quantity.Time;
027    import com.twitter.common.util.Clock;
028    
029    import java.util.concurrent.LinkedBlockingDeque;
030    import java.util.concurrent.atomic.AtomicInteger;
031    import java.util.concurrent.atomic.AtomicLong;
032    
033    /**
034     * Function to compute a windowed per-second rate of a value.
035     *
036     * @author William Farner
037     */
038    public class Rate<T extends Number> extends SampledStat<Double> {
039    
040      private static final int DEFAULT_WINDOW_SIZE = 1;
041      private static final double DEFAULT_SCALE_FACTOR = 1;
042      private static final long NANOS_PER_SEC = Amount.of(1L, Time.SECONDS).as(Time.NANOSECONDS);
043    
044      private final Supplier<T> inputAccessor;
045      private final Clock clock;
046      private final double scaleFactor;
047    
048      private final LinkedBlockingDeque<Pair<Long, Double>> samples;
049    
050      private Rate(String name, Supplier<T> inputAccessor, int windowSize, double scaleFactor,
051          Clock clock) {
052        super(name, 0d);
053    
054        this.inputAccessor = Preconditions.checkNotNull(inputAccessor);
055        this.clock = Preconditions.checkNotNull(clock);
056        samples = new LinkedBlockingDeque<Pair<Long, Double>>(windowSize);
057        Preconditions.checkArgument(scaleFactor != 0, "Scale factor must be non-zero!");
058        this.scaleFactor = scaleFactor;
059      }
060    
061      public static <T extends Number> Builder<T> of(Stat<T> input) {
062        return new Builder<T>(input);
063      }
064    
065      public static Builder<AtomicInteger> of(String name, AtomicInteger input) {
066        return new Builder<AtomicInteger>(name, input);
067      }
068    
069      public static Builder<AtomicLong> of(String name, AtomicLong input) {
070        return new Builder<AtomicLong>(name, input);
071      }
072    
073      @Override
074      public Double doSample() {
075        T newSample = inputAccessor.get();
076        long newTimestamp = clock.nowNanos();
077    
078        double rate = 0;
079        if (!samples.isEmpty()) {
080          Pair<Long, Double> oldestSample = samples.peekLast();
081    
082          double dy = newSample.doubleValue() - oldestSample.getSecond();
083          double dt = newTimestamp - oldestSample.getFirst();
084          rate = dt == 0 ? 0 : (NANOS_PER_SEC * scaleFactor * dy) / dt;
085        }
086    
087        if (samples.remainingCapacity() == 0) samples.removeLast();
088        samples.addFirst(Pair.of(newTimestamp, newSample.doubleValue()));
089    
090        return rate;
091      }
092    
093      public static class Builder<T extends Number> {
094    
095        private String name;
096        private int windowSize = DEFAULT_WINDOW_SIZE;
097        private double scaleFactor = DEFAULT_SCALE_FACTOR;
098        private Supplier<T> inputAccessor;
099        private Clock clock = Clock.SYSTEM_CLOCK;
100    
101        Builder(String name, final T input) {
102          this.name = name;
103          inputAccessor = Suppliers.ofInstance(input);
104        }
105    
106        Builder(final Stat<T> input) {
107          Stats.export(input);
108          this.name = input.getName() + "_per_sec";
109          inputAccessor = new Supplier<T>() {
110            @Override public T get() { return input.read(); }
111          };
112        }
113    
114        public Builder<T> withName(String name) {
115          this.name = name;
116          return this;
117        }
118    
119        public Builder<T> withWindowSize(int windowSize) {
120          this.windowSize = windowSize;
121          return this;
122        }
123    
124        public Builder<T> withScaleFactor(double scaleFactor) {
125          this.scaleFactor = scaleFactor;
126          return this;
127        }
128    
129        @VisibleForTesting
130        Builder<T> withClock(Clock clock ) {
131          this.clock = clock;
132          return this;
133        }
134    
135        public Rate<T> build() {
136          return new Rate<T>(name, inputAccessor, windowSize, scaleFactor, clock);
137        }
138      }
139    }
