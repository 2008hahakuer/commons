018    package com.twitter.common.stats;
019    
020    import com.google.common.annotations.VisibleForTesting;
021    import com.google.common.base.Preconditions;
022    import com.google.common.collect.ImmutableMap;
023    import com.google.common.collect.Lists;
024    import com.twitter.common.base.MorePreconditions;
025    import com.twitter.common.util.Sampler;
026    
027    import java.util.Collections;
028    import java.util.Comparator;
029    import java.util.LinkedList;
030    import java.util.Map;
031    
032    /**
033     * A stats tracker to export percentiles of inputs based on a sampling rate.
034     *
035     * @author William Farner
036     */
037    public class Percentile<T extends Number> {
038    
039      @VisibleForTesting
040      static final int MAX_BUFFER_SIZE = 10000;
041    
042      private final Sampler sampler;
043    
044      private final Map<Integer, SampledStat<Double>> percentiles;
045      @VisibleForTesting
046      final LinkedList<T> samples = Lists.newLinkedList();
047    
048      /**
049       * Creates a new percentile tracker.
050       *
051       * A percentile tracker will randomly sample recorded events with the given sampling rate, and
052       * will automatically register variables to track the percentiles requested.
053       * Percentiles are windowed, in that once the last percentile is sampled, all recorded values are
054       * flushed in preparation for the next window.
055       *
056       * @param name The name of the value whose percentile is being tracked.
057       * @param samplePercent The percent of events to sample [0, 100].
058       * @param percentiles The percentiles to track.
059       */
060      public Percentile(String name, float samplePercent, int... percentiles) {
061        this(name, new Sampler(samplePercent), percentiles);
062      }
063    
064      /**
065       * Creates a new percentile tracker.
066       *
067       * A percentile tracker will randomly sample recorded events with the given sampling rate, and
068       * will automatically register variables to track the percentiles requested.
069       * Percentiles are windowed, in that once the last percentile is sampled, all recorded values are
070       * flushed in preparation for the next window.
071       *
072       * @param name The name of the value whose percentile is being tracked.
073       * @param sampler The sampler to use for selecting recorded events.
074       * @param percentiles The percentiles to track.
075       */
076      public Percentile(String name, Sampler sampler, int... percentiles) {
077        MorePreconditions.checkNotBlank(name);
078        Preconditions.checkNotNull(percentiles);
079        Preconditions.checkArgument(percentiles.length > 0, "Must specify at least one percentile.");
080    
081        this.sampler = Preconditions.checkNotNull(sampler);
082    
083        ImmutableMap.Builder<Integer, SampledStat<Double>> builder =
084            new ImmutableMap.Builder<Integer, SampledStat<Double>>();
085    
086        for (int i = 0; i < percentiles.length; i++) {
087          boolean sortFirst = i == 0;
088          boolean flushAfterSampling = i == percentiles.length - 1;
089          String statName = String.format("%s_%dth_percentile", name, percentiles[i]);
090          SampledStat<Double> stat =
091              new PercentileVar(statName, percentiles[i], sortFirst, flushAfterSampling);
092          Stats.export(stat);
093          builder.put(percentiles[i], stat);
094        }
095    
096        this.percentiles = builder.build();
097      }
098    
099      @VisibleForTesting
100      SampledStat<Double> getPercentile(int percentile) {
101        return percentiles.get(percentile);
102      }
103    
104      /**
105       * Records an event.
106       *
107       * @param value The value to record if it is randomly selected based on the sampling rate.
108       */
109      public void record(T value) {
110        if (sampler.select()) {
111          synchronized (samples) {
112            samples.addLast(value);
113    
114            while (samples.size() > MAX_BUFFER_SIZE) samples.removeFirst();
115          }
116        }
117      }
118    
119      private class PercentileVar extends SampledStat<Double> {
120        private final int percentile;
121        private final boolean sortFirst;
122        private final boolean flushAfterSampling;
123    
124        PercentileVar(String name, int percentile, boolean sortFirst, boolean flushAfterSampling) {
125          super(name, 0d);
126          this.percentile = percentile;
127          this.sortFirst = sortFirst;
128          this.flushAfterSampling = flushAfterSampling;
129        }
130    
131        @Override
132        public Double doSample() {
133          synchronized (samples) {
134            if (samples.isEmpty()) return 0d;
135            if (sortFirst) Collections.sort(samples, NUMBER_COMPARATOR);
136    
137            int selectIndex = ((samples.size() * percentile) / 100) - 1;
138            selectIndex = selectIndex < 0 ? 0 : selectIndex;
139    
140            // Whether we need to interpolate between two values to calculate the percentile.
141            boolean interpolate = ((selectIndex + 1) * 100) / percentile < samples.size();
142    
143            double value = samples.get(selectIndex).doubleValue();
144            if (interpolate) value = (value + samples.get(selectIndex + 1).doubleValue()) / 2;
145    
146            if (flushAfterSampling) samples.clear();
147            return value;
148          }
149        }
150      }
151    
152      private static final Comparator<Number> NUMBER_COMPARATOR = new Comparator<Number>() {
153        @Override
154        public int compare(Number a, Number b) {
155          return (int) Math.signum(a.doubleValue() - b.doubleValue());
156        }
157      };
158    }
