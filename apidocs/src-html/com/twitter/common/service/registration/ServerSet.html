017    package com.twitter.common.service.registration;
018    
019    /**
020     * A server set provides real-time membership status of a set of servers.
021     * One use case is for all servers in a cluster to join a server set
022     * enabling clients to connect to some server in the server set.
023     *
024     * The members of a server set are likely to be up but it is not guaranteed.
025     *
026     * A server set is constructed with a "location".
027     * Any server joining a server set at location will show up in all other server sets
028     * constructed with the same location. If two equal servers join the same server set,
029     * a listener on the server set will see one or the other in the set.
030     *
031     * A server can unjoin a server set to remove itself from the server set.
032     * However, if a server is killed or becomes unresponsive before it can unjoin,
033     * it may automatically be removed from the server set.
034     *
035     * The server set can be "connected" or "disconnected". While disconnected, the server
036     * set membership will remain unchanged.
037     *
038     * @author Patrick Chan
039     */
040    public interface ServerSet {
041      /**
042       * Adds the supplied server to the server set.
043       * This method will succeed and not block, even if this process is currently disconnected
044       * from the network.
045       *
046       * This method cannot be called again, even with a different Server object, until
047       * after unjoin() is called.
048       *
049       * This method is not thread-safe.
050       *
051       * This method tries to catch and handle all connection-related exceptions.
052       * Any thrown runtime exceptions indicate a permanent configuration or programming error.
053       *
054       * @param server    non-null server object representing an endpoint.
055       */
056      void join(Server server);
057    
058      /**
059       * Removes the server from the server set. The supplied server object must equal the
060       * server object supplied to join().
061       *
062       * This method is not thread-safe.
063       *
064       * @param server   non-null server that must be equal to the server object supplied to join().
065       */
066      void unjoin(Server server);
067    
068      /**
069       * Registers a listener to receive change notices for this server set.
070       * The listener will be notified immediately with the current set and then
071       * notified whenever the server set membership changes.
072       *
073       * This method can only be called once.
074       *
075       * @param listener  non-null listener that's notified of server set changes.
076       */
077      void setListener(ServerSetListener listener);
078    }
