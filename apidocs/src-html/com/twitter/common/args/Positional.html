001    package com.twitter.common.args;
002    
003    import java.lang.annotation.Retention;
004    import java.lang.annotation.Target;
005    
006    import static java.lang.annotation.ElementType.FIELD;
007    import static java.lang.annotation.RetentionPolicy.RUNTIME;
008    
009    /**
010     * Annotation to mark an {@link Arg} for gathering the positional command line arguments.
011     */
012    @Retention(RUNTIME)
013    @Target({FIELD})
014    public @interface Positional {
015      /**
016       * The help string to display on the command line in a usage message.
017       *
018       * @return Help string.
019       */
020      String help();
021    
022      /**
023       * The parser class to use for parsing the positional arguments.  The parser must return the same
024       * type as the field being annotated.
025       *
026       * @return Custom parser for this type.
027       */
028      // The default is fully qualified to work around an apt bug:
029      // http://bugs.sun.com/view_bug.do?bug_id=6512707
030      Class<? extends Parser> parser() default com.twitter.common.args.Parser.class;
031    }
