001    package com.twitter.common.args;
002    
003    import java.lang.reflect.Type;
004    
005    /**
006     * An interface to a command line argument parser.
007     *
008     * @param <T> The base class this parser can parse all subtypes of.
009     */
010    public interface Parser<T> {
011    
012      /**
013       * Parses strings as arguments of a given subtype of {@code T}.
014       *
015       * @param parserOracle The registered parserOracle for delegation.
016       * @param type The target type of the parsed argument.
017       * @param raw The raw value of the argument.
018       * @return A value of the given type parsed from the raw value.
019       * @throws IllegalArgumentException if the raw value could not be parsed into a value of the
020       *     given type.
021       */
022      T parse(ParserOracle parserOracle, Type type, String raw) throws IllegalArgumentException;
023    }
