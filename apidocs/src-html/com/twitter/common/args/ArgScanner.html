063    /**
064     * Argument scanning, parsing, and validating system.  This class is designed recursively scan a
065     * package for declared arguments, parse the values based on the declared type, and validate against
066     * any constraints that the arugment is decorated with.
067     *
068     * The supported argument formats are:
069     *   -arg_name=arg_value
070     *   -arg_name arg_value
071     * Where {@code arg_value} may be single or double-quoted if desired or necessary to prevent
072     * splitting by the terminal application.
073     *
074     * A special format for boolean arguments is also supported.  The following syntaxes all set the
075     * {@code bool_arg} to {@code true}:
076     *   -bool_arg
077     *   -bool_arg=true
078     *   -no_bool_arg=false (double negation)
079     *
080     * Likewise, the following would set {@code bool_arg} to {@code false}:
081     *   -no_bool_arg
082     *   -bool_arg=false
083     *   -no_bool_arg=true (negation)
084     *
085     * As with the general argument format, spaces may be used in place of equals for boolean argument
086     * assignment.
087     *
088     * TODO(William Farner): Make default verifier and parser classes package-private and in this package.
089     *
090     * @author William Farner
091     */
092    public class ArgScanner {
093    
094      private static final Logger LOG = Logger.getLogger(ArgScanner.class.getName());
095    
096      // Negation prefix for booleans.
097      private static final String NEGATE_BOOLEAN = "no_";
098    
099      // Regular expression that defines the name format for an argument.
100      private static final String ARG_NAME_RE = "[\\w\\-\\.]+";
101      private static final Pattern ARG_NAME_PATTERN = Pattern.compile(ARG_NAME_RE);
102      private static final Pattern NEGATED_BOOLEAN_PATTERN = Pattern.compile(
103          String.format("(%s)?%s(%s)", ARG_NAME_RE, NEGATE_BOOLEAN, ARG_NAME_RE));
104    
105      // Pattern for the required argument format.
106      private static final Pattern ARG_PATTERN =
107          Pattern.compile(String.format("-(%s)(?:(?:=| +)(.*))?", ARG_NAME_RE));
108    
109      private ArgScanner() {
110        // Utility.
111      }

113      /**
114       * Recursively scans a package prefix for declared arguments, and applies the provided argument
115       * values, after parsing and validating.
116       *
117       * @param packagePrefix Prefix of package to scan.
118       * @param args Argument values to parse, validate, and apply.
119       * @throws IllegalArgumentException If the arguments provided are invalid based on the declared
120       *    arguments found.
121       */
122      public static void parse(Iterable<String> packagePrefix, Map<String, String> args) {
123        process(scan(packagePrefix), args);
124      }
125    
126      /**
127       * Convenience ethod to call {@link #parse(Iterable, Map)} that will handle mapping of argument
128       * keys to values.
129       *
130       * @param packagePrefix Prefix of package to scan.
131       * @param args Argument values to map, parse, validate, and apply.
132       * @throws IllegalArgumentException If the arguments provided are invalid based on the declared
133       *    arguments found.
134       */
135      public static void parse(Iterable<String> packagePrefix, String... args)
136          throws IllegalArgumentException {
137        parse(packagePrefix, mapArguments(args));
138      }

140      // Regular expression to identify a possible dangling assignment.
141      // A dangling assignment occurs in two cases:
142      //   - The command line used spaces between arg names and values, causing the name and value to
143      //     end up in different command line arg array elements.
144      //   - The command line is using the short form for a boolean argument,
145      //     such as -use_feature, or -no_use_feature.
146      private static final String DANGLING_ASSIGNMENT_RE = String.format("^-%s", ARG_NAME_RE);
147      private static final Pattern DANGLING_ASSIGNMENT_PATTERN =
148          Pattern.compile(DANGLING_ASSIGNMENT_RE);
149    
150      // Pattern to identify a full assignment, which would be disassociated from a preceding dangling
151      // assignment.
152      private static final Pattern ASSIGNMENT_PATTERN =
153          Pattern.compile(String.format("%s=.+", DANGLING_ASSIGNMENT_RE));
154    
155      @VisibleForTesting static List<String> joinKeysToValues(String... args) {
156        List<String> joinedArgs = Lists.newArrayList();
157        String unmappedKey = null;
158        for (String arg : args) {
159          if (unmappedKey == null) {
160            if (DANGLING_ASSIGNMENT_PATTERN.matcher(arg).matches()) {
161              // Beginning of a possible dangling assignment.
162              unmappedKey = arg;
163            } else {
164              joinedArgs.add(arg);
165            }
166          } else {
167            if (ASSIGNMENT_PATTERN.matcher(arg).matches()) {
168              // Full assignment, disassociate from dangling assignment.
169              joinedArgs.add(unmappedKey);
170              joinedArgs.add(arg);
171              unmappedKey = null;
172            } else if (DANGLING_ASSIGNMENT_PATTERN.matcher(arg).find()) {
173              // Another dangling assignment, this could be two sequential boolean args.
174              joinedArgs.add(unmappedKey);
175              unmappedKey = arg;
176            } else {
177              // Join the dangling key with its value.
178              joinedArgs.add(unmappedKey + "=" + arg);
179              unmappedKey = null;
180            }
181          }
182        }
183    
184        if (unmappedKey != null) {
185          joinedArgs.add(unmappedKey);
186        }
187    
188        return joinedArgs;
189      }
190    
191      private static final Pattern QUOTE_PATTERN = Pattern.compile("(['\"])([^\\\1]*)\\1");
192    
193      private static String stripQuotes(String str) {
194        Matcher matcher = QUOTE_PATTERN.matcher(str);
195        return matcher.matches() ? matcher.group(2) : str;
196      }
197    
198      /**
199       * Scans through args, mapping keys to values even if the arg values are 'dangling' and reside
200       * in different array entries than the respective keys.
201       *
202       * @param args Arguments to build into a map.
203       * @return Map from argument key (arg name) to value.
204       */
205      public static Map<String, String> mapArguments(String... args) {
206        ImmutableMap.Builder<String, String> argMap = ImmutableMap.builder();
207        for (String arg : joinKeysToValues(args)) {
208          Matcher matcher = ARG_PATTERN.matcher(arg);
209          checkArgument(matcher.matches(),
210              String.format("Argument '%s' does not match required format -arg_name=arg_value", arg));
211    
212          String rawValue = matcher.group(2);
213          // An empty string denotes that the argument was passed with no value.
214          rawValue = rawValue == null ? "" : stripQuotes(rawValue);
215          argMap.put(matcher.group(1), rawValue);
216        }
217    
218        return argMap.build();
219      }

221      /**
222       * Extracts the @Arg annotation from a Field and includes it as a pair.
223       */
224      private static final Function<Field, Pair<Field, CmdLine>> PAIR_WITH_ANNOTATION =
225          new Function<Field, Pair<Field, CmdLine>>() {
226            @Override public Pair<Field, CmdLine> apply(Field field) {
227              CmdLine cmdLineDef = field.getAnnotation(CmdLine.class);
228              Preconditions.checkNotNull(cmdLineDef, "No Arg annotation for field " + field);
229              return Pair.of(field, cmdLineDef);
230            }
231          };
232    
233      /**
234       * Performs multiple checks to ensure the definition of an argument is good.
235       */
236      private static final Closure<Pair<Field, CmdLine>> CHECK_ARG_DEF =
237          new Closure<Pair<Field, CmdLine>>() {
238        @Override public void execute(Pair<Field, CmdLine> argDef) {
239          checkArgument(Modifier.isStatic(argDef.getFirst().getModifiers()),
240              "Non-static argument fields are not supported, found " + argDef.getFirst());
241    
242          String argName = argDef.getSecond().name();
243          checkArgument(ARG_NAME_PATTERN.matcher(argName).matches(),
244              String.format("Argument name '%s' on %s does not match required pattern %s",
245                  argName, argDef.getFirst(), ARG_NAME_RE));
246    
247          checkArgument(getParser(argDef.getSecond(), argDef.getFirst()) != null,
248              "No parser found for type " + TypeUtil.getTypeParamClass(argDef.getFirst())
249                  + ", for arg field " + argDef.getFirst().getName());
250        }
251      };
252    
253      /**
254       * Extracts the name from an @Arg.
255       */
256      private static final Function<CmdLine, String> GET_ARG_NAME = new Function<CmdLine, String>() {
257          @Override public String apply(CmdLine cmdLine) {
258            return cmdLine.name();
259          }
260        };
261    
262      /**
263       * Gets the canonical name for an @Arg, based on the class containing the field it annotates.
264       */
265      private static final Function<Pair<Field, CmdLine>, String> GET_CANONICAL_ARG_NAME =
266          new Function<Pair<Field, CmdLine>, String>() {
267            @Override public String apply(Pair<Field, CmdLine> argDef) {
268              return argDef.getFirst().getDeclaringClass().getCanonicalName()
269                  + "." + argDef.getSecond().name();
270            }
271          };
272    
273      /**
274       * Gets the canonical negated name for an @Arg.
275       */
276      private static final Function<Pair<Field, CmdLine>, String> GET_CANONICAL_NEGATED_ARG_NAME =
277          new Function<Pair<Field, CmdLine>, String>() {
278            @Override public String apply(Pair<Field, CmdLine> argDef) {
279              return argDef.getFirst().getDeclaringClass().getCanonicalName()
280                  + "." + NEGATE_BOOLEAN + argDef.getSecond().name();
281            }
282          };
283    
284      private static final Function<String, String> PREPEND_NEGATION = new Function<String, String>() {
285        @Override public String apply(String name) {
286          return NEGATE_BOOLEAN + name;
287        }
288      };

290      private static <T> Set<T> dropCollisions(Iterable<T> input) {
291        Set<T> copy = Sets.newHashSet();
292        Set<T> collisions = Sets.newHashSet();
293        for (T entry: input) {
294          if (!copy.add(entry)) {
295            collisions.add(entry);
296          }
297        }
298    
299        copy.removeAll(collisions);
300        return copy;
301      }

303      /**
304       * Applies argument values to fields based on their annotations.
305       *
306       * @param argFields Fields to apply argument values to.
307       * @param args Unparsed argument values.
308       */
309      @VisibleForTesting
310      static void process(Set<Field> argFields, Map<String, String> args) {
311    
312        final Set<String> argsFailedToParse = Sets.newHashSet();
313        final Set<String> argsConstraintsFailed = Sets.newHashSet();
314    
315        final Iterable<Pair<Field, CmdLine>> argDefs =
316            Iterables.transform(argFields, PAIR_WITH_ANNOTATION);
317    
318        for (Pair<Field, CmdLine> argDef : argDefs) {
319          CHECK_ARG_DEF.execute(argDef);
320        }
321    
322        Iterable<String> argShortNames = Iterables.transform(argDefs,
323            Functions.compose(GET_ARG_NAME, Pair.<Field, CmdLine>second()));
324        Set<String> argShortNamesNoCollisions = dropCollisions(argShortNames);
325        Set<String> collisionsDropped = Sets.difference(ImmutableSet.copyOf(argShortNames),
326            argShortNamesNoCollisions);
327        if (!collisionsDropped.isEmpty()) {
328          LOG.warning("Found argument name collisions, args must be referenced by canonical names: "
329              + collisionsDropped);
330        }
331    
332        Predicate<Pair<Field, CmdLine>> isBoolean = Predicates.compose(
333            Predicates.<Class>equalTo(Boolean.class),
334            Functions.compose(TypeUtil.GET_TYPE_PARAM_CLASS, Pair.<Field, CmdLine>first()));
335    
336        final Function<Pair<Field, CmdLine>, String> getFieldArgName =
337            Functions.compose(GET_ARG_NAME, Pair.<Field, CmdLine>second());
338    
339        final Map<String, Pair<Field, CmdLine>> argsByName =
340            ImmutableMap.<String, Pair<Field, CmdLine>>builder()
341            // Map by short arg name -> arg def.
342            .putAll(Maps.uniqueIndex(Iterables.filter(argDefs,
343                Predicates.compose(Predicates.in(argShortNamesNoCollisions), getFieldArgName)),
344                getFieldArgName))
345            // Map by canonical arg name -> arg def.
346            .putAll(Maps.uniqueIndex(argDefs, GET_CANONICAL_ARG_NAME))
347            // Map by negated short arg name (for booleans)
348            .putAll(Maps.uniqueIndex(Iterables.filter(argDefs, isBoolean),
349                Functions.compose(PREPEND_NEGATION, getFieldArgName)))
350            // Map by negated canonical arg name (for booleans)
351            .putAll(Maps.uniqueIndex(Iterables.filter(argDefs, isBoolean),
352                GET_CANONICAL_NEGATED_ARG_NAME))
353            .build();
354    
355        // TODO(William Farner): Make sure to disallow duplicate arg specification by short and canonical
356        //    names.
357    
358        // TODO(William Farner): Support non-atomic argument constraints.  @OnlyIfSet, @OnlyIfNotSet,
359        //    @ExclusiveOf to define inter-argument constraints.
360    
361        Set<String> recognizedArgs = Sets.intersection(argsByName.keySet(), args.keySet());
362    
363        for (String argName : recognizedArgs) {
364          String argValue = args.get(argName);
365          Field argField = argsByName.get(argName).getFirst();
366          CmdLine annotation = argsByName.get(argName).getSecond();
367    
368          Object assignmentValue;
369          try {
370            assignmentValue = getParser(annotation, argField)
371                .parse(TypeUtil.getTypeParam(argField), argValue);
372          } catch (IllegalArgumentException e) {
373            argsFailedToParse.add(argName + " - " + e.getMessage());
374            continue;
375          }
376    
377          // If the arg type is boolean, check if the command line uses the negated boolean form.
378          if (TypeUtil.getTypeParamClass(argField) == Boolean.class) {
379            Matcher negatedNameMatcher = NEGATED_BOOLEAN_PATTERN.matcher(argName);
380            if (negatedNameMatcher.matches()) {
381              String argShortName = negatedNameMatcher.group(negatedNameMatcher.groupCount());
382              String argCanonicalName = negatedNameMatcher.groupCount() == 2 ?
383                  negatedNameMatcher.group(1) + argShortName : null;
384              if (Iterables.any(argsByName.keySet(),
385                  Predicates.in(Arrays.asList(argShortName, argCanonicalName)))) {
386                assignmentValue = (!(Boolean) assignmentValue);
387              }
388            }
389          }
390    
391          setField(argField, assignmentValue);
392        }
393    
394        Set<String> commandLineArgumentInfos = Sets.newTreeSet();
395        for (Pair<Field, CmdLine> argDef : argDefs) {
396          Field argField = argDef.getFirst();
397          Arg arg = getArg(argField);
398    
399          commandLineArgumentInfos.add(String.format("%s (%s.%s): %s",
400              argDef.getSecond().name(), argField.getDeclaringClass().getName(), argField.getName(),
401              arg.uncheckedGet()));
402    
403          try {
404            checkConstraints(TypeUtil.getTypeParamClass(argField), arg.uncheckedGet(),
405                argField.getAnnotations());
406          } catch (IllegalArgumentException e) {
407            argsConstraintsFailed.add(argDef.getSecond().name() + " - " + e.getMessage());
408          }
409        }
410        infoLog("-------------------------------------------------------------------------");
411        infoLog("Command line argument values");
412        for (String commandLineArgumentInfo : commandLineArgumentInfos) {
413          infoLog(commandLineArgumentInfo);
414        }
415        infoLog("-------------------------------------------------------------------------");
416    
417        ImmutableMultimap<String, String> warningMessages =
418            ImmutableMultimap.<String, String>builder()
419            .putAll("Unrecognized arguments", Sets.difference(args.keySet(), argsByName.keySet()))
420            .putAll("Failed to parse", argsFailedToParse)
421            .putAll("Value did not meet constraints", argsConstraintsFailed)
422            .build();
423    
424        if (!warningMessages.isEmpty()) {
425          StringBuilder sb = new StringBuilder();
426          for (Map.Entry<String, Collection<String>> warnings : warningMessages.asMap().entrySet()) {
427            sb.append(warnings.getKey()).append(":\n\t").append(Joiner.on("\n\t")
428                .join(warnings.getValue())).append("\n");
429          }
430          throw new IllegalArgumentException(sb.toString());
431        }
432      }

434      private static void infoLog(String msg) {
435        System.out.println(msg);
436      }

438      private static void checkConstraints(Class<?> fieldClass, Object value, Annotation[] annotations) {
439        for (Annotation annotation : maybeFilterNullable(fieldClass, value, annotations)) {
440          verify(fieldClass, value, annotation);
441        }
442      }
443    
444      private static Iterable<Annotation> maybeFilterNullable(Class<?> fieldClass, Object value,
445          Annotation[] annotations) {
446    
447        // Apply all the normal constraint annotations
448        if (value != null) {
449          return Arrays.asList(annotations);
450        }
451    
452        // The value is null so skip verifications unless a @NotNull was specified as a constraint
453        for (Annotation annotation: annotations) {
454          if (annotation instanceof NotNull) {
455            verify(fieldClass, value, annotation); // will throw
456          }
457        }
458        return ImmutableList.of();
459      }
460    
461      @SuppressWarnings("unchecked")
462      private static void verify(Class<?> fieldClass, Object value, Annotation annotation) {
463        Verifier verifier = Constraints.get(fieldClass, annotation);
464        if (verifier != null) {
465          verifier.verify(value, annotation);
466        }
467      }
468    
469      private static Parser getParser(CmdLine cmdLine, Field argField) {
470        Preconditions.checkArgument(argField.getType() == Arg.class,
471            "Field is annotated with @CmdLine but is not of Arg type: " + argField);
472        Class cls = TypeUtil.getTypeParamClass(argField);
473    
474        try {
475          return cmdLine.parser() == Parser.class ? Parsers.get(cls) : cmdLine.parser().newInstance();
476        } catch (InstantiationException e) {
477          throw new RuntimeException("Failed to instantiate parser " + cmdLine.parser());
478        } catch (IllegalAccessException e) {
479          throw new RuntimeException("No access to instantiate parser " + cmdLine.parser());
480        }
481      }
482    
483      @SuppressWarnings("unchecked")
484      private static void setField(Field field, Object value) {
485        getArg(field).set(value);
486      }
487    
488      private static Arg getArg(Field field) {
489        field.setAccessible(true);
490        try {
491          return ((Arg) field.get(null));
492        } catch (IllegalAccessException e) {
493          throw new RuntimeException("Failed to set value for " + field);
494        }
495      }
496    
497      private static Set<Field> scan(Iterable<String> scanPackagePrefixes) {
498        FilterBuilder filterBuilder = new FilterBuilder();
499        ImmutableSet.Builder<URL> urlsBuilder = ImmutableSet.builder();
500        for (String packagePrefix : scanPackagePrefixes) {
501          filterBuilder.add(new FilterBuilder.Include(FilterBuilder.prefix(packagePrefix)));
502          urlsBuilder.addAll(ClasspathHelper.getUrlsForPackagePrefix(packagePrefix));
503        }
504        filterBuilder.add(new FilterBuilder.Exclude(".*Test"));
505    
506        Set<URL> urls = urlsBuilder.build();
507        LOG.info("Scanning resource URLs: " + urls);
508    
509        return new Reflections(new ConfigurationBuilder()
510            .filterInputsBy(filterBuilder)
511            .setUrls(urls)
512            .useParallelExecutor()
513            .setScanners(new FieldAnnotationsScanner()))
514            .getFieldsAnnotatedWith(CmdLine.class);
515      }
