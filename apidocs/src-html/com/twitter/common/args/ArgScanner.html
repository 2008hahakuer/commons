065    /**
066     * Argument scanning, parsing, and validating system.  This class is designed recursively scan a
067     * package for declared arguments, parse the values based on the declared type, and validate against
068     * any constraints that the arugment is decorated with.
069     *
070     * The supported argument formats are:
071     *   -arg_name=arg_value
072     *   -arg_name arg_value
073     * Where {@code arg_value} may be single or double-quoted if desired or necessary to prevent
074     * splitting by the terminal application.
075     *
076     * A special format for boolean arguments is also supported.  The following syntaxes all set the
077     * {@code bool_arg} to {@code true}:
078     *   -bool_arg
079     *   -bool_arg=true
080     *   -no_bool_arg=false (double negation)
081     *
082     * Likewise, the following would set {@code bool_arg} to {@code false}:
083     *   -no_bool_arg
084     *   -bool_arg=false
085     *   -no_bool_arg=true (negation)
086     *
087     * As with the general argument format, spaces may be used in place of equals for boolean argument
088     * assignment.
089     *
090     * TODO(William Farner): Make default verifier and parser classes package-private and in this
091     * package.
092     *
093     * @author William Farner
094     */
095    public final class ArgScanner {

155      /**
156       * Applies the provided argument values to all {@literal @CmdLine} {@code Arg} fields discovered
157       * on the classpath.
158       *
159       * @param args Argument values to map, parse, validate, and apply.
160       * @return {@code true} if the given {@code args} were successfully applied to their corresponding
161       *     {@link Arg} fields.
162       * @throws ArgScanException if there was a problem loading {@literal @CmdLine} argument
163       *    definitions
164       * @throws IllegalArgumentException If the arguments provided are invalid based on the declared
165       *    arguments found.
166       */
167      public boolean parse(Iterable<String> args) {
168        return parse(ArgFilters.SELECT_ALL, args);
169      }
170    
171      /**
172       * Applies the provided argument values to any {@literal @CmdLine} {@code Arg} fields discovered
173       * on the classpath and accepted by the given {@code filter}.
174       *
175       * @param filter A predicate that selects or rejects scanned {@literal @CmdLine} fields for
176       *    argument application.
177       * @param args Argument values to map, parse, validate, and apply.
178       * @return {@code true} if the given {@code args} were successfully applied to their corresponding
179       *     {@link Arg} fields.
180       * @throws ArgScanException if there was a problem loading {@literal @CmdLine} argument
181       *    definitions
182       * @throws IllegalArgumentException If the arguments provided are invalid based on the declared
183       *    arguments found.
184       */
185      public boolean parse(Predicate<Field> filter, Iterable<String> args) {
186        Configuration configuration = load();
187        ArgumentInfo argumentInfo = Args.fromConfiguration(configuration, filter);
188        ParserOracle parserOracle = Parsers.fromConfiguration(configuration);
189        Verifiers verifiers = Verifiers.fromConfiguration(configuration);
190        Pair<ImmutableMap<String, String>, List<String>> results = mapArguments(args);
191        return process(parserOracle, verifiers, argumentInfo, results.getFirst(), results.getSecond());
192      }
193    
194      private Configuration load() {
195        try {
196          return Configuration.load();
197        } catch (IOException e) {
198          throw new ArgScanException(e);
199        }
200      }

202      // Regular expression to identify a possible dangling assignment.
203      // A dangling assignment occurs in two cases:
204      //   - The command line used spaces between arg names and values, causing the name and value to
205      //     end up in different command line arg array elements.
206      //   - The command line is using the short form for a boolean argument,
207      //     such as -use_feature, or -no_use_feature.
208      private static final String DANGLING_ASSIGNMENT_RE = String.format("^-%s", ARG_NAME_RE);
209      private static final Pattern DANGLING_ASSIGNMENT_PATTERN =
210          Pattern.compile(DANGLING_ASSIGNMENT_RE);
211    
212      // Pattern to identify a full assignment, which would be disassociated from a preceding dangling
213      // assignment.
214      private static final Pattern ASSIGNMENT_PATTERN =
215          Pattern.compile(String.format("%s=.+", DANGLING_ASSIGNMENT_RE));
216    
217      @VisibleForTesting static List<String> joinKeysToValues(Iterable<String> args) {
218        List<String> joinedArgs = Lists.newArrayList();
219        String unmappedKey = null;
220        for (String arg : args) {
221          if (unmappedKey == null) {
222            if (DANGLING_ASSIGNMENT_PATTERN.matcher(arg).matches()) {
223              // Beginning of a possible dangling assignment.
224              unmappedKey = arg;
225            } else {
226              joinedArgs.add(arg);
227            }
228          } else {
229            if (ASSIGNMENT_PATTERN.matcher(arg).matches()) {
230              // Full assignment, disassociate from dangling assignment.
231              joinedArgs.add(unmappedKey);
232              joinedArgs.add(arg);
233              unmappedKey = null;
234            } else if (DANGLING_ASSIGNMENT_PATTERN.matcher(arg).find()) {
235              // Another dangling assignment, this could be two sequential boolean args.
236              joinedArgs.add(unmappedKey);
237              unmappedKey = arg;
238            } else {
239              // Join the dangling key with its value.
240              joinedArgs.add(unmappedKey + "=" + arg);
241              unmappedKey = null;
242            }
243          }
244        }
245    
246        if (unmappedKey != null) {
247          joinedArgs.add(unmappedKey);
248        }
249    
250        return joinedArgs;
251      }

253      private static final Pattern QUOTE_PATTERN = Pattern.compile("(['\"])([^\\\1]*)\\1");
254    
255      private static String stripQuotes(String str) {
256        Matcher matcher = QUOTE_PATTERN.matcher(str);
257        return matcher.matches() ? matcher.group(2) : str;
258      }

260      /**
261       * Scans through args, mapping keys to values even if the arg values are 'dangling' and reside
262       * in different array entries than the respective keys.
263       *
264       * @param args Arguments to build into a map.
265       * @return A map from argument key (arg name) to value paired with a list of any leftover
266       *     positional arguments.
267       */
268      private static Pair<ImmutableMap<String, String>, List<String>> mapArguments(
269          Iterable<String> args) {
270    
271        ImmutableMap.Builder<String, String> argMap = ImmutableMap.builder();
272        List<String> positionalArgs = Lists.newArrayList();
273        for (String arg : joinKeysToValues(args)) {
274          if (!arg.startsWith("-")) {
275            positionalArgs.add(arg);
276          } else {
277            Matcher matcher = ARG_PATTERN.matcher(arg);
278            checkArgument(matcher.matches(),
279                String.format("Argument '%s' does not match required format -arg_name=arg_value", arg));
280    
281            String rawValue = matcher.group(2);
282            // An empty string denotes that the argument was passed with no value.
283            rawValue = rawValue == null ? "" : stripQuotes(rawValue);
284            argMap.put(matcher.group(1), rawValue);
285          }
286        }
287    
288        return Pair.of(argMap.build(), positionalArgs);
289      }

294      private static void checkArgDef(ParserOracle parserOracle, Field field, CmdLine cmdLine) {
295        checkArgDef(parserOracle, field, cmdLine.parser());
296    
297        String argName = cmdLine.name();
298        checkArgument(!HELP_ARGS.contains(argName),
299            String.format("Argument name '%s' is reserved for builtin argument help", argName));
300        checkArgument(ARG_NAME_PATTERN.matcher(argName).matches(),
301            String.format("Argument name '%s' on %s does not match required pattern %s",
302                argName, field, ARG_NAME_RE));
303      }

306      private static void checkArgDef(ParserOracle parserOracle, Field field,
307          Class<? extends Parser> custom) {
308    
309        checkArgument(Modifier.isStatic(field.getModifiers()),
310            "Non-static argument fields are not supported, found " + field);
311    
312        checkArgument(getParser(parserOracle, custom, field) != null,
313            "No parser found for type " + TypeUtil.getTypeParamClass(field)
314                + ", for arg field " + field.getName());
315      }

327       * Gets the canonical name for an @Arg, based on the class containing the field it annotates.

330          new Function<OptionInfo, String>() {
331            @Override public String apply(OptionInfo optionInfo) {
332              return optionInfo.field.getDeclaringClass().getCanonicalName()
333                  + "." + optionInfo.cmdLine.name();
334            }

337      /**
338       * Gets the canonical negated name for an @Arg.
339       */
340      private static final Function<OptionInfo, String> GET_CANONICAL_NEGATED_ARG_NAME =
341          new Function<OptionInfo, String>() {
342            @Override public String apply(OptionInfo optionInfo) {
343              return optionInfo.field.getDeclaringClass().getCanonicalName()
344                  + "." + NEGATE_BOOLEAN + optionInfo.cmdLine.name();
345            }
346          };

348      private static final Function<String, String> PREPEND_NEGATION = new Function<String, String>() {
349        @Override public String apply(String name) {
350          return NEGATE_BOOLEAN + name;
351        }

354      private static <T> Set<T> dropCollisions(Iterable<T> input) {
355        Set<T> copy = Sets.newHashSet();
356        Set<T> collisions = Sets.newHashSet();
357        for (T entry: input) {
358          if (!copy.add(entry)) {
359            collisions.add(entry);
360          }
361        }
362    
363        copy.removeAll(collisions);
364        return copy;
365      }

367      private static final Function<OptionInfo, String> GET_FIELD_ARG_NAME =
368          Functions.compose(GET_ARG_NAME, GET_CMD_LINE);

370      /**
371       * Applies argument values to fields based on their annotations.
372       *
373       * @param parserOracle ParserOracle available to parse raw args with.
374       * @param verifiers Verifiers available to verify argument constraints with.
375       * @param argumentInfo Fields to apply argument values to.
376       * @param args Unparsed argument values.
377       * @param positionalArgs The unparsed positional arguments.
378       * @return {@code true} if the given {@code args} were successfully applied to their
379       *     corresponding {@link com.twitter.common.args.Arg} fields.
380       */
381      private boolean process(final ParserOracle parserOracle, Verifiers verifiers,
382          ArgumentInfo argumentInfo, Map<String, String> args,
383          List<String> positionalArgs) {
384    
385        if (!Sets.intersection(args.keySet(), HELP_ARGS).isEmpty()) {
386          printHelp(verifiers, argumentInfo);
387          return false;
388        }
389    
390        Optional<PositionalInfo> positionalInfo = argumentInfo.positionalInfo;
391        checkArgument(positionalInfo.isPresent() || positionalArgs.isEmpty(),
392            "Positional arguments have been supplied but there is no Arg annotated to received them.");
393        if (positionalInfo.isPresent()) {
394          PositionalInfo info = positionalInfo.get();
395          checkArgDef(parserOracle, info.field, info.positional.parser());
396        }
397    
398        Iterable<OptionInfo> optionInfos = argumentInfo.optionInfos;
399        for (OptionInfo optionInfo : optionInfos) {
400          checkArgDef(parserOracle, optionInfo.field, optionInfo.cmdLine);
401        }
402    
403        final Set<String> argsFailedToParse = Sets.newHashSet();
404        final Set<String> argsConstraintsFailed = Sets.newHashSet();
405    
406        Iterable<String> argShortNames = Iterables.transform(optionInfos,
407            Functions.compose(GET_ARG_NAME, GET_CMD_LINE));
408        Set<String> argShortNamesNoCollisions = dropCollisions(argShortNames);
409        Set<String> collisionsDropped = Sets.difference(ImmutableSet.copyOf(argShortNames),
410            argShortNamesNoCollisions);
411        if (!collisionsDropped.isEmpty()) {
412          LOG.warning("Found argument name collisions, args must be referenced by canonical names: "
413              + collisionsDropped);
414        }
415    
416        Predicate<OptionInfo> isBoolean = Predicates.compose(
417            Predicates.<Class>equalTo(Boolean.class),
418            Functions.compose(TypeUtil.GET_TYPE_PARAM_CLASS, GET_FIELD));
419    
420        final Map<String, OptionInfo> argsByName =
421            ImmutableMap.<String, OptionInfo>builder()
422            // Map by short arg name -> arg def.
423            .putAll(Maps.uniqueIndex(Iterables.filter(optionInfos,
424                Predicates.compose(Predicates.in(argShortNamesNoCollisions), GET_FIELD_ARG_NAME)),
425                                   GET_FIELD_ARG_NAME))
426            // Map by canonical arg name -> arg def.
427            .putAll(Maps.uniqueIndex(optionInfos, GET_CANONICAL_ARG_NAME))
428            // Map by negated short arg name (for booleans)
429            .putAll(Maps.uniqueIndex(Iterables.filter(optionInfos, isBoolean),
430                Functions.compose(PREPEND_NEGATION, GET_FIELD_ARG_NAME)))
431            // Map by negated canonical arg name (for booleans)
432            .putAll(Maps.uniqueIndex(Iterables.filter(optionInfos, isBoolean),
433                GET_CANONICAL_NEGATED_ARG_NAME))
434            .build();
435    
436        // TODO(William Farner): Make sure to disallow duplicate arg specification by short and
437        // canonical names.
438    
439        // TODO(William Farner): Support non-atomic argument constraints.  @OnlyIfSet, @OnlyIfNotSet,
440        //    @ExclusiveOf to define inter-argument constraints.
441    
442        Set<String> recognizedArgs = Sets.intersection(argsByName.keySet(), args.keySet());
443    
444        for (String argName : recognizedArgs) {
445          String argValue = args.get(argName);
446          OptionInfo optionInfo = argsByName.get(argName);
447          Field argField = optionInfo.field;
448          Class<? extends Parser> parser = optionInfo.cmdLine.parser();
449    
450          Object assignmentValue;
451          try {
452            assignmentValue =
453                getParser(parserOracle, parser, argField)
454                    .parse(parserOracle, TypeUtil.getTypeParam(argField), argValue);
455          } catch (IllegalArgumentException e) {
456            argsFailedToParse.add(argName + " - " + e.getMessage());
457            continue;
458          }
459    
460          // If the arg type is boolean, check if the command line uses the negated boolean form.
461          if (TypeUtil.getTypeParamClass(argField) == Boolean.class) {
462            Matcher negatedNameMatcher = NEGATED_BOOLEAN_PATTERN.matcher(argName);
463            if (negatedNameMatcher.matches()) {
464              String argShortName = negatedNameMatcher.group(negatedNameMatcher.groupCount());
465              String argCanonicalName = negatedNameMatcher.groupCount() == 2 ?
466                  negatedNameMatcher.group(1) + argShortName : null;
467              if (Iterables.any(argsByName.keySet(),
468                  Predicates.in(Arrays.asList(argShortName, argCanonicalName)))) {
469                assignmentValue = (!(Boolean) assignmentValue);
470              }
471            }
472          }
473    
474          setField(argField, assignmentValue);
475        }
476    
477        if (positionalInfo.isPresent()) {
478          PositionalInfo info = positionalInfo.get();
479          final Field argField = info.field;
480          Class<? extends Parser> custom = info.positional.parser();
481    
482          // We're trying to extract T 2 levels down in: Arg<List<T>>
483          final Type elementType = TypeToken.extractTypeToken(TypeUtil.getTypeParam(argField));
484    
485          final Parser<?> parser = getParser(parserOracle, custom, TypeUtil.getRawType(elementType));
486          List<?> assignmentValue = Lists.newArrayList(Iterables.transform(positionalArgs,
487              new Function<String, Object>() {
488                @Override public Object apply(String argValue) {
489                  return parser.parse(parserOracle, elementType, argValue);
490                }
491              }));
492    
493          setField(argField, assignmentValue);
494        }
495    
496        Set<String> commandLineArgumentInfos = Sets.newTreeSet();
497    
498        Iterable<Pair<String, Field>> argFields = Iterables.transform(optionInfos,
499            new Function<OptionInfo, Pair<String, Field>>() {
500              @Override public Pair<String, Field> apply(OptionInfo optionInfo) {
501                return Pair.of(optionInfo.cmdLine.name(), optionInfo.field);
502              }
503            });
504        if (positionalInfo.isPresent()){
505          PositionalInfo info = positionalInfo.get();
506          argFields = Iterables.concat(argFields,
507              ImmutableList.of(Pair.of("[positional args]", info.field)));
508        }
509    
510        for (Pair<String, Field> argInfo : argFields) {
511          Field argField = argInfo.getSecond();
512          Arg arg = getArg(argField);
513    
514          commandLineArgumentInfos.add(String.format("%s (%s.%s): %s",
515              argInfo.getFirst(), argField.getDeclaringClass().getName(), argField.getName(),
516              arg.uncheckedGet()));
517    
518          try {
519            checkConstraints(verifiers, TypeUtil.getTypeParamClass(argField), arg.uncheckedGet(),
520                argField.getAnnotations());
521          } catch (IllegalArgumentException e) {
522            argsConstraintsFailed.add(argInfo.getFirst() + " - " + e.getMessage());
523          }
524        }
525    
526        ImmutableMultimap<String, String> warningMessages =
527            ImmutableMultimap.<String, String>builder()
528            .putAll("Unrecognized arguments", Sets.difference(args.keySet(), argsByName.keySet()))
529            .putAll("Failed to parse", argsFailedToParse)
530            .putAll("Value did not meet constraints", argsConstraintsFailed)
531            .build();
532    
533        if (!warningMessages.isEmpty()) {
534          printHelp(verifiers, argumentInfo);
535          StringBuilder sb = new StringBuilder();
536          for (Map.Entry<String, Collection<String>> warnings : warningMessages.asMap().entrySet()) {
537            sb.append(warnings.getKey()).append(":\n\t").append(Joiner.on("\n\t")
538                .join(warnings.getValue())).append("\n");
539          }
540          throw new IllegalArgumentException(sb.toString());
541        }
542    
543        infoLog("-------------------------------------------------------------------------");
544        infoLog("Command line argument values");
545        for (String commandLineArgumentInfo : commandLineArgumentInfos) {
546          infoLog(commandLineArgumentInfo);
547        }
548        infoLog("-------------------------------------------------------------------------");
549        return true;
550      }

552      private static final Function<String,String> ARG_NAME_TO_FLAG = new Function<String, String>() {
553        @Override public String apply(String argName) {
554          return "-" + argName;
555        }

558      public static final Ordering<OptionInfo> ORDER_BY_NAME =
559          Ordering.natural().onResultOf(GET_FIELD_ARG_NAME);

561      private void printHelp(Verifiers verifiers, ArgumentInfo argumentInfo) {
562        ImmutableList.Builder<String> requiredHelps = ImmutableList.builder();
563        ImmutableList.Builder<String> optionalHelps = ImmutableList.builder();
564        for (OptionInfo optionInfo : ORDER_BY_NAME.immutableSortedCopy(argumentInfo.optionInfos)) {
565          Field field = optionInfo.field;
566          CmdLine cmdLine = optionInfo.cmdLine;
567          Arg arg = getArg(field);
568          Object defaultValue = arg.uncheckedGet();
569          ImmutableList<String> constraints = collectConstraints(verifiers, field);
570          String help = formatHelp(cmdLine, field, constraints, defaultValue);
571          if (!arg.hasDefault()) {
572            requiredHelps.add(help);
573          } else {
574            optionalHelps.add(help);
575          }
576        }
577    
578        infoLog("-------------------------------------------------------------------------");
579        infoLog(String.format("%s to print this help message",
580            Joiner.on(" or ").join(Iterables.transform(HELP_ARGS, ARG_NAME_TO_FLAG))));
581        Optional<PositionalInfo> positionalInfo = argumentInfo.positionalInfo;
582        if (positionalInfo.isPresent()) {
583          infoLog("\nPositional args:");
584          PositionalInfo info = positionalInfo.get();
585          Field field = info.field;
586          Arg arg = getArg(field);
587          Object defaultValue = arg.uncheckedGet();
588          ImmutableList<String> constraints = collectConstraints(verifiers, field);
589          infoLog(String.format("%s%s\n\t%s\n\t(%s.%s)",
590                                defaultValue != null ? "default " + defaultValue : "",
591                                Iterables.isEmpty(constraints)
592                                    ? ""
593                                    : " [" + Joiner.on(", ").join(constraints) + "]",
594                                info.positional.help(),
595                                field.getDeclaringClass().getName(),
596                                field.getName()));
597        }
598        ImmutableList<String> required = requiredHelps.build();
599        if (!required.isEmpty()) {
600          infoLog("\nRequired flags:"); // yes - this should actually throw!
601          infoLog(Joiner.on('\n').join(required));
602        }
603        ImmutableList<String> optional = optionalHelps.build();
604        if (!optional.isEmpty()) {
605          infoLog("\nOptional flags:");
606          infoLog(Joiner.on('\n').join(optional));
607        }
608        infoLog("-------------------------------------------------------------------------");
609      }
610    
611      private ImmutableList<String> collectConstraints(Verifiers verifiers, Field field) {
612        Builder<String> constraints = ImmutableList.builder();
613        for (Annotation annotation : field.getAnnotations()) {
614          Class<?> argType = TypeUtil.getTypeParamClass(field);
615          @SuppressWarnings("unchecked")
616          Verifier verifier = verifiers.get(argType, annotation);
617          if (verifier != null) {
618            @SuppressWarnings("unchecked")
619            String constraint = verifier.toString(argType, annotation);
620            constraints.add(constraint);
621          }
622        }
623        return constraints.build();
624      }
625    
626      private String formatHelp(CmdLine cmdLine, Field field, Iterable<String> constraints,
627                                @Nullable Object defaultValue) {
628    
629        return String.format("-%s%s%s\n\t%s\n\t(%s.%s)",
630                             cmdLine.name(),
631                             defaultValue != null ? "=" + defaultValue : "",
632                             Iterables.isEmpty(constraints)
633                                 ? ""
634                                 : " [" + Joiner.on(", ").join(constraints) + "]",
635                             cmdLine.help(),
636                             field.getDeclaringClass().getName(),
637                             field.getName());
638      }

640      private void infoLog(String msg) {
641        out.println(msg);
642      }
643    
644      private void checkConstraints(Verifiers verifiers, Class<?> fieldClass, Object value,
645          Annotation[] annotations) {
646        for (Annotation annotation : maybeFilterNullable(verifiers, fieldClass, value, annotations)) {
647          verify(verifiers, fieldClass, value, annotation);
648        }
649      }
650    
651      private Iterable<Annotation> maybeFilterNullable(Verifiers verifiers, Class<?> fieldClass,
652          Object value, Annotation[] annotations) {
653    
654        // Apply all the normal constraint annotations
655        if (value != null) {
656          return Arrays.asList(annotations);
657        }
658    
659        // The value is null so skip verifications unless a @NotNull was specified as a constraint
660        for (Annotation annotation: annotations) {
661          if (annotation instanceof NotNull) {
662            verify(verifiers, fieldClass, value, annotation); // will throw
663          }
664        }
665        return ImmutableList.of();
666      }
667    
668      @SuppressWarnings("unchecked")
669      private void verify(Verifiers verifiers, Class<?> fieldClass, Object value,
670          Annotation annotation) {
671        Verifier verifier = verifiers.get(fieldClass, annotation);
672        if (verifier != null) {
673          verifier.verify(value, annotation);
674        }
675      }

677      private static Parser<?> getParser(ParserOracle parserOracle, Class<? extends Parser> custom,
678          Field argField) {
679    
680        Preconditions.checkArgument(argField.getType() == Arg.class,
681            "Field is annotated for argument parsing but is not of Arg type: " + argField);
682        return getParser(parserOracle, custom, TypeUtil.getTypeParamClass(argField));
683      }
684    
685      private static Parser<?> getParser(ParserOracle parserOracle, Class<? extends Parser> custom,
686          Class<?> cls) {
687        try {
688          return (custom == Parser.class) ? parserOracle.get(cls) : custom.newInstance();
689        } catch (InstantiationException e) {
690          throw new RuntimeException("Failed to instantiate parser " + custom);
691        } catch (IllegalAccessException e) {
692          throw new RuntimeException("No access to instantiate parser " + custom);
693        }
694      }

696      @SuppressWarnings("unchecked")
697      private static void setField(Field field, Object value) {
698        getArg(field).set(value);
699      }
700    
701      private static Arg getArg(Field field) {
702        field.setAccessible(true);
703        try {
704          return ((Arg) field.get(null));
705        } catch (IllegalAccessException e) {
706          throw new RuntimeException("Failed to set value for " + field);
707        }
708      }
