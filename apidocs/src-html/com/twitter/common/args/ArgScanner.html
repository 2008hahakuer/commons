055    /**
056     * Argument scanning, parsing, and validating system.  This class is designed recursively scan a
057     * package for declared arguments, parse the values based on the declared type, and validate against
058     * any constraints that the arugment is decorated with.
059     *
060     * The supported argument formats are:
061     *   -arg_name=arg_value
062     *   -arg_name arg_value
063     * Where {@code arg_value} may be single or double-quoted if desired or necessary to prevent
064     * splitting by the terminal application.
065     *
066     * A special format for boolean arguments is also supported.  The following syntaxes all set the
067     * {@code bool_arg} to {@code true}:
068     *   -bool_arg
069     *   -bool_arg=true
070     *   -no_bool_arg=false (double negation)
071     *
072     * Likewise, the following would set {@code bool_arg} to {@code false}:
073     *   -no_bool_arg
074     *   -bool_arg=false
075     *   -no_bool_arg=true (negation)
076     *
077     * As with the general argument format, spaces may be used in place of equals for boolean argument
078     * assignment.
079     *
080     * TODO(William Farner): Make default verifier and parser classes package-private and in this
081     * package.
082     *
083     * @author William Farner
084     */
085    public final class ArgScanner {
086    
087      private static final Function<OptionInfo, String> GET_OPTIONINFO_NAME =
088          new Function<OptionInfo, String>() {
089            @Override public String apply(OptionInfo optionInfo) {
090              return optionInfo.getName();
091            }
092          };
093    
094      public static final Ordering<OptionInfo> ORDER_BY_NAME =
095          Ordering.natural().onResultOf(GET_OPTIONINFO_NAME);
096    
097      private static final Function<String, String> ARG_NAME_TO_FLAG = new Function<String, String>() {
098        @Override public String apply(String argName) {
099          return "-" + argName;
100        }
101      };
102    
103      private static final Predicate<OptionInfo> IS_BOOLEAN =
104          new Predicate<OptionInfo>() {
105            @Override public boolean apply(OptionInfo optionInfo) {
106              return optionInfo.isBoolean();
107            }
108          };
109    
110      // Regular expression to identify a possible dangling assignment.
111      // A dangling assignment occurs in two cases:
112      //   - The command line used spaces between arg names and values, causing the name and value to
113      //     end up in different command line arg array elements.
114      //   - The command line is using the short form for a boolean argument,
115      //     such as -use_feature, or -no_use_feature.
116      private static final String DANGLING_ASSIGNMENT_RE =
117          String.format("^-%s", OptionInfo.ARG_NAME_RE);
118      private static final Pattern DANGLING_ASSIGNMENT_PATTERN =
119          Pattern.compile(DANGLING_ASSIGNMENT_RE);
120    
121      // Pattern to identify a full assignment, which would be disassociated from a preceding dangling
122      // assignment.
123      private static final Pattern ASSIGNMENT_PATTERN =
124          Pattern.compile(String.format("%s=.+", DANGLING_ASSIGNMENT_RE));
125    
126      /**
127       * Extracts the name from an @OptionInfo.
128       */
129      private static final Function<OptionInfo, String> GET_OPTIONINFO_NEGATED_NAME =
130          new Function<OptionInfo, String>() {
131            @Override public String apply(OptionInfo optionInfo) {
132              return optionInfo.getNegatedName();
133            }
134          };
135    
136      /**
137       * Gets the canonical name for an @Arg, based on the class containing the field it annotates.
138       */
139      private static final Function<OptionInfo, String> GET_CANONICAL_ARG_NAME =
140          new Function<OptionInfo, String>() {
141            @Override public String apply(OptionInfo optionInfo) {
142              return optionInfo.getCanonicalName();
143            }
144          };
145    
146      /**
147       * Gets the canonical negated name for an @Arg.
148       */
149      private static final Function<OptionInfo, String> GET_CANONICAL_NEGATED_ARG_NAME =
150          new Function<OptionInfo, String>() {
151            @Override public String apply(OptionInfo optionInfo) {
152              return optionInfo.getCanonicalNegatedName();
153            }
154          };

156      private static final Logger LOG = Logger.getLogger(ArgScanner.class.getName());

158      // Pattern for the required argument format.
159      private static final Pattern ARG_PATTERN =
160          Pattern.compile(String.format("-(%s)(?:(?:=| +)(.*))?", OptionInfo.ARG_NAME_RE));
161    
162      private static final Pattern QUOTE_PATTERN = Pattern.compile("(['\"])([^\\\1]*)\\1");

184      /**
185       * Applies the provided argument values to all {@literal @CmdLine} {@code Arg} fields discovered
186       * on the classpath.
187       *
188       * @param args Argument values to map, parse, validate, and apply.
189       * @return {@code true} if the given {@code args} were successfully applied to their corresponding
190       *     {@link Arg} fields.
191       * @throws ArgScanException if there was a problem loading {@literal @CmdLine} argument
192       *    definitions
193       * @throws IllegalArgumentException If the arguments provided are invalid based on the declared
194       *    arguments found.
195       */
196      public boolean parse(Iterable<String> args) {
197        return parse(ArgFilters.SELECT_ALL, args);
198      }

200      /**
201       * Applies the provided argument values to any {@literal @CmdLine} or {@literal @Positional}
202       * {@code Arg} fields discovered on the classpath and accepted by the given {@code filter}.
203       *
204       * @param filter A predicate that selects or rejects scanned {@literal @CmdLine} fields for
205       *    argument application.
206       * @param args Argument values to map, parse, validate, and apply.
207       * @return {@code true} if the given {@code args} were successfully applied to their corresponding
208       *     {@link Arg} fields.
209       * @throws ArgScanException if there was a problem loading {@literal @CmdLine} argument
210       *    definitions
211       * @throws IllegalArgumentException If the arguments provided are invalid based on the declared
212       *    arguments found.
213       */
214      public boolean parse(Predicate<Field> filter, Iterable<String> args) {
215        Configuration configuration = load();
216        Args.ArgumentInfo argumentInfo = Args.fromConfiguration(configuration, filter);
217        return parse(argumentInfo, configuration, args);
218      }

220      /**
221       * Parse command line arguments given a {@link Args.ArgumentInfo}
222       *
223       * @param argumentInfo A description of any optional and positional arguments
224       * @param args A sequence of strings from the command-line
225       * @return {@code true} if the given {@code args} were successfully applied to their corresponding
226       *     {@link Arg} fields.
227       * @throws ArgScanException if there was a problem loading {@literal @CmdLine} argument
228       *    definitions
229       * @throws IllegalArgumentException If the arguments provided are invalid based on the declared
230       *    arguments found.
231       */
232      public boolean parse(Args.ArgumentInfo argumentInfo, Iterable<String> args) {
233        Configuration configuration = load();
234        return parse(argumentInfo, configuration, args);
235      }
236    
237      private boolean parse(Args.ArgumentInfo argumentInfo, Configuration configuration,
238        Iterable<String> args) {
239        ParserOracle parserOracle = Parsers.fromConfiguration(configuration);
240        Verifiers verifiers = Verifiers.fromConfiguration(configuration);
241        Pair<ImmutableMap<String, String>, List<String>> results = mapArguments(args);
242        return process(parserOracle, verifiers, argumentInfo, results.getFirst(), results.getSecond());
243      }
244    
245      private Configuration load() {
246        try {
247          return Configuration.load();
248        } catch (IOException e) {
249          throw new ArgScanException(e);
250        }
251      }

253      @VisibleForTesting static List<String> joinKeysToValues(Iterable<String> args) {
254        List<String> joinedArgs = Lists.newArrayList();
255        String unmappedKey = null;
256        for (String arg : args) {
257          if (unmappedKey == null) {
258            if (DANGLING_ASSIGNMENT_PATTERN.matcher(arg).matches()) {
259              // Beginning of a possible dangling assignment.
260              unmappedKey = arg;
261            } else {
262              joinedArgs.add(arg);
263            }
264          } else {
265            if (ASSIGNMENT_PATTERN.matcher(arg).matches()) {
266              // Full assignment, disassociate from dangling assignment.
267              joinedArgs.add(unmappedKey);
268              joinedArgs.add(arg);
269              unmappedKey = null;
270            } else if (DANGLING_ASSIGNMENT_PATTERN.matcher(arg).find()) {
271              // Another dangling assignment, this could be two sequential boolean args.
272              joinedArgs.add(unmappedKey);
273              unmappedKey = arg;
274            } else {
275              // Join the dangling key with its value.
276              joinedArgs.add(unmappedKey + "=" + arg);
277              unmappedKey = null;
278            }
279          }
280        }
281    
282        if (unmappedKey != null) {
283          joinedArgs.add(unmappedKey);
284        }
285    
286        return joinedArgs;
287      }

289      private static String stripQuotes(String str) {
290        Matcher matcher = QUOTE_PATTERN.matcher(str);
291        return matcher.matches() ? matcher.group(2) : str;
292      }

294      /**
295       * Scans through args, mapping keys to values even if the arg values are 'dangling' and reside
296       * in different array entries than the respective keys.
297       *
298       * @param args Arguments to build into a map.
299       * @return A map from argument key (arg name) to value paired with a list of any leftover
300       *     positional arguments.
301       */
302      private static Pair<ImmutableMap<String, String>, List<String>> mapArguments(
303          Iterable<String> args) {
304    
305        ImmutableMap.Builder<String, String> argMap = ImmutableMap.builder();
306        List<String> positionalArgs = Lists.newArrayList();
307        for (String arg : joinKeysToValues(args)) {
308          if (!arg.startsWith("-")) {
309            positionalArgs.add(arg);
310          } else {
311            Matcher matcher = ARG_PATTERN.matcher(arg);
312            checkArgument(matcher.matches(),
313                String.format("Argument '%s' does not match required format -arg_name=arg_value", arg));
314    
315            String rawValue = matcher.group(2);
316            // An empty string denotes that the argument was passed with no value.
317            rawValue = rawValue == null ? "" : stripQuotes(rawValue);
318            argMap.put(matcher.group(1), rawValue);
319          }
320        }
321    
322        return Pair.of(argMap.build(), positionalArgs);
323      }

325      private static <T> Set<T> dropCollisions(Iterable<T> input) {
326        Set<T> copy = Sets.newHashSet();
327        Set<T> collisions = Sets.newHashSet();
328        for (T entry : input) {
329          if (!copy.add(entry)) {
330            collisions.add(entry);
331          }
332        }
333    
334        copy.removeAll(collisions);
335        return copy;
336      }

338      /**
339       * Applies argument values to fields based on their annotations.
340       *
341       * @param parserOracle ParserOracle available to parse raw args with.
342       * @param verifiers Verifiers available to verify argument constraints with.
343       * @param argumentInfo Fields to apply argument values to.
344       * @param args Unparsed argument values.
345       * @param positionalArgs The unparsed positional arguments.
346       * @return {@code true} if the given {@code args} were successfully applied to their
347       *     corresponding {@link com.twitter.common.args.Arg} fields.
348       */
349      private boolean process(final ParserOracle parserOracle,
350          Verifiers verifiers,
351          Args.ArgumentInfo argumentInfo,
352          Map<String, String> args,
353          List<String> positionalArgs) {
354    
355        if (!Sets.intersection(args.keySet(), ArgumentInfo.HELP_ARGS).isEmpty()) {
356          printHelp(verifiers, argumentInfo);
357          return false;
358        }
359    
360        Optional<? extends PositionalInfo<?>> positionalInfoOptional = argumentInfo.getPositionalInfo();
361        checkArgument(positionalInfoOptional.isPresent() || positionalArgs.isEmpty(),
362            "Positional arguments have been supplied but there is no Arg annotated to received them.");
363    
364        Iterable<? extends OptionInfo<?>> optionInfos = argumentInfo.getOptionInfos();
365    
366        final Set<String> argsFailedToParse = Sets.newHashSet();
367        final Set<String> argsConstraintsFailed = Sets.newHashSet();
368    
369        Iterable<String> argShortNames = Iterables.transform(optionInfos, GET_OPTIONINFO_NAME);
370        Set<String> argShortNamesNoCollisions = dropCollisions(argShortNames);
371        Set<String> collisionsDropped = Sets.difference(ImmutableSet.copyOf(argShortNames),
372          argShortNamesNoCollisions);
373        if (!collisionsDropped.isEmpty()) {
374          LOG.warning("Found argument name collisions, args must be referenced by canonical names: "
375              + collisionsDropped);
376        }
377    
378        final Map<String, OptionInfo> argsByName =
379            ImmutableMap.<String, OptionInfo>builder()
380            // Map by short arg name -> arg def.
381            .putAll(Maps.uniqueIndex(Iterables.filter(optionInfos,
382                Predicates.compose(Predicates.in(argShortNamesNoCollisions), GET_OPTIONINFO_NAME)),
383                                   GET_OPTIONINFO_NAME))
384            // Map by canonical arg name -> arg def.
385            .putAll(Maps.uniqueIndex(optionInfos, GET_CANONICAL_ARG_NAME))
386            // Map by negated short arg name (for booleans)
387            .putAll(Maps.uniqueIndex(Iterables.filter(optionInfos, IS_BOOLEAN),
388                GET_OPTIONINFO_NEGATED_NAME))
389            // Map by negated canonical arg name (for booleans)
390            .putAll(Maps.uniqueIndex(Iterables.filter(optionInfos, IS_BOOLEAN),
391                GET_CANONICAL_NEGATED_ARG_NAME))
392            .build();
393    
394        // TODO(William Farner): Make sure to disallow duplicate arg specification by short and
395        // canonical names.
396    
397        // TODO(William Farner): Support non-atomic argument constraints.  @OnlyIfSet, @OnlyIfNotSet,
398        //    @ExclusiveOf to define inter-argument constraints.
399    
400        Set<String> recognizedArgs = Sets.intersection(argsByName.keySet(), args.keySet());
401    
402        for (String argName : recognizedArgs) {
403          String argValue = args.get(argName);
404          OptionInfo optionInfo = argsByName.get(argName);
405    
406          try {
407            optionInfo.load(parserOracle, argName, argValue);
408          } catch (IllegalArgumentException e) {
409            argsFailedToParse.add(argName + " - " + e.getMessage());
410          }
411        }
412    
413        if (positionalInfoOptional.isPresent()) {
414          PositionalInfo<?> positionalInfo = positionalInfoOptional.get();
415          positionalInfo.load(parserOracle, positionalArgs);
416        }
417    
418        Set<String> commandLineArgumentInfos = Sets.newTreeSet();
419    
420        Iterable<? extends ArgumentInfo<?>> allArguments = argumentInfo.getOptionInfos();
421    
422        if (positionalInfoOptional.isPresent()) {
423          PositionalInfo<?> positionalInfo = positionalInfoOptional.get();
424          allArguments = Iterables.concat(optionInfos, ImmutableList.of(positionalInfo));
425        }
426    
427        for (ArgumentInfo<?> anArgumentInfo : allArguments) {
428          Arg<?> arg = anArgumentInfo.getArg();
429    
430          commandLineArgumentInfos.add(String.format("%s (%s): %s",
431              anArgumentInfo.getName(), anArgumentInfo.getCanonicalName(),
432              arg.uncheckedGet()));
433    
434          try {
435            anArgumentInfo.verify(verifiers);
436          } catch (IllegalArgumentException e) {
437            argsConstraintsFailed.add(anArgumentInfo.getName() + " - " + e.getMessage());
438          }
439        }
440    
441        ImmutableMultimap<String, String> warningMessages =
442            ImmutableMultimap.<String, String>builder()
443            .putAll("Unrecognized arguments", Sets.difference(args.keySet(), argsByName.keySet()))
444            .putAll("Failed to parse", argsFailedToParse)
445            .putAll("Value did not meet constraints", argsConstraintsFailed)
446            .build();
447    
448        if (!warningMessages.isEmpty()) {
449          printHelp(verifiers, argumentInfo);
450          StringBuilder sb = new StringBuilder();
451          for (Map.Entry<String, Collection<String>> warnings : warningMessages.asMap().entrySet()) {
452            sb.append(warnings.getKey()).append(":\n\t").append(Joiner.on("\n\t")
453                .join(warnings.getValue())).append("\n");
454          }
455          throw new IllegalArgumentException(sb.toString());
456        }
457    
458        LOG.info("-------------------------------------------------------------------------");
459        LOG.info("Command line argument values");
460        for (String commandLineArgumentInfo : commandLineArgumentInfos) {
461          LOG.info(commandLineArgumentInfo);
462        }
463        LOG.info("-------------------------------------------------------------------------");
464        return true;
465      }

467      private void printHelp(Verifiers verifiers, Args.ArgumentInfo argumentInfo) {
468        ImmutableList.Builder<String> requiredHelps = ImmutableList.builder();
469        ImmutableList.Builder<String> optionalHelps = ImmutableList.builder();
470        for (OptionInfo<?> optionInfo
471            : ORDER_BY_NAME.immutableSortedCopy(argumentInfo.getOptionInfos())) {
472          Arg<?> arg = optionInfo.getArg();
473          Object defaultValue = arg.uncheckedGet();
474          ImmutableList<String> constraints = collectConstraints(verifiers, optionInfo);
475          String help = formatHelp(optionInfo, constraints, defaultValue);
476          if (!arg.hasDefault()) {
477            requiredHelps.add(help);
478          } else {
479            optionalHelps.add(help);
480          }
481        }
482    
483        infoLog("-------------------------------------------------------------------------");
484        infoLog(String.format("%s to print this help message",
485            Joiner.on(" or ").join(Iterables.transform(ArgumentInfo.HELP_ARGS, ARG_NAME_TO_FLAG))));
486        Optional<? extends PositionalInfo<?>> positionalInfoOptional = argumentInfo.getPositionalInfo();
487        if (positionalInfoOptional.isPresent()) {
488          infoLog("\nPositional args:");
489          PositionalInfo<?> positionalInfo = positionalInfoOptional.get();
490          Arg<?> arg = positionalInfo.getArg();
491          Object defaultValue = arg.uncheckedGet();
492          ImmutableList<String> constraints = collectConstraints(verifiers, positionalInfo);
493          infoLog(String.format("%s%s\n\t%s\n\t(%s)",
494                                defaultValue != null ? "default " + defaultValue : "",
495                                Iterables.isEmpty(constraints)
496                                    ? ""
497                                    : " [" + Joiner.on(", ").join(constraints) + "]",
498                                positionalInfo.getHelp(),
499                                positionalInfo.getCanonicalName()));
500        }
501        ImmutableList<String> required = requiredHelps.build();
502        if (!required.isEmpty()) {
503          infoLog("\nRequired flags:"); // yes - this should actually throw!
504          infoLog(Joiner.on('\n').join(required));
505        }
506        ImmutableList<String> optional = optionalHelps.build();
507        if (!optional.isEmpty()) {
508          infoLog("\nOptional flags:");
509          infoLog(Joiner.on('\n').join(optional));
510        }
511        infoLog("-------------------------------------------------------------------------");
512      }
513    
514      private ImmutableList<String> collectConstraints(Verifiers verifierOracle,
515          ArgumentInfo<?> argumentInfo) {
516        Builder<String> builder = ImmutableList.builder();
517        argumentInfo.collectConstraints(verifierOracle, builder);
518        return builder.build();
519      }
520    
521      private String formatHelp(ArgumentInfo<?> argumentInfo, Iterable<String> constraints,
522                                @Nullable Object defaultValue) {
523    
524        return String.format("-%s%s%s\n\t%s\n\t(%s)",
525                             argumentInfo.getName(),
526                             defaultValue != null ? "=" + defaultValue : "",
527                             Iterables.isEmpty(constraints)
528                                 ? ""
529                                 : " [" + Joiner.on(", ").join(constraints) + "]",
530                             argumentInfo.getHelp(),
531                             argumentInfo.getCanonicalName());
532      }
533    
534      private void infoLog(String msg) {
535        out.println(msg);
536      }
