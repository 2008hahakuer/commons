017    package com.twitter.common.args;
018    
019    import java.lang.annotation.Annotation;
020    import java.lang.reflect.Field;
021    import java.lang.reflect.Modifier;
022    import java.util.List;
023    
024    import javax.annotation.Nullable;
025    
026    import com.google.common.base.Function;
027    import com.google.common.base.Preconditions;
028    import com.google.common.base.Predicates;
029    import com.google.common.collect.ImmutableList;
030    import com.google.common.collect.ImmutableSet;
031    import com.google.common.collect.Iterables;
032    
033    import com.twitter.common.args.constraints.NotNullVerifier;
034    import com.twitter.common.base.MorePreconditions;
035    import com.twitter.common.collections.Pair;
036    import com.twitter.common.reflect.TypeToken;
037    
038    import static com.google.common.base.Preconditions.checkArgument;
039    
040    /**
041     * Description of a command line argument.
042     *
043     * @author Nick Kallen
044     */
045    abstract public class ArgumentInfo<T> {
046      static final ImmutableSet<String> HELP_ARGS = ImmutableSet.of("h", "help");
047    
048      private final String help;
049      private final Arg<T> arg;
050      private final TypeToken<T> type;
051      private final List<Annotation> verifierAnnotations;
052      @Nullable private final Class<? extends Parser<? extends T>> parser;
053    
054      public ArgumentInfo(String help, Arg<T> arg, TypeToken<T> type, List<Annotation> verifierAnnotations,
055          @Nullable Class<? extends Parser<? extends T>> parser) {
056        this.help = MorePreconditions.checkNotBlank(help);
057        this.arg = Preconditions.checkNotNull(arg);
058        this.type = Preconditions.checkNotNull(type);
059        this.verifierAnnotations = ImmutableList.copyOf(verifierAnnotations);
060        this.parser = parser;
061      }
062    
063      /**
064       * Return the name of the command line argument. In an optional argument, this is expressed on
065       * the command line by "-name=value"; whereas, for a positional argument, the name indicates
066       * the type/function.
067       */
068      abstract public String getName();
069    
070      protected Parser<? extends T> getParser(ParserOracle parserOracle) {
071        Preconditions.checkNotNull(parserOracle);
072        if (parser == null || Parser.class.equals(parser)) {
073          return parserOracle.get(type);
074        } else {
075          try {
076            return parser.newInstance();
077          } catch (InstantiationException e) {
078            throw new RuntimeException("Failed to instantiate parser " + parser);
079          } catch (IllegalAccessException e) {
080            throw new RuntimeException("No access to instantiate parser " + parser);
081          }
082        }
083      }
084    
085      private Iterable<Pair<? extends Verifier<? super T>, Annotation>> getVerifiers(
086          final Verifiers verifierOracle) {
087        Function<Annotation, Pair<? extends Verifier<? super T>, Annotation>> toVerifier =
088            new Function<Annotation, Pair<? extends Verifier<? super T>, Annotation>>() {
089              @Override public Pair<? extends Verifier<? super T>, Annotation> apply(
090                  Annotation annotation) {
091    
092                @Nullable Verifier<? super T> verifier = verifierOracle.get(type, annotation);
093                return (verifier != null) ? Pair.of(verifier, annotation) : null;
094              }
095            };
096        return Iterables.filter(Iterables.transform(verifierAnnotations, toVerifier),
097            Predicates.<Pair<? extends Verifier<? super T>, Annotation>>notNull());
098      }
099    
100      /**
101       * Returns the instructions for this command-line argument. This is typically used when the
102       * executable is passed the -help flag.
103       */
104      public String getHelp() {
105        return help;
106      }
107    
108      /**
109       * Returns the Arg associated with this command-line argument. The Arg<?> is a mutable container
110       * cell that holds the value passed-in on the command line, after parsing and validation.
111       */
112      public Arg<T> getArg() {
113        return arg;
114      }
115    
116      /**
117       * Returns the TypeToken that represents the type of this command-line argument.
118       */
119      public TypeToken<T> getType() {
120        return type;
121      }
122    
123      protected void setValue(T value) {
124        arg.set(value);
125      }
126    
127      abstract String getCanonicalName();
128    
129      protected static Arg<?> getArgForField(Field field) {
130        Preconditions.checkArgument(field.getType() == Arg.class,
131          "Field is annotated for argument parsing but is not of Arg type: " + field);
132        checkArgument(Modifier.isStatic(field.getModifiers()),
133          "Non-static argument fields are not supported, found " + field);
134    
135        field.setAccessible(true);
136        try {
137          return ((Arg<?>) field.get(null));
138        } catch (IllegalAccessException e) {
139          throw new RuntimeException("Cannot get arg value for " + field);
140        }
141      }
142    
143      void verify(Verifiers verifierOracle) {
144        T value = getArg().uncheckedGet();
145        for (Pair<? extends Verifier<? super T>, Annotation> pair : getVerifiers(verifierOracle)) {
146          Verifier<? super T> verifier = pair.getFirst();
147          Annotation annotation = pair.getSecond();
148          if (value != null || verifier instanceof NotNullVerifier) {
149            verifier.verify(value, annotation);
150          }
151        }
152      }
153    
154      void collectConstraints(Verifiers verifierOracle, ImmutableList.Builder<String> constraints) {
155        for (Pair<? extends Verifier<? super T>, Annotation> pair : getVerifiers(verifierOracle)) {
156          Verifier<? super T> verifier = pair.getFirst();
157          Annotation annotation = pair.getSecond();
158          Class<T> rawType = type.getRawType();
159          String constraint = verifier.toString(rawType, annotation);
160          constraints.add(constraint);
161        }
162      }
163    }
