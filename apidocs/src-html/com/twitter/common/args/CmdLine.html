017    package com.twitter.common.args;
018    
019    import java.lang.annotation.Retention;
020    import java.lang.annotation.Target;
021    
022    import com.twitter.common.args.Parsers.Parser;
023    
024    import static java.lang.annotation.ElementType.FIELD;
025    import static java.lang.annotation.RetentionPolicy.RUNTIME;
026    
027    /**
028     * Annotation for a command line argument.
029     *
030     * @author William Farner
031     */
032    @Target({FIELD})
033    @Retention(RUNTIME)
034    public @interface CmdLine {
035    
036      /**
037       * The short name of the argument, as supplied on the command line.  The argument can also be
038       * accessed by the canonical name, which is {@code com.foo.bar.MyArgClass.arg_name}.
039       * If the global scope contains more than one argument with the same name, all colliding arguments
040       * must be disambiguated with the canonical form.
041       *
042       * The argument name must match the format {@code [\w\-\.]+}.
043       *
044       * @return The argument name.
045       */
046      String name();
047    
048      /**
049       * The help string to display on the command line in a usage message.
050       *
051       * @return Help string.
052       */
053      String help();
054    
055      /**
056       * The parser class to use for parsing this argument.  The parser must return the same type as
057       * the field being annotated.
058       *
059       * @return Custom parser for this type.
060       */
061      // The default is fully qualified to work around an apt bug:
062      // http://bugs.sun.com/view_bug.do?bug_id=6512707
063      Class<? extends Parser> parser() default com.twitter.common.args.Parsers.Parser.class;
064    }
