017    package com.twitter.common.args;
018    
019    import java.lang.annotation.Annotation;
020    import java.lang.reflect.Field;
021    import java.lang.reflect.Type;
022    import java.util.Arrays;
023    import java.util.List;
024    
025    import javax.annotation.Nullable;
026    
027    import com.google.common.base.Function;
028    import com.google.common.base.Preconditions;
029    import com.google.common.collect.Iterables;
030    import com.google.common.collect.Lists;
031    import com.google.common.reflect.TypeToken;
032    
033    import com.twitter.common.args.apt.Configuration;
034    
035    /**
036     * Description of a positional command line argument.
037     *
038     * @author Nick Kallen
039     */
040    public class PositionalInfo<T> extends ArgumentInfo<List<T>> {
041      private final String canonicalName;
042      private final TypeToken<T> elementType;
043    
044      public PositionalInfo(
045          String canonicalName,
046          String help,
047          Arg<List<T>> arg,
048          TypeToken<List<T>> type,
049          TypeToken<T> elementType,
050          List<Annotation> verifierAnnotations,
051          @Nullable Class<? extends Parser<? extends List<T>>> parser) {
052    
053        super(help, arg, type, verifierAnnotations, parser);
054        this.elementType = elementType;
055        this.canonicalName = canonicalName;
056      }
057    
058      /**
059       * Factory method to create a PositionalInfo from a java.lang.reflect.Field.
060       *
061       * @param field The field must contain a Arg<List<?>>. The List<?> represents zero or more
062       *     positional arguments.
063       * @return a PositionalInfo
064       */
065      static PositionalInfo createFromField(Field field) {
066        Preconditions.checkNotNull(field);
067        Positional positional = field.getAnnotation(Positional.class);
068        if (positional == null) {
069          throw new Configuration.ConfigurationException(
070              "No @Positional Arg annotation for field " + field);
071        }
072    
073        Preconditions.checkArgument(
074            TypeUtil.getRawType(TypeUtil.getTypeParam(field)) == List.class,
075            "Field is annotated for positional parsing but is not of Arg<List<?>> type");
076        Type nestedType = TypeUtil.extractTypeToken(TypeUtil.getTypeParam(field));
077    
078        @SuppressWarnings("unchecked")
079        PositionalInfo positionalInfo = new PositionalInfo(
080            field.getDeclaringClass().getCanonicalName() + "." + field.getName(),
081            positional.help(),
082            ArgumentInfo.getArgForField(field),
083            TypeUtil.getTypeParamTypeToken(field),
084            TypeToken.of(nestedType),
085            Arrays.asList(field.getAnnotations()),
086            positional.parser());
087        return positionalInfo;
088      }
089    
090      /**
091       * Get the "name" of the positional argument. Positional arguments, unlike optional arguments,
092       * don't have names (like `-foo=bar`). However, when printing help info, we use the name to
093       * represent the positional argument.
094       *
095       * @return the string "[positional args]"
096       */
097      @Override
098      public String getName() {
099        return "[positional args]";
100      }
101    
102      String getCanonicalName() {
103        return this.canonicalName;
104      }
105    
106      void load(final ParserOracle parserOracle, List<String> positionalArgs) {
107        final Parser<? extends T> parser = parserOracle.get(elementType);
108        List<T> assignmentValue = Lists.newArrayList(Iterables.transform(positionalArgs,
109          new Function<String, T>() {
110            @Override public T apply(String argValue) {
111              return parser.parse(parserOracle, elementType.getType(), argValue);
112            }
113          }));
114        setValue(assignmentValue);
115      }
116    }
