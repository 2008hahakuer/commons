017    package com.twitter.common.args;
018    
019    import java.lang.annotation.Annotation;
020    import java.lang.reflect.Field;
021    import java.lang.reflect.Type;
022    import java.util.Arrays;
023    import java.util.List;
024    
025    import javax.annotation.Nullable;
026    
027    import com.google.common.base.Function;
028    import com.google.common.base.Preconditions;
029    import com.google.common.collect.Iterables;
030    import com.google.common.collect.Lists;
031    
032    import com.twitter.common.args.apt.Configuration;
033    import com.twitter.common.reflect.TypeToken;
034    
035    /**
036     * Description of a positional command line argument.
037     *
038     * @author Nick Kallen
039     */
040    public class PositionalInfo<T> extends ArgumentInfo<List<T>> {
041      private final String canonicalName;
042      private final TypeToken<T> elementType;
043    
044      /**
045       * Factory method to create a PositionalInfo from a java.lang.reflect.Field.
046       *
047       * @param field The field must contain a Arg<List<?>>. The List<?> represents zero or more positional arguments.
048       * @return a PositionalInfo
049       */
050      static PositionalInfo createFromField(Field field) {
051        Preconditions.checkNotNull(field);
052        Positional positional = field.getAnnotation(Positional.class);
053        if (positional == null) {
054          throw new Configuration.ConfigurationException(
055              "No @Positional Arg annotation for field " + field);
056        }
057    
058        Preconditions.checkArgument(
059            TypeUtil.getRawType(TypeUtil.getTypeParam(field)) == List.class,
060            "Field is annotated for positional parsing but is not of Arg<List<?>> type");
061        Type nestedType = TypeToken.extractTypeToken(TypeUtil.getTypeParam(field));
062    
063        @SuppressWarnings("unchecked")
064        PositionalInfo positionalInfo = new PositionalInfo(
065            field.getDeclaringClass().getCanonicalName() + "." + field.getName(),
066            positional.help(),
067            ArgumentInfo.getArgForField(field),
068            TypeUtil.getTypeParamTypeToken(field),
069            TypeToken.create(nestedType),
070            Arrays.asList(field.getAnnotations()),
071            positional.parser());
072        return positionalInfo;
073      }
074    
075      public PositionalInfo(String canonicalName, String help, Arg<List<T>> arg,
076          TypeToken<List<T>> type, TypeToken<T> elementType, List<Annotation> verifierAnnotations,
077          @Nullable Class<? extends Parser<? extends List<T>>> parser) {
078        super(help, arg, type, verifierAnnotations, parser);
079        this.elementType = elementType;
080        this.canonicalName = canonicalName;
081      }
082    
083      /**
084       * Get the "name" of the positional argument. Positional arguments, unlike optional arguments, don't
085       * have names (like `-foo=bar`). However, when printing help info, we use the name to represent the positional
086       * argument.
087       *
088       * @return the string "[positional args]"
089       */
090      @Override
091      public String getName() {
092        return "[positional args]";
093      }
094    
095      String getCanonicalName() {
096        return this.canonicalName;
097      }
098    
099      void load(final ParserOracle parserOracle, List<String> positionalArgs) {
100        final Parser<? extends T> parser = parserOracle.get(elementType);
101        List<T> assignmentValue = Lists.newArrayList(Iterables.transform(positionalArgs,
102          new Function<String, T>() {
103            @Override public T apply(String argValue) {
104              return parser.parse(parserOracle, elementType.getType(), argValue);
105            }
106          }));
107        setValue(assignmentValue);
108      }
109    }
