017    package com.twitter.common.args;
018    
019    import javax.annotation.Nullable;
020    
021    import com.google.common.annotations.VisibleForTesting;
022    import com.google.common.base.Preconditions;
023    
024    /**
025     * Wrapper class for the value of an argument.  For proper behavior, an {@code Arg} should always
026     * be annotated with {@link CmdLine}, which will define the command line interface to the argument.
027     *
028     * @author William Farner
029     */
030    public class Arg<T> {
031    
032      private final T defaultValue;
033      private T value;
034      private boolean hasDefault = true;
035      private boolean valueApplied = false;
036    
037      /**
038       * Creates an arg that has no default value, meaning that its value can only ever be retrieved
039       * if it has been externally set.
040       */
041      public Arg() {
042        this(null);
043        this.hasDefault = false;
044      }
045    
046      /**
047       * Creates an arg that has a default value, and may optionally be set.
048       *
049       * @param defaultValue The default value for the arg.
050       */
051      public Arg(@Nullable T defaultValue) {
052        this.defaultValue = defaultValue;
053        this.value = defaultValue;
054      }
055    
056      synchronized void set(T value) {
057        Preconditions.checkState(!valueApplied, "A value cannot be applied twice to an argument.");
058        valueApplied = true;
059        this.value = value;
060      }
061    
062      @VisibleForTesting
063      synchronized void reset() {
064        valueApplied = false;
065        this.value = hasDefault ? defaultValue : null;
066      }
067    
068      /**
069       * Gets the value of the argument.  If a value has not yet been applied to the argument, or the
070       * argument did not provide a default value, {@link IllegalStateException} will be thrown.
071       *
072       * @return The argument value.
073       */
074      public synchronized T get() {
075        // TODO(William Farner): This has a tendency to break bad-arg reporting by ArgScanner.  Fix.
076        Preconditions.checkState(valueApplied || hasDefault,
077            "A value may only be retrieved from a variable that has a default or has been set.");
078        return uncheckedGet();
079      }
080    
081      /**
082       * Gets the value of the argument, without checking whether a default was available or if a
083       * value was applied.
084       *
085       * @return The argument value.
086       */
087      synchronized T uncheckedGet() {
088        return value;
089      }
090    
091      /**
092       * Convenience factory method to create an arg that has no default value.
093       *
094       * @param <T> Type of arg value.
095       * @return A new arg.
096       */
097      public static <T> Arg<T> create() {
098        return new Arg<T>();
099      }
100    
101      /**
102       * Convenience factory method to create an arg with a default value.
103       *
104       * @param value Default argument value.
105       * @param <T> Type of arg value.
106       * @return A new arg.
107       */
108      public static <T> Arg<T> create(T value) {
109        return new Arg<T>(value);
110      }
111    }
