057    /**
058     * Argument scanning, parsing, and validating system.  This class is designed recursively scan a
059     * package for declared arguments, parse the values based on the declared type, and validate against
060     * any constraints that the arugment is decorated with.
061     *
062     * The supported argument formats are:
063     *   -arg_name=arg_value
064     *   -arg_name arg_value
065     * Where {@code arg_value} may be single or double-quoted if desired or necessary to prevent
066     * splitting by the terminal application.
067     *
068     * A special format for boolean arguments is also supported.  The following syntaxes all set the
069     * {@code bool_arg} to {@code true}:
070     *   -bool_arg
071     *   -bool_arg=true
072     *   -no_bool_arg=false (double negation)
073     *
074     * Likewise, the following would set {@code bool_arg} to {@code false}:
075     *   -no_bool_arg
076     *   -bool_arg=false
077     *   -no_bool_arg=true (negation)
078     *
079     * As with the general argument format, spaces may be used in place of equals for boolean argument
080     * assignment.
081     *
082     * TODO(William Farner): Make default verifier and parser classes package-private and in this
083     * package.
084     *
085     * @author William Farner
086     */
087    public class ArgScanner {

103      // Regular expression that defines the name format for an argument.
104      private static final String ARG_NAME_RE = "[\\w\\-\\.]+";
105      private static final Pattern ARG_NAME_PATTERN = Pattern.compile(ARG_NAME_RE);
106      private static final Pattern NEGATED_BOOLEAN_PATTERN = Pattern.compile(
107          String.format("(%s)?%s(%s)", ARG_NAME_RE, NEGATE_BOOLEAN, ARG_NAME_RE));

111          Pattern.compile(String.format("-(%s)(?:(?:=| +)(.*))?", ARG_NAME_RE));

118       * Applies the provided argument values to all {@literal @CmdLine} {@code Arg} fields discovered
119       * on the classpath.
120       *
121       * @param args Argument values to map, parse, validate, and apply.
122       * @throws ArgScanException if there was a problem loading {@literal @CmdLine} argument
123       *    definitions
124       * @throws IllegalArgumentException If the arguments provided are invalid based on the declared
125       *    arguments found.
126       */
127      public static void parse(String... args) {
128        parse(Predicates.<Field>alwaysTrue(), args);

148      /**
149       * Convenience method to call {@link #parse(Predicate, Map)} that will handle mapping of argument
150       * keys to values.
151       *
152       * @param filter A predicate that selects or rejects scanned {@literal @CmdLine} fields for
153       *    argument application.
154       * @param args Argument values to map, parse, validate, and apply.
155       * @throws ArgScanException if there was a problem loading {@literal @CmdLine} argument
156       *    definitions
157       * @throws IllegalArgumentException If the arguments provided are invalid based on the declared
158       *    arguments found.
159       */
160      public static void parse(Predicate<Field> filter, String... args) {
161        parse(filter, mapArguments(args));
162      }

164      private static void parse(Predicate<Field> filter, Map<String, String> args) {
165        process(scan(filter), args);
166      }

168      // Regular expression to identify a possible dangling assignment.
169      // A dangling assignment occurs in two cases:
170      //   - The command line used spaces between arg names and values, causing the name and value to
171      //     end up in different command line arg array elements.
172      //   - The command line is using the short form for a boolean argument,
173      //     such as -use_feature, or -no_use_feature.
174      private static final String DANGLING_ASSIGNMENT_RE = String.format("^-%s", ARG_NAME_RE);
175      private static final Pattern DANGLING_ASSIGNMENT_PATTERN =
176          Pattern.compile(DANGLING_ASSIGNMENT_RE);
177    
178      // Pattern to identify a full assignment, which would be disassociated from a preceding dangling
179      // assignment.
180      private static final Pattern ASSIGNMENT_PATTERN =
181          Pattern.compile(String.format("%s=.+", DANGLING_ASSIGNMENT_RE));
182    
183      @VisibleForTesting static List<String> joinKeysToValues(String... args) {
184        List<String> joinedArgs = Lists.newArrayList();
185        String unmappedKey = null;
186        for (String arg : args) {
187          if (unmappedKey == null) {
188            if (DANGLING_ASSIGNMENT_PATTERN.matcher(arg).matches()) {
189              // Beginning of a possible dangling assignment.
190              unmappedKey = arg;
191            } else {
192              joinedArgs.add(arg);
193            }
194          } else {
195            if (ASSIGNMENT_PATTERN.matcher(arg).matches()) {
196              // Full assignment, disassociate from dangling assignment.
197              joinedArgs.add(unmappedKey);
198              joinedArgs.add(arg);
199              unmappedKey = null;
200            } else if (DANGLING_ASSIGNMENT_PATTERN.matcher(arg).find()) {
201              // Another dangling assignment, this could be two sequential boolean args.
202              joinedArgs.add(unmappedKey);
203              unmappedKey = arg;
204            } else {
205              // Join the dangling key with its value.
206              joinedArgs.add(unmappedKey + "=" + arg);
207              unmappedKey = null;
208            }
209          }
210        }
211    
212        if (unmappedKey != null) {
213          joinedArgs.add(unmappedKey);
214        }
215    
216        return joinedArgs;
217      }

219      private static final Pattern QUOTE_PATTERN = Pattern.compile("(['\"])([^\\\1]*)\\1");
220    
221      private static String stripQuotes(String str) {
222        Matcher matcher = QUOTE_PATTERN.matcher(str);
223        return matcher.matches() ? matcher.group(2) : str;
224      }

226      /**
227       * Scans through args, mapping keys to values even if the arg values are 'dangling' and reside
228       * in different array entries than the respective keys.
229       *
230       * @param args Arguments to build into a map.
231       * @return Map from argument key (arg name) to value.
232       */
233      public static Map<String, String> mapArguments(String... args) {
234        ImmutableMap.Builder<String, String> argMap = ImmutableMap.builder();
235        for (String arg : joinKeysToValues(args)) {
236          Matcher matcher = ARG_PATTERN.matcher(arg);
237          checkArgument(matcher.matches(),
238              String.format("Argument '%s' does not match required format -arg_name=arg_value", arg));
239    
240          String rawValue = matcher.group(2);
241          // An empty string denotes that the argument was passed with no value.
242          rawValue = rawValue == null ? "" : stripQuotes(rawValue);
243          argMap.put(matcher.group(1), rawValue);
244        }
245    
246        return argMap.build();
247      }

249      /**
250       * Extracts the @Arg annotation from a Field and includes it as a pair.
251       */
252      private static final Function<Field, Pair<Field, CmdLine>> PAIR_WITH_ANNOTATION =
253          new Function<Field, Pair<Field, CmdLine>>() {
254            @Override public Pair<Field, CmdLine> apply(Field field) {
255              CmdLine cmdLineDef = field.getAnnotation(CmdLine.class);
256              Preconditions.checkNotNull(cmdLineDef, "No Arg annotation for field " + field);
257              return Pair.of(field, cmdLineDef);
258            }
259          };

261      /**
262       * Performs multiple checks to ensure the definition of an argument is good.
263       */
264      private static final Closure<Pair<Field, CmdLine>> CHECK_ARG_DEF =
265          new Closure<Pair<Field, CmdLine>>() {
266        @Override public void execute(Pair<Field, CmdLine> argDef) {
267          checkArgument(Modifier.isStatic(argDef.getFirst().getModifiers()),
268              "Non-static argument fields are not supported, found " + argDef.getFirst());
269    
270          String argName = argDef.getSecond().name();
271          checkArgument(ARG_NAME_PATTERN.matcher(argName).matches(),
272              String.format("Argument name '%s' on %s does not match required pattern %s",
273                  argName, argDef.getFirst(), ARG_NAME_RE));
274    
275          checkArgument(getParser(argDef.getSecond(), argDef.getFirst()) != null,
276              "No parser found for type " + TypeUtil.getTypeParamClass(argDef.getFirst())
277                  + ", for arg field " + argDef.getFirst().getName());
278        }
279      };

281      /**
282       * Extracts the name from an @Arg.
283       */
284      private static final Function<CmdLine, String> GET_ARG_NAME = new Function<CmdLine, String>() {
285          @Override public String apply(CmdLine cmdLine) {
286            return cmdLine.name();
287          }
288        };

290      /**
291       * Gets the canonical name for an @Arg, based on the class containing the field it annotates.
292       */
293      private static final Function<Pair<Field, CmdLine>, String> GET_CANONICAL_ARG_NAME =
294          new Function<Pair<Field, CmdLine>, String>() {
295            @Override public String apply(Pair<Field, CmdLine> argDef) {
296              return argDef.getFirst().getDeclaringClass().getCanonicalName()
297                  + "." + argDef.getSecond().name();
298            }
299          };
300    
301      /**
302       * Gets the canonical negated name for an @Arg.
303       */
304      private static final Function<Pair<Field, CmdLine>, String> GET_CANONICAL_NEGATED_ARG_NAME =
305          new Function<Pair<Field, CmdLine>, String>() {
306            @Override public String apply(Pair<Field, CmdLine> argDef) {
307              return argDef.getFirst().getDeclaringClass().getCanonicalName()
308                  + "." + NEGATE_BOOLEAN + argDef.getSecond().name();
309            }
310          };
311    
312      private static final Function<String, String> PREPEND_NEGATION = new Function<String, String>() {
313        @Override public String apply(String name) {
314          return NEGATE_BOOLEAN + name;
315        }
316      };

318      private static <T> Set<T> dropCollisions(Iterable<T> input) {
319        Set<T> copy = Sets.newHashSet();
320        Set<T> collisions = Sets.newHashSet();
321        for (T entry: input) {
322          if (!copy.add(entry)) {
323            collisions.add(entry);
324          }
325        }
326    
327        copy.removeAll(collisions);
328        return copy;
329      }

331      /**
332       * Applies argument values to fields based on their annotations.
333       *
334       * @param argFields Fields to apply argument values to.
335       * @param args Unparsed argument values.
336       */
337      private static void process(Iterable<Field> argFields, Map<String, String> args) {
338        final Set<String> argsFailedToParse = Sets.newHashSet();
339        final Set<String> argsConstraintsFailed = Sets.newHashSet();
340    
341        final Iterable<Pair<Field, CmdLine>> argDefs =
342            Iterables.transform(argFields, PAIR_WITH_ANNOTATION);
343    
344        for (Pair<Field, CmdLine> argDef : argDefs) {
345          CHECK_ARG_DEF.execute(argDef);
346        }
347    
348        Iterable<String> argShortNames = Iterables.transform(argDefs,
349            Functions.compose(GET_ARG_NAME, Pair.<Field, CmdLine>second()));
350        Set<String> argShortNamesNoCollisions = dropCollisions(argShortNames);
351        Set<String> collisionsDropped = Sets.difference(ImmutableSet.copyOf(argShortNames),
352            argShortNamesNoCollisions);
353        if (!collisionsDropped.isEmpty()) {
354          LOG.warning("Found argument name collisions, args must be referenced by canonical names: "
355              + collisionsDropped);
356        }
357    
358        Predicate<Pair<Field, CmdLine>> isBoolean = Predicates.compose(
359            Predicates.<Class>equalTo(Boolean.class),
360            Functions.compose(TypeUtil.GET_TYPE_PARAM_CLASS, Pair.<Field, CmdLine>first()));
361    
362        final Function<Pair<Field, CmdLine>, String> getFieldArgName =
363            Functions.compose(GET_ARG_NAME, Pair.<Field, CmdLine>second());
364    
365        final Map<String, Pair<Field, CmdLine>> argsByName =
366            ImmutableMap.<String, Pair<Field, CmdLine>>builder()
367            // Map by short arg name -> arg def.
368            .putAll(Maps.uniqueIndex(Iterables.filter(argDefs,
369                Predicates.compose(Predicates.in(argShortNamesNoCollisions), getFieldArgName)),
370                getFieldArgName))
371            // Map by canonical arg name -> arg def.
372            .putAll(Maps.uniqueIndex(argDefs, GET_CANONICAL_ARG_NAME))
373            // Map by negated short arg name (for booleans)
374            .putAll(Maps.uniqueIndex(Iterables.filter(argDefs, isBoolean),
375                Functions.compose(PREPEND_NEGATION, getFieldArgName)))
376            // Map by negated canonical arg name (for booleans)
377            .putAll(Maps.uniqueIndex(Iterables.filter(argDefs, isBoolean),
378                GET_CANONICAL_NEGATED_ARG_NAME))
379            .build();
380    
381        // TODO(William Farner): Make sure to disallow duplicate arg specification by short and
382        // canonical names.
383    
384        // TODO(William Farner): Support non-atomic argument constraints.  @OnlyIfSet, @OnlyIfNotSet,
385        //    @ExclusiveOf to define inter-argument constraints.
386    
387        Set<String> recognizedArgs = Sets.intersection(argsByName.keySet(), args.keySet());
388    
389        for (String argName : recognizedArgs) {
390          String argValue = args.get(argName);
391          Field argField = argsByName.get(argName).getFirst();
392          CmdLine annotation = argsByName.get(argName).getSecond();
393    
394          Object assignmentValue;
395          try {
396            assignmentValue = getParser(annotation, argField)
397                .parse(TypeUtil.getTypeParam(argField), argValue);
398          } catch (IllegalArgumentException e) {
399            argsFailedToParse.add(argName + " - " + e.getMessage());
400            continue;
401          }
402    
403          // If the arg type is boolean, check if the command line uses the negated boolean form.
404          if (TypeUtil.getTypeParamClass(argField) == Boolean.class) {
405            Matcher negatedNameMatcher = NEGATED_BOOLEAN_PATTERN.matcher(argName);
406            if (negatedNameMatcher.matches()) {
407              String argShortName = negatedNameMatcher.group(negatedNameMatcher.groupCount());
408              String argCanonicalName = negatedNameMatcher.groupCount() == 2 ?
409                  negatedNameMatcher.group(1) + argShortName : null;
410              if (Iterables.any(argsByName.keySet(),
411                  Predicates.in(Arrays.asList(argShortName, argCanonicalName)))) {
412                assignmentValue = (!(Boolean) assignmentValue);
413              }
414            }
415          }
416    
417          setField(argField, assignmentValue);
418        }
419    
420        Set<String> commandLineArgumentInfos = Sets.newTreeSet();
421        for (Pair<Field, CmdLine> argDef : argDefs) {
422          Field argField = argDef.getFirst();
423          Arg arg = getArg(argField);
424    
425          commandLineArgumentInfos.add(String.format("%s (%s.%s): %s",
426              argDef.getSecond().name(), argField.getDeclaringClass().getName(), argField.getName(),
427              arg.uncheckedGet()));
428    
429          try {
430            checkConstraints(TypeUtil.getTypeParamClass(argField), arg.uncheckedGet(),
431                argField.getAnnotations());
432          } catch (IllegalArgumentException e) {
433            argsConstraintsFailed.add(argDef.getSecond().name() + " - " + e.getMessage());
434          }
435        }
436        infoLog("-------------------------------------------------------------------------");
437        infoLog("Command line argument values");
438        for (String commandLineArgumentInfo : commandLineArgumentInfos) {
439          infoLog(commandLineArgumentInfo);
440        }
441        infoLog("-------------------------------------------------------------------------");
442    
443        ImmutableMultimap<String, String> warningMessages =
444            ImmutableMultimap.<String, String>builder()
445            .putAll("Unrecognized arguments", Sets.difference(args.keySet(), argsByName.keySet()))
446            .putAll("Failed to parse", argsFailedToParse)
447            .putAll("Value did not meet constraints", argsConstraintsFailed)
448            .build();
449    
450        if (!warningMessages.isEmpty()) {
451          StringBuilder sb = new StringBuilder();
452          for (Map.Entry<String, Collection<String>> warnings : warningMessages.asMap().entrySet()) {
453            sb.append(warnings.getKey()).append(":\n\t").append(Joiner.on("\n\t")
454                .join(warnings.getValue())).append("\n");
455          }
456          throw new IllegalArgumentException(sb.toString());
457        }
458      }

460      private static void infoLog(String msg) {
461        System.out.println(msg);
462      }

464      private static void checkConstraints(Class<?> fieldClass, Object value,
465          Annotation[] annotations) {
466        for (Annotation annotation : maybeFilterNullable(fieldClass, value, annotations)) {
467          verify(fieldClass, value, annotation);
468        }
469      }
470    
471      private static Iterable<Annotation> maybeFilterNullable(Class<?> fieldClass, Object value,
472          Annotation[] annotations) {
473    
474        // Apply all the normal constraint annotations
475        if (value != null) {
476          return Arrays.asList(annotations);
477        }
478    
479        // The value is null so skip verifications unless a @NotNull was specified as a constraint
480        for (Annotation annotation: annotations) {
481          if (annotation instanceof NotNull) {
482            verify(fieldClass, value, annotation); // will throw
483          }
484        }
485        return ImmutableList.of();
486      }
487    
488      @SuppressWarnings("unchecked")
489      private static void verify(Class<?> fieldClass, Object value, Annotation annotation) {
490        Verifier verifier = Constraints.get(fieldClass, annotation);
491        if (verifier != null) {
492          verifier.verify(value, annotation);
493        }
494      }
495    
496      private static Parser getParser(CmdLine cmdLine, Field argField) {
497        Preconditions.checkArgument(argField.getType() == Arg.class,
498            "Field is annotated with @CmdLine but is not of Arg type: " + argField);
499        Class cls = TypeUtil.getTypeParamClass(argField);
500    
501        try {
502          return cmdLine.parser() == Parser.class ? Parsers.get(cls) : cmdLine.parser().newInstance();
503        } catch (InstantiationException e) {
504          throw new RuntimeException("Failed to instantiate parser " + cmdLine.parser());
505        } catch (IllegalAccessException e) {
506          throw new RuntimeException("No access to instantiate parser " + cmdLine.parser());
507        }
508      }
509    
510      @SuppressWarnings("unchecked")
511      private static void setField(Field field, Object value) {
512        getArg(field).set(value);
513      }
514    
515      private static Arg getArg(Field field) {
516        field.setAccessible(true);
517        try {
518          return ((Arg) field.get(null));
519        } catch (IllegalAccessException e) {
520          throw new RuntimeException("Failed to set value for " + field);
521        }
522      }

524      private static final Function<String, Predicate<Field>> PREFIX_TO_FILTER =
525          new Function<String, Predicate<Field>>() {
526            @Override public Predicate<Field> apply(final String packagePrefix) {
527              return new Predicate<Field>() {
528                @Override public boolean apply(Field field) {
529                  return field.getDeclaringClass().getName().startsWith(packagePrefix);
530                }
531              };
532            }
533          };

535      private static Iterable<Field> scan(Predicate<Field> filter) {
536        try {
537          Configuration configuration = Configuration.load();
538          return Iterables.filter(configuration.fields(), filter);
539        } catch (IOException e) {
540          throw new ArgScanException(e);
541        }
542      }
