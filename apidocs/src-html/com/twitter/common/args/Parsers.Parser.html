051    /**
052     * Parsers for different supported argument types.
053     *
054     * TODO(William Farner): Add a mechanism for global registration of custom parsers.
055     *
056     * @author William Farner
057     */
058    public final class Parsers {
059    
060      private Parsers() {
061        // Utility
062      }
063    
064      /**
065       * Gets the parser associated with a class.
066       *
067       * @param cls Class to get the parser for.
068       * @return The parser for {@code cls} or {@code null} if no parser was found for that type or any
069       *     of its supertypes.
070       */
071      static Parser get(Class<?> cls) {
072        Parser parser;
073        while (((parser = REGISTRY.get(cls)) == null) && (cls != null)) {
074          cls = cls.getSuperclass();
075        }
076        return parser;
077      }
078    
079      /**
080       * Gets the parser associatd with a class.
081       *
082       * @param cls Class to get the parser for.
083       * @return The parser for {@code cls} or {@code null} if no parser was found for that type.
084       * @throws IllegalArgumentException If no parser was found for {@code cls}.
085       */
086      public static Parser checkedGet(Class<?> cls) throws IllegalArgumentException {
087        Parser parser = get(cls);
088        checkArgument(parser != null, "No parser found for class " + cls);
089        return parser;
090      }
091    
092      /**
093       * An interface to a command line argument parser.
094       *
095       * @param <T> The base class this parser can parse all subtypes of.
096       */
097      public interface Parser<T> {
098    
099        /**
100         * Parses strings as arguments of a given subtype of {@code T}.
101         *
102         * @param type The target type of the parsed argument.
103         * @param raw The raw value of the argument.
104         * @return A value of the given type parsed from the raw value.
105         * @throws IllegalArgumentException if the raw value could not be parsed into a value of the
106         *     given type.
107         */
108        T parse(Type type, String raw) throws IllegalArgumentException;
109    
110        /**
111         * Returns the root of the class hierarchy this parser handles.
112         *
113         * <p>Note that implementations that return a proper supertype class instead of T.class must
114         * be able to handle {@link #parse(java.lang.reflect.Type, String) parsing} raw values into
115         * all possible subtypes in use as command line argument types.
116         *
117         * @return The base class this parser can parse all subtypes of.
118         */
119        Class<? super T> getParsedClass();
120      }
121    
122      private static final Set<Class<? extends Parser>> PARSER_CLASSES =
123          ImmutableSet.<Class<? extends Parser>>builder()
124          .add(AmountParser.class)
125          .add(BooleanParser.class)
126          .add(ByteParser.class)
127          .add(CharacterParser.class)
128          .add(ClassParser.class)
129          .add(DateParser.class)
130          .add(DoubleParser.class)
131          .add(EnumParser.class)
132          .add(FileParser.class)
133          .add(FloatParser.class)
134          .add(InetSocketAddressParser.class)
135          .add(IntegerParser.class)
136          .add(ListParser.class)
137          .add(LongParser.class)
138          .add(MapParser.class)
139          .add(PairParser.class)
140          .add(SetParser.class)
141          .add(ShortParser.class)
142          .add(StringParser.class)
143          .add(UnitParser.class)
144          .add(URIParser.class)
145          .add(URLParser.class)
146          .build();
147      private static final ImmutableMap<Class<?>, Parser> REGISTRY;
148      static {
149        ImmutableMap.Builder<Class<?>, Parser> registryBuilder =
150          ImmutableMap.builder();
151        for (Class<? extends Parser> parserClass : PARSER_CLASSES) {
152          try {
153            Parser parser = parserClass.newInstance();
154            registryBuilder.put(parser.getParsedClass(), parser);
155          } catch (InstantiationException e) {
156            throw new RuntimeException("Failed to instantiate parser class " + parserClass, e);
157          } catch (IllegalAccessException e) {
158            throw new RuntimeException("No access to constructor on parser " + parserClass, e);
159          }
160        }
161        REGISTRY = registryBuilder.build();
162      }
163    
164      public static final Splitter MULTI_VALUE_SPLITTER = Splitter.on(",")
165          .trimResults().omitEmptyStrings();
166    }
