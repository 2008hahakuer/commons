051    /**
052     * Parsers for different supported argument types.
053     *
054     * TODO(William Farner): Add a mechanism for global registration of custom parsers.
055     *
056     * @author William Farner
057     */
058    public final class Parsers {
059    
060      private Parsers() {
061        // Utility
062      }
063    
064      /**
065       * Gets the parser associated with a class.
066       *
067       * @param cls Class to get the parser for.
068       * @return The parser for {@code cls} or {@code null} if no parser was found for that type.
069       */
070      public static Parser get(Class<?> cls) {
071        return REGISTRY.get(cls);
072      }
073    
074      /**
075       * Gets the parser associatd with a class.
076       *
077       * @param cls Class to get the parser for.
078       * @return The parser for {@code cls} or {@code null} if no parser was found for that type.
079       * @throws IllegalArgumentException If no parser was found for {@code cls}.
080       */
081      public static Parser checkedGet(Class<?> cls) throws IllegalArgumentException {
082        Parser parser = get(cls);
083        checkArgument(parser != null, "No parser found for class " + cls);
084        return parser;
085      }
086    
087      public static interface Parser<T> {
088        T parse(Type type, String raw) throws IllegalArgumentException;
089    
090        Class<T> getParsedClass();
091      }
092    
093      private static final Set<Class<? extends Parser<?>>> PARSER_CLASSES =
094          ImmutableSet.<Class<? extends Parser<?>>>builder()
095          .add(AmountParser.class)
096          .add(BooleanParser.class)
097          .add(ByteParser.class)
098          .add(CharacterParser.class)
099          .add(ClassParser.class)
100          .add(DateParser.class)
101          .add(DoubleParser.class)
102          .add(FileParser.class)
103          .add(FloatParser.class)
104          .add(InetSocketAddressParser.class)
105          .add(IntegerParser.class)
106          .add(ListParser.class)
107          .add(LongParser.class)
108          .add(MapParser.class)
109          .add(PairParser.class)
110          .add(SetParser.class)
111          .add(ShortParser.class)
112          .add(StringParser.class)
113          .add(UnitParser.class)
114          .add(URIParser.class)
115          .add(URLParser.class)
116          .build();
117      private static final ImmutableMap<Class<?>, Parser> REGISTRY;
118      static {
119        ImmutableMap.Builder<Class<?>, Parser> registryBuilder =
120          ImmutableMap.builder();
121        for (Class<? extends Parser<?>> parserClass : PARSER_CLASSES) {
122          try {
123            Parser parser = parserClass.newInstance();
124            registryBuilder.put(parser.getParsedClass(), parser);
125          } catch (InstantiationException e) {
126            throw new RuntimeException("Failed to instantiate parser class " + parserClass, e);
127          } catch (IllegalAccessException e) {
128            throw new RuntimeException("No access to constructor on parser " + parserClass, e);
129          }
130        }
131        REGISTRY = registryBuilder.build();
132      }
133    
134      public static final Splitter MULTI_VALUE_SPLITTER = Splitter.on(",")
135          .trimResults().omitEmptyStrings();
136    }
