075    /**
076     * Processes {@literal @CmdLine} annotated fields and {@literal @ArgParser} and
077     * {@literal @VerifierFor} parser and verifier registrations and stores configuration data listing
078     * these fields, parsers and verifiers on the classpath for discovery via
079     * {@link com.twitter.common.args.apt.Configuration#load()}.
080     *
081     * <p>Supports an apt option useful for some build setups that create monolithic jars aggregating
082     * many library jars, one or more of which have embedded arg definitions themselves.  By adding the
083     * following flag to a javac invocation:
084     * <code>-Acom.twitter.common.args.apt.CmdLineProcessor.main</code>
085     * you signal this apt processor that the compilation target is a leaf target that will comprise one
086     * or more executable mains (as opposed to a library jar).  As a result, the embedded arg
087     * definitions generated will occupy a special resource that is always checked for first during
088     * runtime arg parsing.
089     *
090     * @author John Sirois
091     */
092    @SupportedSourceVersion(SourceVersion.RELEASE_6)
093    @SupportedOptions({
094        CmdLineProcessor.MAIN_OPTION,
095        CmdLineProcessor.CHECK_LINKAGE_OPTION
096    })
097    public class CmdLineProcessor extends AbstractProcessor {
098      private static final String OPTION_PREFIX = "com.twitter.common.args.apt.CmdLineProcessor";
099      static final String MAIN_OPTION = OPTION_PREFIX + ".main";
100      static final String CHECK_LINKAGE_OPTION = OPTION_PREFIX + ".check_linkage";
101    
102      private static final Function<Class<?>,String> GET_NAME = new Function<Class<?>, String>() {
103        @Override public String apply(Class<?> type) {
104          return type.getName();
105        }
106      };
107    
108      private static boolean getBooleanOption(Map<String, String> options, String name,
109          boolean defaultValue) {
110    
111        if (!options.containsKey(name)) {
112          return defaultValue;
113        }
114    
115        // We want to map the presence of a boolean option without a value to indicate true, giving the
116        // following accepted boolean option formats:
117        // -Afoo -> true
118        // -Afoo=false -> false
119        // -Afoo=true -> true
120    
121        String isOption = options.get(name);
122        return (isOption == null) || Boolean.parseBoolean(isOption);
123      }

132      @Override
133      public void init(ProcessingEnvironment processingEnv) {
134        super.init(processingEnv);
135    
136        typeUtils = processingEnv.getTypeUtils();
137        elementUtils = processingEnv.getElementUtils();
138    
139        Map<String, String> options = processingEnv.getOptions();
140        isMain = getBooleanOption(options, MAIN_OPTION, false);
141        isCheckLinkage = getBooleanOption(options, CHECK_LINKAGE_OPTION, true);
142      }

144      @Override
145      public Set<String> getSupportedAnnotationTypes() {
146        return ImmutableSet.copyOf(Iterables.transform(
147            ImmutableList.of(Positional.class, CmdLine.class, ArgParser.class, VerifierFor.class),
148            GET_NAME));
149      }

151      @Override
152      public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
153        @Nullable Configuration classpathConfiguration = configSupplier.get();
154    
155        Set<? extends Element> parsers = getAnnotatedElements(roundEnv, ArgParser.class);
156        @Nullable Set<String> parsedTypes = getParsedTypes(classpathConfiguration, parsers);
157    
158        for (ArgInfo cmdLineInfo : processAnnotatedArgs(parsedTypes, roundEnv, CmdLine.class)) {
159          configBuilder.addCmdLineArg(cmdLineInfo);
160        }
161    
162        for (ArgInfo positionalInfo : processAnnotatedArgs(parsedTypes, roundEnv, Positional.class)) {
163          configBuilder.addPositionalInfo(positionalInfo);
164        }
165        checkPositionalArgsAreLists(roundEnv);
166    
167        processParsers(parsers);
168        processVerifiers(roundEnv.getElementsAnnotatedWith(typeElement(VerifierFor.class)));
169    
170        if (roundEnv.processingOver()) {
171          if (classpathConfiguration != null
172              && (!classpathConfiguration.isEmpty() || !configBuilder.isEmpty())) {
173    
174            @Nullable Writer cmdLinePropertiesResource =
175                openCmdLinePropertiesResource(classpathConfiguration);
176            if (cmdLinePropertiesResource != null) {
177              try {
178                configBuilder.build(classpathConfiguration).store(cmdLinePropertiesResource,
179                    "Generated via apt by " + getClass().getName());
180              } finally {
181                Closeables.closeQuietly(cmdLinePropertiesResource);
182              }
183            }
184          }
185        }
186        return true;
187      }
188    
189      private void checkPositionalArgsAreLists(RoundEnvironment roundEnv) {
190        for (Element positionalArg : getAnnotatedElements(roundEnv, Positional.class)) {
191          TypeMirror typeArgument = getTypeArgument(positionalArg.asType(), typeElement(Arg.class));
192          if (!typeUtils.isSubtype(typeElement(List.class).asType(), typeArgument)) {
193            error("Found @Positional %s %s.%s that is not a List",
194                positionalArg.asType(), positionalArg.getEnclosingElement(), positionalArg);
195          }
196        }
197      }

199      @Nullable
200      private Set<String> getParsedTypes(@Nullable Configuration configuration,
201          Set<? extends Element> parsers) {
202    
203        if (!isCheckLinkage) {
204          return null;
205        }
206    
207        Iterable<String> parsersFor = Iterables.transform(parsers,
208            new Function<Element, String>() {
209              @Override public String apply(Element parser) {
210                TypeMirror parsedType = getTypeArgument(parser.asType(), typeElement(Parser.class));
211                if (parsedType == null) {
212                  error("failed to find a type argument for Parser: %s", parser);
213                  return null;
214                }
215                // Equals on TypeMirrors doesn't work - so we compare string representations :/
216                return typeUtils.erasure(parsedType).toString();
217              }
218            });
219        if (configuration != null) {
220          parsersFor = Iterables.concat(parsersFor, Iterables.filter(
221              Iterables.transform(configuration.parserInfo(),
222                  new Function<ParserInfo, String>() {
223                    @Override @Nullable public String apply(ParserInfo parserInfo) {
224                      TypeElement typeElement = elementUtils.getTypeElement(parserInfo.parsedType);
225                      // We may not have a type on the classpath for a previous round - this is fine as
226                      // long as the no Args in this round are of the type.
227                      return (typeElement == null)
228                          ? null : typeUtils.erasure(typeElement.asType()).toString();
229                    }
230                  }), Predicates.notNull()));
231        }
232        return ImmutableSet.copyOf(parsersFor);
233      }

235      private final Supplier<Configuration> configSupplier =
236          Suppliers.memoize(new Supplier<Configuration>() {
237            @Override public Configuration get() {
238              try {
239                Configuration configuration = Configuration.load();
240                for (ArgInfo argInfo : configuration.positionalInfo()) {
241                  configBuilder.addPositionalInfo(argInfo);
242                }
243                for (ArgInfo argInfo : configuration.optionInfo()) {
244                  configBuilder.addCmdLineArg(argInfo);
245                }
246                for (ParserInfo parserInfo : configuration.parserInfo()) {
247                  configBuilder.addParser(parserInfo);
248                }
249                for (VerifierInfo verifierInfo : configuration.verifierInfo()) {
250                  configBuilder.addVerifier(verifierInfo);
251                }
252                return configuration;
253              } catch (IOException e) {
254                error("Problem loading existing flags on compile time classpath: %s",
255                    Throwables.getStackTraceAsString(e));
256                return null;
257              }
258            }
259          });

261      private Iterable<ArgInfo> processAnnotatedArgs(@Nullable final Set<String> parsedTypes,
262          RoundEnvironment roundEnv, final Class<? extends Annotation> argAnnotation) {
263    
264        Set<? extends Element> args = getAnnotatedElements(roundEnv, argAnnotation);
265        return Optional.presentInstances(Iterables.transform(args,
266            new Function<Element, Optional<ArgInfo>>() {
267              @Override public Optional<ArgInfo> apply(Element arg) {
268                @Nullable TypeElement containingType = processArg(parsedTypes, arg, argAnnotation);
269                  if (containingType == null) {
270                    return Optional.absent();
271                  } else {
272                    return Optional.of(new ArgInfo(getBinaryName(containingType),
273                        arg.getSimpleName().toString()));
274                  }
275                }
276            }));
277      }

279      private Set<? extends Element> getAnnotatedElements(RoundEnvironment roundEnv,
280          Class<? extends Annotation> argAnnotation) {
281        return roundEnv.getElementsAnnotatedWith(typeElement(argAnnotation));
282      }

284      @Nullable
285      private TypeElement processArg(@Nullable Set<String> parsedTypes, Element annotationElement,
286          Class<? extends Annotation> annotationType) {
287    
288        TypeElement parserType = typeElement(Parser.class);
289        if (annotationElement.getKind() != ElementKind.FIELD) {
290          error("Found a @%s annotation on a non-field %s",
291              annotationType.getSimpleName(), annotationElement);
292          return null;
293        } else {
294          // Only types contain fields so this cast is safe.
295          TypeElement containingType = (TypeElement) annotationElement.getEnclosingElement();
296    
297          if (!isAssignable(annotationElement.asType(), Arg.class)) {
298            error("Found a @%s annotation on a non-Arg %s.%s",
299                annotationType.getSimpleName(), containingType, annotationElement);
300            return null;
301          }
302          if (!annotationElement.getModifiers().contains(Modifier.STATIC)) {
303            error("Found a @%s annotation on a non-static Arg field %s.%s",
304                annotationType.getSimpleName(), containingType, annotationElement);
305            return null;
306          }
307    
308          if (parsedTypes != null) {
309            // Check Parser<T> linkage for the Arg<T> type T.
310            TypeMirror typeArgument =
311                getTypeArgument(annotationElement.asType(), typeElement(Arg.class));
312            @Nullable AnnotationMirror cmdLine =
313                getAnnotationMirror(annotationElement, typeElement(annotationType));
314            if (cmdLine != null) {
315              TypeMirror customParserType = getClassType(cmdLine, "parser", parserType).asType();
316              if (typeUtils.isSameType(parserType.asType(), customParserType)) {
317                if (!checkTypePresent(parsedTypes, typeArgument)) {
318                  error("No parser registered for %s, %s.%s is un-parseable",
319                      typeArgument, containingType, annotationElement);
320                }
321              } else {
322                TypeMirror customParsedType = getTypeArgument(customParserType, parserType);
323                if (!isAssignable(typeArgument, customParsedType)) {
324                  error("Custom parser %s parses %s but registered for %s.%s with Arg type %s",
325                      customParserType, customParsedType, containingType, annotationElement,
326                      typeArgument);
327                }
328              }
329            }
330          }
331    
332          // TODO(John Sirois): Add additional compile-time @CmdLine verification for:
333          // 1.) for each @CmdLine Arg<T> annotated with @VerifierFor.annotation: T is a subtype of
334          //     V where there is a Verifier<V>
335          // 2.) name checks, including dups
336    
337          return containingType;
338        }
339      }
340    
341      private boolean checkTypePresent(Set<String> types, TypeMirror type) {
342        Iterable<TypeMirror> allTypes = getAllTypes(type);
343        for (TypeMirror t : allTypes) {
344          if (types.contains(typeUtils.erasure(t).toString())) {
345            return true;
346          }
347        }
348        return false;
349      }
350    
351      private void processParsers(Set<? extends Element> elements) {
352        TypeElement parserType = typeElement(Parser.class);
353        for (Element element : elements) {
354          if (element.getKind() != ElementKind.CLASS) {
355            error("Found an @ArgParser annotation on a non-class %s", element);
356          } else {
357            TypeElement parser = (TypeElement) element;
358            if (!isAssignable(parser, Parser.class)) {
359              error("Found an @ArgParser annotation on a non-Parser %s", element);
360              return;
361            }
362    
363            @Nullable String parsedType = getTypeArgument(parser, parserType);
364            if (parsedType != null) {
365              configBuilder.addParser(parsedType, getBinaryName(parser));
366            }
367          }
368        }
369      }
370    
371      private void processVerifiers(Set<? extends Element> elements) {
372        TypeElement verifierType = typeElement(Verifier.class);
373        TypeElement verifierForType = typeElement(VerifierFor.class);
374        for (Element element : elements) {
375          if (element.getKind() != ElementKind.CLASS) {
376            error("Found a @VerifierFor annotation on a non-class %s", element);
377          } else {
378            TypeElement verifier = (TypeElement) element;
379            if (!isAssignable(verifier, Verifier.class)) {
380              error("Found a @Verifier annotation on a non-Verifier %s", element);
381              return;
382            }
383    
384            @Nullable AnnotationMirror verifierFor = getAnnotationMirror(verifier, verifierForType);
385            if (verifierFor != null) {
386              @Nullable TypeElement verifyAnnotationType = getClassType(verifierFor, "value", null);
387              if (verifyAnnotationType != null) {
388                @Nullable String verifiedType = getTypeArgument(verifier, verifierType);
389                if (verifiedType != null) {
390                  String verifyAnnotationClassName =
391                      elementUtils.getBinaryName(verifyAnnotationType).toString();
392                  configBuilder.addVerifier(verifiedType, verifyAnnotationClassName,
393                      getBinaryName(verifier));
394                }
395              }
396            }
397          }
398        }
399      }

401      @Nullable
402      private String getTypeArgument(TypeElement annotatedType, final TypeElement baseType) {
403        TypeMirror typeArgument = getTypeArgument(annotatedType.asType(), baseType);
404        return typeArgument == null
405            ? null
406            : getBinaryName((TypeElement) typeUtils.asElement(typeArgument));
407      }

409      private Iterable<TypeMirror> getAllTypes(TypeMirror type) {
410        return getAllTypes(new HashSet<String>(), Lists.<TypeMirror>newArrayList(), type);
411      }
412    
413      private Iterable<TypeMirror> getAllTypes(Set<String> visitedTypes, List<TypeMirror> types,
414          TypeMirror type) {
415    
416        String typeName = typeUtils.erasure(type).toString();
417        if (!visitedTypes.contains(typeName)) {
418          types.add(type);
419          visitedTypes.add(typeName);
420          for (TypeMirror superType : typeUtils.directSupertypes(type)) {
421            getAllTypes(visitedTypes, types, superType);
422          }
423        }
424        return types;
425      }

427      @Nullable
428      private TypeMirror getTypeArgument(TypeMirror annotatedType, final TypeElement baseType) {
429        for (TypeMirror type : getAllTypes(annotatedType)) {
430          TypeMirror typeArgument = type.accept(new SimpleTypeVisitor6<TypeMirror, Void>() {
431            @Override public TypeMirror visitDeclared(DeclaredType t, Void aVoid) {
432              if (isAssignable(t, baseType)) {
433                List<? extends TypeMirror> typeArguments = t.getTypeArguments();
434                if (!typeArguments.isEmpty()) {
435                  return typeUtils.erasure(typeArguments.get(0));
436                }
437              }
438              return null;
439            }
440          }, null);
441    
442          if (typeArgument != null) {
443            return typeArgument;
444          }
445        }
446        error("Failed to find a type argument for %s in %s", baseType, annotatedType);
447        return null;
448      }

450      @Nullable
451      private AnnotationMirror getAnnotationMirror(Element element, TypeElement annotationType) {
452        for (AnnotationMirror annotationMirror : element.getAnnotationMirrors()) {
453          if (typeUtils.isSameType(annotationMirror.getAnnotationType(),  annotationType.asType())) {
454            return annotationMirror;
455          }
456        }
457        error("Failed to find an annotation of type %s on %s", annotationType, element);
458        return null;
459      }

461      @SuppressWarnings("unchecked")
462      private TypeElement getClassType(AnnotationMirror annotationMirror, String methodName,
463          TypeElement defaultClassType) {
464    
465        for (Map.Entry<? extends ExecutableElement, ? extends AnnotationValue> entry
466            : annotationMirror.getElementValues().entrySet()) {
467          if (entry.getKey().getSimpleName().equals(elementUtils.getName(methodName))) {
468            TypeElement classType = entry.getValue().accept(
469                new SimpleAnnotationValueVisitor6<TypeElement, Void>() {
470                  @Override public TypeElement visitType(TypeMirror t, Void unused) {
471                    return (TypeElement) processingEnv.getTypeUtils().asElement(t);
472                  }
473                }, null);
474    
475            if (classType != null) {
476              return classType;
477            }
478          }
479        }
480        if (defaultClassType == null) {
481          error("Could not find a class type for %s.%s", annotationMirror, methodName);
482        }
483        return defaultClassType;
484      }

486      @Nullable
487      private FileObject createCommandLineDb(Configuration configuration) {
488        String name = isMain ? Configuration.mainResourceName() : configuration.nextResourceName();
489        try {
490          return processingEnv.getFiler().createResource(StandardLocation.CLASS_OUTPUT,
491              Configuration.DEFAULT_RESOURCE_PACKAGE, name);
492        } catch (IOException e) {
493          error("Failed to create resource file to store %s/%s: %s",
494              Configuration.DEFAULT_RESOURCE_PACKAGE, name, Throwables.getStackTraceAsString(e));
495          return null;
496        }
497      }
498    
499      @Nullable
500      private Writer openCmdLinePropertiesResource(Configuration configuration) {
501        FileObject resource = createCommandLineDb(configuration);
502        if (resource == null) {
503          return null;
504        }
505        try {
506          log(Kind.NOTE, "Writing %s", resource.toUri());
507          return resource.openWriter();
508        } catch (IOException e) {
509          if (!resource.delete()) {
510            log(Kind.WARNING, "Failed to clean up %s after a failing to open it for writing",
511                resource.toUri());
512          }
513          error("Failed to open resource file to store %s: %s", resource.toUri(),
514              Throwables.getStackTraceAsString(e));
515          return null;
516        }
517      }
