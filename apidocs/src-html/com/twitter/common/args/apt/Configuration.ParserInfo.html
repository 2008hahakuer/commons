070      private static final CharMatcher IDENTIFIER_START =
071          CharMatcher.forPredicate(new Predicate<Character>() {
072            @Override public boolean apply(Character c) {
073              return Character.isJavaIdentifierStart(c);
074            }
075          });
076    
077      private static final CharMatcher IDENTIFIER_REST =
078          CharMatcher.forPredicate(new Predicate<Character>() {
079            @Override public boolean apply(Character c) {
080              return Character.isJavaIdentifierPart(c);
081            }
082          });

084      private static String checkValidIdentifier(String identifier, boolean compound) {
085        Preconditions.checkNotNull(identifier);
086    
087        String trimmed = identifier.trim();
088        Preconditions.checkArgument(!trimmed.isEmpty(), "Invalid identifier: '%s'", identifier);
089    
090        String[] parts = compound ? trimmed.split("\\.") : new String[] { trimmed };
091        for (String part : parts) {
092          Preconditions.checkArgument(
093              IDENTIFIER_REST.matchesAllOf(IDENTIFIER_START.trimLeadingFrom(part)),
094              "Invalid identifier: '%s'", identifier);
095        }
096    
097        return trimmed;
098      }

100      public static final class ArgInfo {
101        public final String className;
102        public final String fieldName;
103    
104        public ArgInfo(String className, String fieldName) {
105          this.className = checkValidIdentifier(className, true);
106          this.fieldName = checkValidIdentifier(fieldName, false);
107        }
108    
109        @Override
110        public boolean equals(Object obj) {
111          if (this == obj) {
112            return true;
113          }
114    
115          if (!(obj instanceof ArgInfo)) {
116            return false;
117          }
118    
119          ArgInfo other = (ArgInfo) obj;
120    
121          return new EqualsBuilder()
122              .append(className, other.className)
123              .append(fieldName, other.fieldName)
124              .isEquals();
125        }
126    
127        @Override
128        public int hashCode() {
129          return new HashCodeBuilder()
130              .append(className)
131              .append(fieldName)
132              .toHashCode();
133        }
134    
135        @Override public String toString() {
136          return new ToStringBuilder(this)
137              .append("className", className)
138              .append("fieldName", fieldName)
139              .toString();
140        }
141      }

236        private final Set<ArgInfo> argInfos = Sets.newHashSet();
237        private final Set<ParserInfo> parserInfos = Sets.newHashSet();
238        private final Set<VerifierInfo> verifierInfos = Sets.newHashSet();

242              && argInfos.isEmpty()
243              && parserInfos.isEmpty()
244              && verifierInfos.isEmpty();

265        }

277      private static final Function<URL, InputSupplier<? extends InputStream>> URL_TO_INPUT =
278          new Function<URL, InputSupplier<? extends InputStream>>() {
279            @Override public InputSupplier<? extends InputStream> apply(final URL resource) {
280              return new InputSupplier<InputStream>() {
281                @Override public InputStream getInput() throws IOException {
282                  return resource.openStream();
283                }
284              };
285            }
286          };
287    
288      private static final Function<InputSupplier<? extends InputStream>,
289                                    InputSupplier<? extends Reader>> INPUT_TO_READER =
290          new Function<InputSupplier<? extends InputStream>, InputSupplier<? extends Reader>>() {
291            @Override public InputSupplier<? extends Reader> apply(
292                final InputSupplier<? extends InputStream> input) {
293              return CharStreams.newReaderSupplier(input, Charsets.UTF_8);
294            }
295          };
296    
297      private static final Function<URL, InputSupplier<? extends Reader>> URL_TO_READER =
298          Functions.compose(INPUT_TO_READER, URL_TO_INPUT);

315        final int nextResourceIndex;
316        final Iterator<URL> resources;

319          this.nextResourceIndex = nextResourceIndex;
320          this.resources = resources;

324      /**
325       * Loads the {@literal @CmdLine} argument configuration data stored in the classpath.
326       *
327       * @return The {@literal @CmdLine} argument configuration materialized from the classpath.
328       * @throws ConfigurationException if any configuration data is malformed.
329       * @throws IOException if the configuration data can not be read from the classpath.
330       */
331      public static Configuration load() throws ConfigurationException, IOException {
332        ConfigurationResources allResources = getAllResources();
333        List<URL> configs = ImmutableList.copyOf(allResources.resources);
334        if (configs.isEmpty()) {
335          LOG.info("No @CmdLine arg configs found on the classpath");
336        } else {
337          LOG.info("Loading @CmdLine config from: " + configs);
338        }
339        return load(allResources.nextResourceIndex, configs);
340      }
341    
342      private static ConfigurationResources getAllResources() throws IOException {
343        Iterator<URL> allResources = getResources(0); // Try for a main
344        for (int nextResourceIndex = 1; /* until no resources */; nextResourceIndex++) {
345          Iterator<URL> resources = getResources(nextResourceIndex);
346          if (resources.hasNext()) {
347            allResources = Iterators.concat(allResources, resources);
348          } else {
349            return new ConfigurationResources(nextResourceIndex, allResources);
350          }
351        }
352      }
353    
354      private static Iterator<URL> getResources(int index) throws IOException {
355        return Iterators.forEnumeration(
356            Configuration.class.getClassLoader().getResources(getResourcePath(index)));
357      }

359      private static class ConfigurationParser implements LineProcessor<Configuration> {
360        private final ImmutableList.Builder<ArgInfo> positionalInfo = ImmutableList.builder();
361        private final ImmutableList.Builder<ArgInfo> fieldInfoBuilder = ImmutableList.builder();
362        private final ImmutableList.Builder<ParserInfo> parserInfoBuilder = ImmutableList.builder();
363        private final ImmutableList.Builder<VerifierInfo> verifierInfoBuilder = ImmutableList.builder();
364    
365        final int nextIndex;
366        int lineNumber = 0;
367    
368        private ConfigurationParser(int nextIndex) {
369          this.nextIndex = nextIndex;
370        }
371    
372        @Override
373        public boolean processLine(String line) throws IOException {
374          ++lineNumber;
375          line = line.trim();
376          if (!line.isEmpty() && !line.startsWith("#")) {
377            List<String> parts = Lists.newArrayList(line.split(" "));
378            if (parts.size() < 1) {
379              throw new ConfigurationException("Invalid line: %s @%d", line, lineNumber);
380            }
381    
382            String type = parts.remove(0);
383            if ("positional".equals(type)) {
384              if (parts.size() != 2) {
385                throw new ConfigurationException("Invalid positional line: %s @%d", line, lineNumber);
386              }
387              positionalInfo.add(new ArgInfo(parts.get(0), parts.get(1)));
388            } else if ("field".equals(type)) {
389              if (parts.size() != 2) {
390                throw new ConfigurationException("Invalid field line: %s @%d", line, lineNumber);
391              }
392              fieldInfoBuilder.add(new ArgInfo(parts.get(0), parts.get(1)));
393            } else if ("parser".equals(type)) {
394              if (parts.size() != 2) {
395                throw new ConfigurationException("Invalid parser line: %s @%d", line, lineNumber);
396              }
397              parserInfoBuilder.add(new ParserInfo(parts.get(0), parts.get(1)));
398            } else if ("verifier".equals(type)) {
399              if (parts.size() != 3) {
400                throw new ConfigurationException("Invalid verifier line: %s @%d", line, lineNumber);
401              }
402              verifierInfoBuilder.add(new VerifierInfo(parts.get(0), parts.get(1), parts.get(2)));
403            } else {
404              LOG.warning(String.format("Did not recognize entry type %s for line: %s @%d",
405                  type, line, lineNumber));
406            }
407          }
408          return true;
409        }
410    
411        @Override
412        public Configuration getResult() {
413          return new Configuration(nextIndex, positionalInfo.build(),
414              fieldInfoBuilder.build(), parserInfoBuilder.build(), verifierInfoBuilder.build());
415        }
416      }
417    
418      private static Configuration load(int nextIndex, List<URL> configs)
419          throws ConfigurationException, IOException {
420        InputSupplier<Reader> input = CharStreams.join(Iterables.transform(configs, URL_TO_READER));
421        return CharStreams.readLines(input, new ConfigurationParser(nextIndex));
422      }

425      private final ImmutableSet<ArgInfo> positionalInfos;

431          Iterable<ArgInfo> positionalInfos, Iterable<ArgInfo> cmdLineInfos,

493      void store(Writer output, String message) {
494        PrintWriter writer = new PrintWriter(output);
495        writer.printf("# %s\n", new Date());
496        writer.printf("# %s\n ", message);
497    
498        writer.println();
499        for (ArgInfo info : positionalInfos) {
500          writer.printf("positional %s %s\n", info.className, info.fieldName);
501        }
502    
503        writer.println();
504        for (ArgInfo info : cmdLineInfos) {
505          writer.printf("field %s %s\n", info.className, info.fieldName);
506        }
507    
508        writer.println();
509        for (ParserInfo info : parserInfos) {
510          writer.printf("parser %s %s\n", info.parsedType, info.parserClass);
511        }
512    
513        writer.println();
514        for (VerifierInfo info : verifierInfos) {
515          writer.printf("verifier %s %s %s\n",
516              info.verifiedType, info.verifyingAnnotation, info.verifierClass);
517        }
518      }
