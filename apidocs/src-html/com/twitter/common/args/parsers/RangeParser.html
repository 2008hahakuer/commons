001    package com.twitter.common.args.parsers;
002    
003    import com.google.common.base.Splitter;
004    import com.google.common.collect.ImmutableList;
005    import com.google.common.collect.Range;
006    import com.google.common.collect.Ranges;
007    
008    import com.twitter.common.args.ArgParser;
009    
010    /**
011     * A parser that handles closed ranges. For the input "4-6", it will capture [4, 5, 6].
012     */
013    @ArgParser
014    public class RangeParser extends NonParameterizedTypeParser<Range<Integer>> {
015      @Override
016      public Range<Integer> doParse(String raw) throws IllegalArgumentException {
017        ImmutableList<String> numbers =
018            ImmutableList.copyOf(Splitter.on('-').omitEmptyStrings().split(raw));
019        try {
020          int from = Integer.parseInt(numbers.get(0));
021          int to = Integer.parseInt(numbers.get(1));
022          if (numbers.size() != 2) {
023            throw new IllegalArgumentException("Failed to parse the range:" + raw);
024          }
025          if (to < from) {
026            return Ranges.closed(to, from);
027          } else {
028            return Ranges.closed(from, to);
029          }
030        } catch (NumberFormatException e) {
031          throw new IllegalArgumentException("Failed to parse the range:" + raw, e);
032        }
033      }
034    }
