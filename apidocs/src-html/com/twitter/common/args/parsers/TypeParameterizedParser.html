018    package com.twitter.common.args.parsers;
019    
020    import java.lang.reflect.Type;
021    import java.util.List;
022    
023    import com.twitter.common.args.TypeUtil;
024    
025    import static com.google.common.base.Preconditions.checkArgument;
026    
027    /**
028     * Parser that makes implementation of parsers for parameterized types simpler.
029     *
030     * This only works for top-level parameters.  Behavior is undefined for nested parameterized types.
031     * For example, this will handle {@code Map<String, Integer>} fine but will not handle
032     * {@code Map<String, List<String>}.
033     *
034     * @author William Farner
035     */
036    public abstract class TypeParameterizedParser<T> extends BaseParser<T> {
037    
038      private final int typeParamCount;
039    
040      /**
041       * Creates a new type parameterized parser.
042       *
043       * @param parsedClass The parsed class.
044       * @param typeParamCount Strict number of type parameters to allow on the assigned type.
045       */
046      TypeParameterizedParser(Class<T> parsedClass, int typeParamCount) {
047        super(parsedClass);
048        this.typeParamCount = typeParamCount;
049      }
050    
051      /**
052       * Performs the actual parsing.
053       *
054       * @param raw The raw value to parse.
055       * @param paramParsers Parser classes for the parameter types, in the same order as the type
056       *    parameters.
057       * @return The parsed value.
058       * @throws IllegalArgumentException If the value could not be parsed.
059       */
060      abstract T doParse(String raw, List<Class<?>> paramParsers) throws IllegalArgumentException;
061    
062      @Override public T parse(Type type, String raw) {
063        List<Class<?>> typeParamClasses = TypeUtil.getTypeParamClasses(type);
064        checkArgument(typeParamClasses.size() == typeParamCount, String.format(
065            "Expected %d type parameters for %s but got %d",
066            typeParamCount, type, typeParamClasses.size()));
067    
068        return doParse(raw, typeParamClasses);
069      }
070    }
