017    package com.twitter.common.args.parsers;
018    
019    import java.lang.reflect.Type;
020    import java.util.Collections;
021    import java.util.List;
022    import java.util.StringTokenizer;
023    
024    import com.google.common.base.Preconditions;
025    
026    import com.twitter.common.args.ParserOracle;
027    import com.twitter.common.quantity.Amount;
028    import com.twitter.common.quantity.Time;
029    
030    /**
031     * Utility class for parsing durations of the form "1d23h59m59s" (as well as subvariants, i.e.
032     * "10h5s" would also work, as would "2d"). These values are useful representations in HTTP query
033     * parameters for durations.
034     *
035     */
036    public class DurationParser extends TypeParameterizedParser<Amount> {
037    
038      public DurationParser() {
039        super(2);
040      }
041    
042      private static final String SUFFIXES = "dhms";
043      private static final Time[] TIME_UNITS = { Time.DAYS, Time.HOURS, Time.MINUTES, Time.SECONDS };
044    
045      @Override
046      Amount doParse(ParserOracle parserOracle, String raw, List<Type> paramParsers)
047          throws IllegalArgumentException {
048        Type secondParamClass = paramParsers.get(1);
049        Preconditions.checkArgument(
050            secondParamClass == Time.class,
051            String.format("Expected %s for "
052                + "second type parameter, but got got %s", Time.class.getName(),
053                secondParamClass));
054        return parse(raw);
055      }
056    
057      /**
058       * Parses a duration of the form "1d23h59m59s" (as well as subvariants, i.e. "10h5s" would also
059       * work, as would "2d").
060       *
061       * @param spec the textual duration specification
062       * @return the parsed form
063       * @throws IllegalArgumentException if the specification can not be parsed
064       */
065      public static Amount<Long, Time> parse(String spec) {
066        long time = 0L;
067        final List<Object> tokens = Collections.list(new StringTokenizer(spec, SUFFIXES, true));
068        Preconditions.checkArgument(tokens.size() > 1);
069        for (int i = 1; i < tokens.size(); i += 2) {
070          final String token = (String) tokens.get(i);
071          Preconditions.checkArgument(token.length() == 1, "Too long suffix '%s'", token);
072          final int suffixIndex = SUFFIXES.indexOf(token.charAt(0));
073          Preconditions.checkArgument(suffixIndex != -1, "Unrecognized suffix '%s'", token);
074          try {
075            final int value = Integer.parseInt((String) tokens.get(i - 1));
076            time += Amount.of(value, TIME_UNITS[suffixIndex]).as(Time.SECONDS);
077          } catch (NumberFormatException e) {
078            Preconditions.checkArgument(false, "Invalid number %s", tokens.get(i - 1));
079          }
080        }
081        return Amount.of(time, Time.SECONDS);
082      }
083    }
