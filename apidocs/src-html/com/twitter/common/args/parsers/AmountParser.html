017    package com.twitter.common.args.parsers;
018    
019    import java.util.List;
020    import java.util.regex.Matcher;
021    import java.util.regex.Pattern;
022    
023    import com.twitter.common.args.Parsers.Parser;
024    import com.twitter.common.quantity.Amount;
025    import com.twitter.common.quantity.Unit;
026    
027    import static com.google.common.base.Preconditions.checkArgument;
028    import static com.twitter.common.args.Parsers.checkedGet;
029    
030    /**
031     * amount parser.
032     *
033     * @author William Farner
034     */
035    public class AmountParser extends TypeParameterizedParser<Amount> {
036    
037      public AmountParser() {
038        super(Amount.class, 2);
039      }
040    
041      @SuppressWarnings("unchecked")
042      @Override
043      Amount doParse(String raw, List<Class<?>> paramParsers) {
044        final Parser parser = checkedGet(paramParsers.get(0));
045    
046        Matcher matcher = Pattern.compile("(\\d+)([A-Za-z]+)").matcher(raw);
047        checkArgument(matcher.matches(),
048            "Value '" + raw + "' must be of the format 1ns, 4mb, etc.");
049    
050        Number number = (Number) parser.parse(null, matcher.group(1));
051        String unitRaw = matcher.group(2);
052    
053        Unit unit = (Unit) checkedGet(Unit.class).parse(null, unitRaw);
054        checkArgument(unit.getClass() == paramParsers.get(1), String.format(
055            "Unit type (%s) does not match argument type (%s).",
056            unit.getClass(), paramParsers.get(1)));
057    
058        Class numberClass = paramParsers.get(0);
059        if (numberClass == Integer.class) {
060          return Amount.of(number.intValue(), unit);
061        } else if (numberClass == Double.class) {
062          return Amount.of(number.doubleValue(), unit);
063        } else if (numberClass == Long.class) {
064          return Amount.of(number.longValue(), unit);
065        } else if (numberClass == Byte.class) {
066          return Amount.of(number.byteValue(), unit);
067        } else if (numberClass == Short.class) {
068          return Amount.of(number.shortValue(), unit);
069        } else if (numberClass == Float.class) {
070          return Amount.of(number.floatValue(), unit);
071        }
072    
073        throw new IllegalArgumentException("Unrecognized number class "
074            + numberClass.getCanonicalName());
075      }
076    }
