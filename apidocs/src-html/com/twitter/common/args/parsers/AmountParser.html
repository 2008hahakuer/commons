017    package com.twitter.common.args.parsers;
018    
019    import java.lang.reflect.Type;
020    import java.util.List;
021    import java.util.regex.Matcher;
022    import java.util.regex.Pattern;
023    
024    import com.google.common.reflect.TypeToken;
025    
026    import com.twitter.common.args.ArgParser;
027    import com.twitter.common.args.Parser;
028    import com.twitter.common.args.ParserOracle;
029    import com.twitter.common.quantity.Amount;
030    import com.twitter.common.quantity.Unit;
031    
032    import static com.google.common.base.Preconditions.checkArgument;
033    
034    /**
035     * Amount parser.
036     *
037     * @author William Farner
038     */
039    @ArgParser
040    public class AmountParser extends TypeParameterizedParser<Amount> {
041    
042      private static final Pattern AMOUNT_PATTERN = Pattern.compile("(\\d+)([A-Za-z]+)");
043    
044      public AmountParser() {
045        super(2);
046      }
047    
048      @Override
049      Amount doParse(ParserOracle parserOracle, String raw, List<Type> typeParams) {
050        Type valueType = typeParams.get(0);
051        Parser parser = parserOracle.get(TypeToken.of(valueType));
052    
053        Matcher matcher = AMOUNT_PATTERN.matcher(raw);
054        checkArgument(matcher.matches(),
055            "Value '" + raw + "' must be of the format 1ns, 4mb, etc.");
056    
057        Number number = (Number) parser.parse(parserOracle, valueType, matcher.group(1));
058        String unitRaw = matcher.group(2);
059    
060        Type unitType = typeParams.get(1);
061        Parser<Unit> unitParser = parserOracle.get(TypeToken.of(Unit.class));
062        Unit unit = unitParser.parse(parserOracle, unitType, unitRaw);
063        checkArgument(unit.getClass() == unitType, String.format(
064            "Unit type (%s) does not match argument type (%s).",
065            unit.getClass(), unitType));
066    
067        return create(valueType, number, unit);
068      }
069    
070      @SuppressWarnings("unchecked")
071      private static Amount create(Type valueType, Number number, Unit unit) {
072        if (valueType == Integer.class) {
073          return Amount.of(number.intValue(), unit);
074        } else if (valueType == Double.class) {
075          return Amount.of(number.doubleValue(), unit);
076        } else if (valueType == Long.class) {
077          return Amount.of(number.longValue(), unit);
078        } else if (valueType == Byte.class) {
079          return Amount.of(number.byteValue(), unit);
080        } else if (valueType == Short.class) {
081          return Amount.of(number.shortValue(), unit);
082        } else if (valueType == Float.class) {
083          return Amount.of(number.floatValue(), unit);
084        }
085        throw new IllegalArgumentException("Unrecognized number class " + valueType);
086      }
087    }
