017    package com.twitter.common.args.parsers;
018    
019    import java.lang.reflect.Type;
020    
021    import com.twitter.common.args.ArgParser;
022    import com.twitter.common.args.Parser;
023    import com.twitter.common.args.ParserOracle;
024    import com.twitter.common.quantity.Amount;
025    import com.twitter.common.quantity.Time;
026    import com.twitter.common.reflect.TypeToken;
027    import com.twitter.util.Duration;
028    
029    
030    /**
031     * A parser for Duration objects.
032     *
033     * Accepts in input the same format as the Amount&lt;Long, Time&gt; parser,
034     * &lt;number&gt;&lt;unit&gt; i.e. 10ns, 1days
035     *
036     * @author Ugo Di Girolamo
037     */
038    @ArgParser
039    public class TimeDurationParser implements Parser<Duration> {
040      private static final Type TIME_AMOUNT_TYPE = new TypeToken<Amount<Long, Time>>() { }.getType();
041    
042      @Override
043      public Duration parse(ParserOracle parserOracle, Type type, String raw)
044          throws IllegalArgumentException {
045        Parser amountParser = parserOracle.get(TypeToken.create(Amount.class));
046    
047        @SuppressWarnings("unchecked")
048        Amount<Long, Time> parsed = (Amount<Long, Time>)
049            amountParser.parse(parserOracle, TIME_AMOUNT_TYPE, raw);
050        return Duration.fromTimeUnit(parsed.getValue(), parsed.getUnit().getTimeUnit());
051      }
052    }
