017    package com.twitter.common.args;
018    
019    import java.lang.reflect.Field;
020    import java.lang.reflect.ParameterizedType;
021    import java.lang.reflect.Type;
022    import java.lang.reflect.WildcardType;
023    import java.util.Arrays;
024    import java.util.List;
025    
026    import com.google.common.base.Function;
027    import com.google.common.base.Preconditions;
028    import com.google.common.collect.Lists;
029    import com.google.common.reflect.TypeToken;
030    
031    /**
032     * Utility class to extract generic type information.
033     *
034     * TODO(William Farner): Move this into a common library, integrate with EasyMockTest.Clazz.
035     *
036     * @author William Farner
037     */
038    public final class TypeUtil {
039    
040      private static final Function<Type, Type> GET_TYPE = new Function<Type, Type>() {
041        @Override public Type apply(Type type) {
042          if (type instanceof WildcardType) {
043            return apply(((WildcardType) type).getUpperBounds()[0]);
044          }
045          return type;
046        }
047      };
048    
049      private TypeUtil() {
050        // Utility.
051      }
052    
053      /**
054       * Gets the types that a type is type-parameterized with, in declaration order.
055       *
056       * @param type The type to extract type parameters from.
057       * @return The types that {@code type} is parameterized with.
058       */
059      public static List<Type> getTypeParams(Type type) {
060        if (type instanceof WildcardType) {
061          return getTypeParams(GET_TYPE.apply(type));
062        }
063        return Lists.transform(Arrays.asList(
064            ((ParameterizedType) type).getActualTypeArguments()), GET_TYPE);
065      }
066    
067      /**
068       * Finds the raw class of type.
069       *
070       * @param type The type to get the raw class of.
071       * @return The raw class of type.
072       */
073      public static Class<?> getRawType(Type type) {
074        if (type instanceof ParameterizedType) {
075          return getRawType(((ParameterizedType) type).getRawType());
076        }
077        if (type instanceof WildcardType) {
078          return getRawType(((WildcardType) type).getUpperBounds()[0]);
079        }
080        return (Class<?>) type;
081      }
082    
083      /**
084       * Convenience method to call {@link #getTypeParam(Field)}, with the requirement that there
085       * is exactly one type parameter on the field.
086       *
087       * @param field The field to extract type parameters from.
088       * @return The raw classes of types that {@code field} is parameterized with.
089       */
090      public static TypeToken<?> getTypeParamTypeToken(Field field) {
091        List<Type> typeParams = getTypeParams(field.getGenericType());
092        Preconditions.checkArgument(typeParams.size() == 1,
093            "Expected exactly one type parameter for field " + field);
094        return TypeToken.of(typeParams.get(0));
095      }
096    
097      /**
098       * Gets the type parameter from a field.  Assumes that there is at least one type parameter.
099       *
100       * @param field The field to extract the type parameter from.
101       * @return The field type parameter.
102       */
103      public static Type getTypeParam(Field field) {
104        return extractTypeToken(field.getGenericType());
105      }
106    
107      /**
108       * Extracts the actual type parameter for a singly parameterized type.
109       *
110       * @param type The parameterized type to extract the type argument from.
111       * @return The type of the single specified type parameter for {@code type}.
112       * @throws IllegalArgumentException if the supplied type does not have exactly one specified type
113       *     parameter
114       */
115      public static Type extractTypeToken(Type type) {
116        Preconditions.checkNotNull(type);
117        Preconditions.checkArgument(type instanceof ParameterizedType, "Missing type parameter.");
118        Type[] typeArguments = ((ParameterizedType) type).getActualTypeArguments();
119        Preconditions.checkArgument(typeArguments.length == 1,
120            "Expected a type with exactly 1 type argument");
121        return typeArguments[0];
122      }
123    }
