017    package com.twitter.common.args;
018    
019    import java.lang.reflect.Field;
020    import java.lang.reflect.ParameterizedType;
021    import java.lang.reflect.Type;
022    import java.lang.reflect.WildcardType;
023    import java.util.Arrays;
024    import java.util.List;
025    
026    import com.google.common.base.Function;
027    import com.google.common.base.Preconditions;
028    import com.google.common.collect.Lists;
029    
030    import com.twitter.common.reflect.TypeToken;
031    
032    /**
033     * Utility class to extract generic type information.
034     *
035     * TODO(William Farner): Move this into a common library, integrate with EasyMockTest.Clazz.
036     *
037     * @author William Farner
038     */
039    public class TypeUtil {
040    
041      private static final Function<Type, Class<?>> GET_CLASS = new Function<Type, Class<?>>() {
042        @Override public Class<?> apply(Type type) {
043          return getRawType(type);
044        }
045      };
046    
047      private static final Function<Type, Type> GET_TYPE = new Function<Type, Type>() {
048        @Override public Type apply(Type type) {
049          if (type instanceof WildcardType) {
050            return apply(((WildcardType) type).getUpperBounds()[0]);
051          }
052          return type;
053        }
054      };
055    
056      private TypeUtil() {
057        // Utility.
058      }
059    
060      /**
061       * Gets the classes that a field is type-parameterized with, in declaration order.
062       *
063       * @param field The field to extract type parameters from.
064       * @return The raw classes of types that {@code field} is parameterized with.
065       */
066      public static List<Class<?>> getTypeParamClasses(Field field) {
067        return Lists.transform(getTypeParams(field.getGenericType()), GET_CLASS);
068      }
069    
070      /**
071       * Gets the types that a type is type-parameterized with, in declaration order.
072       *
073       * @param type The type to extract type parameters from.
074       * @return The types that {@code type} is parameterized with.
075       */
076      public static List<Type> getTypeParams(Type type) {
077        if (type instanceof WildcardType) {
078          return getTypeParams(GET_TYPE.apply(type));
079        }
080        return Lists.transform(Arrays.asList(
081            ((ParameterizedType) type).getActualTypeArguments()), GET_TYPE);
082      }
083    
084      /**
085       * Finds the raw class of type.
086       *
087       * @param type The type to get the raw class of.
088       * @return The raw class of type.
089       */
090      public static Class<?> getRawType(Type type) {
091        if (type instanceof ParameterizedType) {
092          return getRawType(((ParameterizedType) type).getRawType());
093        }
094        if (type instanceof WildcardType) {
095          return getRawType(((WildcardType) type).getUpperBounds()[0]);
096        }
097        return (Class<?>) type;
098      }
099    
100      /**
101       * Convenience method to call {@link #getTypeParamClasses(Field)}, with the requirement that there
102       * is exactly one type parameter on the field.
103       *
104       * @param field The field to extract type parameters from.
105       * @return The raw classes of types that {@code field} is parameterized with.
106       */
107      public static Class<?> getTypeParamClass(Field field) {
108        List<Class<?>> typeParams = getTypeParamClasses(field);
109        Preconditions.checkArgument(typeParams.size() == 1,
110            "Expected exactly one type parameter for field " + field);
111        return typeParams.get(0);
112      }
113    
114      /**
115       * Gets the type parameter from a field.  Assumes that there is at least one type parameter.
116       *
117       * @param field The field to extract the type parameter from.
118       * @return The field type parameter.
119       */
120      public static Type getTypeParam(Field field) {
121        return TypeToken.extractTypeToken(field.getGenericType());
122      }
123    
124      /**
125       * A function to extract the type parameter class from fields.
126       */
127      public static Function<Field, Class<?>> GET_TYPE_PARAM_CLASS = new Function<Field, Class<?>>() {
128        @Override public Class<?> apply(Field field) {
129          return getTypeParamClass(field);
130        }
131      };
132    }
