017    package com.twitter.common.args;
018    
019    import java.lang.reflect.Field;
020    import java.lang.reflect.ParameterizedType;
021    import java.lang.reflect.Type;
022    import java.lang.reflect.WildcardType;
023    import java.util.Arrays;
024    import java.util.List;
025    
026    import com.google.common.base.Function;
027    import com.google.common.base.Preconditions;
028    import com.google.common.collect.Lists;
029    
030    import com.twitter.common.reflect.TypeToken;
031    
032    /**
033     * Utility class to extract generic type information.
034     *
035     * TODO(William Farner): Move this into a common library, integrate with EasyMockTest.Clazz.
036     *
037     * @author William Farner
038     */
039    public class TypeUtil {
040    
041      private static final Function<Type, Type> GET_TYPE = new Function<Type, Type>() {
042        @Override public Type apply(Type type) {
043          if (type instanceof WildcardType) {
044            return apply(((WildcardType) type).getUpperBounds()[0]);
045          }
046          return type;
047        }
048      };
049    
050      private TypeUtil() {
051        // Utility.
052      }
053    
054      /**
055       * Gets the types that a type is type-parameterized with, in declaration order.
056       *
057       * @param type The type to extract type parameters from.
058       * @return The types that {@code type} is parameterized with.
059       */
060      public static List<Type> getTypeParams(Type type) {
061        if (type instanceof WildcardType) {
062          return getTypeParams(GET_TYPE.apply(type));
063        }
064        return Lists.transform(Arrays.asList(
065            ((ParameterizedType) type).getActualTypeArguments()), GET_TYPE);
066      }
067    
068      /**
069       * Finds the raw class of type.
070       *
071       * @param type The type to get the raw class of.
072       * @return The raw class of type.
073       */
074      public static Class<?> getRawType(Type type) {
075        if (type instanceof ParameterizedType) {
076          return getRawType(((ParameterizedType) type).getRawType());
077        }
078        if (type instanceof WildcardType) {
079          return getRawType(((WildcardType) type).getUpperBounds()[0]);
080        }
081        return (Class<?>) type;
082      }
083    
084      /**
085       * Convenience method to call {@link #getTypeParams(Field)}, with the requirement that there
086       * is exactly one type parameter on the field.
087       *
088       * @param field The field to extract type parameters from.
089       * @return The raw classes of types that {@code field} is parameterized with.
090       */
091      public static TypeToken<?> getTypeParamTypeToken(Field field) {
092        List<Type> typeParams = getTypeParams(field.getGenericType());
093        Preconditions.checkArgument(typeParams.size() == 1,
094            "Expected exactly one type parameter for field " + field);
095        return TypeToken.create(typeParams.get(0));
096      }
097    
098      /**
099       * Gets the type parameter from a field.  Assumes that there is at least one type parameter.
100       *
101       * @param field The field to extract the type parameter from.
102       * @return The field type parameter.
103       */
104      public static Type getTypeParam(Field field) {
105        return TypeToken.extractTypeToken(field.getGenericType());
106      }
107    }
