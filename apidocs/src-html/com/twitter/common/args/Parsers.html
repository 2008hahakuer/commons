017    package com.twitter.common.args;
018    
019    import java.util.Map;
020    
021    import com.google.common.base.Function;
022    import com.google.common.base.Preconditions;
023    import com.google.common.base.Splitter;
024    import com.google.common.collect.ImmutableMap;
025    import com.google.common.collect.Maps;
026    
027    import com.twitter.common.args.apt.Configuration;
028    import com.twitter.common.args.apt.Configuration.ParserInfo;
029    import com.twitter.common.reflect.TypeToken;
030    
031    import static com.google.common.base.Preconditions.checkArgument;
032    import static com.twitter.common.args.apt.Configuration.ConfigurationException;
033    
034    /**
035     * A registry of Parsers for different supported argument types.
036     *
037     * @author William Farner
038     */
039    public final class Parsers implements ParserOracle {
040    
041      private final ImmutableMap<Class<?>, Parser<?>> registry;
042    
043      /**
044       * Creates a new parser registry over the specified {@code parsers}.
045       *
046       * @param parsers The parsers to register.
047       */
048      public Parsers(Map<Class<?>, Parser<?>> parsers) {
049        Preconditions.checkNotNull(parsers);
050        registry = ImmutableMap.copyOf(parsers);
051      }
052    
053      @Override
054      public <T> Parser<T> get(TypeToken<T> type) throws IllegalArgumentException {
055        Parser parser;
056        Class<?> explicitClass = type.getRawType();
057        while (((parser = registry.get(explicitClass)) == null) && (explicitClass != null)) {
058          explicitClass = explicitClass.getSuperclass();
059        }
060        checkArgument(parser != null, "No parser found for " + type);
061    
062        // We control loading of the registry which ensures a proper mapping of class -> parser
063        @SuppressWarnings("unchecked")
064        Parser<T> parserT = parser;
065    
066        return parserT;
067      }
068    
069      private static final Function<ParserInfo, Class<?>> INFO_TO_PARSED_TYPE =
070          new Function<ParserInfo, Class<?>>() {
071            @Override public Class<?> apply(ParserInfo parserInfo) {
072              try {
073                return Class.forName(parserInfo.parsedType);
074              } catch (ClassNotFoundException e) {
075                throw new ConfigurationException(e);
076              }
077            }
078          };
079    
080      private static final Function<ParserInfo, Parser<?>> INFO_TO_PARSER =
081          new Function<ParserInfo, Parser<?>>() {
082            @Override public Parser apply(ParserInfo parserInfo) {
083              try {
084                return (Parser<?>) Class.forName(parserInfo.parserClass).newInstance();
085              } catch (ClassNotFoundException e) {
086                throw new ConfigurationException(e);
087              } catch (InstantiationException e) {
088                throw new ConfigurationException(e);
089              } catch (IllegalAccessException e) {
090                throw new ConfigurationException(e);
091              }
092            }
093          };
094    
095      static Parsers fromConfiguration(Configuration configuration) {
096        Map<Class<?>, Parser<?>> parsers =
097            Maps.transformValues(
098                Maps.uniqueIndex(configuration.parserInfo(), INFO_TO_PARSED_TYPE),
099                INFO_TO_PARSER);
100        return new Parsers(parsers);
101      }
102    
103      public static final Splitter MULTI_VALUE_SPLITTER =
104            Splitter.on(",").trimResults().omitEmptyStrings();
105    }
