017    package com.twitter.common.args.constraints;
018    
019    import java.lang.annotation.Annotation;
020    import java.math.BigDecimal;
021    
022    import com.google.common.base.Function;
023    import com.google.common.base.Functions;
024    
025    import com.twitter.common.args.Verifier;
026    import com.twitter.common.args.VerifierFor;
027    
028    import static com.google.common.base.Preconditions.checkArgument;
029    
030    /**
031     * Verifies that a number (inclusively) lies within a range.
032     *
033     * @author William Farner
034     */
035    @VerifierFor(Range.class)
036    public class RangeNumberVerifier implements Verifier<Number> {
037      @Override
038      public void verify(Number value, Annotation annotation) {
039        Range range = getRange(annotation);
040    
041        checkArgument(range.lower() < range.upper(),
042            "Range lower bound must be greater than upper bound.");
043    
044        double dblValue = value.doubleValue();
045        checkArgument(dblValue >= range.lower() && dblValue <= range.upper(),
046            String.format("Value must be in range [%f, %f]", range.lower(), range.upper()));
047      }
048    
049      @Override
050      public String toString(Class<? extends Number> argType, Annotation annotation) {
051        Range range = getRange(annotation);
052    
053        Function<Number, Number> converter;
054        if (Float.class.isAssignableFrom(argType)
055            || Double.class.isAssignableFrom(argType)
056            || BigDecimal.class.isAssignableFrom(argType)) {
057    
058          converter = Functions.identity();
059        } else {
060          converter = new Function<Number, Number>() {
061            @Override public Number apply(Number item) {
062              return item.longValue();
063            }
064          };
065        }
066    
067        return String.format("must be >= %s and <= %s",
068                             converter.apply(range.lower()),
069                             converter.apply(range.upper()));
070      }
071    
072      private Range getRange(Annotation annotation) {
073        checkArgument(annotation instanceof Range, "Annotation is not a range: " + annotation);
074    
075        return (Range) annotation;
076      }
077    }
