018    package com.twitter.common.args.constraints;
019    
020    import static com.google.common.base.Preconditions.checkArgument;
021    
022    /**
023     * Verifier to ensure that a string is not empty.
024     *
025     * @author William Farner
026     */
027    public class NotEmptyStringVerifier extends ScalarVerifier<String> {
028      @Override
029      public void verify(String s) {
030        checkArgument(!s.isEmpty(), "Value must not be empty.");
031      }
032    }
