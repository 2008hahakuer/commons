074      // Binary protocol is used here since it is currently (5/18/10) compatible all thrift languages.
075      private static final Codec<ServiceInstance> THRIFT_BINARY_PROTOCOL_CODEC =
076          new ThriftCodec<ServiceInstance>(ServiceInstance.class, ThriftCodec.BINARY_PROTOCOL);
077    
078      private final ZooKeeperClient zkClient;
079      private final Group group;
080      private final Codec<ServiceInstance> codec;

083      /**
084       * Creates a new ServerSet using open ZooKeeper node ACLs.
085       *
086       * @param zkClient the client to use for interactions with ZooKeeper
087       * @param path the name-service path of the service to connect to
088       */
089      public ServerSetImpl(ZooKeeperClient zkClient, String path) {
090        this(zkClient, ZooDefs.Ids.OPEN_ACL_UNSAFE, path);
091      }
092    
093      /**
094       * Creates a new ServerSet for the given service {@code path}.
095       *
096       * @param zkClient the client to use for interactions with ZooKeeper
097       * @param acl the ACL to use for creating the persistent group path if it does not already exist
098       * @param path the name-service path of the service to connect to
099       */
100      public ServerSetImpl(ZooKeeperClient zkClient, List<ACL> acl, String path) {
101        this(zkClient, new Group(zkClient, acl, path), THRIFT_BINARY_PROTOCOL_CODEC);
102      }

114      /**
115       * Creates a new ServerSet using the given service {@code group} and a custom {@code codec}.
116       *
117       * @param zkClient the client to use for interactions with ZooKeeper
118       * @param group the server group
119       * @param codec a codec to use for serializing and de-serializing the ServiceInstance data to and
120       *     from a byte array
121       */
122      public ServerSetImpl(ZooKeeperClient zkClient, Group group, Codec<ServiceInstance> codec) {
123        this.zkClient = Preconditions.checkNotNull(zkClient);
124        this.group = Preconditions.checkNotNull(group);
125        this.codec = codec;
126    
127        // TODO(John Sirois): Inject the helper so that backoff strategy can be configurable.
128        backoffHelper = new BackoffHelper();
129      }
130    
131      @VisibleForTesting
132      ZooKeeperClient getZkClient() {
133        return zkClient;
134      }

136      @Override
137      public EndpointStatus join(InetSocketAddress endpoint,
138          Map<String, InetSocketAddress> additionalEndpoints, Status status)
139          throws JoinException, InterruptedException {
140        Preconditions.checkNotNull(endpoint);
141        Preconditions.checkNotNull(additionalEndpoints);
142        Preconditions.checkNotNull(status);
143    
144        final MemberStatus memberStatus = new MemberStatus(endpoint, additionalEndpoints, status);
145        Supplier<byte[]> serviceInstanceSupplier = new Supplier<byte[]>() {
146          @Override public byte[] get() {
147            return memberStatus.serializeServiceInstance();
148          }
149        };
150        final Membership membership = group.join(serviceInstanceSupplier);
151    
152        return new EndpointStatus() {
153          @Override public void update(Status status) throws UpdateException {
154            Preconditions.checkNotNull(status);
155            memberStatus.updateStatus(membership, status);
156          }
157        };
158      }
159    
160      @Override
161      public void monitor(final HostChangeMonitor<ServiceInstance> monitor) throws MonitorException {
162        ServerSetWatcher serverSetWatcher = new ServerSetWatcher(zkClient, monitor);
163        try {
164          serverSetWatcher.watch();
165        } catch (WatchException e) {
166          throw new MonitorException("ZooKeeper watch failed.", e);
167        } catch (InterruptedException e) {
168          throw new MonitorException("Interrupted while watching ZooKeeper.", e);
169        }
170      }
171    
172      private class MemberStatus {
173        private final InetSocketAddress endpoint;
174        private final Map<String, InetSocketAddress> additionalEndpoints;
175        private volatile Status status;
176    
177        private MemberStatus(InetSocketAddress endpoint,
178            Map<String, InetSocketAddress> additionalEndpoints, Status status) {
179    
180          this.endpoint = endpoint;
181          this.additionalEndpoints = additionalEndpoints;
182          this.status = status;
183        }
184    
185        synchronized void updateStatus(Membership membership, Status status) throws UpdateException {
186          if (this.status != status) {
187            this.status = status;
188            if (Status.DEAD == status) {
189              try {
190                membership.cancel();
191              } catch (JoinException e) {
192                throw new UpdateException(
193                    "Failed to auto-cancel group membership on transition to DEAD status", e);
194              }
195            } else {
196              try {
197                membership.updateMemberData();
198              } catch (Group.UpdateException e) {
199                throw new UpdateException(
200                    "Failed to update service data for: " + membership.getMemberPath(), e);
201              }
202            }
203          }
204        }
205    
206        byte[] serializeServiceInstance() {
207          ServiceInstance serviceInstance =
208              new ServiceInstance(toEndpoint(endpoint),
209                  Maps.transformValues(additionalEndpoints, TO_ENDPOINT), status);
210          LOG.info("updating endpoint data to:\n\t" + serviceInstance);
211          ByteArrayOutputStream output = new ByteArrayOutputStream();
212          try {
213            codec.serialize(serviceInstance, output);
214          } catch (IOException e) {
215            throw new IllegalStateException("Unexpected problem serializing thrift struct: " +
216                                            serviceInstance + " to a byte[]", e);
217          }
218          return output.toByteArray();
219        }
220      }
221    
222      private static final Function<InetSocketAddress, Endpoint> TO_ENDPOINT =
223          new Function<InetSocketAddress, Endpoint>() {
224            @Override public Endpoint apply(InetSocketAddress address) {
225              return toEndpoint(address);
226            }
227          };
228    
229      private static Endpoint toEndpoint(InetSocketAddress address) {
230        return new Endpoint(address.getHostName(), address.getPort());
231      }
232    
233      private static class ServiceInstanceFetchException extends RuntimeException {
234        ServiceInstanceFetchException(String message, Throwable cause) {
235          super(message, cause);
236        }
237      }
238    
239      private static class ServiceInstanceDeletedException extends RuntimeException {
240        ServiceInstanceDeletedException(String path) {
241          super(path);
242        }
243      }
244    
245      private class ServerSetWatcher {
246        private final ZooKeeperClient zkClient;
247        private final HostChangeMonitor<ServiceInstance> monitor;
248        private ImmutableSet<ServiceInstance> serverSet;
249    
250        ServerSetWatcher(ZooKeeperClient zkClient, HostChangeMonitor<ServiceInstance> monitor) {
251          this.zkClient = zkClient;
252          this.monitor = monitor;
253        }
254    
255        public void watch() throws WatchException, InterruptedException {
256          zkClient.registerExpirationHandler(new Command() {
257            @Override public void execute() throws RuntimeException {
258              // Servers may have changed Status while we were disconnected from ZooKeeper, check and
259              // re-register our node watches.
260              rebuildServerSet();
261            }
262          });
263    
264          group.watch(new GroupChangeListener() {
265            @Override public void onGroupChange(Iterable<String> memberIds) {
266              notifyGroupChange(memberIds);
267            }
268          });
269        }
270    
271        private Watcher serviceInstanceWatcher = new Watcher() {
272          @Override public void process(WatchedEvent event) {
273            if (event.getState() == KeeperState.SyncConnected) {
274              switch (event.getType()) {
275                case None:
276                  // Ignore re-connects that happen while we're watching
277                  break;
278                case NodeDeleted:
279                  // Ignore deletes since these trigger a group change through the group node watch.
280                  break;
281                case NodeDataChanged:
282                  notifyNodeChange(event.getPath());
283                  break;
284                default:
285                  LOG.severe("Unexpected event watching service node: " + event);
286              }
287            }
288          }
289        };
290    
291        private ServiceInstance getServiceInstance(final String nodePath) {
292          try {
293            return backoffHelper.doUntilResult(new Supplier<ServiceInstance>() {
294              @Override public ServiceInstance get() {
295                try {
296                  byte[] data = zkClient.get().getData(nodePath, serviceInstanceWatcher, null);
297                  return codec.deserialize(new ByteArrayInputStream(data));
298                } catch (InterruptedException e) {
299                  Thread.currentThread().interrupt();
300                  throw new ServiceInstanceFetchException(
301                      "Interrupted updating service data for: " + nodePath, e);
302                } catch (ZooKeeperConnectionException e) {
303                  LOG.log(Level.WARNING,
304                      "Temporary error trying to updating service data for: " + nodePath, e);
305                  return null;
306                } catch (NoNodeException e) {
307                  invalidateNodePath(nodePath);
308                  throw new ServiceInstanceDeletedException(nodePath);
309                } catch (KeeperException e) {
310                  if (zkClient.shouldRetry(e)) {
311                    LOG.log(Level.WARNING,
312                        "Temporary error trying to update service data for: " + nodePath, e);
313                    return null;
314                  } else {
315                    throw new ServiceInstanceFetchException(
316                        "Failed to update service data for: " + nodePath, e);
317                  }
318                } catch (IOException e) {
319                  throw new ServiceInstanceFetchException(
320                      "Failed to deserialize the ServiceInstance data for: " + nodePath, e);
321                }
322              }
323            });
324          } catch (InterruptedException e) {
325            Thread.currentThread().interrupt();
326            throw new ServiceInstanceFetchException(
327                "Interrupted trying to update service data for: " + nodePath, e);
328          }
329        }
330    
331        private final Map<String, ServiceInstance> servicesByMemberId = new MapMaker()
332            .makeComputingMap(new Function<String, ServiceInstance>() {
333              @Override public ServiceInstance apply(String memberId) {
334                return getServiceInstance(group.getMemberPath(memberId));
335              }
336            });
337    
338        private void rebuildServerSet() {
339          Set<String> memberIds = ImmutableSet.copyOf(servicesByMemberId.keySet());
340          servicesByMemberId.clear();
341          notifyGroupChange(memberIds);
342        }
343    
344        private void notifyNodeChange(String changedPath) {
345          // Invalidate the associated ServiceInstance to trigger a fetch on group notify.
346          String memberId = invalidateNodePath(changedPath);
347          notifyGroupChange(Iterables.concat(servicesByMemberId.keySet(), ImmutableList.of(memberId)));
348        }
349    
350        private String invalidateNodePath(String deletedPath) {
351          String memberId = group.getMemberId(deletedPath);
352          servicesByMemberId.remove(memberId);
353          return memberId;
354        }
355    
356        private synchronized void notifyGroupChange(Iterable<String> memberIds) {
357          Set<String> currentMemberIds = ImmutableSet.copyOf(memberIds);
358          Set<String> oldMemberIds = servicesByMemberId.keySet();
359    
360          // Ignore no-op state changes except for the 1st when we've seen no group yet.
361          if ((serverSet == null) || !currentMemberIds.equals(oldMemberIds)) {
362            SetView<String> deletedMemberIds = Sets.difference(oldMemberIds, currentMemberIds);
363            oldMemberIds.removeAll(deletedMemberIds);
364    
365            SetView<String> unchangedMemberIds = Sets.intersection(oldMemberIds, currentMemberIds);
366            ImmutableSet.Builder<ServiceInstance> services = ImmutableSet.builder();
367            for (String unchangedMemberId : unchangedMemberIds) {
368              services.add(servicesByMemberId.get(unchangedMemberId));
369            }
370    
371            // TODO(John Sirois): consider parallelizing fetches
372            SetView<String> newMemberIds = Sets.difference(currentMemberIds, oldMemberIds);
373            for (String newMemberId : newMemberIds) {
374              // This get will trigger a fetch
375              try {
376                services.add(servicesByMemberId.get(newMemberId));
377              } catch (ComputationException e) {
378                Throwable cause = e.getCause();
379                if (!(cause instanceof ServiceInstanceDeletedException)) {
380                  Throwables.propagateIfInstanceOf(cause, ServiceInstanceFetchException.class);
381                  throw new IllegalStateException(
382                      "Unexpected error fetching member data for: " + newMemberId, e);
383                }
384              }
385            }
386    
387            notifyServerSetChange(services.build());
388          }
389        }
390    
391        private void notifyServerSetChange(ImmutableSet<ServiceInstance> currentServerSet) {
392          // ZK nodes may have changed if there was a session expiry for a server in the server set, but
393          // if the server's status has not changed, we can skip any onChange updates.
394          if (!currentServerSet.equals(serverSet)) {
395            serverSet = currentServerSet;
396            if (serverSet.isEmpty()) {
397              LOG.warning("server set empty!");
398            } else {
399              LOG.info("server set change to:\n\t" + Joiner.on("\n\t").join(serverSet));
400            }
401            monitor.onChange(serverSet);
402          }
403        }
404      }
