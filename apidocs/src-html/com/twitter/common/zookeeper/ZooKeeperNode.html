017    package com.twitter.common.zookeeper;
018    
019    import com.google.common.annotations.VisibleForTesting;
020    import com.google.common.base.Function;
021    import com.google.common.base.Functions;
022    import com.google.common.base.Preconditions;
023    import com.google.common.base.Supplier;
024    import com.twitter.common.base.*;
025    import com.twitter.common.util.BackoffHelper;
026    import com.twitter.common.zookeeper.ZooKeeperClient.ZooKeeperConnectionException;
027    import org.apache.zookeeper.KeeperException;
028    import org.apache.zookeeper.WatchedEvent;
029    import org.apache.zookeeper.Watcher;
030    
031    import java.util.logging.Level;
032    import java.util.logging.Logger;
033    import javax.annotation.Nullable;
034    
035    /**
036     * An implementation of {@link Supplier} that offers a readonly view of a
037     * zookeeper data node.  This class is thread-safe.
038     *
039     * Instances of this class each maintain a zookeeper watch for the remote data node.  Instances
040     * of this class should be created via the {@link #create} factory method.
041     *
042     * Please see zookeeper documentation and talk to your cluster administrator for guidance on
043     * appropriate node size and total number of nodes you should be using.
044     *
045     * @param <T> the type of data this node stores
046     *
047     * @author Adam Samet
048     */
049    public class ZooKeeperNode<T> implements Supplier<T> {
050      /**
051       * Deserializer for the constructor if you want to simply store the zookeeper byte[] data
052       * as-is.
053       */
054      public static final Function<byte[], byte[]> BYTE_ARRAY_VALUE = Functions.identity();
055    
056      private static final Logger LOG = Logger.getLogger(ZooKeeperNode.class.getName());
057    
058      private final ZooKeeperClient zkClient;
059      private final String nodePath;
060      private final Function<byte[], T> deserializer;
061    
062      private final BackoffHelper backoffHelper;
063    
064      // Whether it's safe to re-establish watches if our zookeeper session has expired.
065      private final Object safeToRewatchLock;
066      private volatile boolean safeToRewatch;
067    
068      private final T NO_DATA = null;
069      @Nullable private volatile T nodeData;
070      private final Closure<T> dataUpdateListener;
071    
072      /**
073       * Returns an initialized ZooKeeperNode.  The given node must exist at the time of object
074       * creation or a {@link KeeperException} will be thrown.
075       *
076       * @param zkClient a zookeeper client
077       * @param nodePath path to a node whose data will be watched
078       * @param deserializer a function that converts byte[] data from a zk node to this supplier's
079       *     type T
080       *
081       * @throws InterruptedException if the underlying zookeeper server transaction is interrupted
082       * @throws KeeperException.NoNodeException if the given nodePath doesn't exist
083       * @throws KeeperException if the server signals an error
084       * @throws ZooKeeperConnectionException if there was a problem connecting to the zookeeper
085       *     cluster
086       */
087      public static <T> ZooKeeperNode<T> create(ZooKeeperClient zkClient, String nodePath,
088          Function<byte[], T> deserializer) throws InterruptedException, KeeperException,
089          ZooKeeperConnectionException {
090        return create(zkClient, nodePath, deserializer, Closures.<T>noop());
091      }
092    
093      /**
094       * Like the above, but optionally takes in a {@link Closure} that will get notified
095       * whenever the data is updated from the remote node.
096       *
097       * @param dataUpdateListener a {@link Closure} to receive data update notifications.
098       */
099      public static <T> ZooKeeperNode<T> create(ZooKeeperClient zkClient, String nodePath,
100          Function<byte[], T> deserializer, Closure<T> dataUpdateListener) throws InterruptedException,
101          KeeperException, ZooKeeperConnectionException {
102        ZooKeeperNode<T> zkNode =
103            new ZooKeeperNode<T>(zkClient, nodePath, deserializer, dataUpdateListener);
104        zkNode.init();
105        return zkNode;
106      }
107    
108      /**
109       * Initializes a ZooKeeperNode.  The given node must exist at the time of object creation or
110       * a {@link KeeperException} will be thrown.
111       *
112       * Please note that this object will not track any remote zookeeper data until {@link #init()}
113       * is successfully called.  After construction and before that call, this {@link Supplier} will
114       * return null.
115       *
116       * @param zkClient a zookeeper client
117       * @param nodePath path to a node whose data will be watched
118       * @param deserializer a function that converts byte[] data from a zk node to this supplier's
119       *     type T
120       * @param dataUpdateListener a {@link Closure} to receive data update notifications.
121       */
122      @VisibleForTesting
123      ZooKeeperNode(ZooKeeperClient zkClient, String nodePath,
124          Function<byte[], T> deserializer, Closure<T> dataUpdateListener) {
125        this.zkClient = Preconditions.checkNotNull(zkClient);
126        this.nodePath = MorePreconditions.checkNotBlank(nodePath);
127        this.deserializer = Preconditions.checkNotNull(deserializer);
128        this.dataUpdateListener = Preconditions.checkNotNull(dataUpdateListener);
129    
130        backoffHelper = new BackoffHelper();
131        safeToRewatchLock = new Object();
132        safeToRewatch = false;
133        nodeData = NO_DATA;
134      }
135    
136      /**
137       * Initialize zookeeper tracking for this {@link Supplier}.  Once this call returns, this object
138       * will be tracking data in zookeeper.
139       *
140       * @throws InterruptedException if the underlying zookeeper server transaction is interrupted
141       * @throws KeeperException if the server signals an error
142       * @throws ZooKeeperConnectionException if there was a problem connecting to the zookeeper
143       *     cluster
144       */
145      @VisibleForTesting
146      void init() throws InterruptedException, KeeperException,
147          ZooKeeperConnectionException {
148        Watcher watcher = zkClient.registerExpirationHandler(new Command() {
149          @Override public void execute() {
150            try {
151              synchronized (safeToRewatchLock) {
152                if (safeToRewatch) {
153                  tryWatchDataNode();
154                }
155              }
156            } catch (InterruptedException e) {
157              LOG.log(Level.WARNING, "Interrupted while trying to re-establish watch.", e);
158            }
159          }
160        });
161    
162        try {
163          /*
164           * Synchronize to prevent the race of watchDataNode completing and then the session expiring
165           * before we update safeToRewatch.
166           */
167          synchronized (safeToRewatchLock) {
168            watchDataNode();
169            safeToRewatch = true;
170          }
171        } catch (InterruptedException e) {
172          zkClient.unregister(watcher);
173          throw e;
174        } catch (KeeperException e) {
175          zkClient.unregister(watcher);
176          throw e;
177        } catch (ZooKeeperConnectionException e) {
178          zkClient.unregister(watcher);
179          throw e;
180        }
181      }
182    
183      /**
184       * Returns the data corresponding to a byte array in a remote zookeeper node.  This data is
185       * cached locally and updated in the background on watch notifications.
186       *
187       * @return the data currently cached locally or null if {@link #init()} hasn't been called
188       *   or the backing node has no data or does not exist anymore.
189       */
190      @Override
191      public @Nullable T get() {
192        return nodeData;
193      }
194    
195      @VisibleForTesting
196      void updateData(@Nullable T newData) {
197        nodeData = newData;
198        dataUpdateListener.execute(newData);
199      }
200    
201      private void tryWatchDataNode() throws InterruptedException {
202        backoffHelper.doUntilSuccess(new ExceptionalSupplier<Boolean, InterruptedException>() {
203          @Override public Boolean get() throws InterruptedException {
204            try {
205              watchDataNode();
206              return true;
207            } catch (KeeperException e) {
208              return false;
209            } catch (ZooKeeperConnectionException e) {
210              return false;
211            }
212          }
213        });
214      }
215    
216      private void watchDataNode() throws InterruptedException, KeeperException,
217          ZooKeeperConnectionException {
218        final Watcher nodeWatcher = new Watcher() {
219          @Override
220          public void process(WatchedEvent event) {
221            if (event.getType() == Watcher.Event.EventType.NodeDataChanged ||
222                event.getType() == Watcher.Event.EventType.NodeDeleted) {
223              try {
224                tryWatchDataNode();
225              } catch (InterruptedException e) {
226                LOG.log(Level.WARNING, "Interrupted while trying to watch a data node.", e);
227              }
228            }
229          }
230        };
231    
232        try {
233          T newData = deserializer.apply(zkClient.get().getData(nodePath, nodeWatcher, null));
234          updateData(newData);
235        } catch (KeeperException.NoNodeException e) {
236          /*
237           * This node doesn't exist right now, reflect that locally and then create a watch to wait
238           * for its recreation.
239           */
240          updateData(NO_DATA);
241          watchForExistence();
242        }
243      }
244    
245      private void watchForExistence() throws InterruptedException, KeeperException,
246          ZooKeeperConnectionException {
247        final Watcher watcher = new Watcher() {
248          @Override
249          public void process(WatchedEvent event) {
250            if (event.getType() == Watcher.Event.EventType.NodeCreated) {
251              try {
252                tryWatchDataNode();
253              } catch (InterruptedException e) {
254                LOG.log(Level.WARNING, "Interrupted while trying to watch a data node.", e);
255              }
256            }
257          }
258        };
259    
260        /*
261         * If the node was created between the getData call and this call, just try watching it.
262         * We'll have an extra exists watch on it that goes off on its next deletion, which will
263         * be a no-op.
264         * Otherwise, just let the exists watch wait for its creation.
265         */
266        if (zkClient.get().exists(nodePath, watcher) != null) {
267          tryWatchDataNode();
268        }
269      }
270    }
