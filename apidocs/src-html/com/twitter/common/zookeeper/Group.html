050    /**
051     * This class exposes methods for joining and monitoring distributed groups.  The groups this class
052     * monitors are realized as persistent paths in ZooKeeper with ephemeral sequential child nodes for
053     * each member of a group.
054     *
055     * @author John Sirois
056     */
057    public class Group {
058      private static final Logger LOG = Logger.getLogger(Partitioner.class.getName());
059    
060      private static final Supplier<byte[]> NO_MEMBER_DATA = Suppliers.ofInstance(null);
061      private static final String DEFAULT_NODE_NAME_PREFIX = "member_";
062    
063      private Predicate<String> nodeNameFilter;
064    
065      private final ZooKeeperClient zkClient;
066      private final List<ACL> acl;
067      private final String path;
068      private final String nodeNamePrefix;
069    
070      private final BackoffHelper backoffHelper;
071    
072      /**
073       * @param zkClient the client to use for interactions with ZooKeeper
074       * @param acl the ACL to use for creating the persistent group path if it does not already exist
075       * @param path the persistent path that represents this group
076       * @param nodeNamePrefix Node name prefix that denotes group membership.
077       */
078      public Group(ZooKeeperClient zkClient, List<ACL> acl, String path, String nodeNamePrefix) {
079        this.zkClient = Preconditions.checkNotNull(zkClient);
080        this.acl = Preconditions.checkNotNull(acl);
081        this.path = MorePreconditions.checkNotBlank(path);
082        this.nodeNamePrefix = MorePreconditions.checkNotBlank(nodeNamePrefix);
083    
084        final Pattern groupNodeNamePattern = Pattern.compile(
085            "^" + Pattern.quote(nodeNamePrefix) + "[0-9]+$");
086        nodeNameFilter = new Predicate<String>() {
087            @Override public boolean apply(String childNodeName) {
088              return groupNodeNamePattern.matcher(childNodeName).matches();
089            }
090          };
091    
092        backoffHelper = new BackoffHelper();
093      }
094    
095      /**
096       * @param zkClient the client to use for interactions with ZooKeeper
097       * @param acl the ACL to use for creating the persistent group path if it does not already exist
098       * @param path the persistent path that represents this group
099       */
100      public Group(ZooKeeperClient zkClient, List<ACL> acl, String path) {
101        this(zkClient, acl, path, DEFAULT_NODE_NAME_PREFIX);
102      }
103    
104      public String getMemberPath(String memberId) {
105        return path + "/" + MorePreconditions.checkNotBlank(memberId);
106      }
107    
108      public String getMemberId(String nodePath) {
109        MorePreconditions.checkNotBlank(nodePath);
110        Preconditions.checkArgument(nodePath.startsWith(path + "/"),
111            "Not a member of this group[%s]: %s", path, nodePath);
112        return extractMemberId(nodePath);
113      }
114    
115      private String extractMemberId(String nodePath) {
116        String memberId = StringUtils.substringAfterLast(nodePath, "/");
117        Preconditions.checkArgument(nodeNameFilter.apply(memberId), "Not a group member: %s", memberId);
118        return memberId;
119      }
120    
121      /**
122       * Returns the current list of group member ids by querying ZooKeeper synchronously.
123       *
124       * @return the ids of all the present members of this group
125       * @throws ZooKeeperConnectionException if there was a problem connecting to ZooKeeper
126       * @throws KeeperException if there was a problem reading this group's member ids
127       * @throws InterruptedException if this thread is interrupted listing the group members
128       */
129      public Iterable<String> getMemberIds()
130          throws ZooKeeperConnectionException, KeeperException, InterruptedException {
131        return Iterables.filter(zkClient.get().getChildren(path, false), nodeNameFilter);
132      }
133    
134      /**
135       * Gets the data for one of this groups members by querying ZooKeeper synchronously.
136       *
137       * @param memberId the id of the member whose data to retrieve
138       * @return the data associated with the {@code memberId}
139       * @throws ZooKeeperConnectionException if there was a problem connecting to ZooKeeper
140       * @throws KeeperException if there was a problem reading this member's data
141       * @throws InterruptedException if this thread is interrupted retrieving the member data
142       */
143      public byte[] getMemberData(String memberId)
144          throws ZooKeeperConnectionException, KeeperException, InterruptedException {
145        return zkClient.get().getData(getMemberPath(memberId), false, null);
146      }

148      /**
149       * Represents membership in a distributed group.
150       */
151      public interface Membership {
152    
153        /**
154         * Returns the persistent ZooKeeper path that represents this group.
155         */
156        String getGroupPath();
157    
158        /**
159         * Returns the id (ZooKeeper node name) of this group member.  May change over time if the
160         * ZooKeeper session expires.
161         */
162        String getMemberId();
163    
164        /**
165         * Returns the full ZooKeeper path to this group member.  May change over time if the
166         * ZooKeeper session expires.
167         */
168        String getMemberPath();
169    
170        /**
171         * Updates the membership data synchronously using the {@code Supplier<byte[]>} passed to
172         * {@link Group#join()}.
173         *
174         * @return the new membership data
175         * @throws UpdateException if there was a problem updating the membership data
176         */
177        byte[] updateMemberData() throws UpdateException;
178    
179        /**
180         * Cancels group membership by deleting the associated ZooKeeper member node.
181         *
182         * @throws JoinException if there is a problem deleting the node
183         */
184        void cancel() throws JoinException;
185      }

230      /**
231       * Joins this group and returns the resulting Membership when successful.  Membership will be
232       * automatically cancelled when the current jvm process dies; however the returned Membership
233       * object can be used to cancel membership earlier.  Unless
234       * {@link com.twitter.common.zookeeper.Group.Membership#cancel()} is called the membership will
235       * be maintained by re-establishing it silently in the background.
236       *
237       * <p>Any {@code memberData} given is persisted in the member node in ZooKeeper.  If an
238       * {@code onLoseMembership} callback is supplied, it will be notified each time this member loses
239       * membership in the group.
240       *
241       * @param memberData a supplier of the data to store in the member node
242       * @param onLoseMembership a callback to notify when membership is lost
243       * @return a Membership object with the member details
244       * @throws JoinException if there was a problem joining the group
245       * @throws InterruptedException if this thread is interrupted awaiting completion of the join
246       */
247      public final Membership join(Supplier<byte[]> memberData, @Nullable Command onLoseMembership)
248          throws JoinException, InterruptedException {
249    
250        Preconditions.checkNotNull(memberData);
251        ensurePersistentGroupPath();
252    
253        final ActiveMembership groupJoiner = new ActiveMembership(memberData, onLoseMembership);
254        return backoffHelper.doUntilResult(new ExceptionalSupplier<Membership, JoinException>() {
255          @Override public Membership get() throws JoinException {
256            try {
257              return groupJoiner.join();
258            } catch (InterruptedException e) {
259              Thread.currentThread().interrupt();
260              throw new JoinException("Interrupted trying to join group at path: " + path, e);
261            } catch (ZooKeeperConnectionException e) {
262              LOG.log(Level.WARNING, "Temporary error trying to join group at path: " + path, e);
263              return null;
264            } catch (KeeperException e) {
265              if (zkClient.shouldRetry(e)) {
266                LOG.log(Level.WARNING, "Temporary error trying to join group at path: " + path, e);
267                return null;
268              } else {
269                throw new JoinException("Problem joining partition group at path: " + path, e);
270              }
271            }
272          }
273        });
274      }
275    
276      private void ensurePersistentGroupPath() throws JoinException, InterruptedException {
277        backoffHelper.doUntilSuccess(new ExceptionalSupplier<Boolean, JoinException>() {
278          @Override public Boolean get() throws JoinException {
279            try {
280              ZooKeeperUtils.ensurePath(zkClient, acl, path);
281              return true;
282            } catch (InterruptedException e) {
283              Thread.currentThread().interrupt();
284              throw new JoinException("Interrupted trying to ensure group at path: " + path, e);
285            } catch (ZooKeeperConnectionException e) {
286              LOG.log(Level.WARNING, "Problem connecting to ZooKeeper, retrying", e);
287              return false;
288            } catch (KeeperException e) {
289              if (zkClient.shouldRetry(e)) {
290                LOG.log(Level.WARNING, "Temporary error ensuring path: " + path, e);
291                return false;
292              } else {
293                throw new JoinException("Problem ensuring group at path: " + path, e);
294              }
295            }
296          }
297        });
298      }

300      private class ActiveMembership implements Membership {
301        private final Supplier<byte[]> memberData;
302        private final Command onLoseMembership;
303        private String nodePath;
304        private String memberId;
305        private volatile boolean cancelled;
306        private byte[] membershipData;
307    
308        public ActiveMembership(Supplier<byte[]> memberData, @Nullable Command onLoseMembership) {
309          this.memberData = memberData;
310          this.onLoseMembership = (onLoseMembership == null) ? Commands.NOOP : onLoseMembership;
311        }
312    
313        @Override
314        public String getGroupPath() {
315          return path;
316        }
317    
318        @Override
319        public synchronized String getMemberId() {
320          return memberId;
321        }
322    
323        @Override
324        public synchronized String getMemberPath() {
325          return nodePath;
326        }
327    
328        @Override
329        public synchronized byte[] updateMemberData() throws UpdateException {
330          byte[] membershipData = memberData.get();
331          if (!ArrayUtils.isEquals(this.membershipData, membershipData)) {
332            try {
333              zkClient.get().setData(nodePath, membershipData, ZooKeeperUtils.ANY_VERSION);
334              this.membershipData = membershipData;
335            } catch (KeeperException e) {
336              throw new UpdateException("Problem updating membership data.", e);
337            } catch (InterruptedException e) {
338              throw new UpdateException("Interrupted attempting to update membership data.", e);
339            } catch (ZooKeeperConnectionException e) {
340              throw new UpdateException(
341                  "Could not connect to the ZooKeeper cluster to update membership data.", e);
342            }
343          }
344          return membershipData;
345        }
346    
347        @Override
348        public synchronized void cancel() throws JoinException {
349          if (!cancelled) {
350            try {
351              backoffHelper.doUntilSuccess(new ExceptionalSupplier<Boolean, JoinException>() {
352                @Override public Boolean get() throws JoinException {
353                  try {
354                    zkClient.get().delete(nodePath, ZooKeeperUtils.ANY_VERSION);
355                    return true;
356                  } catch (InterruptedException e) {
357                    Thread.currentThread().interrupt();
358                    throw new JoinException("Interrupted trying to cancel membership: " + nodePath, e);
359                  } catch (ZooKeeperConnectionException e) {
360                    LOG.log(Level.WARNING, "Problem connecting to ZooKeeper, retrying", e);
361                    return false;
362                  } catch (NoNodeException e) {
363                    LOG.info("Membership already cancelled, node at path: " + nodePath +
364                             " has been deleted");
365                    return true;
366                  } catch (KeeperException e) {
367                    if (zkClient.shouldRetry(e)) {
368                      LOG.log(Level.WARNING, "Temporary error cancelling membership: " + nodePath, e);
369                      return false;
370                    } else {
371                      throw new JoinException("Problem cancelling membership: " + nodePath, e);
372                    }
373                  }
374                }
375              });
376              cancelled = true; // Prevent auto-re-join logic from undoing this cancel.
377            } catch (InterruptedException e) {
378              Thread.currentThread().interrupt();
379              throw new JoinException("Problem cancelling membership: " + nodePath, e);
380            }
381          }
382        }
383    
384        private class CancelledException extends IllegalStateException { /* marker */ }
385    
386        synchronized Membership join()
387            throws ZooKeeperConnectionException, InterruptedException, KeeperException {
388    
389          if (cancelled) {
390            throw new CancelledException();
391          }
392    
393          if (nodePath == null) {
394            // Re-join if our ephemeral node goes away due to session expiry - only needs to be
395            // registered once.
396            zkClient.registerExpirationHandler(new Command() {
397              @Override public void execute() {
398                tryJoin();
399              }
400            });
401          }
402    
403          byte[] membershipData = memberData.get();
404          nodePath = zkClient.get().create(path + "/" + nodeNamePrefix, membershipData, acl,
405              CreateMode.EPHEMERAL_SEQUENTIAL);
406          memberId = extractMemberId(nodePath);
407          this.membershipData = membershipData;
408    
409          // Re-join if our ephemeral node goes away due to maliciousness.
410          zkClient.get().exists(nodePath, new Watcher() {
411            @Override public void process(WatchedEvent event) {
412              if (event.getType() == EventType.NodeDeleted) {
413                tryJoin();
414              }
415            }
416          });
417    
418          return this;
419        }
420    
421        private final ExceptionalSupplier<Boolean, InterruptedException> tryJoin =
422            new ExceptionalSupplier<Boolean, InterruptedException>() {
423              @Override public Boolean get() throws InterruptedException {
424                try {
425                  join();
426                  return true;
427                } catch (CancelledException e) {
428                  // Lost a cancel race - that's ok.
429                  return true;
430                } catch (ZooKeeperConnectionException e) {
431                  LOG.log(Level.WARNING, "Problem connecting to ZooKeeper, retrying", e);
432                  return false;
433                } catch (KeeperException e) {
434                  if (zkClient.shouldRetry(e)) {
435                    LOG.log(Level.WARNING, "Temporary error re-joining group: " + path, e);
436                    return false;
437                  } else {
438                    throw new IllegalStateException("Permanent problem re-joining group: " + path, e);
439                  }
440                }
441              }
442            };
443    
444        private synchronized void tryJoin() {
445          onLoseMembership.execute();
446          try {
447            backoffHelper.doUntilSuccess(tryJoin);
448          } catch (InterruptedException e) {
449            Thread.currentThread().interrupt();
450            throw new RuntimeException(
451                String.format("Interrupted while trying to re-join group: %s, giving up", path), e);
452          }
453        }
454      }

457       * An interface to an object that listens for changes to a group's membership.
458       */
459      public interface GroupChangeListener {
460    
461        /**
462         * Called whenever group membership changes with the new list of member ids.
463         *
464         * @param memberIds the current member ids
465         */
466        void onGroupChange(Iterable<String> memberIds);
467      }

469      /**
470       * Indicates an error watching a group.
471       */
472      public static class WatchException extends Exception {
473        public WatchException(String message, Throwable cause) {
474          super(message, cause);
475        }
476      }
477    
478      /**
479       * Watches this group for the lifetime of this jvm process.  This method will block until the
480       * current group members are available, notify the {@code groupChangeListener} and then return.
481       * All further changes to the group membership will cause notifications on a background thread.
482       *
483       * @param groupChangeListener the listener to notify of group membership change events
484       * @throws WatchException if there is a problem generating the 1st group membership list
485       * @throws InterruptedException if interrupted waiting to gather the 1st group membership list
486       */
487      public final void watch(final GroupChangeListener groupChangeListener)
488          throws WatchException, InterruptedException {
489        Preconditions.checkNotNull(groupChangeListener);
490    
491        try {
492          ensurePersistentGroupPath();
493        } catch (JoinException e) {
494          throw new WatchException("Failed to create group path: " + path, e);
495        }
496    
497        final GroupMonitor groupMonitor = new GroupMonitor(groupChangeListener);
498        backoffHelper.doUntilSuccess(new ExceptionalSupplier<Boolean, WatchException>() {
499          @Override public Boolean get() throws WatchException {
500            try {
501              groupMonitor.watchGroup();
502              return true;
503            } catch (InterruptedException e) {
504              Thread.currentThread().interrupt();
505              throw new WatchException("Interrupted trying to watch group at path: " + path, e);
506            } catch (ZooKeeperConnectionException e) {
507              LOG.log(Level.WARNING, "Temporary error trying to watch group at path: " + path, e);
508              return null;
509            } catch (KeeperException e) {
510              if (zkClient.shouldRetry(e)) {
511                LOG.log(Level.WARNING, "Temporary error trying to watch group at path: " + path, e);
512                return null;
513              } else {
514                throw new WatchException("Problem trying to watch group at path: " + path, e);
515              }
516            }
517          }
518        });
519      }
520    
521      /**
522       * Helps continuously monitor a group for membership changes.
523       */
524      private class GroupMonitor {
525        private final GroupChangeListener groupChangeListener;
526        private Set<String> members;
527    
528        GroupMonitor(GroupChangeListener groupChangeListener) {
529          this.groupChangeListener = groupChangeListener;
530        }
531    
532        private final Watcher groupWatcher = new Watcher() {
533          @Override public final void process(WatchedEvent event) {
534            if (event.getType() == EventType.NodeChildrenChanged) {
535              tryWatchGroup();
536            }
537          }
538        };
539    
540        private final ExceptionalSupplier<Boolean, InterruptedException> tryWatchGroup =
541            new ExceptionalSupplier<Boolean, InterruptedException>() {
542              @Override public Boolean get() throws InterruptedException {
543                try {
544                  watchGroup();
545                  return true;
546                } catch (ZooKeeperConnectionException e) {
547                  LOG.log(Level.WARNING, "Problem connecting to ZooKeeper, retrying", e);
548                  return false;
549                } catch (KeeperException e) {
550                  if (zkClient.shouldRetry(e)) {
551                    LOG.log(Level.WARNING, "Temporary error re-watching group: " + path, e);
552                    return false;
553                  } else {
554                    throw new IllegalStateException("Permanent problem re-watching group: " + path, e);
555                  }
556                }
557              }
558            };
559    
560        private void tryWatchGroup() {
561          try {
562            backoffHelper.doUntilSuccess(tryWatchGroup);
563          } catch (InterruptedException e) {
564            Thread.currentThread().interrupt();
565            throw new RuntimeException(
566                String.format("Interrupted while trying to re-watch group: %s, giving up", path), e);
567          }
568        }
569    
570        private void watchGroup()
571            throws ZooKeeperConnectionException, InterruptedException, KeeperException {
572    
573          List<String> children = zkClient.get().getChildren(path, groupWatcher);
574          setMembers(Iterables.filter(children, nodeNameFilter));
575        }
576    
577        synchronized void setMembers(Iterable<String> members) {
578          if (this.members == null) {
579            // Reset our watch on the group if session expires - only needs to be registered once.
580            zkClient.registerExpirationHandler(new Command() {
581              @Override public void execute() {
582                tryWatchGroup();
583              }
584            });
585          }
586    
587          Set<String> membership = ImmutableSet.copyOf(members);
588          if (!membership.equals(this.members)) {
589            groupChangeListener.onGroupChange(members);
590            this.members = membership;
591          }
592        }
593      }
594    
595      @Override
596      public String toString() {
597        return "Group " + path;
598      }
