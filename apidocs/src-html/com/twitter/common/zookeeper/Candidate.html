018    package com.twitter.common.zookeeper;
019    
020    import com.google.common.base.Supplier;
021    import com.twitter.common.base.ExceptionalCommand;
022    import com.twitter.common.zookeeper.Group.JoinException;
023    import com.twitter.common.zookeeper.Group.WatchException;
024    import com.twitter.common.zookeeper.ZooKeeperClient.ZooKeeperConnectionException;
025    import org.apache.zookeeper.KeeperException;
026    
027    /**
028     * Interface definition for becoming or querying for a ZooKeeper-based group leader.
029     *
030     * @author William Farner
031     */
032    public interface Candidate {
033    
034      /**
035       * Returns the current group leader by querying ZooKeeper synchronously.
036       *
037       * @return the id of the current group leader or {@code null} if there is none
038       * @throws ZooKeeperConnectionException if there was a problem connecting to ZooKeeper
039       * @throws KeeperException if there was a problem reading the leader information
040       * @throws InterruptedException if this thread is interrupted getting the leader
041       */
042      public String getLeaderId()
043          throws ZooKeeperConnectionException, KeeperException, InterruptedException;
044    
045      /**
046       * Encapsulates a leader that can be elected and subsequently defeated.
047       */
048      interface Leader {
049    
050        /**
051         * Called when this leader has been elected.
052         *
053         * @param abdicate a command that can be used to abdicate leadership and force a new election
054         */
055        void onElected(ExceptionalCommand<JoinException> abdicate);
056    
057        /**
058         * Called when the leader has been ousted.  Can occur either if the leader abdicates or if an
059         * external event causes the leader to lose its leadership role (session expiration).
060         */
061        void onDefeated();
062      }
063    
064      /**
065       * Offers this candidate in leadership elections for as long as the current jvm process is alive.
066       * Upon election, the {@code onElected} callback will be executed and a command that can be used
067       * to abdicate leadership will be passed in.  If the elected leader jvm process dies or the
068       * elected leader successfully abdicates then a new leader will be elected.  Leaders that
069       * successfully abdicate are removed from the group and will not be eligible for leadership
070       * election unless {@link #offerLeadership(Leader)} is called again.
071       *
072       * @param leader the leader to notify of election and defeat events
073       * @throws JoinException if there was a problem joining the group
074       * @throws WatchException if there is a problem generating the 1st group membership list
075       * @throws InterruptedException if interrupted waiting to join the group and determine initial
076       *     election results
077       * @return a supplier that can be queried to find out if this leader is currently elected
078       */
079      public Supplier<Boolean> offerLeadership(Leader leader)
080            throws JoinException, WatchException, InterruptedException;
081    }
