052    /**
053     * This class exposes methods for joining and monitoring distributed groups.  The groups this class
054     * monitors are realized as persistent paths in ZooKeeper with ephemeral sequential child nodes for
055     * each member of a group.
056     *
057     * @author John Sirois
058     */
059    public class Group {
060      private static final Logger LOG = Logger.getLogger(Partitioner.class.getName());
061    
062      private static final Supplier<byte[]> NO_MEMBER_DATA = Suppliers.ofInstance(null);
063      private static final String DEFAULT_NODE_NAME_PREFIX = "member_";

065      private Predicate<String> nodeNameFilter;
066

068      private final List<ACL> acl;
069      private final String path;
070      private final String nodeNamePrefix;

074      @VisibleForTesting static String normalizePath(String path) {
075        String normalizedPath = path.replaceAll("//+", "/").replaceFirst("(.+)/$", "$1");
076        PathUtils.validatePath(normalizedPath);
077        return normalizedPath;
078      }

080      /**
081       * Creates a group rooted at the given {@code path}.  Paths must be absolute and trailing or
082       * duplicate slashes will be normalized.  For example, all the following paths would create a
083       * group at the normalized path /my/distributed/group:
084       * <ul>
085       *   <li>/my/distributed/group
086       *   <li>/my/distributed/group/
087       *   <li>/my/distributed//group
088       * </ul>
089       *
090       * @param zkClient the client to use for interactions with ZooKeeper
091       * @param acl the ACL to use for creating the persistent group path if it does not already exist
092       * @param path the absolute persistent path that represents this group
093       * @param nodeNamePrefix Node name prefix that denotes group membership.
094       */
095      public Group(ZooKeeperClient zkClient, List<ACL> acl, String path, String nodeNamePrefix) {
096        this.zkClient = Preconditions.checkNotNull(zkClient);
097        this.acl = Preconditions.checkNotNull(acl);
098        this.path = normalizePath(Preconditions.checkNotNull(path));
099        this.nodeNamePrefix = MorePreconditions.checkNotBlank(nodeNamePrefix);
100    
101        final Pattern groupNodeNamePattern = Pattern.compile(
102            "^" + Pattern.quote(nodeNamePrefix) + "[0-9]+$");
103        nodeNameFilter = new Predicate<String>() {
104            @Override public boolean apply(String childNodeName) {
105              return groupNodeNamePattern.matcher(childNodeName).matches();
106            }
107          };
108    
109        backoffHelper = new BackoffHelper();
110      }
111    
112      /**
113       * Equivalent to {@link #Group(ZooKeeperClient, java.util.List, String, String)} with a default
114       * {@code nodeNamePrefix} of 'member_'.
115       */
116      public Group(ZooKeeperClient zkClient, List<ACL> acl, String path) {
117        this(zkClient, acl, path, DEFAULT_NODE_NAME_PREFIX);
118      }
119    
120      public String getMemberPath(String memberId) {
121        return path + "/" + MorePreconditions.checkNotBlank(memberId);
122      }
123    
124      public String getMemberId(String nodePath) {
125        MorePreconditions.checkNotBlank(nodePath);
126        Preconditions.checkArgument(nodePath.startsWith(path + "/"),
127            "Not a member of this group[%s]: %s", path, nodePath);
128        return extractMemberId(nodePath);
129      }
130    
131      private String extractMemberId(String nodePath) {
132        String memberId = StringUtils.substringAfterLast(nodePath, "/");
133        Preconditions.checkArgument(nodeNameFilter.apply(memberId), "Not a group member: %s", memberId);
134        return memberId;
135      }

137      /**
138       * Returns the current list of group member ids by querying ZooKeeper synchronously.
139       *
140       * @return the ids of all the present members of this group
141       * @throws ZooKeeperConnectionException if there was a problem connecting to ZooKeeper
142       * @throws KeeperException if there was a problem reading this group's member ids
143       * @throws InterruptedException if this thread is interrupted listing the group members
144       */
145      public Iterable<String> getMemberIds()
146          throws ZooKeeperConnectionException, KeeperException, InterruptedException {
147        return Iterables.filter(zkClient.get().getChildren(path, false), nodeNameFilter);
148      }
149    
150      /**
151       * Gets the data for one of this groups members by querying ZooKeeper synchronously.
152       *
153       * @param memberId the id of the member whose data to retrieve
154       * @return the data associated with the {@code memberId}
155       * @throws ZooKeeperConnectionException if there was a problem connecting to ZooKeeper
156       * @throws KeeperException if there was a problem reading this member's data
157       * @throws InterruptedException if this thread is interrupted retrieving the member data
158       */
159      public byte[] getMemberData(String memberId)
160          throws ZooKeeperConnectionException, KeeperException, InterruptedException {
161        return zkClient.get().getData(getMemberPath(memberId), false, null);
162      }

164      /**
165       * Represents membership in a distributed group.
166       */
167      public interface Membership {
168    
169        /**
170         * Returns the persistent ZooKeeper path that represents this group.
171         */
172        String getGroupPath();
173    
174        /**
175         * Returns the id (ZooKeeper node name) of this group member.  May change over time if the
176         * ZooKeeper session expires.
177         */
178        String getMemberId();
179    
180        /**
181         * Returns the full ZooKeeper path to this group member.  May change over time if the
182         * ZooKeeper session expires.
183         */
184        String getMemberPath();
185    
186        /**
187         * Updates the membership data synchronously using the {@code Supplier<byte[]>} passed to
188         * {@link Group#join()}.
189         *
190         * @return the new membership data
191         * @throws UpdateException if there was a problem updating the membership data
192         */
193        byte[] updateMemberData() throws UpdateException;
194    
195        /**
196         * Cancels group membership by deleting the associated ZooKeeper member node.
197         *
198         * @throws JoinException if there is a problem deleting the node
199         */
200        void cancel() throws JoinException;
201      }

246      /**
247       * Joins this group and returns the resulting Membership when successful.  Membership will be
248       * automatically cancelled when the current jvm process dies; however the returned Membership
249       * object can be used to cancel membership earlier.  Unless
250       * {@link com.twitter.common.zookeeper.Group.Membership#cancel()} is called the membership will
251       * be maintained by re-establishing it silently in the background.
252       *
253       * <p>Any {@code memberData} given is persisted in the member node in ZooKeeper.  If an
254       * {@code onLoseMembership} callback is supplied, it will be notified each time this member loses
255       * membership in the group.
256       *
257       * @param memberData a supplier of the data to store in the member node
258       * @param onLoseMembership a callback to notify when membership is lost
259       * @return a Membership object with the member details
260       * @throws JoinException if there was a problem joining the group
261       * @throws InterruptedException if this thread is interrupted awaiting completion of the join
262       */
263      public final Membership join(Supplier<byte[]> memberData, @Nullable Command onLoseMembership)
264          throws JoinException, InterruptedException {
265    
266        Preconditions.checkNotNull(memberData);
267        ensurePersistentGroupPath();
268    
269        final ActiveMembership groupJoiner = new ActiveMembership(memberData, onLoseMembership);
270        return backoffHelper.doUntilResult(new ExceptionalSupplier<Membership, JoinException>() {
271          @Override public Membership get() throws JoinException {
272            try {
273              return groupJoiner.join();
274            } catch (InterruptedException e) {
275              Thread.currentThread().interrupt();
276              throw new JoinException("Interrupted trying to join group at path: " + path, e);
277            } catch (ZooKeeperConnectionException e) {
278              LOG.log(Level.WARNING, "Temporary error trying to join group at path: " + path, e);
279              return null;
280            } catch (KeeperException e) {
281              if (zkClient.shouldRetry(e)) {
282                LOG.log(Level.WARNING, "Temporary error trying to join group at path: " + path, e);
283                return null;
284              } else {
285                throw new JoinException("Problem joining partition group at path: " + path, e);
286              }
287            }
288          }
289        });
290      }
291    
292      private void ensurePersistentGroupPath() throws JoinException, InterruptedException {
293        backoffHelper.doUntilSuccess(new ExceptionalSupplier<Boolean, JoinException>() {
294          @Override public Boolean get() throws JoinException {
295            try {
296              ZooKeeperUtils.ensurePath(zkClient, acl, path);
297              return true;
298            } catch (InterruptedException e) {
299              Thread.currentThread().interrupt();
300              throw new JoinException("Interrupted trying to ensure group at path: " + path, e);
301            } catch (ZooKeeperConnectionException e) {
302              LOG.log(Level.WARNING, "Problem connecting to ZooKeeper, retrying", e);
303              return false;
304            } catch (KeeperException e) {
305              if (zkClient.shouldRetry(e)) {
306                LOG.log(Level.WARNING, "Temporary error ensuring path: " + path, e);
307                return false;
308              } else {
309                throw new JoinException("Problem ensuring group at path: " + path, e);
310              }
311            }
312          }
313        });
314      }

316      private class ActiveMembership implements Membership {
317        private final Supplier<byte[]> memberData;
318        private final Command onLoseMembership;
319        private String nodePath;
320        private String memberId;
321        private volatile boolean cancelled;
322        private byte[] membershipData;
323    
324        public ActiveMembership(Supplier<byte[]> memberData, @Nullable Command onLoseMembership) {
325          this.memberData = memberData;
326          this.onLoseMembership = (onLoseMembership == null) ? Commands.NOOP : onLoseMembership;
327        }
328    
329        @Override
330        public String getGroupPath() {
331          return path;
332        }
333    
334        @Override
335        public synchronized String getMemberId() {
336          return memberId;
337        }
338    
339        @Override
340        public synchronized String getMemberPath() {
341          return nodePath;
342        }
343    
344        @Override
345        public synchronized byte[] updateMemberData() throws UpdateException {
346          byte[] membershipData = memberData.get();
347          if (!ArrayUtils.isEquals(this.membershipData, membershipData)) {
348            try {
349              zkClient.get().setData(nodePath, membershipData, ZooKeeperUtils.ANY_VERSION);
350              this.membershipData = membershipData;
351            } catch (KeeperException e) {
352              throw new UpdateException("Problem updating membership data.", e);
353            } catch (InterruptedException e) {
354              throw new UpdateException("Interrupted attempting to update membership data.", e);
355            } catch (ZooKeeperConnectionException e) {
356              throw new UpdateException(
357                  "Could not connect to the ZooKeeper cluster to update membership data.", e);
358            }
359          }
360          return membershipData;
361        }
362    
363        @Override
364        public synchronized void cancel() throws JoinException {
365          if (!cancelled) {
366            try {
367              backoffHelper.doUntilSuccess(new ExceptionalSupplier<Boolean, JoinException>() {
368                @Override public Boolean get() throws JoinException {
369                  try {
370                    zkClient.get().delete(nodePath, ZooKeeperUtils.ANY_VERSION);
371                    return true;
372                  } catch (InterruptedException e) {
373                    Thread.currentThread().interrupt();
374                    throw new JoinException("Interrupted trying to cancel membership: " + nodePath, e);
375                  } catch (ZooKeeperConnectionException e) {
376                    LOG.log(Level.WARNING, "Problem connecting to ZooKeeper, retrying", e);
377                    return false;
378                  } catch (NoNodeException e) {
379                    LOG.info("Membership already cancelled, node at path: " + nodePath +
380                             " has been deleted");
381                    return true;
382                  } catch (KeeperException e) {
383                    if (zkClient.shouldRetry(e)) {
384                      LOG.log(Level.WARNING, "Temporary error cancelling membership: " + nodePath, e);
385                      return false;
386                    } else {
387                      throw new JoinException("Problem cancelling membership: " + nodePath, e);
388                    }
389                  }
390                }
391              });
392              cancelled = true; // Prevent auto-re-join logic from undoing this cancel.
393            } catch (InterruptedException e) {
394              Thread.currentThread().interrupt();
395              throw new JoinException("Problem cancelling membership: " + nodePath, e);
396            }
397          }
398        }
399    
400        private class CancelledException extends IllegalStateException { /* marker */ }
401    
402        synchronized Membership join()
403            throws ZooKeeperConnectionException, InterruptedException, KeeperException {
404    
405          if (cancelled) {
406            throw new CancelledException();
407          }
408    
409          if (nodePath == null) {
410            // Re-join if our ephemeral node goes away due to session expiry - only needs to be
411            // registered once.
412            zkClient.registerExpirationHandler(new Command() {
413              @Override public void execute() {
414                tryJoin();
415              }
416            });
417          }
418    
419          byte[] membershipData = memberData.get();
420          nodePath = zkClient.get().create(path + "/" + nodeNamePrefix, membershipData, acl,
421              CreateMode.EPHEMERAL_SEQUENTIAL);
422          memberId = extractMemberId(nodePath);
423          LOG.info("Set group member ID to " + memberId);
424          this.membershipData = membershipData;
425    
426          // Re-join if our ephemeral node goes away due to maliciousness.
427          zkClient.get().exists(nodePath, new Watcher() {
428            @Override public void process(WatchedEvent event) {
429              if (event.getType() == EventType.NodeDeleted) {
430                LOG.info("Member ID deleted. Rejoining. Event: " + event);
431                tryJoin();
432              }
433            }
434          });
435    
436          return this;
437        }
438    
439        private final ExceptionalSupplier<Boolean, InterruptedException> tryJoin =
440            new ExceptionalSupplier<Boolean, InterruptedException>() {
441              @Override public Boolean get() throws InterruptedException {
442                try {
443                  join();
444                  return true;
445                } catch (CancelledException e) {
446                  // Lost a cancel race - that's ok.
447                  return true;
448                } catch (ZooKeeperConnectionException e) {
449                  LOG.log(Level.WARNING, "Problem connecting to ZooKeeper, retrying", e);
450                  return false;
451                } catch (KeeperException e) {
452                  if (zkClient.shouldRetry(e)) {
453                    LOG.log(Level.WARNING, "Temporary error re-joining group: " + path, e);
454                    return false;
455                  } else {
456                    throw new IllegalStateException("Permanent problem re-joining group: " + path, e);
457                  }
458                }
459              }
460            };
461    
462        private synchronized void tryJoin() {
463          onLoseMembership.execute();
464          try {
465            backoffHelper.doUntilSuccess(tryJoin);
466          } catch (InterruptedException e) {
467            Thread.currentThread().interrupt();
468            throw new RuntimeException(
469                String.format("Interrupted while trying to re-join group: %s, giving up", path), e);
470          }
471        }
472      }

475       * An interface to an object that listens for changes to a group's membership.
476       */
477      public interface GroupChangeListener {
478    
479        /**
480         * Called whenever group membership changes with the new list of member ids.
481         *
482         * @param memberIds the current member ids
483         */
484        void onGroupChange(Iterable<String> memberIds);
485      }

487      /**
488       * Indicates an error watching a group.
489       */
490      public static class WatchException extends Exception {
491        public WatchException(String message, Throwable cause) {
492          super(message, cause);
493        }
494      }
495    
496      /**
497       * Watches this group for the lifetime of this jvm process.  This method will block until the
498       * current group members are available, notify the {@code groupChangeListener} and then return.
499       * All further changes to the group membership will cause notifications on a background thread.
500       *
501       * @param groupChangeListener the listener to notify of group membership change events
502       * @throws WatchException if there is a problem generating the 1st group membership list
503       * @throws InterruptedException if interrupted waiting to gather the 1st group membership list
504       */
505      public final void watch(final GroupChangeListener groupChangeListener)
506          throws WatchException, InterruptedException {
507        Preconditions.checkNotNull(groupChangeListener);
508    
509        try {
510          ensurePersistentGroupPath();
511        } catch (JoinException e) {
512          throw new WatchException("Failed to create group path: " + path, e);
513        }
514    
515        final GroupMonitor groupMonitor = new GroupMonitor(groupChangeListener);
516        backoffHelper.doUntilSuccess(new ExceptionalSupplier<Boolean, WatchException>() {
517          @Override public Boolean get() throws WatchException {
518            try {
519              groupMonitor.watchGroup();
520              return true;
521            } catch (InterruptedException e) {
522              Thread.currentThread().interrupt();
523              throw new WatchException("Interrupted trying to watch group at path: " + path, e);
524            } catch (ZooKeeperConnectionException e) {
525              LOG.log(Level.WARNING, "Temporary error trying to watch group at path: " + path, e);
526              return null;
527            } catch (KeeperException e) {
528              if (zkClient.shouldRetry(e)) {
529                LOG.log(Level.WARNING, "Temporary error trying to watch group at path: " + path, e);
530                return null;
531              } else {
532                throw new WatchException("Problem trying to watch group at path: " + path, e);
533              }
534            }
535          }
536        });
537      }
538    
539      /**
540       * Helps continuously monitor a group for membership changes.
541       */
542      private class GroupMonitor {
543        private final GroupChangeListener groupChangeListener;
544        private Set<String> members;
545    
546        GroupMonitor(GroupChangeListener groupChangeListener) {
547          this.groupChangeListener = groupChangeListener;
548        }
549    
550        private final Watcher groupWatcher = new Watcher() {
551          @Override public final void process(WatchedEvent event) {
552            if (event.getType() == EventType.NodeChildrenChanged) {
553              tryWatchGroup();
554            }
555          }
556        };
557    
558        private final ExceptionalSupplier<Boolean, InterruptedException> tryWatchGroup =
559            new ExceptionalSupplier<Boolean, InterruptedException>() {
560              @Override public Boolean get() throws InterruptedException {
561                try {
562                  watchGroup();
563                  return true;
564                } catch (ZooKeeperConnectionException e) {
565                  LOG.log(Level.WARNING, "Problem connecting to ZooKeeper, retrying", e);
566                  return false;
567                } catch (KeeperException e) {
568                  if (zkClient.shouldRetry(e)) {
569                    LOG.log(Level.WARNING, "Temporary error re-watching group: " + path, e);
570                    return false;
571                  } else {
572                    throw new IllegalStateException("Permanent problem re-watching group: " + path, e);
573                  }
574                }
575              }
576            };
577    
578        private void tryWatchGroup() {
579          try {
580            backoffHelper.doUntilSuccess(tryWatchGroup);
581          } catch (InterruptedException e) {
582            Thread.currentThread().interrupt();
583            throw new RuntimeException(
584                String.format("Interrupted while trying to re-watch group: %s, giving up", path), e);
585          }
586        }
587    
588        private void watchGroup()
589            throws ZooKeeperConnectionException, InterruptedException, KeeperException {
590    
591          List<String> children = zkClient.get().getChildren(path, groupWatcher);
592          setMembers(Iterables.filter(children, nodeNameFilter));
593        }
594    
595        synchronized void setMembers(Iterable<String> members) {
596          if (this.members == null) {
597            // Reset our watch on the group if session expires - only needs to be registered once.
598            zkClient.registerExpirationHandler(new Command() {
599              @Override public void execute() {
600                tryWatchGroup();
601              }
602            });
603          }
604    
605          Set<String> membership = ImmutableSet.copyOf(members);
606          if (!membership.equals(this.members)) {
607            groupChangeListener.onGroupChange(members);
608            this.members = membership;
609          }
610        }
611      }
612    
613      @Override
614      public String toString() {
615        return "Group " + path;
616      }
