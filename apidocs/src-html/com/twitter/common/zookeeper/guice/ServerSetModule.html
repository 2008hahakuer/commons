017    package com.twitter.common.zookeeper.guice;
018    
019    import java.net.InetSocketAddress;
020    import java.util.logging.Level;
021    import java.util.logging.Logger;
022    
023    import com.google.common.base.Optional;
024    import com.google.inject.AbstractModule;
025    import com.google.inject.Inject;
026    
027    import com.twitter.common.application.ShutdownRegistry;
028    import com.twitter.common.application.modules.LifecycleModule;
029    import com.twitter.common.application.modules.LocalServiceRegistry;
030    import com.twitter.common.args.Arg;
031    import com.twitter.common.args.CmdLine;
032    import com.twitter.common.args.constraints.NotEmpty;
033    import com.twitter.common.args.constraints.NotNull;
034    import com.twitter.common.base.Command;
035    import com.twitter.common.base.ExceptionalCommand;
036    import com.twitter.common.zookeeper.Group.JoinException;
037    import com.twitter.common.zookeeper.ServerSet;
038    import com.twitter.common.zookeeper.ServerSet.EndpointStatus;
039    import com.twitter.common.zookeeper.ServerSet.UpdateException;
040    import com.twitter.common.zookeeper.ServerSetImpl;
041    import com.twitter.common.zookeeper.ZooKeeperClient;
042    import com.twitter.thrift.Status;
043    
044    import static com.google.common.base.Preconditions.checkNotNull;
045    
046    /**
047     * A module that registers all ports in the {@link LocalServiceRegistry} in an {@link ServerSet}.
048     *
049     * Required bindings:
050     * <ul>
051     *   <li> {@link ZooKeeperClient}
052     *   <li> {@link ShutdownRegistry}
053     *   <li> {@link LocalServiceRegistry}
054     * </ul>
055     * {@link LifecycleModule} must also be included by users so a startup action may be registered.
056     *
057     * @author William Farner
058     */
059    public class ServerSetModule extends AbstractModule {
060    
061      private static final Logger LOG = Logger.getLogger(ServerSetModule.class.getName());
062    
063      @NotNull
064      @NotEmpty
065      @CmdLine(name = "serverset_path", help = "ServerSet registration path")
066      protected static final Arg<String> SERVERSET_PATH = Arg.create(null);
067    
068      @Override
069      protected void configure() {
070        requireBinding(ZooKeeperClient.class);
071        requireBinding(ShutdownRegistry.class);
072        requireBinding(LocalServiceRegistry.class);
073        LifecycleModule.bindStartupAction(binder(), ServerSetJoiner.class);
074      }
075    
076      private static class ServerSetJoiner implements Command {
077        private final ZooKeeperClient zkClient;
078        private final LocalServiceRegistry serviceRegistry;
079        private final ShutdownRegistry shutdownRegistry;
080    
081        @Inject
082        ServerSetJoiner(
083            ZooKeeperClient zkClient,
084            LocalServiceRegistry serviceRegistry,
085            ShutdownRegistry shutdownRegistry) {
086          this.zkClient = checkNotNull(zkClient);
087          this.serviceRegistry = checkNotNull(serviceRegistry);
088          this.shutdownRegistry = checkNotNull(shutdownRegistry);
089        }
090    
091        @Override public void execute() throws RuntimeException {
092          ServerSet serverSet = new ServerSetImpl(zkClient, SERVERSET_PATH.get());
093    
094          Optional<InetSocketAddress> primarySocket = serviceRegistry.getPrimarySocket();
095          if (!primarySocket.isPresent()) {
096            throw new IllegalStateException("No primary service registered with LocalServiceRegistry.");
097          }
098    
099          final EndpointStatus status;
100          try {
101            status = serverSet.join(
102                primarySocket.get(),
103                serviceRegistry.getAuxiliarySockets(),
104                Status.ALIVE);
105          } catch (JoinException e) {
106            LOG.log(Level.WARNING, "Failed to join ServerSet.", e);
107            throw new RuntimeException(e);
108          } catch (InterruptedException e) {
109            LOG.log(Level.WARNING, "Interrupted while joining ServerSet.", e);
110            Thread.currentThread().interrupt();
111            throw new RuntimeException(e);
112          }
113    
114          shutdownRegistry.addAction(new ExceptionalCommand<UpdateException>() {
115            @Override public void execute() throws UpdateException {
116              LOG.info("Leaving ServerSet.");
117              status.update(Status.DEAD);
118            }
119          });
120        }
121      }
122    }
