001    package com.twitter.common.zookeeper.testing.angrybird;
002    
003    import java.io.IOException;
004    
005    import com.google.common.base.Optional;
006    
007    import com.twitter.common.zookeeper.testing.angrybird.gen.ExpireEndpointRequest;
008    import com.twitter.common.zookeeper.testing.angrybird.gen.ExpireFollowerRequest;
009    import com.twitter.common.zookeeper.testing.angrybird.gen.ExpireLeaderRequest;
010    import com.twitter.common.zookeeper.testing.angrybird.gen.ExpireSessionRequest;
011    import com.twitter.common.zookeeper.testing.angrybird.gen.ExpireResponse;
012    import com.twitter.common.zookeeper.testing.angrybird.gen.RestartResponse;
013    import com.twitter.common.zookeeper.testing.angrybird.gen.ServerPortResponse;
014    import com.twitter.common.zookeeper.testing.angrybird.gen.ShutdownResponse;
015    import com.twitter.common.zookeeper.testing.angrybird.gen.StartupResponse;
016    import com.twitter.common.zookeeper.testing.angrybird.gen.ZooKeeperThriftServer;
017    import com.twitter.util.Future;
018    
019    import static com.twitter.common.zookeeper.testing.angrybird.gen.ResponseCode.ERROR;
020    import static com.twitter.common.zookeeper.testing.angrybird.gen.ResponseCode.OK;
021    
022    /**
023     * Thrift interface for the angrybird ZooKeeper server.
024     */
025    public class AngryBirdZooKeeperThriftService implements ZooKeeperThriftServer.ServiceIface {
026    
027      private final AngryBirdZooKeeperServer zkServer;
028    
029      private Future<ExpireResponse> sessionIdToExpireResponse(Optional<Long> sessionId) {
030        ExpireResponse response = new ExpireResponse();
031    
032        if (sessionId.isPresent()) {
033          response.setResponseCode(OK).setSessionId(sessionId.get().longValue());
034        } else {
035          response.setResponseCode(ERROR).setSessionId(0);
036        }
037    
038        return Future.value(response);
039      }
040    
041      /**
042       * Creates a new angrybird thrift server
043       *
044       * @param zkServer Thrift server to interact with.
045       */
046      public AngryBirdZooKeeperThriftService(AngryBirdZooKeeperServer zkServer) {
047        this.zkServer = zkServer;
048      }
049    
050      @Override
051      public Future<ServerPortResponse> getZooKeeperServerPort() {
052        ServerPortResponse response = new ServerPortResponse();
053        int port = zkServer.getPort();
054        return Future.value(response.setResponseCode(OK).setPort(port));
055      }
056    
057      @Override
058      public Future<ExpireResponse> expireSession(ExpireSessionRequest expireRequest) {
059        return sessionIdToExpireResponse(zkServer.expireSession(expireRequest.sessionId));
060      }
061    
062      @Override
063      public Future<ExpireResponse> expireEndpoint(ExpireEndpointRequest expireRequest) {
064        return sessionIdToExpireResponse(
065            zkServer.expireEndpoint(expireRequest.host, expireRequest.port));
066      }
067    
068      @Override
069      public Future<ExpireResponse> expireLeader(ExpireLeaderRequest request) {
070        return sessionIdToExpireResponse(zkServer.expireLeader(request.zkPath));
071      }
072    
073      @Override
074      public Future<ExpireResponse> expireFollower(ExpireFollowerRequest request) {
075        return sessionIdToExpireResponse(
076            zkServer.expireFollower(request.zkPath, Optional.of(request.nodeId)));
077      }
078    
079      @Override
080      public Future<StartupResponse> startup() {
081        StartupResponse response = new StartupResponse();
082    
083        try {
084          zkServer.startNetwork();
085          response.setResponseCode(OK);
086          response.setResponseMessage("OK");
087        } catch (IOException e) {
088          response.setResponseCode(ERROR).setResponseMessage(e.getMessage());
089        } catch (InterruptedException e) {
090          response.setResponseCode(ERROR).setResponseMessage(e.getMessage());
091        }
092        return Future.value(response);
093      }
094    
095      @Override
096      public Future<ShutdownResponse> shutdown() {
097        ShutdownResponse response = new ShutdownResponse();
098        zkServer.shutdownNetwork();
099        return Future.value(response.setResponseCode(OK).setResponseMessage("OK"));
100      }
101    
102      @Override
103      public Future<RestartResponse> restart() {
104        RestartResponse response = new RestartResponse();
105    
106        try {
107          zkServer.shutdownNetwork();
108          zkServer.restartNetwork();
109          response.setResponseCode(OK).setResponseMessage("OK");
110        } catch (IOException e) {
111          response.setResponseCode(ERROR).setResponseMessage(e.getMessage());
112        } catch (InterruptedException e) {
113          response.setResponseCode(ERROR).setResponseMessage(e.getMessage());
114        }
115        return Future.value(response);
116      }
117    }
