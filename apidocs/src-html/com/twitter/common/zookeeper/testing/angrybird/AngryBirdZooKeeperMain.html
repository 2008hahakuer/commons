001    package com.twitter.common.zookeeper.testing.angrybird;
002    
003    import java.io.IOException;
004    import java.net.InetSocketAddress;
005    import java.util.Arrays;
006    import java.util.logging.Level;
007    import java.util.logging.Logger;
008    
009    import javax.annotation.Nonnegative;
010    
011    import com.google.inject.AbstractModule;
012    import com.google.inject.Inject;
013    import com.google.inject.Module;
014    
015    import org.apache.thrift.protocol.TBinaryProtocol;
016    
017    import com.twitter.common.application.AbstractApplication;
018    import com.twitter.common.application.AppLauncher;
019    import com.twitter.common.application.Lifecycle;
020    import com.twitter.common.application.ShutdownRegistry;
021    import com.twitter.common.application.modules.LifecycleModule;
022    import com.twitter.common.args.Arg;
023    import com.twitter.common.args.CmdLine;
024    import com.twitter.common.args.constraints.NotNull;
025    import com.twitter.common.args.constraints.Positive;
026    import com.twitter.common.base.Command;
027    import com.twitter.finagle.builder.Server;
028    import com.twitter.finagle.builder.ServerBuilder;
029    import com.twitter.finagle.thrift.ThriftServerFramedCodec;
030    import com.twitter.common.zookeeper.testing.angrybird.gen.ZooKeeperThriftServer;
031    import com.twitter.util.Duration;
032    
033    import static com.google.common.base.Preconditions.checkNotNull;
034    
035    /**
036     * Angrybird ZooKeeper server launcher.
037     */
038    public class AngryBirdZooKeeperMain extends AbstractApplication {
039    
040      private static final Logger LOG = Logger.getLogger(AngryBirdZooKeeperMain.class.getName());
041    
042      @NotNull
043      @Positive
044      @CmdLine(name = "thrift_port", help = "Thrift server port.")
045      private static final Arg<Integer> THRIFT_PORT = Arg.create();
046    
047      @Nonnegative
048      @CmdLine(name = "zk_port", help = "Zookeeper server port")
049      private static final Arg<Integer> ZK_PORT = Arg.create(0);
050    
051      @Inject private Lifecycle lifecycle;
052    
053      private static class ThriftServiceStarter implements Command {
054        private final ShutdownRegistry shutdownRegistry;
055    
056        @Inject ThriftServiceStarter(ShutdownRegistry shutdownRegistry) {
057          this.shutdownRegistry = checkNotNull(shutdownRegistry);
058        }
059    
060        @Override public void execute() {
061          AngryBirdZooKeeperServer zooKeeperServer;
062    
063          try {
064            zooKeeperServer = new AngryBirdZooKeeperServer(ZK_PORT.get(), shutdownRegistry);
065            int port = zooKeeperServer.startNetwork();
066            LOG.info(String.format("ZooKeeper server started on: %d", port));
067          } catch (IOException e) {
068            throw new RuntimeException("Failed to start server: " + e);
069          } catch (InterruptedException e) {
070            throw new RuntimeException("Failed to start server: " + e);
071          }
072    
073          final Server thriftServer = ServerBuilder.safeBuild(
074              new ZooKeeperThriftServer.Service(new AngryBirdZooKeeperThriftService(zooKeeperServer),
075                  new TBinaryProtocol.Factory()),
076              ServerBuilder.get()
077                  .name("AngryBirdZooKeeperServer")
078                  .codec(ThriftServerFramedCodec.get())
079                  .bindTo(new InetSocketAddress(THRIFT_PORT.get())));
080    
081          shutdownRegistry.addAction(new Command() {
082            @Override public void execute() {
083              thriftServer.close(Duration.forever());
084            }
085          });
086        }
087      }
088    
089      private static class AngryBirdZooKeeperModule extends AbstractModule {
090        @Override
091        public void configure() {
092          LifecycleModule.bindStartupAction(binder(), ThriftServiceStarter.class);
093        }
094      }
095    
096      @Override
097      public void run() {
098        LOG.info("Starting AngryBird ZooKeeper Server");
099        lifecycle.awaitShutdown();
100      }
101    
102      @Override
103      public Iterable<? extends Module> getModules() {
104        return Arrays.asList(new AngryBirdZooKeeperModule());
105      }
106    
107      public static void main(String[] args) {
108        AppLauncher.launch(AngryBirdZooKeeperMain.class, args);
109      }
110    }
