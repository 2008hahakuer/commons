017    package com.twitter.common.zookeeper.testing;
018    
019    import java.io.IOException;
020    
021    import com.google.common.base.Preconditions;
022    import com.google.common.testing.TearDown;
023    import com.google.common.testing.junit4.TearDownTestCase;
024    
025    import org.junit.Before;
026    
027    import com.twitter.common.application.ShutdownRegistry;
028    import com.twitter.common.application.ShutdownRegistry.ShutdownRegistryImpl;
029    import com.twitter.common.quantity.Amount;
030    import com.twitter.common.quantity.Time;
031    import com.twitter.common.zookeeper.ZooKeeperClient;
032    import com.twitter.common.zookeeper.ZooKeeperClient.Credentials;
033    import com.twitter.common.zookeeper.ZooKeeperClient.ZooKeeperConnectionException;
034    
035    /**
036     * A baseclass for in-process zookeeper tests.
037     * Uses ZooKeeperTestHelper to start the server and create clients: new tests should directly use
038     * that helper class instead of extending this class.
039     *
040     * @author John Sirois
041     */
042    public abstract class BaseZooKeeperTest extends TearDownTestCase {
043    
044      private final Amount<Integer, Time> defaultSessionTimeout;
045      private ZooKeeperTestServer zkTestServer;
046    
047      /**
048       * Creates a test case where the test server uses its
049       * {@link ZooKeeperTestServer#DEFAULT_SESSION_TIMEOUT} for clients created without an explicit
050       * session timeout.
051       */
052      public BaseZooKeeperTest() {
053        this(ZooKeeperTestServer.DEFAULT_SESSION_TIMEOUT);
054      }
055    
056      /**
057       * Creates a test case where the test server uses the given {@code defaultSessionTimeout} for
058       * clients created without an explicit session timeout.
059       */
060      public BaseZooKeeperTest(Amount<Integer, Time> defaultSessionTimeout) {
061        this.defaultSessionTimeout = Preconditions.checkNotNull(defaultSessionTimeout);
062      }
063    
064      @Before
065      public final void setUp() throws Exception {
066        final ShutdownRegistryImpl shutdownRegistry = new ShutdownRegistryImpl();
067        addTearDown(new TearDown() {
068          @Override public void tearDown() {
069            shutdownRegistry.execute();
070          }
071        });
072        zkTestServer = new ZooKeeperTestServer(0, shutdownRegistry, defaultSessionTimeout);
073        zkTestServer.startNetwork();
074      }
075    
076      /**
077       * Starts zookeeper back up on the last used port.
078       */
079      protected final void restartNetwork() throws IOException, InterruptedException {
080        zkTestServer.restartNetwork();
081      }
082    
083      /**
084       * Shuts down the in-process zookeeper network server.
085       */
086      protected final void shutdownNetwork() {
087        zkTestServer.shutdownNetwork();
088      }
089    
090      /**
091       * Expires the active session for the given client.  The client should be one returned from
092       * {@link #createZkClient}.
093       *
094       * @param zkClient the client to expire
095       * @throws ZooKeeperClient.ZooKeeperConnectionException if a problem is encountered connecting to
096       *    the local zk server while trying to expire the session
097       * @throws InterruptedException if interrupted while requesting expiration
098       */
099      protected final void expireSession(ZooKeeperClient zkClient)
100          throws ZooKeeperConnectionException, InterruptedException {
101        zkTestServer.expireClientSession(zkClient);
102      }
103    
104      /**
105       * Returns the current port to connect to the in-process zookeeper instance.
106       */
107      protected final int getPort() {
108        return zkTestServer.getPort();
109      }
110    
111      /**
112       * Returns a new unauthenticated zookeeper client connected to the in-process zookeeper server
113       * with the default session timeout.
114       */
115      protected final ZooKeeperClient createZkClient() {
116        return zkTestServer.createClient();
117      }
118    
119      /**
120       * Returns a new authenticated zookeeper client connected to the in-process zookeeper server with
121       * the default session timeout.
122       */
123      protected final ZooKeeperClient createZkClient(Credentials credentials) {
124        return zkTestServer.createClient(credentials);
125      }
126    
127      /**
128       * Returns a new authenticated zookeeper client connected to the in-process zookeeper server with
129       * the default session timeout.  The client is authenticated in the digest authentication scheme
130       * with the given {@code username} and {@code password}.
131       */
132      protected final ZooKeeperClient createZkClient(String username, String password) {
133        return createZkClient(ZooKeeperClient.digestCredentials(username, password));
134      }
135    
136      /**
137       * Returns a new unauthenticated zookeeper client connected to the in-process zookeeper server
138       * with a custom {@code sessionTimeout}.
139       */
140      protected final ZooKeeperClient createZkClient(Amount<Integer, Time> sessionTimeout) {
141        return zkTestServer.createClient(sessionTimeout);
142      }
143    
144      /**
145       * Returns a new authenticated zookeeper client connected to the in-process zookeeper server with
146       * a custom {@code sessionTimeout}.
147       */
148      protected final ZooKeeperClient createZkClient(Amount<Integer, Time> sessionTimeout,
149          Credentials credentials) {
150        return zkTestServer.createClient(sessionTimeout, credentials);
151      }
152    }
