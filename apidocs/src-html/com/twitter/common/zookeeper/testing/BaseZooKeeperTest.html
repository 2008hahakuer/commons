018    package com.twitter.common.zookeeper.testing;
019    
020    import com.google.common.base.Preconditions;
021    import com.google.common.testing.TearDown;
022    import com.google.common.testing.junit4.TearDownTestCase;
023    import com.twitter.common.io.FileUtils;
024    import com.twitter.common.quantity.Amount;
025    import com.twitter.common.quantity.Time;
026    import com.twitter.common.zookeeper.ZooKeeperClient;
027    import com.twitter.common.zookeeper.ZooKeeperClient.ZooKeeperConnectionException;
028    import org.apache.zookeeper.server.NIOServerCnxn;
029    import org.apache.zookeeper.server.ZooKeeperServer;
030    import org.apache.zookeeper.server.ZooKeeperServer.BasicDataTreeBuilder;
031    import org.apache.zookeeper.server.persistence.FileTxnSnapLog;
032    import org.junit.Before;
033    
034    import java.io.File;
035    import java.io.IOException;
036    import java.net.InetSocketAddress;
037    
038    /**
039     * A baseclass for in-process zookeeper tests.  Ensures
040     *
041     * @author John Sirois
042     */
043    public abstract class BaseZooKeeperTest extends TearDownTestCase {
044      private static final Amount<Integer, Time> DEFAULT_SESSION_TIMEOUT =
045          Amount.of(100, Time.MILLISECONDS);
046    
047      private ZooKeeperServer zooKeeperServer;
048      private NIOServerCnxn.Factory connectionFactory;
049      private int zkPort;
050    
051      @Before
052      public final void setUp() throws Exception {
053        zooKeeperServer =
054            new ZooKeeperServer(new FileTxnSnapLog(createTempDir(), createTempDir()),
055                new BasicDataTreeBuilder());
056    
057        startupNetwork(0);
058        addTearDown(new TearDown() {
059          @Override public void tearDown() {
060            shutdownNetwork();
061          }
062        });
063    
064        zkPort = zooKeeperServer.getClientPort();
065      }
066    
067      private void startupNetwork(int port) throws Exception {
068        connectionFactory = new NIOServerCnxn.Factory(new InetSocketAddress(port));
069        connectionFactory.startup(zooKeeperServer);
070        zkPort = zooKeeperServer.getClientPort();
071      }
072    
073      /**
074       * Starts zookeeper back up on the last used port.
075       */
076      protected final void restartNetwork() throws Exception {
077        Preconditions.checkState(zkPort > 0);
078        startupNetwork(zkPort);
079      }
080    
081      /**
082       * Shuts down the in-process zookeeper network server.
083       */
084      protected final void shutdownNetwork() {
085        if (connectionFactory.isAlive()) {
086          connectionFactory.shutdown();
087        }
088      }
089    
090      protected final void expireSession(ZooKeeperClient zkClient)
091          throws ZooKeeperConnectionException, InterruptedException {
092        zooKeeperServer.closeSession(zkClient.get().getSessionId());
093      }
094    
095      private File createTempDir() {
096        final File tempDir = FileUtils.createTempDir();
097        addTearDown(new TearDown() {
098          @Override public void tearDown() throws IOException {
099            org.apache.commons.io.FileUtils.deleteDirectory(tempDir);
100          }
101        });
102        return tempDir;
103      }
104    
105      /**
106       * Returns the current port to connect to the in-process zookeeper instance.
107       */
108      protected final int getPort() {
109        return zkPort;
110      }
111    
112      /**
113       * Returns a new zookeeper client connected to the in-process zookeeper server with the
114       * {@link #DEFAULT_SESSION_TIMEOUT}.
115       */
116      protected final ZooKeeperClient createZkClient() throws IOException {
117        return createZkClient(DEFAULT_SESSION_TIMEOUT);
118      }
119    
120      /**
121       * Returns a new zookeeper client connected to the in-process zookeeper server with a custom
122       * {@code sessionTimeout}.
123       */
124      protected final ZooKeeperClient createZkClient(Amount<Integer, Time> sessionTimeout)
125          throws IOException {
126    
127        final ZooKeeperClient client = new ZooKeeperClient(sessionTimeout,
128            InetSocketAddress.createUnresolved("127.0.0.1", zkPort));
129        addTearDown(new TearDown() {
130          @Override public void tearDown() throws InterruptedException {
131            client.close();
132          }
133        });
134        return client;
135      }
136    }
