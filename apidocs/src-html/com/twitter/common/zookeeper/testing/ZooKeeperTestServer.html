017    package com.twitter.common.zookeeper.testing;
018    
019    import java.io.File;
020    import java.io.IOException;
021    import java.net.InetSocketAddress;
022    
023    import com.google.common.base.Preconditions;
024    
025    import org.apache.zookeeper.server.NIOServerCnxn;
026    import org.apache.zookeeper.server.ZooKeeperServer;
027    import org.apache.zookeeper.server.ZooKeeperServer.BasicDataTreeBuilder;
028    import org.apache.zookeeper.server.persistence.FileTxnSnapLog;
029    
030    import com.twitter.common.application.ShutdownRegistry;
031    import com.twitter.common.base.Command;
032    import com.twitter.common.base.ExceptionalCommand;
033    import com.twitter.common.io.FileUtils;
034    import com.twitter.common.quantity.Amount;
035    import com.twitter.common.quantity.Time;
036    import com.twitter.common.zookeeper.ZooKeeperClient;
037    import com.twitter.common.zookeeper.ZooKeeperClient.Credentials;
038    
039    /**
040     * A helper class for starting in-process ZooKeeper server and clients.
041     *
042     * <p>This is ONLY meant to be used for testing.
043     */
044    public class ZooKeeperTestServer {
045    
046      /**
047       * The default session timeout for clients created by servers constructed with
048       * {@link #ZooKeeperTestServer(int, ShutdownRegistry)}.
049       */
050      public static final Amount<Integer, Time> DEFAULT_SESSION_TIMEOUT =
051          Amount.of(100, Time.MILLISECONDS);
052    
053      protected final ZooKeeperServer zooKeeperServer;
054      private final ShutdownRegistry shutdownRegistry;
055      private NIOServerCnxn.Factory connectionFactory;
056      private int port;
057      private final Amount<Integer, Time> defaultSessionTimeout;
058    
059      /**
060       * @param port the port to start the zoo keeper server on - {@code 0} picks an ephemeral port
061       * @param shutdownRegistry a registry that will be used to register client and server shutdown
062       *     commands.  It is up to the caller to execute the registered actions at an appropriate time.
063       * @throws IOException if there was aproblem creating the server's database
064       */
065      public ZooKeeperTestServer(int port, ShutdownRegistry shutdownRegistry) throws IOException {
066        this(port, shutdownRegistry, DEFAULT_SESSION_TIMEOUT);
067      }
068    
069      /**
070       * @param port the port to start the zoo keeper server on - {@code 0} picks an ephemeral port
071       * @param shutdownRegistry a registry that will be used to register client and server shutdown
072       *     commands.  It is up to the caller to execute the registered actions at an appropriate time.
073       * @param defaultSessionTimeout the default session timeout for clients created with
074       *     {@link #createClient()}.
075       * @throws IOException if there was aproblem creating the server's database
076       */
077      public ZooKeeperTestServer(int port, ShutdownRegistry shutdownRegistry,
078          Amount<Integer, Time> defaultSessionTimeout) throws IOException {
079        Preconditions.checkArgument(0 <= port && port <= 0xFFFF);
080        this.port = port;
081        this.shutdownRegistry = Preconditions.checkNotNull(shutdownRegistry);
082        this.defaultSessionTimeout = Preconditions.checkNotNull(defaultSessionTimeout);
083    
084        zooKeeperServer =
085            new ZooKeeperServer(new FileTxnSnapLog(createTempDir(), createTempDir()),
086                new BasicDataTreeBuilder());
087      }
088    
089      /**
090       * Starts zookeeper up on the configured port.  If the configured port is the ephemeral port
091       * (@{code 0}), then the actual chosen port is returned.
092       */
093      public final int startNetwork() throws IOException, InterruptedException {
094        connectionFactory = new NIOServerCnxn.Factory(new InetSocketAddress(port));
095        connectionFactory.startup(zooKeeperServer);
096        shutdownRegistry.addAction(new Command() {
097          @Override public void execute() {
098            shutdownNetwork();
099          }
100        });
101        port = zooKeeperServer.getClientPort();
102        return port;
103      }
104    
105      /**
106       * Starts zookeeper back up on the last used port.
107       */
108      public final void restartNetwork() throws IOException, InterruptedException {
109        checkEphemeralPortAssigned();
110        Preconditions.checkState(!connectionFactory.isAlive());
111        startNetwork();
112      }
113    
114      /**
115       * Shuts down the in-process zookeeper network server.
116       */
117      public final void shutdownNetwork() {
118        if (connectionFactory != null && connectionFactory.isAlive()) {
119          connectionFactory.shutdown();
120        }
121      }
122    
123      /**
124       * Expires the active session for the given client.  The client should be one returned from
125       * {@link #createClient}.
126       *
127       * @param zkClient the client to expire
128       * @throws ZooKeeperClient.ZooKeeperConnectionException if a problem is encountered connecting to
129       *    the local zk server while trying to expire the session
130       * @throws InterruptedException if interrupted while requesting expiration
131       */
132      public final void expireClientSession(ZooKeeperClient zkClient)
133          throws ZooKeeperClient.ZooKeeperConnectionException, InterruptedException {
134        zooKeeperServer.closeSession(zkClient.get().getSessionId());
135      }
136    
137      /**
138       * Returns the current port to connect to the in-process zookeeper instance.
139       */
140      public final int getPort() {
141        checkEphemeralPortAssigned();
142        return port;
143      }
144    
145      /**
146       * Returns a new unauthenticated zookeeper client connected to the in-process zookeeper server
147       * with the default session timeout.
148       */
149      public final ZooKeeperClient createClient() {
150        return createClient(defaultSessionTimeout);
151      }
152    
153      /**
154       * Returns a new authenticated zookeeper client connected to the in-process zookeeper server with
155       * the default session timeout.
156       */
157      public final ZooKeeperClient createClient(Credentials credentials) {
158        return createClient(defaultSessionTimeout, credentials);
159      }
160    
161      /**
162       * Returns a new unauthenticated zookeeper client connected to the in-process zookeeper server
163       * with a custom {@code sessionTimeout}.
164       */
165      public final ZooKeeperClient createClient(Amount<Integer, Time> sessionTimeout) {
166        return createClient(sessionTimeout, Credentials.NONE);
167      }
168    
169      /**
170       * Returns a new authenticated zookeeper client connected to the in-process zookeeper server with
171       * a custom {@code sessionTimeout}.
172       */
173      public final ZooKeeperClient createClient(Amount<Integer, Time> sessionTimeout,
174          Credentials credentials) {
175        final ZooKeeperClient client = new ZooKeeperClient(sessionTimeout, credentials,
176            InetSocketAddress.createUnresolved("127.0.0.1", port));
177        shutdownRegistry.addAction(new ExceptionalCommand<InterruptedException>() {
178          @Override public void execute() {
179            client.close();
180          }
181        });
182        return client;
183      }
184    
185      private void checkEphemeralPortAssigned() {
186        Preconditions.checkState(port > 0, "startNetwork must be called first");
187      }
188    
189      private File createTempDir() {
190        final File tempDir = FileUtils.createTempDir();
191        shutdownRegistry.addAction(new ExceptionalCommand<IOException>() {
192          @Override public void execute() throws IOException {
193            org.apache.commons.io.FileUtils.deleteDirectory(tempDir);
194          }
195        });
196        return tempDir;
197      }
198    }
