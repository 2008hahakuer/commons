047    /**
048     * Manages a connection to a ZooKeeper cluster.
049     *
050     * <p>TODO(John Sirois): evaluate the need for ACLs and consider moving to zkClient:
051     * http://github.com/phunt/zkclient
052     *
053     * @author John Sirois
054     */
055    public class ZooKeeperClient {
056    
057      /**
058       * Indicates an error connecting to a zookeeper cluster.
059       *
060       * @author John Sirois
061       */
062      public class ZooKeeperConnectionException extends Exception {
063        public ZooKeeperConnectionException(String message, Throwable cause) {
064          super(message, cause);
065        }
066      }
067    
068      private final class SessionState {
069        private final long sessionId;
070        private final byte[] sessionPasswd;
071    
072        private SessionState(long sessionId, byte[] sessionPasswd) {
073          this.sessionId = sessionId;
074          this.sessionPasswd = sessionPasswd;
075        }
076      }
077    
078      private static final Logger LOG = Logger.getLogger(ZooKeeperClient.class.getName());
079    
080      private static final Amount<Long,Time> WAIT_FOREVER = Amount.of(0L, Time.MILLISECONDS);
081    
082      private final int sessionTimeoutMs;
083      private final String zooKeeperServers;
084      private ZooKeeper zooKeeper;
085      private SessionState sessionState;
086      private boolean closed = true;
087    
088      private final Set<Watcher> watchers = Collections.synchronizedSet(new HashSet<Watcher>());
089    
090      /**
091       * Creates an unconnected client that will lazily attempt to connect on the first call to
092       * {@link #get()}.
093       *
094       * @param sessionTimeout the ZK session timeout
095       * @param zooKeeperServer the first, required ZK server
096       * @param zooKeeperServers any additional servers forming the ZK cluster
097       */
098      public ZooKeeperClient(Amount<Integer, Time> sessionTimeout, InetSocketAddress zooKeeperServer,
099          InetSocketAddress... zooKeeperServers) {
100        this(sessionTimeout, ImmutableSet.<InetSocketAddress>builder()
101            .add(Preconditions.checkNotNull(zooKeeperServer))
102            .addAll(ImmutableList.copyOf(zooKeeperServers)).build());
103      }
104    
105      /**
106       * Creates an unconnected client that will lazily attempt to connect on the first call to
107       * {@link #get}.
108       *
109       * @param sessionTimeout the ZK session timeout
110       * @param zooKeeperServers the set of servers forming the ZK cluster
111       */
112      public ZooKeeperClient(Amount<Integer, Time> sessionTimeout,
113          Iterable<InetSocketAddress> zooKeeperServers) {
114        this.sessionTimeoutMs = Preconditions.checkNotNull(sessionTimeout).as(Time.MILLISECONDS);
115        Preconditions.checkNotNull(zooKeeperServers);
116        Preconditions.checkArgument(!Iterables.isEmpty(zooKeeperServers),
117            "Must present at least 1 ZK server");
118    
119        Iterable<String> servers =
120            Iterables.transform(ImmutableSet.copyOf(zooKeeperServers),
121                InetSocketAddressHelper.INET_TO_STR);
122        this.zooKeeperServers = Joiner.on(',').join(servers);
123      }
124    
125      /**
126       * Returns the current active ZK connection or establishes a new one if none has yet been
127       * established or a previous connection was disconnected or had its session time out.  This method
128       * will attempt to re-use sessions when possible.  Equivalent to:
129       * <pre>get(Amount.of(0L, ...)</pre>.
130       *
131       * @return a connected ZooKeeper client
132       * @throws ZooKeeperConnectionException if there was a problem connecting to the ZK cluster
133       * @throws InterruptedException if interrupted while waiting for a connection to be established
134       * @throws IllegalStateException if this client has already been {@link #close() closed}
135       */
136      public synchronized ZooKeeper get() throws ZooKeeperConnectionException, InterruptedException {
137        try {
138          return get(WAIT_FOREVER);
139        } catch (TimeoutException e) {
140          InterruptedException interruptedException =
141              new InterruptedException("Got an unexpected TimeoutException for 0 wait");
142          interruptedException.initCause(e);
143          throw interruptedException;
144        }
145      }
146    
147      /**
148       * Returns the current active ZK connection or establishes a new one if none has yet been
149       * established or a previous connection was disconnected or had its session time out.  This
150       * method will attempt to re-use sessions when possible.
151       *
152       * @param connectionTimeout the maximum amount of time to wait for the connection to the ZK
153       *     cluster to be established; 0 to wait forever
154       * @return a connected ZooKeeper client
155       * @throws ZooKeeperConnectionException if there was a problem connecting to the ZK cluster
156       * @throws InterruptedException if interrupted while waiting for a connection to be established
157       * @throws TimeoutException if a connection could not be established within the configured
158       *     session timeout
159       */
160      public synchronized ZooKeeper get(Amount<Long, Time> connectionTimeout)
161          throws ZooKeeperConnectionException, InterruptedException, TimeoutException {
162    
163        if (zooKeeper == null) {
164          final CountDownLatch connected = new CountDownLatch(1);
165          Watcher watcher = new Watcher() {
166            @Override public void process(WatchedEvent event) {
167              switch (event.getType()) {
168                // Guard the None type since this watch may be used as the default watch on calls by
169                // the client outside our control.
170                case None:
171                  switch (event.getState()) {
172                    case Expired:
173                      LOG.info("Zookeeper session expired. Event: " + event);
174                      close();
175                      break;
176                    case SyncConnected:
177                      connected.countDown();
178                      break;
179                  }
180              }
181    
182              synchronized (watchers) {
183                for (Watcher watcher : watchers) {
184                  watcher.process(event);
185                }
186              }
187            }
188          };
189    
190          try {
191            zooKeeper = (sessionState != null)
192              ? new ZooKeeper(zooKeeperServers, sessionTimeoutMs, watcher, sessionState.sessionId,
193                sessionState.sessionPasswd)
194              : new ZooKeeper(zooKeeperServers, sessionTimeoutMs, watcher);
195          } catch (IOException e) {
196            throw new ZooKeeperConnectionException(
197                "Problem connecting to servers: " + zooKeeperServers, e);
198          }
199    
200          if (connectionTimeout.getValue() > 0) {
201            if(!connected.await(connectionTimeout.as(Time.MILLISECONDS), TimeUnit.MILLISECONDS)) {
202              throw new TimeoutException("Timed out waiting for a ZK connection after "
203                                         + sessionTimeoutMs + "ms");
204            }
205          } else {
206            connected.await();
207          }
208    
209          sessionState = new SessionState(zooKeeper.getSessionId(), zooKeeper.getSessionPasswd());
210          closed = false;
211        }
212        return zooKeeper;
213      }
214    
215      /**
216       * Clients that need to re-establish state after session expiration can register an
217       * {@code onExpired} command to execute.
218       *
219       * @param onExpired the {@code Command} to register
220       * @return the new {@link Watcher} which can later be passed to {@link #unregister} for
221       *     removal.
222       */
223      public Watcher registerExpirationHandler(final Command onExpired) {
224        Watcher watcher = new Watcher() {
225          @Override public void process(WatchedEvent event) {
226            if (event.getType() == EventType.None && event.getState() == KeeperState.Expired) {
227              onExpired.execute();
228            }
229          }
230        };
231        register(watcher);
232        return watcher;
233      }
234    
235      /**
236       * Clients that need to register a top-level {@code Watcher} should do so using this method.  The
237       * registered {@code watcher} will remain registered across re-connects and session expiration
238       * events.
239       *
240       * @param watcher the {@code Watcher to register}
241       */
242      public void register(Watcher watcher) {
243        watchers.add(watcher);
244      }
245    
246      /**
247       * Clients can attempt to unregister a top-level {@code Watcher} that has previously been
248       * registered.
249       *
250       * @param watcher the {@code Watcher} to unregister as a top-level, persistent watch
251       * @return whether the given {@code Watcher} was found and removed from the active set
252       */
253      public boolean unregister(Watcher watcher) {
254        return watchers.remove(watcher);
255      }
256    
257      /**
258       * Checks to see if the client might reasonably re-try an operation given the exception thrown
259       * while attempting it.  If the ZooKeeper session should be expired to enable the re-try to
260       * succeed this method will expire it as a side-effect.
261       *
262       * @param e the exception to test
263       * @return true if a retry can be attempted
264       */
265      public boolean shouldRetry(KeeperException e) {
266        if (e instanceof SessionExpiredException) {
267          close();
268        }
269        return ZooKeeperUtils.isRetryable(e);
270      }
271    
272      /**
273       * Closes the current connection if any expiring the current ZooKeeper session.  Any subsequent
274       * calls to this method will no-op until the next successful {@link #get}.
275       */
276      public synchronized void close() {
277        if (closed) {
278          return;
279        }
280    
281        closed = true;
282        if (zooKeeper != null) {
283          try {
284            zooKeeper.close();
285            zooKeeper = null;
286            sessionState = null;
287          } catch (InterruptedException e) {
288            Thread.currentThread().interrupt();
289            LOG.warning("Interrupted trying to close zooKeeper");
290          }
291        }
292      }
