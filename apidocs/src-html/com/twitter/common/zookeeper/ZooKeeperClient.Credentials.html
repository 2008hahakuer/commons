069      /**
070       * Encapsulates a user's credentials and has the ability to authenticate them through a
071       * {@link ZooKeeper} client.
072       */
073      public interface Credentials {
074    
075        /**
076         * A set of {@code Credentials} that performs no authentication.
077         */
078        Credentials NONE = new Credentials() {
079          @Override public void authenticate(ZooKeeper zooKeeper) {
080            // noop
081          }
082    
083          @Override public String scheme() {
084            return null;
085          }
086    
087          @Override public byte[] authToken() {
088            return null;
089          }
090        };
091    
092        /**
093         * Authenticates these credentials against the given {@code ZooKeeper} client.
094         *
095         * @param zooKeeper the client to authenticate
096         */
097        void authenticate(ZooKeeper zooKeeper);
098    
099        /**
100         * Returns the authentication scheme these credentials are for.
101         *
102         * @return the scheme these credentials are for or {@code null} if no authentication is
103         *     intended.
104         */
105        @Nullable
106        String scheme();
107    
108        /**
109         * Returns the authentication token.
110         *
111         * @return the authentication token or {@code null} if no authentication is intended.
112         */
113        @Nullable
114        byte[] authToken();
115      }
116    
117      /**
118       * Creates a set of credentials for the zoo keeper digest authentication mechanism.
119       *
120       * @param username the username to authenticate with
121       * @param password the password to authenticate with
122       * @return a set of credentials that can be used to authenticate the zoo keeper client
123       */
124      public static Credentials digestCredentials(String username, String password) {
125        MorePreconditions.checkNotBlank(username);
126        Preconditions.checkNotNull(password);
127    
128        // TODO(John Sirois): DigestAuthenticationProvider is broken - uses platform default charset
129        // (on server) and so we just have to hope here that clients are deployed in compatible jvms.
130        // Consider writing and installing a version of DigestAuthenticationProvider that controls its
131        // Charset explicitly.
132        return credentials("digest", (username + ":" + password).getBytes());
133      }
134    
135      /**
136       * Creates a set of credentials for the given authentication {@code scheme}.
137       *
138       * @param scheme the scheme to authenticate with
139       * @param authToken the authentication token
140       * @return a set of credentials that can be used to authenticate the zoo keeper client
141       */
142      public static Credentials credentials(final String scheme, final byte[] authToken) {
143        MorePreconditions.checkNotBlank(scheme);
144        Preconditions.checkNotNull(authToken);
145    
146        return new Credentials() {
147          @Override public void authenticate(ZooKeeper zooKeeper) {
148            zooKeeper.addAuthInfo(scheme, authToken);
149          }
150    
151          @Override public String scheme() {
152            return scheme;
153          }
154    
155          @Override public byte[] authToken() {
156            return authToken;
157          }
158        };
159      }

161      private final class SessionState {
162        private final long sessionId;
163        private final byte[] sessionPasswd;
164    
165        private SessionState(long sessionId, byte[] sessionPasswd) {
166          this.sessionId = sessionId;
167          this.sessionPasswd = sessionPasswd;
168        }
169      }

171      private static final Logger LOG = Logger.getLogger(ZooKeeperClient.class.getName());
172    
173      private static final Amount<Long,Time> WAIT_FOREVER = Amount.of(0L, Time.MILLISECONDS);
174    
175      private final int sessionTimeoutMs;
176      private final Credentials credentials;
177      private final String zooKeeperServers;
178      private ZooKeeper zooKeeper;
179      private SessionState sessionState;
180      private boolean closed = true;

182      private final Set<Watcher> watchers = Collections.synchronizedSet(new HashSet<Watcher>());
183    
184      private static Iterable<InetSocketAddress> combine(InetSocketAddress address,
185          InetSocketAddress... addresses) {
186        return ImmutableSet.<InetSocketAddress>builder().add(address).add(addresses).build();
187      }

189      /**
190       * Creates an unconnected client that will lazily attempt to connect on the first call to
191       * {@link #get()}.
192       *
193       * @param sessionTimeout the ZK session timeout
194       * @param zooKeeperServer the first, required ZK server
195       * @param zooKeeperServers any additional servers forming the ZK cluster
196       */
197      public ZooKeeperClient(Amount<Integer, Time> sessionTimeout, InetSocketAddress zooKeeperServer,
198          InetSocketAddress... zooKeeperServers) {
199        this(sessionTimeout, combine(zooKeeperServer, zooKeeperServers));
200      }

203       * Creates an unconnected client that will lazily attempt to connect on the first call to
204       * {@link #get}.
205       *
206       * @param sessionTimeout the ZK session timeout
207       * @param zooKeeperServers the set of servers forming the ZK cluster
208       */
209      public ZooKeeperClient(Amount<Integer, Time> sessionTimeout,
210          Iterable<InetSocketAddress> zooKeeperServers) {
211        this(sessionTimeout, Credentials.NONE, zooKeeperServers);
212      }

214      /**
215       * Creates an unconnected client that will lazily attempt to connect on the first call to
216       * {@link #get()}.  All successful connections will be authenticated with the given
217       * {@code credentials}.
218       *
219       * @param sessionTimeout the ZK session timeout
220       * @param credentials the credentials to authenticate with
221       * @param zooKeeperServer the first, required ZK server
222       * @param zooKeeperServers any additional servers forming the ZK cluster
223       */
224      public ZooKeeperClient(Amount<Integer, Time> sessionTimeout, Credentials credentials,
225          InetSocketAddress zooKeeperServer, InetSocketAddress... zooKeeperServers) {
226        this(sessionTimeout, credentials, combine(zooKeeperServer, zooKeeperServers));
227      }
228    
229      /**
230       * Creates an unconnected client that will lazily attempt to connect on the first call to
231       * {@link #get}.  All successful connections will be authenticated with the given
232       * {@code credentials}.
233       *
234       * @param sessionTimeout the ZK session timeout
235       * @param credentials the credentials to authenticate with
236       * @param zooKeeperServers the set of servers forming the ZK cluster
237       */
238      public ZooKeeperClient(Amount<Integer, Time> sessionTimeout, Credentials credentials,
239          Iterable<InetSocketAddress> zooKeeperServers) {
240        this.sessionTimeoutMs = Preconditions.checkNotNull(sessionTimeout).as(Time.MILLISECONDS);
241        this.credentials = Preconditions.checkNotNull(credentials);
242    
243        Preconditions.checkNotNull(zooKeeperServers);
244        Preconditions.checkArgument(!Iterables.isEmpty(zooKeeperServers),
245            "Must present at least 1 ZK server");
246    
247        Iterable<String> servers =
248            Iterables.transform(ImmutableSet.copyOf(zooKeeperServers),
249                InetSocketAddressHelper.INET_TO_STR);
250        this.zooKeeperServers = Joiner.on(',').join(servers);
251      }
252    
253      /**
254       * Returns the current active ZK connection or establishes a new one if none has yet been
255       * established or a previous connection was disconnected or had its session time out.  This method
256       * will attempt to re-use sessions when possible.  Equivalent to:
257       * <pre>get(Amount.of(0L, ...)</pre>.
258       *
259       * @return a connected ZooKeeper client
260       * @throws ZooKeeperConnectionException if there was a problem connecting to the ZK cluster
261       * @throws InterruptedException if interrupted while waiting for a connection to be established
262       * @throws IllegalStateException if this client has already been {@link #close() closed}
263       */
264      public synchronized ZooKeeper get() throws ZooKeeperConnectionException, InterruptedException {
265        try {
266          return get(WAIT_FOREVER);
267        } catch (TimeoutException e) {
268          InterruptedException interruptedException =
269              new InterruptedException("Got an unexpected TimeoutException for 0 wait");
270          interruptedException.initCause(e);
271          throw interruptedException;
272        }
273      }
274    
275      /**
276       * Returns the current active ZK connection or establishes a new one if none has yet been
277       * established or a previous connection was disconnected or had its session time out.  This
278       * method will attempt to re-use sessions when possible.
279       *
280       * @param connectionTimeout the maximum amount of time to wait for the connection to the ZK
281       *     cluster to be established; 0 to wait forever
282       * @return a connected ZooKeeper client
283       * @throws ZooKeeperConnectionException if there was a problem connecting to the ZK cluster
284       * @throws InterruptedException if interrupted while waiting for a connection to be established
285       * @throws TimeoutException if a connection could not be established within the configured
286       *     session timeout
287       */
288      public synchronized ZooKeeper get(Amount<Long, Time> connectionTimeout)
289          throws ZooKeeperConnectionException, InterruptedException, TimeoutException {
290    
291        if (zooKeeper == null) {
292          final CountDownLatch connected = new CountDownLatch(1);
293          Watcher watcher = new Watcher() {
294            @Override public void process(WatchedEvent event) {
295              switch (event.getType()) {
296                // Guard the None type since this watch may be used as the default watch on calls by
297                // the client outside our control.
298                case None:
299                  switch (event.getState()) {
300                    case Expired:
301                      LOG.info("Zookeeper session expired. Event: " + event);
302                      close();
303                      break;
304                    case SyncConnected:
305                      connected.countDown();
306                      break;
307                  }
308              }
309    
310              synchronized (watchers) {
311                for (Watcher watcher : watchers) {
312                  watcher.process(event);
313                }
314              }
315            }
316          };
317    
318          try {
319            zooKeeper = (sessionState != null)
320              ? new ZooKeeper(zooKeeperServers, sessionTimeoutMs, watcher, sessionState.sessionId,
321                sessionState.sessionPasswd)
322              : new ZooKeeper(zooKeeperServers, sessionTimeoutMs, watcher);
323          } catch (IOException e) {
324            throw new ZooKeeperConnectionException(
325                "Problem connecting to servers: " + zooKeeperServers, e);
326          }
327    
328          if (connectionTimeout.getValue() > 0) {
329            if (!connected.await(connectionTimeout.as(Time.MILLISECONDS), TimeUnit.MILLISECONDS)) {
330              close();
331              throw new TimeoutException("Timed out waiting for a ZK connection after "
332                                         + connectionTimeout);
333            }
334          } else {
335            connected.await();
336          }
337          credentials.authenticate(zooKeeper);
338    
339          sessionState = new SessionState(zooKeeper.getSessionId(), zooKeeper.getSessionPasswd());
340          closed = false;
341        }
342        return zooKeeper;
343      }
344    
345      /**
346       * Clients that need to re-establish state after session expiration can register an
347       * {@code onExpired} command to execute.
348       *
349       * @param onExpired the {@code Command} to register
350       * @return the new {@link Watcher} which can later be passed to {@link #unregister} for
351       *     removal.
352       */
353      public Watcher registerExpirationHandler(final Command onExpired) {
354        Watcher watcher = new Watcher() {
355          @Override public void process(WatchedEvent event) {
356            if (event.getType() == EventType.None && event.getState() == KeeperState.Expired) {
357              onExpired.execute();
358            }
359          }
360        };
361        register(watcher);
362        return watcher;
363      }
364    
365      /**
366       * Clients that need to register a top-level {@code Watcher} should do so using this method.  The
367       * registered {@code watcher} will remain registered across re-connects and session expiration
368       * events.
369       *
370       * @param watcher the {@code Watcher to register}
371       */
372      public void register(Watcher watcher) {
373        watchers.add(watcher);
374      }
375    
376      /**
377       * Clients can attempt to unregister a top-level {@code Watcher} that has previously been
378       * registered.
379       *
380       * @param watcher the {@code Watcher} to unregister as a top-level, persistent watch
381       * @return whether the given {@code Watcher} was found and removed from the active set
382       */
383      public boolean unregister(Watcher watcher) {
384        return watchers.remove(watcher);
385      }
386    
387      /**
388       * Checks to see if the client might reasonably re-try an operation given the exception thrown
389       * while attempting it.  If the ZooKeeper session should be expired to enable the re-try to
390       * succeed this method will expire it as a side-effect.
391       *
392       * @param e the exception to test
393       * @return true if a retry can be attempted
394       */
395      public boolean shouldRetry(KeeperException e) {
396        if (e instanceof SessionExpiredException) {
397          close();
398        }
399        return ZooKeeperUtils.isRetryable(e);
400      }
401    
402      /**
403       * Closes the current connection if any expiring the current ZooKeeper session.  Any subsequent
404       * calls to this method will no-op until the next successful {@link #get}.
405       */
406      public synchronized void close() {
407        if (closed) {
408          return;
409        }
410    
411        closed = true;
412        if (zooKeeper != null) {
413          try {
414            zooKeeper.close();
415            zooKeeper = null;
416            sessionState = null;
417          } catch (InterruptedException e) {
418            Thread.currentThread().interrupt();
419            LOG.warning("Interrupted trying to close zooKeeper");
420          }
421        }
422      }
423    
424      @VisibleForTesting
425      synchronized boolean isClosed() {
426        return closed;
427      }
