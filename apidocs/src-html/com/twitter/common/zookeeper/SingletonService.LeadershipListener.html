017    package com.twitter.common.zookeeper;
018    
019    import java.net.InetSocketAddress;
020    import java.util.Map;
021    import java.util.logging.Level;
022    import java.util.logging.Logger;
023    
024    import javax.annotation.Nullable;
025    
026    import com.google.common.annotations.VisibleForTesting;
027    import com.google.common.base.Preconditions;
028    
029    import org.apache.zookeeper.ZooDefs;
030    import org.apache.zookeeper.data.ACL;
031    
032    import com.twitter.common.base.ExceptionalCommand;
033    import com.twitter.common.zookeeper.Candidate.Leader;
034    import com.twitter.common.zookeeper.Group.JoinException;
035    import com.twitter.common.zookeeper.ServerSet.EndpointStatus;
036    import com.twitter.thrift.Status;
037    
038    /**
039     * A service that uses master election to only allow a single instance of the server to join
040     * the {@link ServerSet} at a time.
041     *
042     * @author William Farner
043     */
044    public class SingletonService {
045    
046      private static final Logger LOG = Logger.getLogger(SingletonService.class.getName());
047      private static final String LEADER_ELECT_NODE_PREFIX = "singleton_candidate_";
048    
049      private final ServerSet serverSet;
050      private final Candidate candidate;
051    
052      /**
053       * Equivalent to {@link #SingletonService(ZooKeeperClient, String, Iterable)} with a default
054       * wide open {@code acl} ({@link ZooDefs.Ids#OPEN_ACL_UNSAFE}).
055       */
056      public SingletonService(ZooKeeperClient zkClient, String servicePath) {
057        this(zkClient, servicePath, ZooDefs.Ids.OPEN_ACL_UNSAFE);
058      }
059    
060      /**
061       * Creates a new singleton service, identified by {@code servicePath}.  All nodes related to the
062       * service (for both leader election and service registration) will live under the path and each
063       * node will be created with the supplied {@code acl}. Internally, two ZooKeeper {@code Group}s
064       * are used to manage a singleton service - one for leader election, and another for the
065       * {@code ServerSet} where the leader's endpoints are registered.  Leadership election should
066       * guarantee that at most one instance will ever exist in the ServerSet at once.
067       *
068       * @param zkClient The ZooKeeper client to use.
069       * @param servicePath The path where service nodes live.
070       * @param acl The acl to apply to newly created candidate nodes and serverset nodes.
071       */
072      public SingletonService(ZooKeeperClient zkClient, String servicePath, Iterable<ACL> acl) {
073        this(new ServerSetImpl(zkClient, new Group(zkClient, acl, servicePath)),
074            new CandidateImpl(new Group(zkClient, acl, servicePath, LEADER_ELECT_NODE_PREFIX)));
075      }
076    
077      @VisibleForTesting
078      SingletonService(ServerSet serverSet, Candidate candidate) {
079        this.serverSet = Preconditions.checkNotNull(serverSet);
080        this.candidate = Preconditions.checkNotNull(candidate);
081      }
082    
083      /**
084       * Attempts to lead the singleton service.
085       *
086       * @param endpoint The primary endpoint to register as a leader candidate in the service.
087       * @param additionalEndpoints Additional endpoints that are available on the host.
088       * @param status Current status of the candidate.
089       * @param listener Handler to call when the candidate is elected or defeated.
090       * @throws Group.WatchException If there was a problem watching the ZooKeeper group.
091       * @throws Group.JoinException If there was a problem joining the ZooKeeper group.
092       * @throws InterruptedException If the thread watching/joining the group was interrupted.
093       */
094      public void lead(final InetSocketAddress endpoint,
095          final Map<String, InetSocketAddress> additionalEndpoints,
096          final Status status, final LeadershipListener listener)
097          throws Group.WatchException, Group.JoinException, InterruptedException {
098        Preconditions.checkNotNull(listener);
099    
100        candidate.offerLeadership(new Leader() {
101          private EndpointStatus endpointStatus = null;
102          @Override public void onElected(ExceptionalCommand<JoinException> abdicate) {
103            try {
104              endpointStatus = serverSet.join(endpoint, additionalEndpoints, status);
105              listener.onLeading(endpointStatus);
106            } catch (Group.JoinException e) {
107              LOG.log(Level.SEVERE, "Failed to join group.", e);
108            } catch (InterruptedException e) {
109              LOG.log(Level.SEVERE, "Interrupted while joining group.", e);
110              Thread.currentThread().interrupt();
111            }
112          }
113    
114          @Override public void onDefeated() {
115            listener.onDefeated(endpointStatus);
116          }
117        });
118      }
119    
120      public static interface LeadershipListener {
121        public void onLeading(EndpointStatus status);
122    
123        public void onDefeated(@Nullable EndpointStatus status);
124      }
125    }
