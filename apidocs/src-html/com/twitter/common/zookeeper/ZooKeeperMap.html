042    /**
043     * A ZooKeeper backed {@link Map}.  Initialized with a node path, this map represents child nodes
044     * under that path as keys, with the data in those nodes as values.  This map is readonly from
045     * clients of this class, and only can be modified via direct zookeeper operations.
046     *
047     * Note that instances of this class maintain a zookeeper watch for each zookeeper node under the
048     * parent, as well as on the parent itself.  Instances of this class should be created via the
049     * {@link #create} factory method.
050     *
051     * As of ZooKeeper Version 3.1, the maxiumum allowable size of a data node is 1 MB  A single
052     * client should be able to hold up to maintain several thousand watches, but this depends on rate
053     * of data change as well.
054     *
055     * Talk to your zookeeper cluster administrator if you expect number of map entries times number
056     * of live clients to exceed a thousand, as a zookeeper cluster is limited by total number of
057     * server-side watches enabled.
058     *
059     * For an example of a set of tools to maintain one of these maps, please see
060     * src/scripts/HenAccess.py in the hen repository.
061     *
062     * @param <V> the type of values this map stores
063     *
064     * @author Adam Samet
065     */
066    public class ZooKeeperMap<V> extends ForwardingMap<String, V> {
067    
068      /**
069       * Default deserializer for the constructor if you want to simply store the zookeeper byte[] data
070       * in this map.
071       */
072      public static final Function<byte[], byte[]> BYTE_ARRAY_VALUES = Functions.identity();
073    
074      private static final Logger LOG = Logger.getLogger(ZooKeeperMap.class.getName());
075    
076      private final ZooKeeperClient zkClient;
077      private final String nodePath;
078      private final Function<byte[], V> deserializer;
079    
080      private final ConcurrentMap<String, V> localMap;
081      private final BackoffHelper backoffHelper;
082    
083      // Whether it's safe to re-establish watches if our zookeeper session has expired.
084      private final Object safeToRewatchLock;
085      private volatile boolean safeToRewatch;
086    
087      /**
088       * Returns an initialized ZooKeeperMap.  The given path must exist at the time of
089       * creation or a {@link KeeperException} will be thrown.
090       *
091       * @param zkClient a zookeeper client
092       * @param nodePath path to a node whose data will be watched
093       * @param deserializer a function that converts byte[] data from a zk node to this map's
094       *     value type V
095       *
096       * @throws InterruptedException if the underlying zookeeper server transaction is interrupted
097       * @throws KeeperException.NoNodeException if the given nodePath doesn't exist
098       * @throws KeeperException if the server signals an error
099       * @throws ZooKeeperConnectionException if there was a problem connecting to the zookeeper
100       *     cluster
101       */
102      public static <V> ZooKeeperMap<V> create(ZooKeeperClient zkClient, String nodePath,
103          Function<byte[], V> deserializer) throws InterruptedException, KeeperException,
104          ZooKeeperConnectionException {
105        ZooKeeperMap<V> zkMap = new ZooKeeperMap<V>(zkClient, nodePath, deserializer);
106        zkMap.init();
107        return zkMap;
108      }
109    
110      /**
111       * Initializes a ZooKeeperMap.  The given path must exist at the time of object creation or
112       * a {@link KeeperException} will be thrown.
113       *
114       * Please note that this object will not track any remote zookeeper data until {@link #init()}
115       * is successfully called.  After construction and before that call, this {@link Map} will
116       * be empty.
117       *
118       * @param zkClient a zookeeper client
119       * @param nodePath top-level node path under which the map data lives
120       * @param deserializer a function that converts byte[] data from a zk node to this map's
121       *     value type V
122       *
123       * @throws InterruptedException if the underlying zookeeper server transaction is interrupted
124       * @throws KeeperException.NoNodeException if the given nodePath doesn't exist
125       * @throws KeeperException if the server signals an error
126       * @throws ZooKeeperConnectionException if there was a problem connecting to the zookeeper
127       *     cluster
128       */
129      @VisibleForTesting
130      ZooKeeperMap(ZooKeeperClient zkClient, String nodePath,
131          Function<byte[], V> deserializer) throws InterruptedException, KeeperException,
132          ZooKeeperConnectionException {
133        super();
134        this.zkClient = Preconditions.checkNotNull(zkClient);
135        this.nodePath = MorePreconditions.checkNotBlank(nodePath);
136        this.deserializer = Preconditions.checkNotNull(deserializer);
137    
138        localMap = new ConcurrentHashMap<String, V>();
139        backoffHelper = new BackoffHelper();
140        safeToRewatchLock = new Object();
141        safeToRewatch = false;
142    
143        if (zkClient.get().exists(nodePath, null) == null) {
144          throw new KeeperException.NoNodeException();
145        }
146      }
147    
148      /**
149       * Initialize zookeeper tracking for this {@link Map}.  Once this call returns, this object
150       * will be tracking data in zookeeper.
151       *
152       * @throws InterruptedException if the underlying zookeeper server transaction is interrupted
153       * @throws KeeperException if the server signals an error
154       * @throws ZooKeeperConnectionException if there was a problem connecting to the zookeeper
155       *     cluster
156       */
157      @VisibleForTesting
158      void init() throws InterruptedException, KeeperException,
159          ZooKeeperConnectionException {
160        Watcher watcher = zkClient.registerExpirationHandler(new Command() {
161          @Override public void execute() {
162            /*
163             * First rewatch all of our locally cached children.  Some of them may not exist anymore,
164             * which will lead to caught KeeperException.NoNode whereafter we'll remove that child
165             * from the cached map.
166             *
167             * Next, we'll establish our top level child watch and add any new nodes that might exist.
168             */
169            try {
170              synchronized (safeToRewatchLock) {
171                if (safeToRewatch) {
172                  rewatchDataNodes();
173                  tryWatchChildren();
174                }
175              }
176            } catch (InterruptedException e) {
177              LOG.log(Level.WARNING, "Interrupted while trying to re-establish watch.", e);
178            }
179          }
180        });
181    
182        try {
183          // Synchronize to prevent the race of watchChildren completing and then the session expiring
184          // before we update safeToRewatch.
185          synchronized (safeToRewatchLock) {
186            watchChildren();
187            safeToRewatch = true;
188          }
189        } catch (InterruptedException e) {
190          zkClient.unregister(watcher);
191          throw e;
192        } catch (KeeperException e) {
193          zkClient.unregister(watcher);
194          throw e;
195        } catch (ZooKeeperConnectionException e) {
196          zkClient.unregister(watcher);
197          throw e;
198        }
199      }
200    
201      @Override
202      protected Map<String, V> delegate() {
203        return localMap;
204      }

238      private void tryWatchChildren() throws InterruptedException {
239        backoffHelper.doUntilSuccess(new ExceptionalSupplier<Boolean, InterruptedException>() {
240          @Override public Boolean get() throws InterruptedException {
241            try {
242              watchChildren();
243              return true;
244            } catch (KeeperException e) {
245              return false;
246            } catch (ZooKeeperConnectionException e) {
247              return false;
248            }
249          }
250        });
251      }
252    
253      private synchronized void watchChildren() throws InterruptedException, KeeperException,
254          ZooKeeperConnectionException {
255        /*
256         * Add a watch on the parent node itself, and attempt to rewatch if it
257         * gets deleted
258         */
259        zkClient.get().exists(nodePath, new Watcher() {
260          @Override public void process(WatchedEvent event) {
261            if (event.getType() == Watcher.Event.EventType.NodeDeleted) {
262              // If the parent node no longer exists
263              localMap.clear();
264              try {
265                tryWatchChildren();
266              } catch (InterruptedException e) {
267                LOG.log(Level.WARNING, "Interrupted while trying to watch children.", e);
268              }
269            }
270          }});
271    
272        final Watcher childWatcher = new Watcher() {
273          @Override
274          public void process(WatchedEvent event) {
275            if (event.getType() == Watcher.Event.EventType.NodeChildrenChanged) {
276              try {
277                tryWatchChildren();
278              } catch (InterruptedException e) {
279                LOG.log(Level.WARNING, "Interrupted while trying to watch children.", e);
280              }
281            }
282          }
283        };
284    
285        List<String> children = zkClient.get().getChildren(nodePath, childWatcher);
286        updateChildren(Sets.newHashSet(children));
287      }
288    
289      private void tryAddChild(final String child) throws InterruptedException{
290        backoffHelper.doUntilSuccess(new ExceptionalSupplier<Boolean, InterruptedException>() {
291          @Override public Boolean get() throws InterruptedException {
292            try {
293              addChild(child);
294              return true;
295            } catch (KeeperException e) {
296              return false;
297            } catch (ZooKeeperConnectionException e) {
298              return false;
299            }
300          }
301        });
302      }
303    
304      // TODO(Adam Samet) - Make this use the ZooKeeperNode class.
305      private void addChild(final String child) throws InterruptedException, KeeperException,
306          ZooKeeperConnectionException {
307        final Watcher nodeWatcher = new Watcher() {
308          @Override
309          public void process(WatchedEvent event) {
310            if (event.getType() == Watcher.Event.EventType.NodeDataChanged) {
311              try {
312                tryAddChild(child);
313              } catch (InterruptedException e) {
314                LOG.log(Level.WARNING, "Interrupted while trying to add a child.", e);
315              }
316            } else if (event.getType() == Watcher.Event.EventType.NodeDeleted) {
317              removeEntry(child);
318            }
319          }
320        };
321    
322        try {
323          V value = deserializer.apply(zkClient.get().getData(makePath(child), nodeWatcher, null));
324          putEntry(child, value);
325        } catch (KeeperException.NoNodeException e) {
326          // This node doesn't exist anymore, remove it from the map and we're done.
327          removeEntry(child);
328        }
329      }
330    
331      @VisibleForTesting
332      void removeEntry(String key) {
333        localMap.remove(key);
334      }
335    
336      @VisibleForTesting
337      void putEntry(String key, V value) {
338        localMap.put(key, value);
339      }
340    
341      private void rewatchDataNodes() throws InterruptedException {
342        for (String child : keySet()) {
343          tryAddChild(child);
344        }
345      }
346    
347      private String makePath(final String child) {
348        return nodePath + "/" + child;
349      }
350    
351      private void updateChildren(Set<String> zkChildren) throws InterruptedException {
352        Set<String> addedChildren = Sets.difference(zkChildren, keySet());
353        Set<String> removedChildren = Sets.difference(keySet(), zkChildren);
354        for (String child : addedChildren) {
355          tryAddChild(child);
356        }
357        for (String child : removedChildren) {
358          removeEntry(child);
359        }
360      }
361    }
