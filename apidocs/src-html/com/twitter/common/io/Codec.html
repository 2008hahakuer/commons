017    package com.twitter.common.io;
018    
019    import java.io.IOException;
020    import java.io.InputStream;
021    import java.io.OutputStream;
022    
023    /**
024     * A Codec represents a reversible encoding for a given type.  Codecs are able to both
025     * {@link #deserialize(java.io.InputStream) read} items from streams and
026     * {@link #serialize(Object, java.io.OutputStream) write} items to streams.
027     *
028     * <p> TODO(John Sirois): consider whether this interface should optionally support null items to be
029     * read and written.
030     *
031     * @param <T> The type of object the Codec can handle.
032     *
033     * @author John Sirois
034     */
035    public interface Codec<T> {
036    
037      /**
038       * Writes a representation of {@code item} to the {@code sink} that can be read back by
039       * {@link #deserialize(java.io.InputStream)}.
040       *
041       * @param item the item to serialize
042       * @param sink the stream to write the item out to
043       * @throws IOException if there is a problem serializing the item
044       */
045      void serialize(T item, OutputStream sink) throws IOException;
046    
047      /**
048       * Reads an item from the {@code source} stream that was written by
049       * {@link #serialize(Object, java.io.OutputStream)}.
050       *
051       * @param source the stream to read an item from
052       * @return the deserialized item
053       * @throws IOException if there is a problem reading an item
054       */
055      T deserialize(InputStream source) throws IOException;
056    }
