018    package com.twitter.common.io;
019    
020    import java.io.IOException;
021    import java.io.InputStream;
022    import java.io.OutputStream;
023    
024    /**
025     * A Codec represents a reversible encoding for a given type.  Codecs are able to both
026     * {@link #deserialize(java.io.InputStream) read} items from streams and
027     * {@link #serialize(Object, java.io.OutputStream) write} items to streams.
028     *
029     * <p> TODO(John Sirois): consider whether this interface should optionally support null items to be
030     * read and written.
031     *
032     * @param <T> The type of object the Codec can handle.
033     *
034     * @author John Sirois
035     */
036    public interface Codec<T> {
037    
038      /**
039       * Writes a representation of {@code item} to the {@code sink} that can be read back by
040       * {@link #deserialize(java.io.InputStream)}.
041       *
042       * @param item the item to serialize
043       * @param sink the stream to write the item out to
044       * @throws IOException if there is a problem serializing the item
045       */
046      void serialize(T item, OutputStream sink) throws IOException;
047    
048      /**
049       * Reads an item from the {@code source} stream that was written by
050       * {@link #serialize(Object, java.io.OutputStream)}.
051       *
052       * @param source the stream to read an item from
053       * @return the deserialized item
054       * @throws IOException if there is a problem reading an item
055       */
056      T deserialize(InputStream source) throws IOException;
057    }
