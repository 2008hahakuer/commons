018    package com.twitter.common.io;
019    
020    import org.apache.commons.lang.SystemUtils;
021    
022    import java.io.File;
023    import java.util.UUID;
024    
025    /**
026     * Utility methods for working with files and directories.
027     *
028     * @author John Sirois
029     */
030    public final class FileUtils {
031      private static final int MAX_TMP_DIR_TRIES = 5;
032    
033      /**
034       * Returns a new empty temporary directory.
035       */
036      public static File createTempDir() {
037        File tempDir;
038        int tries = 0;
039        do {
040          // For sanity sake, die eventually if we keep failing to pick a new unique directory name.
041          if (++tries > MAX_TMP_DIR_TRIES) {
042            throw new IllegalStateException("Failed to create a new temp directory in "
043                                            + MAX_TMP_DIR_TRIES + " attempts, giving up");
044          }
045          tempDir = new File(SystemUtils.getJavaIoTmpDir(), UUID.randomUUID().toString());
046        } while (!tempDir.mkdir());
047        return tempDir;
048      }
049    
050      private FileUtils() {
051        // utility
052      }
053    }
