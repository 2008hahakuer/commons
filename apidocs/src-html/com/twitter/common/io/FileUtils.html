017    package com.twitter.common.io;
018    
019    import org.apache.commons.lang.SystemUtils;
020    
021    import java.io.File;
022    import java.util.UUID;
023    
024    /**
025     * Utility methods for working with files and directories.
026     *
027     * @author John Sirois
028     */
029    public final class FileUtils {
030      private static final int MAX_TMP_DIR_TRIES = 5;
031    
032      /**
033       * Returns a new empty temporary directory.
034       */
035      public static File createTempDir() {
036        File tempDir;
037        int tries = 0;
038        do {
039          // For sanity sake, die eventually if we keep failing to pick a new unique directory name.
040          if (++tries > MAX_TMP_DIR_TRIES) {
041            throw new IllegalStateException("Failed to create a new temp directory in "
042                                            + MAX_TMP_DIR_TRIES + " attempts, giving up");
043          }
044          tempDir = new File(SystemUtils.getJavaIoTmpDir(), UUID.randomUUID().toString());
045        } while (!tempDir.mkdir());
046        return tempDir;
047      }
048    
049      private FileUtils() {
050        // utility
051      }
052    }
