018    package com.twitter.common.io;
019    
020    import com.google.common.base.Preconditions;
021    import com.google.common.base.Predicate;
022    import com.google.common.base.Predicates;
023    import com.twitter.common.base.Closure;
024    
025    import java.io.EOFException;
026    import java.io.IOException;
027    import java.io.ObjectInputStream;
028    import java.io.Serializable;
029    import java.util.logging.Level;
030    import java.util.logging.Logger;
031    
032    /**
033     * A streamer that reads from serialized files.
034     *
035     * @author Gilad Mishne
036     */
037    public class SerializedFileStreamer<T extends Serializable> implements Streamer<T> {
038      private static final Logger LOG = Logger.getLogger(SerializedFileStreamer.class.getName());
039    
040      private final Iterable<? extends ObjectInputStream> inputs;
041      private Predicate<T> filter = Predicates.alwaysTrue();
042      private Predicate<T> endCondition = Predicates.alwaysFalse();
043    
044      /**
045       * Returns a streamer that will deserialize objects of type T from a set of files in the order
046       * the files are given.
047       *
048       * @param inputs The input streams to read from.
049       */
050      public SerializedFileStreamer (Iterable<? extends ObjectInputStream> inputs) {
051        this.inputs = Preconditions.checkNotNull(inputs);
052      }
053    
054      /**
055       * Deserialize {@code inputStream}, executing {@code work} on each object.
056       */
057      private void process(ObjectInputStream inputStream, Closure<T> work) {
058        Object o;
059        int count = 0;
060        try {
061          while ((o = inputStream.readObject()) != null) {
062            try {
063              T t = (T) o;
064              if (endCondition.apply(t)) {
065                break;
066              }
067              if (filter.apply(t)) {
068                work.execute(t);
069                ++count;
070              }
071            } catch (ClassCastException e) {
072              LOG.log(Level.SEVERE, "Unexpected object " + o.toString() + " in stream: " , e);
073            }
074          }
075        } catch (EOFException e) {
076          LOG.log(Level.INFO, "Read " + count + " objects from stream.");
077        } catch (ClassNotFoundException e) {
078          LOG.log(Level.SEVERE, "Unexpected class in stream.");
079          throw new RuntimeException(e);
080        } catch (IOException e) {
081          LOG.log(Level.SEVERE, "Error reading from stream.");
082          throw new RuntimeException(e);
083        } catch (NullPointerException e) {
084          LOG.log(Level.SEVERE, "Unexpected data in status file stream.");
085          throw new RuntimeException(e);
086        } finally {
087          try {
088            inputStream.close();
089          } catch (IOException e) {
090            LOG.log(Level.SEVERE, "Can't close stream.", e);
091          }
092        }
093      }
094    
095      @Override
096      public Streamer<T> endOn(Predicate<T> cond) {
097        Preconditions.checkNotNull(cond);
098        endCondition = Predicates.or(cond);
099        return this;
100      }
101    
102      @Override public void process(Closure<T> work) {
103        Preconditions.checkNotNull(work);
104        for (ObjectInputStream input : inputs) {
105          process(input, work);
106        }
107      }
108    
109      @Override public Streamer<T> filter(Predicate<T> filter) {
110        this.filter = Preconditions.checkNotNull(filter);
111        return this;
112      }
113    }
