018    package com.twitter.common.io;
019    
020    import com.google.common.base.Predicate;
021    import com.twitter.common.base.Closure;
022    
023    /**
024     * Encapsulates iteration over a typed data stream that can be filtered.
025     *
026     * @author John Sirois
027     */
028    public interface Streamer<T> {
029    
030      /**
031       * Processes a stream fully.  This may cause a database query to be executed, a file to be read
032       * or even just call {@link Iterable#iterator()} depending on the implementation.  Implementations
033       * guaranty that any resources allocated opening the stream will be closed whether or not process
034       * completes normally.
035       *
036       * @param work a closure over the work to be done for each item in the stream.
037       */
038      void process(Closure<T> work);
039    
040      /**
041       * Returns a {@code Streamer} that will process the same stream as this streamer, but will stop
042       * processing when encountering the first item for which {@code cond} is true.
043       *
044       * @param cond a predicate that returns {@code false} as long as the stream should keep being
045       *     processed.
046       * @return a streamer that will process items until the condition triggers.
047       */
048      Streamer<T> endOn(Predicate<T> cond);
049    
050      /**
051       * Returns a {@code Streamer} that will process the same stream as this streamer, but with any
052       * items failing the filter to be omitted from processing.
053       * @param filter a predicate that returns {@code true} if an item in the stream should be
054       *     processed
055       * @return a filtered streamer
056       */
057      Streamer<T> filter(Predicate<T> filter);
058    }
