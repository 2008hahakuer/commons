017    package com.twitter.common.inject;
018    
019    import java.lang.annotation.ElementType;
020    import java.lang.annotation.Retention;
021    import java.lang.annotation.RetentionPolicy;
022    import java.lang.annotation.Target;
023    import java.lang.reflect.Method;
024    
025    import com.google.common.base.Preconditions;
026    import com.google.common.cache.CacheBuilder;
027    import com.google.common.cache.CacheLoader;
028    import com.google.common.cache.LoadingCache;
029    import com.google.inject.Binder;
030    import com.google.inject.matcher.Matchers;
031    
032    import org.aopalliance.intercept.MethodInterceptor;
033    import org.aopalliance.intercept.MethodInvocation;
034    import org.apache.commons.lang.StringUtils;
035    
036    import com.twitter.common.stats.SlidingStats;
037    import com.twitter.common.stats.TimeSeriesRepository;
038    
039    /**
040     * A method interceptor that exports timing information for methods annotated with
041     * {@literal @Timed}.
042     *
043     * @author John Sirois
044     */
045    public final class TimedInterceptor implements MethodInterceptor {
046    
047      /**
048       * Marks a method as a target for timing.
049       */
050      @Retention(RetentionPolicy.RUNTIME)
051      @Target(ElementType.METHOD)
052      public @interface Timed {
053    
054        /**
055         * The base name to export timing data with; empty to use the annotated method's name.
056         */
057        String value() default "";
058      }
059    
060      private final LoadingCache<Method, SlidingStats> stats =
061          CacheBuilder.newBuilder().build(new CacheLoader<Method, SlidingStats>() {
062            @Override public SlidingStats load(Method method) {
063              return createStats(method);
064            }
065          });
066    
067      private TimedInterceptor() {
068        // preserve for guice
069      }
070    
071      private SlidingStats createStats(Method method) {
072        Timed timed = method.getAnnotation(Timed.class);
073        Preconditions.checkArgument(timed != null,
074            "TimedInterceptor can only be applied to @Timed methods");
075    
076        String name = timed.value();
077        String statName = !StringUtils.isEmpty(name) ? name : method.getName();
078        return new SlidingStats(statName, "nanos");
079      }
080    
081      @Override
082      public Object invoke(MethodInvocation methodInvocation) throws Throwable {
083        // TODO(John Sirois): consider including a SlidingRate tracking thrown exceptions
084        SlidingStats stat = stats.get(methodInvocation.getMethod());
085        long start = System.nanoTime();
086        try {
087          return methodInvocation.proceed();
088        } finally {
089          stat.accumulate(System.nanoTime() - start);
090        }
091      }
092    
093      /**
094       * Installs an interceptor in a guice {@link com.google.inject.Injector}, enabling
095       * {@literal @Timed} method interception in guice-provided instances.  Requires that a
096       * {@link TimeSeriesRepository} is bound elsewhere.
097       *
098       * @param binder a guice binder to require bindings against
099       */
100      public static void bind(Binder binder) {
101        Preconditions.checkNotNull(binder);
102    
103        Bindings.requireBinding(binder, TimeSeriesRepository.class);
104    
105        TimedInterceptor interceptor = new TimedInterceptor();
106        binder.requestInjection(interceptor);
107        binder.bindInterceptor(Matchers.any(), Matchers.annotatedWith(Timed.class), interceptor);
108      }
109    }
