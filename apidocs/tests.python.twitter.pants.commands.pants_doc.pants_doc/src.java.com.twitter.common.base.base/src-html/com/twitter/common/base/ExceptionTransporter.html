017    package com.twitter.common.base;
018    
019    import com.google.common.base.Function;
020    
021    /**
022     * A utility for transporting checked exceptions across boundaries that do not allow for checked
023     * exception propagation.
024     *
025     * @param <E> The type of checked exception the ExceptionTransported can transport
026     *
027     * @author John Sirois
028     */
029    public class ExceptionTransporter<E extends Exception> {
030    
031      /**
032       * An exception wrapper used to transport checked exceptions.  Never leaves an
033       * {@link ExceptionTransporter#guard(com.google.common.base.Function)} call.
034       */
035      private static final class TransportingException extends RuntimeException {
036        private TransportingException(Exception cause) {
037          super("It is a usage error to see this message!", cause);
038        }
039      }
040    
041      /**
042       * Guards a unit of work that internally can generate checked exceptions.  Callers wrap up the
043       * work in a function that rethrows any checked exceptions using the supplied
044       * ExceptionTransporter.  Guard will ensure the original exception is unwrapped an re-thrown.
045       *
046       * @param work The unit of work that guards its checked exceptions.
047       * @param <T> The type returned by the unit of work when it successfully completes.
048       * @param <X> The type of checked exception that the unit of work wishes to guard.
049       * @return the result of the unit of work if no excpetions are thrown
050       * @throws X when the unit of work uses the ExceptionTransporter to throw a checked exception
051       */
052      public static <T, X extends Exception> T guard(Function<ExceptionTransporter<X>, T> work)
053        throws X {
054    
055        try {
056          return work.apply(new ExceptionTransporter<X>());
057        } catch (TransportingException e) {
058          @SuppressWarnings("unchecked")
059          X cause = (X) e.getCause();
060          throw cause;
061        }
062      }
063    
064      /**
065       * Throws the given {@code checked} exception across a boundary that does not allow checked
066       * exceptions.  Although a RuntimeException is returned by this method signature, the method never
067       * actually completes normally.  The return type does allow the following usage idiom however:
068       * <pre>
069       * public String apply(ExceptionTransporter transporter) {
070       *   try {
071       *     return doChecked();
072       *   } catch (CheckedException e) {
073       *     // Although transport internally throws and does not return, we satisfy the compiler that
074       *     // our method returns a value or throws by pretending to throw the RuntimeException that
075       *     // never actually gets returned by transporter.transport(...)
076       *     throw transporter.transport(e);
077       *   }
078       * }
079       * </pre>
080       *
081       * @param checked The checked exception to transport.
082       * @return A RuntimeException that can be thrown to satisfy the compiler at the call site
083       */
084      public RuntimeException transport(E checked) {
085        throw new TransportingException(checked);
086      }
087    }
